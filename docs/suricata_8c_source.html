313static void SignalHandlerUnexpected(int sig_num, siginfo_t *info, void *context)
314{
315 char msg[SC_LOG_MAX_LOG_MSG_LEN];
316 unw_cursor_t cursor;
317 /* Restore defaults for signals to avoid loops */
318 signal(SIGABRT, SIG_DFL);
319 signal(SIGSEGV, SIG_DFL);
320 int r;
321 if ((r = unw_init_local(&cursor, (unw_context_t *)(context)) != 0)) {
322 fprintf(stderr, "unable to obtain stack trace: unw_init_local: %s\n", unw_strerror(r));
323 goto terminate;
324 }
325
326 char *temp = msg;
327 int cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - msg), "stacktrace:sig %d:", sig_num);
328 temp += cw;
329 r = 1;
330 while (r > 0) {
331 if (unw_is_signal_frame(&cursor) == 0) {
332 unw_word_t off;
333 char name[256];
334 if (unw_get_proc_name(&cursor, name, sizeof(name), &off) == UNW_ENOMEM) {
335 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - msg), "[unknown]:");
336 } else {
337 cw = snprintf(
338 temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - msg), "%s+0x%08" PRIx64, name, off);
339 }
340 temp += cw;
341 }
342
343 r = unw_step(&cursor);
344 if (r > 0) {
345 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - msg), ";");
346 temp += cw;
347 }
348 }
349 SCLogError(SC_ERR_SIGNAL, "%s", msg);
350
351terminate:
352 // Propagate signal to watchers, if any
353 kill(getpid(), sig_num);
354}
355#undef UNW_LOCAL_ONLY
356#endif /* HAVE_LIBUNWIND */
357#endif /* !OS_WIN32 */
358#endif

472static int SetBpfString(int argc, char *argv[])
473{
474 char *bpf_filter = NULL;
475 uint32_t bpf_len = 0;
476 int tmpindex = 0;
477
478 /* attempt to parse remaining args as bpf filter */
479 tmpindex = argc;
480 while(argv[tmpindex] != NULL) {
481 bpf_len+=strlen(argv[tmpindex]) + 1;
482 tmpindex++;
483 }
484
485 if (bpf_len == 0)
486 return TM_ECODE_OK;
487
488 if (EngineModeIsIPS()) {
489 SCLogError(SC_ERR_NOT_SUPPORTED,
490 "BPF filter not available in IPS mode."
491 " Use firewall filtering if possible.");
492 return TM_ECODE_FAILED;
493 }
494
495 bpf_filter = SCMalloc(bpf_len);
496 if (unlikely(bpf_filter == NULL))
497 return TM_ECODE_OK;
498 memset(bpf_filter, 0x00, bpf_len);
499
500 tmpindex = optind;
501 while(argv[tmpindex] != NULL) {
502 strlcat(bpf_filter, argv[tmpindex],bpf_len);
503 if(argv[tmpindex + 1] != NULL) {
504 strlcat(bpf_filter," ", bpf_len);
505 }
506 tmpindex++;
507 }
508
509 if(strlen(bpf_filter) > 0) {
510 if (ConfSetFinal("bpf-filter", bpf_filter) != 1) {
511 SCLogError(SC_ERR_FATAL, "Failed to set bpf filter.");
512 SCFree(bpf_filter);
513 return TM_ECODE_FAILED;
514 }
515 }
516 SCFree(bpf_filter);
517
518 return TM_ECODE_OK;
519}

521static void SetBpfStringFromFile(char *filename)
522{
523 char *bpf_filter = NULL;
524 char *bpf_comment_tmp = NULL;
525 char *bpf_comment_start = NULL;
526 uint32_t bpf_len = 0;
527#ifdef OS_WIN32
528 struct _stat st;
529#else
530 struct stat st;
531#endif /* OS_WIN32 */
532 FILE *fp = NULL;
533 size_t nm = 0;
534
535 if (EngineModeIsIPS()) {
536 FatalError(SC_ERR_FATAL,
537 "BPF filter not available in IPS mode."
538 " Use firewall filtering if possible.");
539 }
540
541 fp = fopen(filename, "r");
542 if (fp == NULL) {
543 SCLogError(SC_ERR_FOPEN, "Failed to open file %s", filename);
544 exit(EXIT_FAILURE);
545 }
546
547#ifdef OS_WIN32
548 if (_fstat(_fileno(fp), &st) != 0) {
549#else
550 if (fstat(fileno(fp), &st) != 0) {
551#endif /* OS_WIN32 */
552 SCLogError(SC_ERR_FOPEN, "Failed to stat file %s", filename);
553 exit(EXIT_FAILURE);
554 }
555 bpf_len = st.st_size + 1;
556
557 bpf_filter = SCMalloc(bpf_len * sizeof(char));
558 if (unlikely(bpf_filter == NULL)) {
559 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate buffer for bpf filter in file %s", filename);
560 exit(EXIT_FAILURE);
561 }
562 memset(bpf_filter, 0x00, bpf_len);
563
564 nm = fread(bpf_filter, 1, bpf_len - 1, fp);
565 if ((ferror(fp) != 0) || (nm != (bpf_len - 1))) {
566 SCLogError(SC_ERR_BPF, "Failed to read complete BPF file %s", filename);
567 SCFree(bpf_filter);
568 fclose(fp);
569 exit(EXIT_FAILURE);
570 }
571 fclose(fp);
572 bpf_filter[nm] = '\0';
573
574 if(strlen(bpf_filter) > 0) {
575 /*replace comments with space*/
576 bpf_comment_start = bpf_filter;
577 while((bpf_comment_tmp = strchr(bpf_comment_start, '#')) != NULL) {
578 while((*bpf_comment_tmp !='\0') &&
579 (*bpf_comment_tmp != '\r') && (*bpf_comment_tmp != '\n'))
580 {
581 *bpf_comment_tmp++ = ' ';
582 }
583 bpf_comment_start = bpf_comment_tmp;
584 }
585 /*remove remaining '\r' and '\n' */
586 while((bpf_comment_tmp = strchr(bpf_filter, '\r')) != NULL) {
587 *bpf_comment_tmp = ' ';
588 }
589 while((bpf_comment_tmp = strchr(bpf_filter, '\n')) != NULL) {
590 *bpf_comment_tmp = ' ';
591 }
592 /* cut trailing spaces */
593 while (strlen(bpf_filter) > 0 &&
594 bpf_filter[strlen(bpf_filter)-1] == ' ')
595 {
596 bpf_filter[strlen(bpf_filter)-1] = '\0';
597 }
598 if (strlen(bpf_filter) > 0) {
599 if(ConfSetFinal("bpf-filter", bpf_filter) != 1) {
600 SCLogError(SC_ERR_FOPEN, "ERROR: Failed to set bpf filter!");
601 SCFree(bpf_filter);
602 exit(EXIT_FAILURE);
603 }
604 }
605 }
606 SCFree(bpf_filter);
607}
