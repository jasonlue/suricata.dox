18/**
19 * \file
20 *
21 * \author Victor Julien <victor@inliniac.net>
22 *
23 * Prefilter engine
24 *
25 * Prefilter engines have as purpose to check for a critical common part of
26 * a set of rules. If the condition is present in the traffic, the rules
27 * will have to be inspected individually. Otherwise, the rules can be
28 * skipped.
29 *
30 * The best example of this is the MPM. From each rule take a pattern and
31 * add it to the MPM state machine. Inspect that in one step and only
32 * individually inspect the rules that had a match in MPM.
33 *
34 * This prefilter API is designed to abstract this logic so that it becomes
35 * easier to add other types of prefilters.
36 *
37 * The prefilter engines are structured as a simple list of engines. Each
38 * engine checks for a condition using it's callback function and private
39 * data. It then adds the rule match candidates to the PrefilterRuleStore
40 * structure.
41 *
42 * After the engines have run the resulting list of match candidates is
43 * sorted by the rule id's so that the individual inspection happens in
44 * the correct order.
45 */

66static inline void QuickSortSigIntId(SigIntId *sids, uint32_t n)
67{
68 if (n < 2)
69 return;
70 SigIntId p = sids[n / 2];
71 SigIntId *l = sids;
72 SigIntId *r = sids + n - 1;
73 while (l <= r) {
74 if (*l < p)
75 l++;
76 else if (*r > p)
77 r--;
78 else {
79 SigIntId t = *l;
80 *l = *r;
81 *r = t;
82 l++;
83 r--;
84 }
85 }
86 QuickSortSigIntId(sids, r - sids + 1);
87 QuickSortSigIntId(l, sids + n - l);
88}

90/**
91 * \brief run prefilter engines on a transaction
92 */
93void DetectRunPrefilterTx(DetectEngineThreadCtx *det_ctx,
94 const SigGroupHead *sgh,
95 Packet *p,
96 const uint8_t ipproto,
97 const uint8_t flow_flags,
98 const AppProto alproto,
99 void *alstate,
100 DetectTransaction *tx)
101{
102 /* reset rule store */
103 det_ctx->pmq.rule_id_array_cnt = 0;
104
105 SCLogDebug("packet %" PRIu64 " tx %p progress %d tx->prefilter_flags %" PRIx64, p->pcap_cnt,
106 tx->tx_ptr, tx->tx_progress, tx->prefilter_flags);
107
108 PrefilterEngine *engine = sgh->tx_engines;
109 do {
110 if (engine->alproto != alproto)
111 goto next;
112 if (engine->ctx.tx_min_progress > tx->tx_progress)
113 break;
114 if (tx->tx_progress > engine->ctx.tx_min_progress) {
115 if (tx->prefilter_flags & BIT_U64(engine->ctx.tx_min_progress)) {
116 goto next;
117 }
118 }
119
120 PREFILTER_PROFILING_START(det_ctx);
121 engine->cb.PrefilterTx(det_ctx, engine->pectx,
122 p, p->flow, tx->tx_ptr, tx->tx_id, flow_flags);
123 PREFILTER_PROFILING_END(det_ctx, engine->gid);
124
125 if (tx->tx_progress > engine->ctx.tx_min_progress && engine->is_last_for_progress) {
126 tx->prefilter_flags |= BIT_U64(engine->ctx.tx_min_progress);
127 }
128 next:
129 if (engine->is_last)
130 break;
131 engine++;
132 } while (1);
133
134 /* Sort the rule list to lets look at pmq.
135 * NOTE due to merging of 'stream' pmqs we *MAY* have duplicate entries */
136 if (likely(det_ctx->pmq.rule_id_array_cnt > 1)) {
137 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_SORT1);
138 QuickSortSigIntId(det_ctx->pmq.rule_id_array, det_ctx->pmq.rule_id_array_cnt);
139 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_SORT1);
140 }
141}

143void Prefilter(DetectEngineThreadCtx *det_ctx, const SigGroupHead *sgh,
144 Packet *p, const uint8_t flags)
145{
146 SCEnter();
147#if 0
148 /* TODO review this check */
149 SCLogDebug("sgh %p frame_engines %p", sgh, sgh->frame_engines);
150 if (p->proto == IPPROTO_TCP && sgh->frame_engines && p->flow &&
151 p->flow->alproto != ALPROTO_UNKNOWN && p->flow->alparser != NULL) {
152 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_RECORD);
153 PrefilterFrames(det_ctx, sgh, p, flags, p->flow->alproto);
154 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_RECORD);
155 }
156#endif
157 if (sgh->pkt_engines) {
158 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_PKT);
159 /* run packet engines */
160 PrefilterEngine *engine = sgh->pkt_engines;
161 do {
162 PREFILTER_PROFILING_START(det_ctx);
163 engine->cb.Prefilter(det_ctx, p, engine->pectx);
164 PREFILTER_PROFILING_END(det_ctx, engine->gid);
165
166 if (engine->is_last)
167 break;
168 engine++;
169 } while (1);
170 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_PKT);
171 }
172
173 /* run payload inspecting engines */
174 if (sgh->payload_engines &&
175 (p->payload_len || (p->flags & PKT_DETECT_HAS_STREAMDATA)) &&
176 !(p->flags & PKT_NOPAYLOAD_INSPECTION))
177 {
178 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_PAYLOAD);
179 PrefilterEngine *engine = sgh->payload_engines;
180 while (1) {
181 PREFILTER_PROFILING_START(det_ctx);
182 engine->cb.Prefilter(det_ctx, p, engine->pectx);
183 PREFILTER_PROFILING_END(det_ctx, engine->gid);
184
185 if (engine->is_last)
186 break;
187 engine++;
188 }
189 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_PAYLOAD);
190 }
191
192 /* Sort the rule list to lets look at pmq.
193 * NOTE due to merging of 'stream' pmqs we *MAY* have duplicate entries */
194 if (likely(det_ctx->pmq.rule_id_array_cnt > 1)) {
195 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_SORT1);
196 QuickSortSigIntId(det_ctx->pmq.rule_id_array, det_ctx->pmq.rule_id_array_cnt);
197 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_SORT1);
198 }
199 SCReturn;
200}

202int PrefilterAppendEngine(DetectEngineCtx *de_ctx, SigGroupHead *sgh,
203 void (*PrefilterFunc)(DetectEngineThreadCtx *det_ctx, Packet *p, const void *pectx),
204 void *pectx, void (*FreeFunc)(void *pectx),
205 const char *name)
206{
207 if (sgh == NULL || PrefilterFunc == NULL || pectx == NULL)
208 return -1;
209
210 PrefilterEngineList *e = SCMallocAligned(sizeof(*e), CLS);
211 if (e == NULL)
212 return -1;
213 memset(e, 0x00, sizeof(*e));
214
215 e->Prefilter = PrefilterFunc;
216 e->pectx = pectx;
217 e->Free = FreeFunc;
218
219 if (sgh->init->pkt_engines == NULL) {
220 sgh->init->pkt_engines = e;
221 } else {
222 PrefilterEngineList *t = sgh->init->pkt_engines;
223 while (t->next != NULL) {
224 t = t->next;
225 }
226
227 t->next = e;
228 e->id = t->id + 1;
229 }
230
231 e->name = name;
232 e->gid = PrefilterStoreGetId(de_ctx, e->name, e->Free);
233 return 0;
234}

365static void PrefilterFreeEngines(const DetectEngineCtx *de_ctx, PrefilterEngine *list)
366{
367 PrefilterEngine *t = list;
368
369 while (1) {
370 const PrefilterStore *s = PrefilterStoreGetStore(de_ctx, t->gid);
371 if (s && s->FreeFunc && t->pectx) {
372 s->FreeFunc(t->pectx);
373 }
374
375 if (t->is_last)
376 break;
377 t++;
378 }
379 SCFreeAligned(list);
380}

402static int PrefilterSetupRuleGroupSortHelper(const void *a, const void *b)
403{
404 const PrefilterEngine *s0 = a;
405 const PrefilterEngine *s1 = b;
406 if (s1->ctx.tx_min_progress == s0->ctx.tx_min_progress) {
407 if (s1->alproto == s0->alproto) {
408 return s0->local_id > s1->local_id ? 1 : -1;
409 } else {
410 return s0->alproto > s1->alproto ? 1 : -1;
411 }
412 } else {
413 return s0->ctx.tx_min_progress > s1->ctx.tx_min_progress ? 1 : -1;
414 }
415}

417void PrefilterSetupRuleGroup(DetectEngineCtx *de_ctx, SigGroupHead *sgh)
418{
419 int r = PatternMatchPrepareGroup(de_ctx, sgh);
420 if (r != 0) {
421 FatalError(SC_ERR_INITIALIZATION, "failed to set up pattern matching");
422 }
423
424 /* set up engines if needed - when prefilter is set to auto we run
425 * all engines, otherwise only those that have been forced by the
426 * prefilter keyword. */
427 const enum DetectEnginePrefilterSetting setting = de_ctx->prefilter_setting;
428 for (int i = 0; i < DETECT_TBLSIZE; i++)
429 {
430 if (sigmatch_table[i].SetupPrefilter != NULL &&
431 (setting == DETECT_PREFILTER_AUTO ||
432 de_ctx->sm_types_prefilter[i]))
433 {
434 sigmatch_table[i].SetupPrefilter(de_ctx, sgh);
435 }
436 }
437
438 /* we have lists of engines in sgh->init now. Lets setup the
439 * match arrays */
440 PrefilterEngineList *el;
441 if (sgh->init->pkt_engines != NULL) {
442 uint32_t cnt = 0;
443 for (el = sgh->init->pkt_engines ; el != NULL; el = el->next) {
444 cnt++;
445 de_ctx->prefilter_maxid = MAX(de_ctx->prefilter_maxid, el->gid);
446 }
447 sgh->pkt_engines = SCMallocAligned(cnt * sizeof(PrefilterEngine), CLS);
448 if (sgh->pkt_engines == NULL) {
449 return;
450 }
451 memset(sgh->pkt_engines, 0x00, (cnt * sizeof(PrefilterEngine)));
452
453 PrefilterEngine *e = sgh->pkt_engines;
454 for (el = sgh->init->pkt_engines ; el != NULL; el = el->next) {
455 e->local_id = el->id;
456 e->cb.Prefilter = el->Prefilter;
457 e->pectx = el->pectx;
458 el->pectx = NULL; // e now owns the ctx
459 e->gid = el->gid;
460 if (el->next == NULL) {
461 e->is_last = TRUE;
462 }
463 e++;
464 }
465 }
466 if (sgh->init->payload_engines != NULL) {
467 uint32_t cnt = 0;
468 for (el = sgh->init->payload_engines ; el != NULL; el = el->next) {
469 cnt++;
470 de_ctx->prefilter_maxid = MAX(de_ctx->prefilter_maxid, el->gid);
471 }
472 sgh->payload_engines = SCMallocAligned(cnt * sizeof(PrefilterEngine), CLS);
473 if (sgh->payload_engines == NULL) {
474 return;
475 }
476 memset(sgh->payload_engines, 0x00, (cnt * sizeof(PrefilterEngine)));
477
478 PrefilterEngine *e = sgh->payload_engines;
479 for (el = sgh->init->payload_engines ; el != NULL; el = el->next) {
480 e->local_id = el->id;
481 e->cb.Prefilter = el->Prefilter;
482 e->pectx = el->pectx;
483 el->pectx = NULL; // e now owns the ctx
484 e->gid = el->gid;
485 if (el->next == NULL) {
486 e->is_last = TRUE;
487 }
488 e++;
489 }
490 }
491 if (sgh->init->tx_engines != NULL) {
492 uint32_t cnt = 0;
493 for (el = sgh->init->tx_engines ; el != NULL; el = el->next) {
494 cnt++;
495 de_ctx->prefilter_maxid = MAX(de_ctx->prefilter_maxid, el->gid);
496 }
497 sgh->tx_engines = SCMallocAligned(cnt * sizeof(PrefilterEngine), CLS);
498 if (sgh->tx_engines == NULL) {
499 return;
500 }
501 memset(sgh->tx_engines, 0x00, (cnt * sizeof(PrefilterEngine)));
502
503 uint16_t local_id = 0;
504 PrefilterEngine *e = sgh->tx_engines;
505 for (el = sgh->init->tx_engines ; el != NULL; el = el->next) {
506 e->local_id = local_id++;
507 e->alproto = el->alproto;
508 e->ctx.tx_min_progress = el->tx_min_progress;
509 e->cb.PrefilterTx = el->PrefilterTx;
510 e->pectx = el->pectx;
511 el->pectx = NULL; // e now owns the ctx
512 e->gid = el->gid;
513 e++;
514 }
515
516 /* sort by tx_min_progress, then alproto, then local_id */
517 qsort(sgh->tx_engines, local_id, sizeof(PrefilterEngine),
518 PrefilterSetupRuleGroupSortHelper);
519 sgh->tx_engines[local_id - 1].is_last = true;
520 sgh->tx_engines[local_id - 1].is_last_for_progress = true;
521
522 PrefilterEngine *engine;
523
524 /* per alproto to set is_last_for_progress per alproto because the inspect
525 * loop skips over engines that are not the correct alproto */
526 for (AppProto a = 1; a < ALPROTO_FAILED; a++) {
527 int last_tx_progress = 0;
528 bool last_tx_progress_set = false;
529 PrefilterEngine *prev_engine = NULL;
530 engine = sgh->tx_engines;
531 do {
532 BUG_ON(engine->ctx.tx_min_progress < last_tx_progress);
533 if (engine->alproto == a) {
534 if (last_tx_progress_set && engine->ctx.tx_min_progress > last_tx_progress) {
535 if (prev_engine) {
536 prev_engine->is_last_for_progress = true;
537 }
538 }
539
540 last_tx_progress_set = true;
541 prev_engine = engine;
542 } else {
543 if (prev_engine) {
544 prev_engine->is_last_for_progress = true;
545 }
546 }
547 last_tx_progress = engine->ctx.tx_min_progress;
548 if (engine->is_last)
549 break;
550 engine++;
551 } while (1);
552 }
553#ifdef DEBUG
554 SCLogDebug("sgh %p", sgh);
555 engine = sgh->tx_engines;
556 do {
557 SCLogDebug("engine: gid %u alproto %s tx_min_progress %d is_last %s "
558 "is_last_for_progress %s",
559 engine->gid, AppProtoToString(engine->alproto), engine->ctx.tx_min_progress,
560 engine->is_last ? "true" : "false",
561 engine->is_last_for_progress ? "true" : "false");
562 if (engine->is_last)
563 break;
564 engine++;
565 } while (1);
566#endif
567 }
568 if (sgh->init->frame_engines != NULL) {
569 uint32_t cnt = 0;
570 for (el = sgh->init->frame_engines; el != NULL; el = el->next) {
571 cnt++;
572 de_ctx->prefilter_maxid = MAX(de_ctx->prefilter_maxid, el->gid);
573 }
574 sgh->frame_engines = SCMallocAligned(cnt * sizeof(PrefilterEngine), CLS);
575 if (sgh->frame_engines == NULL) {
576 return;
577 }
578 memset(sgh->frame_engines, 0x00, (cnt * sizeof(PrefilterEngine)));
579
580 PrefilterEngine *e = sgh->frame_engines;
581 for (el = sgh->init->frame_engines; el != NULL; el = el->next) {
582 e->local_id = el->id;
583 e->ctx.frame_type = el->frame_type;
584 e->cb.PrefilterFrame = el->PrefilterFrame;
585 e->alproto = el->alproto;
586 e->pectx = el->pectx;
587 el->pectx = NULL; // e now owns the ctx
588 e->gid = el->gid;
589 if (el->next == NULL) {
590 e->is_last = TRUE;
591 }
592 e++;
593 }
594 }
595}

597/* hash table for assigning a unique id to each engine type. */

599static uint32_t PrefilterStoreHashFunc(HashListTable *ht, void *data, uint16_t datalen)
600{
601 PrefilterStore *ctx = data;
602
603 uint32_t hash = strlen(ctx->name);
604
605 for (size_t u = 0; u < strlen(ctx->name); u++) {
606 hash += ctx->name[u];
607 }
608
609 hash %= ht->array_size;
610 return hash;
611}

613static char PrefilterStoreCompareFunc(void *data1, uint16_t len1,
614 void *data2, uint16_t len2)
615{
616 PrefilterStore *ctx1 = data1;
617 PrefilterStore *ctx2 = data2;
618 return (strcmp(ctx1->name, ctx2->name) == 0);
619}

644static int PrefilterStoreGetId(DetectEngineCtx *de_ctx,
645 const char *name, void (*FreeFunc)(void *))
646{
647 PrefilterStore ctx = { name, FreeFunc, 0 };
648
649 BUG_ON(de_ctx->prefilter_hash_table == NULL);
650
651 SCLogDebug("looking up %s", name);
652
653 PrefilterStore *rctx = HashListTableLookup(de_ctx->prefilter_hash_table, (void *)&ctx, 0);
654 if (rctx != NULL) {
655 return rctx->id;
656 }
657
658 PrefilterStore *actx = SCCalloc(1, sizeof(*actx));
659 if (actx == NULL) {
660 return -1;
661 }
662
663 actx->name = name;
664 actx->FreeFunc = FreeFunc;
