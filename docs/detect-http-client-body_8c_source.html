79/**
80 * \brief Registers the keyword handlers for the "http_client_body" keyword.
81 */
82void DetectHttpClientBodyRegister(void)
83{
84 /* http_client_body content modifier */
85 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].name = "http_client_body";
86 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].desc = "content modifier to match only on HTTP request-body";
87 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].url = "/rules/http-keywords.html#http-client-body";
88 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].Setup = DetectHttpClientBodySetup;
89#ifdef UNITTESTS
90 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].RegisterTests = DetectHttpClientBodyRegisterTests;
91#endif
92 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].flags |= SIGMATCH_NOOPT ;
93 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].flags |= SIGMATCH_INFO_CONTENT_MODIFIER;
94 sigmatch_table[DETECT_AL_HTTP_CLIENT_BODY].alternative = DETECT_HTTP_REQUEST_BODY;
95
96 /* http.request_body sticky buffer */
97 sigmatch_table[DETECT_HTTP_REQUEST_BODY].name = "http.request_body";
98 sigmatch_table[DETECT_HTTP_REQUEST_BODY].desc = "sticky buffer to match the HTTP request body buffer";
99 sigmatch_table[DETECT_HTTP_REQUEST_BODY].url = "/rules/http-keywords.html#http-client-body";
100 sigmatch_table[DETECT_HTTP_REQUEST_BODY].Setup = DetectHttpClientBodySetupSticky;
101 sigmatch_table[DETECT_HTTP_REQUEST_BODY].flags |= SIGMATCH_NOOPT;
102 sigmatch_table[DETECT_HTTP_REQUEST_BODY].flags |= SIGMATCH_INFO_STICKY_BUFFER;
103
104 DetectAppLayerInspectEngineRegister2("http_client_body", ALPROTO_HTTP1, SIG_FLAG_TOSERVER,
105 HTP_REQUEST_BODY, DetectEngineInspectBufferHttpBody, NULL);
106
107 DetectAppLayerMpmRegister2("http_client_body", SIG_FLAG_TOSERVER, 2,
108 PrefilterMpmHttpRequestBodyRegister, NULL, ALPROTO_HTTP1, HTP_REQUEST_BODY);
109
110 DetectBufferTypeSetDescriptionByName("http_client_body",
111 "http request body");
112
113 DetectBufferTypeRegisterSetupCallback("http_client_body",
114 DetectHttpClientBodySetupCallback);
115
116 g_http_client_body_buffer_id = DetectBufferTypeGetByName("http_client_body");
117}

166static inline HtpBody *GetRequestBody(htp_tx_t *tx)
167{
168 HtpTxUserData *htud = (HtpTxUserData *)htp_tx_get_user_data(tx);
169 if (htud == NULL) {
170 SCLogDebug("no htud");
171 return NULL;
172 }
173
174 return &htud->request_body;
175}

190 const DetectEngineTransforms *transforms, const int list_id, InspectionBuffer *base_buffer)
191{
192 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);

197 buffer->inspect_offset = base_buffer->inspect_offset;

203static InspectionBuffer *HttpRequestBodyGetDataCallback(DetectEngineThreadCtx *det_ctx,
204 const DetectEngineTransforms *transforms, Flow *f, const uint8_t flow_flags, void *txv,
205 const int list_id, const int base_id)
206{
207 SCEnter();
208
209 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, base_id);
210 if (base_id != list_id && buffer->inspect != NULL)
211 return HttpRequestBodyXformsGetDataCallback(det_ctx, transforms, list_id, buffer);
212 else if (buffer->inspect != NULL)
213 return buffer;
214
215 htp_tx_t *tx = txv;
216 HtpState *htp_state = f->alstate;
217 const uint8_t flags = flow_flags;
218
219 HtpBody *body = GetRequestBody(tx);
220 if (body == NULL) {
221 return NULL;
222 }
223
224 /* no new data */
225 if (body->body_inspected == body->content_len_so_far) {
226 SCLogDebug("no new data");
227 return NULL;
228 }
229
230 HtpBodyChunk *cur = body->first;
231 if (cur == NULL) {
232 SCLogDebug("No http chunks to inspect for this transaction");
233 return NULL;
234 }
235
236 SCLogDebug("request.body_limit %u request_body.content_len_so_far %" PRIu64
237 ", request.inspect_min_size %" PRIu32 ", EOF %s, progress > body? %s",
238 htp_state->cfg->request.body_limit, body->content_len_so_far,
239 htp_state->cfg->request.inspect_min_size, flags & STREAM_EOF ? "true" : "false",
240 (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, tx, flags) >
241 HTP_REQUEST_BODY)
242 ? "true"
243 : "false");
244
245 if (!htp_state->cfg->http_body_inline) {
246 /* inspect the body if the transfer is complete or we have hit
247 * our body size limit */
248 if ((htp_state->cfg->request.body_limit == 0 ||
249 body->content_len_so_far < htp_state->cfg->request.body_limit) &&
250 body->content_len_so_far < htp_state->cfg->request.inspect_min_size &&
251 !(AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, tx, flags) >
252 HTP_REQUEST_BODY) &&
253 !(flags & STREAM_EOF)) {
254 SCLogDebug("we still haven't seen the entire request body. "
255 "Let's defer body inspection till we see the "
256 "entire body.");
257 return NULL;
258 }
259 }
260
261 /* get the inspect buffer
262 *
263 * make sure that we have at least the configured inspect_win size.
264 * If we have more, take at least 1/4 of the inspect win size before
265 * the new data.
266 */
267 uint64_t offset = 0;
268 if (body->body_inspected > htp_state->cfg->request.inspect_min_size) {
269 BUG_ON(body->content_len_so_far < body->body_inspected);
270 uint64_t inspect_win = body->content_len_so_far - body->body_inspected;
271 SCLogDebug("inspect_win %"PRIu64, inspect_win);
272 if (inspect_win < htp_state->cfg->request.inspect_window) {
273 uint64_t inspect_short = htp_state->cfg->request.inspect_window - inspect_win;
274 if (body->body_inspected < inspect_short)
275 offset = 0;
276 else
277 offset = body->body_inspected - inspect_short;
278 } else {
279 offset = body->body_inspected - (htp_state->cfg->request.inspect_window / 4);
280 }
281 }
282
283 const uint8_t *data;
284 uint32_t data_len;
285
286 StreamingBufferGetDataAtOffset(body->sb,
287 &data, &data_len, offset);
288 InspectionBufferSetup(det_ctx, base_id, buffer, data, data_len);
289 buffer->inspect_offset = offset;
290 body->body_inspected = body->content_len_so_far;
291 SCLogDebug("body->body_inspected now: %" PRIu64, body->body_inspected);
292
293 if (base_id != list_id) {
294 buffer = HttpRequestBodyXformsGetDataCallback(det_ctx, transforms, list_id, buffer);
295 }
296 SCReturnPtr(buffer, "InspectionBuffer");
297}

299static uint8_t DetectEngineInspectBufferHttpBody(DetectEngineCtx *de_ctx,
300 DetectEngineThreadCtx *det_ctx, const DetectEngineAppInspectionEngine *engine,
301 const Signature *s, Flow *f, uint8_t flags, void *alstate, void *txv, uint64_t tx_id)
302{
303 bool eof =
304 (AppLayerParserGetStateProgress(f->proto, f->alproto, txv, flags) > engine->progress);
305 const InspectionBuffer *buffer = HttpRequestBodyGetDataCallback(
306 det_ctx, engine->v2.transforms, f, flags, txv, engine->sm_list, engine->sm_list_base);
307 if (buffer == NULL || buffer->inspect == NULL) {
308 return eof ? DETECT_ENGINE_INSPECT_SIG_CANT_MATCH : DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
309 }
310
311 const uint32_t data_len = buffer->inspect_len;
312 const uint8_t *data = buffer->inspect;
313 const uint64_t offset = buffer->inspect_offset;
314
315 uint8_t ci_flags = eof ? DETECT_CI_FLAGS_END : 0;
316 ci_flags |= (offset == 0 ? DETECT_CI_FLAGS_START : 0);
317 ci_flags |= buffer->flags;
318
319 det_ctx->discontinue_matching = 0;
320 det_ctx->buffer_offset = 0;
321 det_ctx->inspection_recursion_counter = 0;
322
323 /* Inspect all the uricontents fetched on each
324 * transaction at the app layer */
325 int r = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd, NULL, f, (uint8_t *)data,
326 data_len, offset, ci_flags, DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
327 if (r == 1) {
328 return DETECT_ENGINE_INSPECT_SIG_MATCH;
329 }
330
331 if (flags & STREAM_TOSERVER) {
332 if (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, txv, flags) >
333 HTP_REQUEST_BODY)
334 return DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
335 } else {
336 if (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, txv, flags) >
337 HTP_RESPONSE_BODY)
338 return DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
339 }
340 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
341}

343/** \brief HTTP Request body callback
344 *
345 * \param det_ctx detection engine thread ctx
346 * \param pectx inspection context
347 * \param p packet to inspect
348 * \param f flow to inspect
349 * \param txv tx to inspect
350 * \param idx transaction id
351 * \param flags STREAM_* flags including direction
352 */
353static void PrefilterTxHttpRequestBody(DetectEngineThreadCtx *det_ctx, const void *pectx, Packet *p,
354 Flow *f, void *txv, const uint64_t idx, const uint8_t flags)
355{
356 SCEnter();
357
358 const PrefilterMpmHttpRequestBody *ctx = (const PrefilterMpmHttpRequestBody *)pectx;
359 const MpmCtx *mpm_ctx = ctx->mpm_ctx;
360 const int list_id = ctx->list_id;
361
362 InspectionBuffer *buffer = HttpRequestBodyGetDataCallback(
363 det_ctx, ctx->transforms, f, flags, txv, list_id, ctx->base_list_id);
364 if (buffer == NULL)
365 return;
366
367 if (buffer->inspect_len >= mpm_ctx->minlen) {
368 (void)mpm_table[mpm_ctx->mpm_type].Search(
369 mpm_ctx, &det_ctx->mtcu, &det_ctx->pmq, buffer->inspect, buffer->inspect_len);
370 PREFILTER_PROFILING_ADD_BYTES(det_ctx, buffer->inspect_len);
371 }
372}

374static int PrefilterMpmHttpRequestBodyRegister(DetectEngineCtx *de_ctx, SigGroupHead *sgh,
375 MpmCtx *mpm_ctx, const DetectBufferMpmRegistery *mpm_reg, int list_id)
376{
377 PrefilterMpmHttpRequestBody *pectx = SCCalloc(1, sizeof(*pectx));
378 if (pectx == NULL)
379 return -1;
380 pectx->list_id = list_id;
381 pectx->base_list_id = mpm_reg->sm_list_base;
382 SCLogDebug("list_id %d base_list_id %d", list_id, pectx->base_list_id);
383 pectx->mpm_ctx = mpm_ctx;
384 pectx->transforms = &mpm_reg->transforms;
385
386 return PrefilterAppendTxEngine(de_ctx, sgh, PrefilterTxHttpRequestBody, mpm_reg->app_v2.alproto,
387 mpm_reg->app_v2.tx_min_progress, pectx, PrefilterMpmHttpRequestBodyFree,
388 mpm_reg->pname);
389}
