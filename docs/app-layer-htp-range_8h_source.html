51/** Item in hash table for a file in multiple ranges
52 * Thread-safety is ensured with the thread-safe hash table cf THashData
53 * The number of use is increased for each flow opening a new HttpRangeContainerBlock
54 * until it closes this HttpRangeContainerBlock
55 * The design goal is to have concurrency only on opening and closing a range request
56 * and have a lock-free data structure belonging to one Flow
57 * (see HttpRangeContainerBlock below)
58 * for every append in between (we suppose we have many appends per range request)
59 */
60typedef struct HttpRangeContainerFile {
61 /** key for hashtable */
62 uint8_t *key;
63 /** key length */
64 uint32_t len;
65 /** expire time in epoch */
66 uint32_t expire;
67 /** pointer to hashtable data, for locking and use count */
68 THashData *hdata;
69 /** total expected size of the file in ranges */
70 uint64_t totalsize;
71 /** size of the file after last sync */
72 uint64_t lastsize;
73 /** file container, with only one file */
74 FileContainer *files;
75 /** red and black tree list of ranges which came out of order */
76 struct HTTP_RANGES fragment_tree;
77 /** file flags */
78 uint16_t flags;
79 /** error condition for this range. Its up to timeout handling to cleanup */
80 bool error;
81} HttpRangeContainerFile;

83/** A structure representing a single range request :
84 * either skipping, buffering, or appending
85 * As this belongs to a flow, appending data to it is ensured to be thread-safe
86 * Only one block per file has the pointer to the container
87 */
88typedef struct HttpRangeContainerBlock {
89 /** state where we skip content */
90 uint64_t toskip;
91 /** current out of order range to write into */
92 HttpRangeContainerBuffer *current;
93 /** pointer to the main file container, where to directly append data */
94 HttpRangeContainerFile *container;
95 /** file container we are owning for now */
96 FileContainer *files;
97} HttpRangeContainerBlock;
