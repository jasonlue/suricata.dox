99/**
100 * \brief apply the post match filestore with options
101 */
102static int FilestorePostMatchWithOptions(Packet *p, Flow *f, const DetectFilestoreData *filestore,
103 FileContainer *fc, uint32_t file_id, uint64_t tx_id)
104{
105 if (filestore == NULL) {
106 SCReturnInt(0);
107 }
108
109 int this_file = 0;
110 int this_tx = 0;
111 int this_flow = 0;
112 int rule_dir = 0;
113 int toserver_dir = 0;
114 int toclient_dir = 0;
115
116 switch (filestore->direction) {
117 case FILESTORE_DIR_DEFAULT:
118 rule_dir = 1;
119 break;
120 case FILESTORE_DIR_BOTH:
121 toserver_dir = 1;
122 toclient_dir = 1;
123 break;
124 case FILESTORE_DIR_TOSERVER:
125 toserver_dir = 1;
126 break;
127 case FILESTORE_DIR_TOCLIENT:
128 toclient_dir = 1;
129 break;
130 }
131
132 switch (filestore->scope) {
133 case FILESTORE_SCOPE_DEFAULT:
134 if (rule_dir) {
135 this_file = 1;
136 } else if ((p->flowflags & FLOW_PKT_TOCLIENT) && toclient_dir) {
137 this_file = 1;
138 } else if ((p->flowflags & FLOW_PKT_TOSERVER) && toserver_dir) {
139 this_file = 1;
140 }
141 break;
142 case FILESTORE_SCOPE_TX:
143 this_tx = 1;
144 break;
145 case FILESTORE_SCOPE_SSN:
146 this_flow = 1;
147 break;
148 }
149
150 if (this_file) {
151 FileStoreFileById(fc, file_id);
152 } else if (this_tx) {
153 /* flag tx all files will be stored */
154 if (f->alproto == ALPROTO_HTTP1 && f->alstate != NULL) {
155 HtpState *htp_state = f->alstate;
156 if (toserver_dir) {
157 htp_state->flags |= HTP_FLAG_STORE_FILES_TX_TS;
158 FileStoreAllFilesForTx(htp_state->files_ts, tx_id);
159 }
160 if (toclient_dir) {
161 htp_state->flags |= HTP_FLAG_STORE_FILES_TX_TC;
162 FileStoreAllFilesForTx(htp_state->files_tc, tx_id);
163 }
164 htp_state->store_tx_id = tx_id;
165 }
166 } else if (this_flow) {
167 /* flag flow all files will be stored */
168 if (f->alproto == ALPROTO_HTTP1 && f->alstate != NULL) {
169 HtpState *htp_state = f->alstate;
170 if (toserver_dir) {
171 htp_state->flags |= HTP_FLAG_STORE_FILES_TS;
172 FileStoreAllFiles(htp_state->files_ts);
173 }
174 if (toclient_dir) {
175 htp_state->flags |= HTP_FLAG_STORE_FILES_TC;
176 FileStoreAllFiles(htp_state->files_tc);
177 }
178 }
179 } else {
180 FileStoreFileById(fc, file_id);
181 }
182
183 SCReturnInt(0);
184}

186/**
187 * \brief post-match function for filestore
188 *
189 * \param t thread local vars
190 * \param det_ctx pattern matcher thread local data
191 * \param p packet
192 *
193 * The match function for filestore records store candidates in the det_ctx.
194 * When we are sure all parts of the signature matched, we run this function
195 * to finalize the filestore.
196 */
197static int DetectFilestorePostMatch(DetectEngineThreadCtx *det_ctx,
198 Packet *p, const Signature *s, const SigMatchCtx *ctx)
199{
200 SCEnter();
201
202 if (det_ctx->filestore_cnt == 0) {
203 SCReturnInt(0);
204 }
205
206 if ((s->filestore_ctx == NULL && !(s->flags & SIG_FLAG_FILESTORE)) || p->flow == NULL) {
207#ifndef DEBUG
208 SCReturnInt(0);
209#else
210 BUG_ON(1);
211#endif
212 }
213
214 if (p->proto == IPPROTO_TCP && p->flow->protoctx != NULL) {
215 /* set filestore depth for stream reassembling */
216 TcpSession *ssn = (TcpSession *)p->flow->protoctx;
217 TcpSessionSetReassemblyDepth(ssn, FileReassemblyDepth());
218 }
219
220 const uint8_t flags = STREAM_FLAGS_FOR_PACKET(p);
221 for (uint16_t u = 0; u < det_ctx->filestore_cnt; u++) {
222 AppLayerParserSetStreamDepthFlag(p->flow->proto, p->flow->alproto,
223 FlowGetAppState(p->flow),
224 det_ctx->filestore[u].tx_id,
225 flags);
226 }
227
228 FileContainer *ffc = AppLayerParserGetFiles(p->flow, flags);
229
230 /* filestore for single files only */
231 if (s->filestore_ctx == NULL) {
232 for (uint16_t u = 0; u < det_ctx->filestore_cnt; u++) {
233 FileStoreFileById(ffc, det_ctx->filestore[u].file_id);
234 }
235 } else {
236 for (uint16_t u = 0; u < det_ctx->filestore_cnt; u++) {
237 FilestorePostMatchWithOptions(p, p->flow, s->filestore_ctx, ffc,
238 det_ctx->filestore[u].file_id, det_ctx->filestore[u].tx_id);
239 }
240 }
241
242 SCReturnInt(0);
243}

245/**
246 * \brief match the specified filestore
247 *
248 * \param t thread local vars
249 * \param det_ctx pattern matcher thread local data
250 * \param f *LOCKED* flow
251 * \param flags direction flags
252 * \param file file being inspected
253 * \param s signature being inspected
254 * \param m sigmatch that we will cast into DetectFilestoreData
255 *
256 * \retval 0 no match
257 * \retval 1 match
258 *
259 * \todo when we start supporting more protocols, the logic in this function
260 * needs to be put behind a api.
261 */
262static int DetectFilestoreMatch (DetectEngineThreadCtx *det_ctx, Flow *f,
263 uint8_t flags, File *file, const Signature *s, const SigMatchCtx *m)
264{
265 uint32_t file_id = 0;
266
267 SCEnter();
268
269 if (det_ctx->filestore_cnt >= DETECT_FILESTORE_MAX) {
270 SCReturnInt(1);
271 }
272
273 /* file can be NULL when a rule with filestore scope > file
274 * matches. */
275 if (file != NULL) {
276 file_id = file->file_track_id;
277 if (file->sid != NULL && s->id > 0) {
278 if (file->sid_cnt >= file->sid_max) {
279 void *p = SCRealloc(file->sid, sizeof(uint32_t) * (file->sid_max + 8));
280 if (p == NULL) {
281 SCFree(file->sid);
282 file->sid = NULL;
283 file->sid_cnt = 0;
284 file->sid_max = 0;
285 goto continue_after_realloc_fail;
286 } else {
287 file->sid = p;
288 file->sid_max += 8;
289 }
290 }
291 file->sid[file->sid_cnt] = s->id;
292 file->sid_cnt++;
293 }
294 }
295
296continue_after_realloc_fail:
297
298 det_ctx->filestore[det_ctx->filestore_cnt].file_id = file_id;
299 det_ctx->filestore[det_ctx->filestore_cnt].tx_id = det_ctx->tx_id;
300
301 SCLogDebug("%u, file %u, tx %"PRIu64, det_ctx->filestore_cnt,
302 det_ctx->filestore[det_ctx->filestore_cnt].file_id,
303 det_ctx->filestore[det_ctx->filestore_cnt].tx_id);
304
305 det_ctx->filestore_cnt++;
306 SCReturnInt(1);
307}

309/**
310 * \brief this function is used to parse filestore options
311 * \brief into the current signature
312 *
313 * \param de_ctx pointer to the Detection Engine Context
314 * \param s pointer to the Current Signature
315 * \param str pointer to the user provided "filestore" option
316 *
317 * \retval 0 on Success
318 * \retval -1 on Failure
319 */
320static int DetectFilestoreSetup (DetectEngineCtx *de_ctx, Signature *s, const char *str)
321{
322 SCEnter();
323
324 static bool warn_not_configured = false;
325 static uint32_t de_version = 0;
326
327 /* Check on first-time loads (includes following a reload) */
328 if (!warn_not_configured || (de_ctx->version != de_version)) {
329 if (de_version != de_ctx->version) {
330 SCLogDebug("reload-detected; re-checking feature presence; DE version now %"PRIu32,
331 de_ctx->version);
332 }
333 if (!RequiresFeature(FEATURE_OUTPUT_FILESTORE)) {
334 SCLogWarning(SC_WARN_ALERT_CONFIG, "One or more rule(s) depends on the "
335 "file-store output log which is not enabled. "
336 "Enable the output \"file-store\".");
337 }
338 warn_not_configured = true;
339 de_version = de_ctx->version;
340 }
341
342 DetectFilestoreData *fd = NULL;
343 SigMatch *sm = NULL;
344 char *args[3] = {NULL,NULL,NULL};
345 int ret = 0, res = 0;
346 size_t pcre2len;
347
348 /* filestore and bypass keywords can't work together */
349 if (s->flags & SIG_FLAG_BYPASS) {
350 SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS,
351 "filestore can't work with bypass keyword");
352 return -1;
353 }
354
355 sm = SigMatchAlloc();
356 if (sm == NULL)
357 goto error;
358
359 sm->type = DETECT_FILESTORE;
360
361 if (str != NULL && strlen(str) > 0) {
362 char str_0[32];
363 char str_1[32];
364 char str_2[32];
365 SCLogDebug("str %s", str);
366
367 ret = DetectParsePcreExec(&parse_regex, str, 0, 0);
368 if (ret < 1 || ret > 4) {
369 SCLogError(SC_ERR_PCRE_MATCH, "parse error, ret %" PRId32 ", string %s", ret, str);
370 goto error;
371 }
372
373 if (ret > 1) {
374 pcre2len = sizeof(str_0);
375 res = pcre2_substring_copy_bynumber(
376 parse_regex.match, 1, (PCRE2_UCHAR8 *)str_0, &pcre2len);
377 if (res < 0) {
378 SCLogError(SC_ERR_PCRE_COPY_SUBSTRING, "pcre2_substring_copy_bynumber failed");
379 goto error;
380 }
381 args[0] = (char *)str_0;
382
383 if (ret > 2) {
384 pcre2len = sizeof(str_1);
385 res = pcre2_substring_copy_bynumber(
386 parse_regex.match, 2, (PCRE2_UCHAR8 *)str_1, &pcre2len);
387 if (res < 0) {
388 SCLogError(SC_ERR_PCRE_COPY_SUBSTRING, "pcre2_substring_copy_bynumber failed");
389 goto error;
390 }
391 args[1] = (char *)str_1;
392 }
393 if (ret > 3) {
394 pcre2len = sizeof(str_2);
395 res = pcre2_substring_copy_bynumber(
396 parse_regex.match, 3, (PCRE2_UCHAR8 *)str_2, &pcre2len);
397 if (res < 0) {
398 SCLogError(SC_ERR_PCRE_COPY_SUBSTRING, "pcre2_substring_copy_bynumber failed");
399 goto error;
400 }
401 args[2] = (char *)str_2;
402 }
403 }
404
405 fd = SCMalloc(sizeof(DetectFilestoreData));
406 if (unlikely(fd == NULL))
407 goto error;
408 memset(fd, 0x00, sizeof(DetectFilestoreData));
409
410 if (args[0] != NULL) {
411 SCLogDebug("first arg %s", args[0]);
412
413 if (strcasecmp(args[0], "request") == 0 ||
414 strcasecmp(args[0], "to_server") == 0)
415 {
416 fd->direction = FILESTORE_DIR_TOSERVER;
417 fd->scope = FILESTORE_SCOPE_TX;
418 }
419 else if (strcasecmp(args[0], "response") == 0 ||
420 strcasecmp(args[0], "to_client") == 0)
421 {
422 fd->direction = FILESTORE_DIR_TOCLIENT;
423 fd->scope = FILESTORE_SCOPE_TX;
424 }
425 else if (strcasecmp(args[0], "both") == 0)
426 {
427 fd->direction = FILESTORE_DIR_BOTH;
428 fd->scope = FILESTORE_SCOPE_TX;
429 }
430 } else {
431 fd->direction = FILESTORE_DIR_DEFAULT;
432 }
433
434 if (args[1] != NULL) {
435 SCLogDebug("second arg %s", args[1]);
436
437 if (strcasecmp(args[1], "file") == 0)
438 {
439 fd->scope = FILESTORE_SCOPE_DEFAULT;
440 } else if (strcasecmp(args[1], "tx") == 0)
441 {
442 fd->scope = FILESTORE_SCOPE_TX;
443 } else if (strcasecmp(args[1], "ssn") == 0 ||
444 strcasecmp(args[1], "flow") == 0)
445 {
446 fd->scope = FILESTORE_SCOPE_SSN;
447 }
448 } else {
449 if (fd->scope == 0)
450 fd->scope = FILESTORE_SCOPE_DEFAULT;
451 }
452
453 sm->ctx = (SigMatchCtx*)fd;
454 } else {
455 sm->ctx = (SigMatchCtx*)NULL;
456 }
457
458 if (s->alproto == ALPROTO_HTTP1 || s->alproto == ALPROTO_HTTP) {
459 AppLayerHtpNeedFileInspection();
460 }
461
462 SigMatchAppendSMToList(s, sm, g_file_match_list_id);
463 s->filestore_ctx = (const DetectFilestoreData *)sm->ctx;
464
465 sm = SigMatchAlloc();
466 if (unlikely(sm == NULL))
467 goto error;
468 sm->type = DETECT_FILESTORE_POSTMATCH;
469 sm->ctx = NULL;
470 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_POSTMATCH);
471
472
473 s->flags |= SIG_FLAG_FILESTORE;
474 return 0;
475
476error:
477 if (sm != NULL)
478 SCFree(sm);
479 return -1;
480}
