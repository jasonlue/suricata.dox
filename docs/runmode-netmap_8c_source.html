84static int ParseNetmapSettings(NetmapIfaceSettings *ns, const char *iface,
85 ConfNode *if_root, ConfNode *if_default)
86{
87 ns->threads = 0;
88 ns->promisc = true;
89 ns->checksum_mode = CHECKSUM_VALIDATION_AUTO;
90 ns->copy_mode = NETMAP_COPY_MODE_NONE;
91 strlcpy(ns->iface, iface, sizeof(ns->iface));
92
93 if (ns->iface[0]) {
94 size_t len = strlen(ns->iface);
95 if (ns->iface[len-1] == '+') {
96 SCLogWarning(SC_WARN_OPTION_OBSOLETE,
97 "netmap interface %s uses obsolete '+' notation. "
98 "Using '^' instead.", ns->iface);
99 ns->iface[len-1] = '^';
100 ns->sw_ring = true;
101 } else if (ns->iface[len-1] == '^') {
102 ns->sw_ring = true;
103 }
104 }
105
106 /* we will need the base interface name for later */
107 char base_name[IFNAMSIZ];
108 strlcpy(base_name, ns->iface, sizeof(base_name));
109 if (strlen(base_name) > 0 &&
110 (base_name[strlen(base_name) - 1] == '^' || base_name[strlen(base_name) - 1] == '*')) {
111 base_name[strlen(base_name) - 1] = '\0';
112 }
113
114 /* prefixed with netmap or vale means it's not a real interface
115 * and we don't check offloading. */
116 if (strncmp(ns->iface, "netmap:", 7) != 0 &&
117 strncmp(ns->iface, "vale", 4) != 0) {
118 ns->real = true;
119 }
120
121 const char *bpf_filter = NULL;
122 if (ConfGet("bpf-filter", &bpf_filter) == 1) {
123 if (strlen(bpf_filter) > 0) {
124 ns->bpf_filter = bpf_filter;
125 SCLogInfo("Going to use command-line provided bpf filter '%s'",
126 ns->bpf_filter);
127 }
128 }
129
130 if (if_root == NULL && if_default == NULL) {
131 SCLogInfo("Unable to find netmap config for "
132 "interface \"%s\" or \"default\", using default values",
133 iface);
134 goto finalize;
135
136 /* If there is no setting for current interface use default one as main iface */
137 } else if (if_root == NULL) {
138 if_root = if_default;
139 if_default = NULL;
140 }
141
142 const char *threadsstr = NULL;
143 if (ConfGetChildValueWithDefault(if_root, if_default, "threads", &threadsstr) != 1) {
144 ns->threads = 0;
145 ns->threads_auto = true;
146 } else {
147 if (strcmp(threadsstr, "auto") == 0) {
148 ns->threads = 0;
149 ns->threads_auto = true;
150 } else {
151 if (StringParseUint16(&ns->threads, 10, 0, threadsstr) < 0) {
152 SCLogWarning(SC_ERR_INVALID_VALUE, "Invalid config value for "
153 "threads: %s, resetting to 0", threadsstr);
154 ns->threads = 0;
155 }
156 }
157 }
158
159 /* load netmap bpf filter */
160 /* command line value has precedence */
161 if (ns->bpf_filter == NULL) {
162 if (ConfGetChildValueWithDefault(if_root, if_default, "bpf-filter", &bpf_filter) == 1) {
163 if (strlen(bpf_filter) > 0) {
164 ns->bpf_filter = bpf_filter;
165 SCLogInfo("Going to use bpf filter %s", ns->bpf_filter);
166 }
167 }
168 }
169
170 int boolval = 0;
171 (void)ConfGetChildValueBoolWithDefault(if_root, if_default, "disable-promisc", (int *)&boolval);
172 if (boolval) {
173 SCLogInfo("Disabling promiscuous mode on iface %s", ns->iface);
174 ns->promisc = false;
175 }
176
177 const char *tmpctype;
178 if (ConfGetChildValueWithDefault(if_root, if_default,
179 "checksum-checks", &tmpctype) == 1)
180 {
181 if (strcmp(tmpctype, "auto") == 0) {
182 ns->checksum_mode = CHECKSUM_VALIDATION_AUTO;
183 } else if (ConfValIsTrue(tmpctype)) {
184 ns->checksum_mode = CHECKSUM_VALIDATION_ENABLE;
185 } else if (ConfValIsFalse(tmpctype)) {
186 ns->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
187 } else {
188 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid value for "
189 "checksum-checks for %s", iface);
190 }
191 }
192
193 const char *copymodestr;
194 if (ConfGetChildValueWithDefault(if_root, if_default,
195 "copy-mode", &copymodestr) == 1)
196 {
197 if (strcmp(copymodestr, "ips") == 0) {
198 ns->copy_mode = NETMAP_COPY_MODE_IPS;
199 } else if (strcmp(copymodestr, "tap") == 0) {
200 ns->copy_mode = NETMAP_COPY_MODE_TAP;
201 } else {
202 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid copy-mode "
203 "(valid are tap, ips)");
204 }
205 }
206
207finalize:
208
209 ns->ips = (ns->copy_mode != NETMAP_COPY_MODE_NONE);
210
211 if (ns->threads_auto) {
212 /* As NetmapGetRSSCount used to be broken on Linux,
213 * fall back to GetIfaceRSSQueuesNum if needed. */
214 ns->threads = NetmapGetRSSCount(ns->iface);
215 if (ns->threads == 0) {
216 /* need to use base_name of interface here */
217 ns->threads = GetIfaceRSSQueuesNum(base_name);
218 }
219 }
220 if (ns->threads <= 0) {
221 ns->threads = 1;
222 }
223
224 return 0;
225}

227/**
228 * \brief extract information from config file
229 *
230 * The returned structure will be freed by the thread init function.
231 * This is thus necessary to copy the structure before giving it
232 * to thread or to reparse the file for each thread (and thus have
233 * new structure.
234 *
235 * \return a NetmapIfaceConfig corresponding to the interface name
236 */
237static void *ParseNetmapConfig(const char *iface_name)
238{
239 ConfNode *if_root = NULL;
240 ConfNode *if_default = NULL;
241 const char *out_iface = NULL;
242
243 if (iface_name == NULL) {
244 return NULL;
245 }
246
247 NetmapIfaceConfig *aconf = SCCalloc(1, sizeof(*aconf));
248 if (unlikely(aconf == NULL)) {
249 return NULL;
250 }
251
252 aconf->DerefFunc = NetmapDerefConfig;
253 strlcpy(aconf->iface_name, iface_name, sizeof(aconf->iface_name));
254 SC_ATOMIC_INIT(aconf->ref);
255 (void) SC_ATOMIC_ADD(aconf->ref, 1);
256
257 /* Find initial node */
258 ConfNode *netmap_node = ConfGetNode("netmap");
259 if (netmap_node == NULL) {
260 SCLogInfo("Unable to find netmap config using default value");
261 } else {
262 if_root = ConfFindDeviceConfig(netmap_node, aconf->iface_name);
263 if_default = ConfFindDeviceConfig(netmap_node, "default");
264 }
265
266 /* parse settings for capture iface */
267 ParseNetmapSettings(&aconf->in, aconf->iface_name, if_root, if_default);
268
269 /* if we have a copy iface, parse that as well */
270 if (netmap_node != NULL &&
271 ConfGetChildValueWithDefault(if_root, if_default, "copy-iface", &out_iface) == 1)
272 {
273 if (strlen(out_iface) > 0) {
274 if_root = ConfFindDeviceConfig(netmap_node, out_iface);
275 ParseNetmapSettings(&aconf->out, out_iface, if_root, if_default);
276 }
277 }
278
279 int ring_count = NetmapGetRSSCount(aconf->iface_name);
280 if (strlen(aconf->iface_name) > 0 &&
281 (aconf->iface_name[strlen(aconf->iface_name) - 1] == '^' ||
282 aconf->iface_name[strlen(aconf->iface_name) - 1] == '*')) {
283 SCLogDebug("%s -- using %d netmap host ring pair%s", aconf->iface_name, ring_count,
284 ring_count == 1 ? "" : "s");
285 } else {
286 SCLogDebug("%s -- using %d netmap ring pair%s", aconf->iface_name, ring_count,
287 ring_count == 1 ? "" : "s");
288 }
289
290 for (int i = 0; i < ring_count; i++) {
291 char live_buf[32] = { 0 };
292 snprintf(live_buf, sizeof(live_buf), "netmap%d", i);
293 LiveRegisterDevice(live_buf);
294 }
295
296 /* netmap needs all offloading to be disabled */
297 if (aconf->in.real) {
298 char base_name[sizeof(aconf->in.iface)];
299 strlcpy(base_name, aconf->in.iface, sizeof(base_name));
300 /* for a sw_ring enabled device name, strip the trailing char */
301 if (aconf->in.sw_ring) {
302 base_name[strlen(base_name) - 1] = '\0';
303 }
304
305 if (LiveGetOffload() == 0) {
306 (void)GetIfaceOffloading(base_name, 1, 1);
307 } else {
308 DisableIfaceOffloading(LiveGetDevice(base_name), 1, 1);
309 }
310 }
311
312 SC_ATOMIC_RESET(aconf->ref);
313 (void) SC_ATOMIC_ADD(aconf->ref, aconf->in.threads);
314 SCLogPerf("Using %d threads for interface %s", aconf->in.threads,
315 aconf->iface_name);
316
317 LiveDeviceHasNoStats();
318 return aconf;
319}

327int NetmapRunModeIsIPS()
328{
329 int nlive = LiveGetDeviceCount();
330 int ldev;
331 ConfNode *if_root;
332 ConfNode *if_default = NULL;
333 ConfNode *netmap_node;
334 int has_ips = 0;
335 int has_ids = 0;
336
337 /* Find initial node */
338 netmap_node = ConfGetNode("netmap");
339 if (netmap_node == NULL) {
340 return 0;
341 }
342
343 if_default = ConfNodeLookupKeyValue(netmap_node, "interface", "default");
344
345 for (ldev = 0; ldev < nlive; ldev++) {
346 const char *live_dev = LiveGetDeviceName(ldev);
347 if (live_dev == NULL) {
348 SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
349 return 0;
350 }
351 const char *copymodestr = NULL;
352 if_root = ConfNodeLookupKeyValue(netmap_node, "interface", live_dev);
353
354 if (if_root == NULL) {
355 if (if_default == NULL) {
356 SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
357 return 0;
358 }
359 if_root = if_default;
360 }
361
362 if (ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) {
363 if (strcmp(copymodestr, "ips") == 0) {
364 has_ips = 1;
365 } else {
366 has_ids = 1;
367 }
368 } else {
369 has_ids = 1;
370 }
371 }
372
373 if (has_ids && has_ips) {
374 SCLogInfo("Netmap mode using IPS and IDS mode");
375 for (ldev = 0; ldev < nlive; ldev++) {
376 const char *live_dev = LiveGetDeviceName(ldev);
377 if (live_dev == NULL) {
378 SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
379 return 0;
380 }
381 if_root = ConfNodeLookupKeyValue(netmap_node, "interface", live_dev);
382 const char *copymodestr = NULL;
383
384 if (if_root == NULL) {
385 if (if_default == NULL) {
386 SCLogError(SC_ERR_INVALID_VALUE, "Problem with config file");
387 return 0;
388 }
389 if_root = if_default;
390 }
391
392 if (! ((ConfGetChildValueWithDefault(if_root, if_default, "copy-mode", &copymodestr) == 1) &&
393 (strcmp(copymodestr, "ips") == 0))) {
394 SCLogError(SC_ERR_INVALID_ARGUMENT,
395 "Netmap IPS mode used and interface '%s' is in IDS or TAP mode. "
396 "Sniffing '%s' but expect bad result as stream-inline is activated.",
397 live_dev, live_dev);
398 }
399 }
400 }
401
402 return has_ips;
403}

405typedef enum { NETMAP_AUTOFP, NETMAP_WORKERS, NETMAP_SINGLE } NetmapRunMode_t;

407static int NetmapRunModeInit(NetmapRunMode_t runmode)
408{
409 SCEnter();
410
411 RunModeInitialize();
412 TimeModeSetLive();
413
414 const char *live_dev = NULL;
415 (void)ConfGet("netmap.live-interface", &live_dev);
416
417 int ret;
418 switch (runmode) {
419 case NETMAP_AUTOFP:
420 ret = RunModeSetLiveCaptureAutoFp(ParseNetmapConfig, NetmapConfigGeThreadsCount,
421 "ReceiveNetmap", "DecodeNetmap", thread_name_autofp, live_dev);
422 break;
423 case NETMAP_WORKERS:
424 ret = RunModeSetLiveCaptureWorkers(ParseNetmapConfig, NetmapConfigGeThreadsCount,
425 "ReceiveNetmap", "DecodeNetmap", thread_name_workers, live_dev);
426 break;
427 case NETMAP_SINGLE:
428 ret = RunModeSetLiveCaptureSingle(ParseNetmapConfig, NetmapConfigGeThreadsCount,
429 "ReceiveNetmap", "DecodeNetmap", thread_name_single, live_dev);
430 break;
431 }
432 if (ret != 0) {
433 FatalError(SC_ERR_FATAL, "Unable to start runmode %s",
434 runmode == NETMAP_AUTOFP ? "autofp"
435 : runmode == NETMAP_WORKERS ? "workers" : "single");
436 }
437
438 SCLogDebug("%s initialized",
439 runmode == NETMAP_AUTOFP ? "autofp" : runmode == NETMAP_WORKERS ? "workers" : "single");
440
441 SCReturnInt(0);
442}
