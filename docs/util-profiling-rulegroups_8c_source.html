72void SCProfilingSghsGlobalInit(void)
73{
74 ConfNode *conf;
75
76 conf = ConfGetNode("profiling.rulegroups");
77 if (conf != NULL) {
78 if (ConfNodeChildValueIsTrue(conf, "enabled")) {
79 profiling_sghs_enabled = 1;
80 const char *filename = ConfNodeLookupChildValue(conf, "filename");
81 if (filename != NULL) {
82 const char *log_dir;
83 log_dir = ConfigGetLogDirectory();
84
85 snprintf(profiling_file_name, sizeof(profiling_file_name),
86 "%s/%s", log_dir, filename);
87
88 const char *v = ConfNodeLookupChildValue(conf, "append");
89 if (v == NULL || ConfValIsTrue(v)) {
90 profiling_file_mode = "a";
91 } else {
92 profiling_file_mode = "w";
93 }
94
95 profiling_sghs_output_to_file = 1;
96 }
97 if (ConfNodeChildValueIsTrue(conf, "json")) {
98 profiling_rulegroup_json = 1;
99 }
100 }
101 }

104static void DoDumpJSON(SCProfileSghDetectCtx *rules_ctx, FILE *fp, const char *name)
105{
106 char timebuf[64];
107 uint32_t i;
108 struct timeval tval;
109
110 json_t *js = json_object();
111 if (js == NULL)
112 return;
113 json_t *jsa = json_array();
114 if (jsa == NULL) {
115 json_decref(js);
116 return;
117 }
118
119 gettimeofday(&tval, NULL);
120 CreateIsoTimeString(&tval, timebuf, sizeof(timebuf));
121 json_object_set_new(js, "timestamp", json_string(timebuf));
122
123 for (i = 0; i < rules_ctx->cnt; i++) {
124 SCProfileSghData *d = &rules_ctx->data[i];
125 if (d == NULL || d->checks == 0)
126 continue;
127
128 double avgsigs = 0;
129 double avgmpms = 0;
130
131 if (d->post_prefilter_sigs_total && d->checks) {
132 avgsigs = (double)((double)d->post_prefilter_sigs_total / (double)d->checks);
133 }
134 if (d->mpm_match_cnt_total && d->checks) {
135 avgmpms = (double)((double)d->mpm_match_cnt_total / (double)d->checks);
136 }
137
138 json_t *jsm = json_object();
139 if (jsm) {
140 json_object_set_new(jsm, "id", json_integer(i));
141 json_object_set_new(jsm, "checks", json_integer(d->checks));
142 json_object_set_new(jsm, "non_mpm_generic", json_integer(d->non_mpm_generic));
143 json_object_set_new(jsm, "non_mpm_syn", json_integer(d->non_mpm_syn));
144 json_object_set_new(jsm, "avgmpms", json_real(avgmpms));
145 json_object_set_new(jsm, "mpm_match_cnt_max", json_integer(d->mpm_match_cnt_max));
146 json_object_set_new(jsm, "avgsigs", json_real(avgsigs));
147 json_object_set_new(jsm, "post_prefilter_sigs_max", json_integer(d->post_prefilter_sigs_max));
148 json_array_append_new(jsa, jsm);
149 }
150 }
151 json_object_set_new(js, "rule_groups", jsa);
152
153 char *js_s = json_dumps(js,
154 JSON_PRESERVE_ORDER|JSON_COMPACT|JSON_ENSURE_ASCII|
155 JSON_ESCAPE_SLASH);
156 if (likely(js_s != NULL)) {
157 fprintf(fp, "%s", js_s);
158 free(js_s);
159 }
160 json_decref(js);
161}

163static void DoDump(SCProfileSghDetectCtx *rules_ctx, FILE *fp, const char *name)
164{
165 uint32_t i;
166 struct timeval tval;
167 struct tm *tms;
168 struct tm local_tm;
169
170 gettimeofday(&tval, NULL);
171 tms = SCLocalTime(tval.tv_sec, &local_tm);
172
173 fprintf(fp, " ----------------------------------------------"
174 "------------------------------------------------------"
175 "----------------------------\n");
176 fprintf(fp, " Date: %" PRId32 "/%" PRId32 "/%04d -- "
177 "%02d:%02d:%02d\n", tms->tm_mon + 1, tms->tm_mday, tms->tm_year + 1900,
178 tms->tm_hour,tms->tm_min, tms->tm_sec);
179
180 fprintf(fp, " ----------------------------------------------"
181 "------------------------------------------------------"
182 "----------------------------\n");
183 fprintf(fp, " Stats for: %s %u\n", name, rules_ctx->cnt);
184 fprintf(fp, " ----------------------------------------------"
185 "------------------------------------------------------"
186 "----------------------------\n");
187 fprintf(fp, " %-16s %-15s %-15s %-15s %-15s %-15s %-15s %-15s\n", "Sgh", "Checks", "Non-MPM(gen)", "Non-Mpm(syn)", "MPM Matches", "MPM Match Max", "Post-Filter", "Post-Filter Max");
188 fprintf(fp, " ---------------- "
189 "--------------- "
190 "--------------- "
191 "--------------- "
192 "--------------- "
193 "--------------- "
194 "--------------- "
195 "--------------- "
196 "\n");
197 for (i = 0; i < rules_ctx->cnt; i++) {
198 SCProfileSghData *d = &rules_ctx->data[i];
199 if (d == NULL || d->checks == 0)
200 continue;
201
202 double avgsigs = 0;
203 double avgmpms = 0;
204
205 if (d->post_prefilter_sigs_total && d->checks) {
206 avgsigs = (double)((double)d->post_prefilter_sigs_total / (double)d->checks);
207 }
208 if (d->mpm_match_cnt_total && d->checks) {
209 avgmpms = (double)((double)d->mpm_match_cnt_total / (double)d->checks);
210 }
211
212 fprintf(fp,
213 " %-16u %-15"PRIu64" %-15"PRIu64" %-15"PRIu64" %-15.2f %-15"PRIu64" %-15.2f %-15"PRIu64"\n",
214 i,
215 d->checks,
216 d->non_mpm_generic,
217 d->non_mpm_syn,
218 avgmpms,
219 d->mpm_match_cnt_max,
220 avgsigs,
221 d->post_prefilter_sigs_max);
222 }
223 fprintf(fp,"\n");
224}

226static void
227SCProfilingSghDump(DetectEngineCtx *de_ctx)
228{
229 FILE *fp;
230
231 if (profiling_sghs_enabled == 0)
232 return;
233
234 if (profiling_sghs_output_to_file == 1) {
235 SCLogDebug("file %s mode %s", profiling_file_name, profiling_file_mode);
236
237 fp = fopen(profiling_file_name, profiling_file_mode);
238
239 if (fp == NULL) {
240 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", profiling_file_name,
241 strerror(errno));
242 return;
243 }
244 } else {
245 fp = stdout;
246 }
247
248 if (profiling_rulegroup_json) {
249 DoDumpJSON(de_ctx->profile_sgh_ctx, fp, "rule groups");
250 } else {
251 DoDump(de_ctx->profile_sgh_ctx, fp, "rule groups");
252 }
253
254 if (fp != stdout)
255 fclose(fp);
256
257 SCLogPerf("Done dumping rulegroup profiling data.");
258}

260/**
261 * \brief Update a rule counter.
262 *
263 * \param id The ID of this counter.
264 * \param ticks Number of CPU ticks for this rule.
265 * \param match Did the rule match?
266 */
267void
268SCProfilingSghUpdateCounter(DetectEngineThreadCtx *det_ctx, const SigGroupHead *sgh)
269{
270 if (det_ctx != NULL && det_ctx->sgh_perf_data != NULL && sgh->id < det_ctx->de_ctx->sgh_array_cnt) {
271 SCProfileSghData *p = &det_ctx->sgh_perf_data[sgh->id];
272 p->checks++;
273
274 if (det_ctx->non_pf_store_cnt > 0) {
275 if (det_ctx->non_pf_store_ptr == sgh->non_pf_syn_store_array)
276 p->non_mpm_syn++;
277 else
278 p->non_mpm_generic++;
279 }
280 p->post_prefilter_sigs_total += det_ctx->match_array_cnt;
281 if (det_ctx->match_array_cnt > p->post_prefilter_sigs_max)
282 p->post_prefilter_sigs_max = det_ctx->match_array_cnt;
283 p->mpm_match_cnt_total += det_ctx->pmq.rule_id_array_cnt;
284 if (det_ctx->pmq.rule_id_array_cnt > p->mpm_match_cnt_max)
285 p->mpm_match_cnt_max = det_ctx->pmq.rule_id_array_cnt;
286 }
287}

289static SCProfileSghDetectCtx *SCProfilingSghInitCtx(void)
290{
291 SCProfileSghDetectCtx *ctx = SCCalloc(1, sizeof(SCProfileSghDetectCtx));
292 if (ctx != NULL) {
293 if (pthread_mutex_init(&ctx->data_m, NULL) != 0) {
294 FatalError(SC_ERR_FATAL, "Failed to initialize mutex.");
295 }
296 }
297
298 return ctx;

333static void SCProfilingSghThreadMerge(DetectEngineCtx *de_ctx, const DetectEngineThreadCtx *det_ctx)
334{
335 if (de_ctx == NULL || de_ctx->profile_sgh_ctx == NULL ||
336 de_ctx->profile_sgh_ctx->data == NULL || det_ctx == NULL ||
337 det_ctx->sgh_perf_data == NULL)
338 return;
339
340#define ADD(name) de_ctx->profile_sgh_ctx->data[i].name += det_ctx->sgh_perf_data[i].name
341 uint32_t i;
342 for (i = 0; i < de_ctx->sgh_array_cnt; i++) {
343 ADD(checks);
344 ADD(non_mpm_generic);
345 ADD(non_mpm_syn);
346 ADD(post_prefilter_sigs_total);
347 ADD(mpm_match_cnt_total);
348
349 if (det_ctx->sgh_perf_data[i].mpm_match_cnt_max > de_ctx->profile_sgh_ctx->data[i].mpm_match_cnt_max)
350 de_ctx->profile_sgh_ctx->data[i].mpm_match_cnt_max = det_ctx->sgh_perf_data[i].mpm_match_cnt_max;
351 if (det_ctx->sgh_perf_data[i].post_prefilter_sigs_max > de_ctx->profile_sgh_ctx->data[i].post_prefilter_sigs_max)
352 de_ctx->profile_sgh_ctx->data[i].post_prefilter_sigs_max = det_ctx->sgh_perf_data[i].post_prefilter_sigs_max;
353 }
354#undef ADD
355}

357void SCProfilingSghThreadCleanup(DetectEngineThreadCtx *det_ctx)
358{
359 if (det_ctx == NULL || det_ctx->de_ctx == NULL || det_ctx->sgh_perf_data == NULL)
360 return;
361
362 pthread_mutex_lock(&det_ctx->de_ctx->profile_sgh_ctx->data_m);
363 SCProfilingSghThreadMerge(det_ctx->de_ctx, det_ctx);
364 pthread_mutex_unlock(&det_ctx->de_ctx->profile_sgh_ctx->data_m);
365
366 SCFree(det_ctx->sgh_perf_data);
367 det_ctx->sgh_perf_data = NULL;
368}

370/**
371 * \brief Register the keyword profiling counters.
372 *
373 * \param de_ctx The active DetectEngineCtx, used to get at the loaded rules.
374 */
375void
376SCProfilingSghInitCounters(DetectEngineCtx *de_ctx)
377{
378 if (profiling_sghs_enabled == 0)
379 return;
380
381 de_ctx->profile_sgh_ctx = SCProfilingSghInitCtx();
382 BUG_ON(de_ctx->profile_sgh_ctx == NULL);
383
384 de_ctx->profile_sgh_ctx->data = SCCalloc(de_ctx->sgh_array_cnt, sizeof(SCProfileSghData));
385 BUG_ON(de_ctx->profile_sgh_ctx->data == NULL);
386
387 de_ctx->profile_sgh_ctx->cnt = de_ctx->sgh_array_cnt;
388
389 SCLogPerf("Registered %"PRIu32" rulegroup profiling counters.", de_ctx->sgh_array_cnt);
390}
