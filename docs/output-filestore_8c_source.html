91/**
92 * \brief Update the timestamps on a file to match those of another
93 * file.
94 *
95 * \param src_filename Filename to use as timestamp source.
96 * \param filename Filename to apply timestamps to.
97 */
98static void OutputFilestoreUpdateFileTime(const char *src_filename,
99 const char *filename)
100{
101 struct stat sb;
102 if (stat(src_filename, &sb) != 0) {
103 SCLogDebug("Failed to stat %s: %s", filename, strerror(errno));
104 return;
105 }
106 struct utimbuf utimbuf = {
107 .actime = sb.st_atime,
108 .modtime = sb.st_mtime,
109 };
110 if (utime(filename, &utimbuf) != 0) {
111 SCLogDebug("Failed to update file timestamps: %s: %s", filename,
112 strerror(errno));
113 }
114}

116static void OutputFilestoreFinalizeFiles(ThreadVars *tv,
117 const OutputFilestoreLogThread *oft, const OutputFilestoreCtx *ctx,
118 const Packet *p, File *ff, uint8_t dir) {
119 /* Stringify the SHA256 which will be used in the final
120 * filename. */
121 char sha256string[(SC_SHA256_LEN * 2) + 1];
122 PrintHexString(sha256string, sizeof(sha256string), ff->sha256,
123 sizeof(ff->sha256));
124
125 char tmp_filename[PATH_MAX] = "";
126 snprintf(tmp_filename, sizeof(tmp_filename), "%s/file.%u", ctx->tmpdir,
127 ff->file_store_id);
128
129 char final_filename[PATH_MAX] = "";
130 snprintf(final_filename, sizeof(final_filename), "%s/%c%c/%s",
131 ctx->prefix, sha256string[0], sha256string[1], sha256string);
132
133 if (SCPathExists(final_filename)) {
134 OutputFilestoreUpdateFileTime(tmp_filename, final_filename);
135 if (unlink(tmp_filename) != 0) {
136 StatsIncr(tv, oft->fs_error_counter);
137 WARN_ONCE(SC_WARN_REMOVE_FILE,
138 "Failed to remove temporary file %s: %s", tmp_filename,
139 strerror(errno));
140 }
141 } else if (rename(tmp_filename, final_filename) != 0) {
142 StatsIncr(tv, oft->fs_error_counter);
143 WARN_ONCE(SC_WARN_RENAMING_FILE, "Failed to rename %s to %s: %s",
144 tmp_filename, final_filename, strerror(errno));
145 if (unlink(tmp_filename) != 0) {
146 /* Just increment, don't log as has_fs_errors would
147 * already be set above. */
148 StatsIncr(tv, oft->fs_error_counter);
149 }
150 return;
151 }
152
153 if (ctx->fileinfo) {
154 char js_metadata_filename[PATH_MAX];
155 if (snprintf(js_metadata_filename, sizeof(js_metadata_filename),
156 "%s.%"PRIuMAX".%u.json", final_filename,
157 (uintmax_t)p->ts.tv_sec, ff->file_store_id)
158 == (int)sizeof(js_metadata_filename)) {
159 WARN_ONCE(SC_ERR_SPRINTF,
160 "Failed to write file info record. Output filename truncated.");
161 } else {
162 JsonBuilder *js_fileinfo =
163 JsonBuildFileInfoRecord(p, ff, true, dir, ctx->xff_cfg, NULL);
164 if (likely(js_fileinfo != NULL)) {
165 jb_close(js_fileinfo);
166 FILE *out = fopen(js_metadata_filename, "w");
167 if (out != NULL) {
168 size_t js_len = jb_len(js_fileinfo);
169 fwrite(jb_ptr(js_fileinfo), js_len, 1, out);
170 fclose(out);
171 }
172 jb_free(js_fileinfo);
173 }
174 }
175 }
176}

178static int OutputFilestoreLogger(ThreadVars *tv, void *thread_data,
179 const Packet *p, File *ff, const uint8_t *data, uint32_t data_len,
180 uint8_t flags, uint8_t dir)
181{
182 SCEnter();
183 OutputFilestoreLogThread *aft = (OutputFilestoreLogThread *)thread_data;
184 OutputFilestoreCtx *ctx = aft->ctx;
185 char filename[PATH_MAX] = "";
186 int file_fd = -1;
187
188 /* no flow, no files */
189 if (p->flow == NULL) {
190 SCReturnInt(TM_ECODE_OK);
191 }
192
193 if (!(PKT_IS_IPV4(p) || PKT_IS_IPV6(p))) {
194 return 0;
195 }
196
197 SCLogDebug("ff %p, data %p, data_len %u", ff, data, data_len);
198
199 char base_filename[PATH_MAX] = "";
200 snprintf(base_filename, sizeof(base_filename), "%s/file.%u",
201 ctx->tmpdir, ff->file_store_id);
202 snprintf(filename, sizeof(filename), "%s", base_filename);
203
204 if (flags & OUTPUT_FILEDATA_FLAG_OPEN) {
205 file_fd = open(filename, O_CREAT | O_TRUNC | O_NOFOLLOW | O_WRONLY,
206 0644);
207 if (file_fd == -1) {
208 StatsIncr(tv, aft->fs_error_counter);
209 SCLogWarning(SC_ERR_OPENING_FILE,
210 "Filestore (v2) failed to create %s: %s", filename,
211 strerror(errno));
212 return -1;
213 }
214
215 if (SC_ATOMIC_GET(filestore_open_file_cnt) < FileGetMaxOpenFiles()) {
216 SC_ATOMIC_ADD(filestore_open_file_cnt, 1);
217 ff->fd = file_fd;
218 } else {
219 if (FileGetMaxOpenFiles() > 0) {
220 StatsIncr(tv, aft->counter_max_hits);
221 }
222 ff->fd = -1;
223 }
224 /* we can get called with a NULL ffd when we need to close */
225 } else if (data != NULL) {
226 if (ff->fd == -1) {
227 file_fd = open(filename, O_APPEND | O_NOFOLLOW | O_WRONLY);
228 if (file_fd == -1) {
229 StatsIncr(tv, aft->fs_error_counter);
230 WARN_ONCE(SC_ERR_OPENING_FILE,
231 "Filestore (v2) failed to open file %s: %s",
232 filename, strerror(errno));
233 return -1;
234 }
235 } else {
236 file_fd = ff->fd;
237 }
238 }
239
240 if (file_fd != -1) {
241 ssize_t r = write(file_fd, (const void *)data, (size_t)data_len);
242 if (r == -1) {
243 StatsIncr(tv, aft->fs_error_counter);
244 WARN_ONCE(SC_ERR_FWRITE,
245 "Filestore (v2) failed to write to %s: %s",
246 filename, strerror(errno));
247 if (ff->fd != -1) {
248 SC_ATOMIC_SUB(filestore_open_file_cnt, 1);
249 }
250 ff->fd = -1;
251 }
252 if (ff->fd == -1) {
253 close(file_fd);
254 }
255 }
256
257 if (flags & OUTPUT_FILEDATA_FLAG_CLOSE) {
258 if (ff->fd != -1) {
259 close(ff->fd);
260 ff->fd = -1;
261 SC_ATOMIC_SUB(filestore_open_file_cnt, 1);
262 }
263 OutputFilestoreFinalizeFiles(tv, aft, ctx, p, ff, dir);
264 }
265
266 return 0;
267}

269static TmEcode OutputFilestoreLogThreadInit(ThreadVars *t, const void *initdata,
270 void **data)
271{
272 OutputFilestoreLogThread *aft = SCMalloc(sizeof(OutputFilestoreLogThread));
273 if (unlikely(aft == NULL))
274 return TM_ECODE_FAILED;
275 memset(aft, 0, sizeof(OutputFilestoreLogThread));
276
277 if (initdata == NULL) {
278 SCLogDebug("Error getting context for LogFileStore. \"initdata\" argument NULL");
279 SCFree(aft);
280 return TM_ECODE_FAILED;
281 }
282
283 OutputFilestoreCtx *ctx = ((OutputCtx *)initdata)->data;
284 aft->ctx = ctx;
285
286 aft->counter_max_hits =
287 StatsRegisterCounter("file_store.open_files_max_hit", t);
288
289 /* File system type errors (open, write, rename) will only be
290 * logged once. But this stat will be incremented for every
291 * occurence. */
292 aft->fs_error_counter = StatsRegisterCounter("file_store.fs_errors", t);
293
294 *data = (void *)aft;
295 return TM_ECODE_OK;
296}

322static void GetLogDirectory(const ConfNode *conf, char *out, size_t out_size)
323{
324 const char *log_base_dir = ConfNodeLookupChildValue(conf, "dir");
325 if (log_base_dir == NULL) {
326 SCLogConfig("Filestore (v2) default log directory %s", default_log_dir);
327 log_base_dir = default_log_dir;

333 snprintf(out, out_size, "%s/%s", default_log_prefix, log_base_dir);

337static bool InitFilestoreDirectory(const char *dir)
338{
339 const uint8_t dir_count = 0xff;
340
341 if (!SCPathExists(dir)) {
342 SCLogInfo("Filestore (v2) creating directory %s", dir);
343 if (SCCreateDirectoryTree(dir, true) != 0) {
344 SCLogError(SC_ERR_CREATE_DIRECTORY,
345 "Filestore (v2) failed to create directory %s: %s", dir,
346 strerror(errno));
347 return false;
348 }
349 }
350
351 for (int i = 0; i <= dir_count; i++) {
352 char leaf[PATH_MAX];
353 int n = snprintf(leaf, sizeof(leaf), "%s/%02x", dir, i);
354 if (n < 0 || n >= PATH_MAX) {
355 SCLogError(SC_ERR_CREATE_DIRECTORY,
356 "Filestore (v2) failed to create leaf directory: "
357 "path too long");
358 return false;
359 }
360 if (!SCPathExists(leaf)) {
361 SCLogInfo("Filestore (v2) creating directory %s", leaf);
362 if (SCDefaultMkDir(leaf) != 0) {
363 SCLogError(SC_ERR_CREATE_DIRECTORY,
364 "Filestore (v2) failed to create directory %s: %s",
365 leaf, strerror(errno));
366 return false;
367 }
368 }
369 }
370
371 /* Make sure the tmp directory exists. */
372 char tmpdir[PATH_MAX];
373 int n = snprintf(tmpdir, sizeof(tmpdir), "%s/tmp", dir);
374 if (n < 0 || n >= PATH_MAX) {
375 SCLogError(SC_ERR_CREATE_DIRECTORY,
376 "Filestore (v2) failed to create tmp directory: path too long");
377 return false;
378 }
379 if (!SCPathExists(tmpdir)) {
380 SCLogInfo("Filestore (v2) creating directory %s", tmpdir);
381 if (SCDefaultMkDir(tmpdir) != 0) {
382 SCLogError(SC_ERR_CREATE_DIRECTORY,
383 "Filestore (v2) failed to create directory %s: %s", tmpdir,
384 strerror(errno));
385 return false;
386 }
387 }
388
389 return true;
390}

392/** \brief Create a new http log OutputFilestoreCtx.
393 * \param conf Pointer to ConfNode containing this loggers configuration.
394 * \return NULL if failure, OutputFilestoreCtx* to the file_ctx if succesful
395 * */
396static OutputInitResult OutputFilestoreLogInitCtx(ConfNode *conf)
397{
398 OutputInitResult result = { NULL, false };
399
400 intmax_t version = 0;
401 if (!ConfGetChildValueInt(conf, "version", &version) || version < 2) {
402 SCLogWarning(SC_WARN_DEPRECATED,
403 "File-store v1 has been removed. Please update to file-store v2.");
404 return result;
405 }
406
407 if (RunModeOutputFiledataEnabled()) {
408 SCLogWarning(SC_ERR_NOT_SUPPORTED,
409 "A file data logger is already enabled. Filestore (v2) "
410 "will not be enabled.");
411 return result;
412 }
413
414 char log_directory[PATH_MAX] = "";
415 GetLogDirectory(conf, log_directory, sizeof(log_directory));
416 if (!InitFilestoreDirectory(log_directory)) {
417 return result;
418 }
419
420 OutputFilestoreCtx *ctx = SCCalloc(1, sizeof(*ctx));
421 if (unlikely(ctx == NULL)) {
422 return result;
423 }
424
425 strlcpy(ctx->prefix, log_directory, sizeof(ctx->prefix));
426 int written = snprintf(ctx->tmpdir, sizeof(ctx->tmpdir) - 1, "%s/tmp",
427 log_directory);
428 if (written == sizeof(ctx->tmpdir)) {
429 SCLogError(SC_ERR_SPRINTF, "File-store output directory overflow.");
430 SCFree(ctx);
431 return result;
432 }
433
434 ctx->xff_cfg = SCCalloc(1, sizeof(HttpXFFCfg));
435 if (ctx->xff_cfg != NULL) {
436 HttpXFFGetCfg(conf, ctx->xff_cfg);
437 }
438
439 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
440 if (unlikely(output_ctx == NULL)) {
441 SCFree(ctx);
442 return result;
443 }
444
445 output_ctx->data = ctx;
446 output_ctx->DeInit = OutputFilestoreLogDeInitCtx;
447
448 const char *write_fileinfo = ConfNodeLookupChildValue(conf,
449 "write-fileinfo");
450 if (write_fileinfo != NULL && ConfValIsTrue(write_fileinfo)) {
451 SCLogConfig("Filestore (v2) will output fileinfo records.");
452 ctx->fileinfo = true;
453 }
454
455 const char *force_filestore = ConfNodeLookupChildValue(conf,
456 "force-filestore");
457 if (force_filestore != NULL && ConfValIsTrue(force_filestore)) {
458 FileForceFilestoreEnable();
459 SCLogInfo("forcing filestore of all files");
460 }
461
462 const char *force_magic = ConfNodeLookupChildValue(conf, "force-magic");
463 if (force_magic != NULL && ConfValIsTrue(force_magic)) {
464 FileForceMagicEnable();
465 SCLogConfig("Filestore (v2) forcing magic lookup for stored files");
466 }
467
468 FileForceHashParseCfg(conf);
469
470 /* The new filestore requires SHA256. */
471 FileForceSha256Enable();
472
473 ProvidesFeature(FEATURE_OUTPUT_FILESTORE);
474
475 const char *stream_depth_str = ConfNodeLookupChildValue(conf,
476 "stream-depth");
477 if (stream_depth_str != NULL && strcmp(stream_depth_str, "no")) {
478 uint32_t stream_depth = 0;
479 if (ParseSizeStringU32(stream_depth_str,
480 &stream_depth) < 0) {
481 SCLogError(SC_ERR_SIZE_PARSE, "Error parsing "
482 "file-store.stream-depth "
483 "from conf file - %s. Killing engine",
484 stream_depth_str);
485 exit(EXIT_FAILURE);
486 }
487 if (stream_depth) {
488 if (stream_depth <= stream_config.reassembly_depth) {
489 SCLogWarning(SC_WARN_FILESTORE_CONFIG,
490 "file-store.stream-depth value %" PRIu32 " has "
491 "no effect since it's less than stream.reassembly.depth "
492 "value.", stream_depth);
493 } else {
494 FileReassemblyDepthEnable(stream_depth);
495 }
496 }
497 }
498
499 const char *file_count_str = ConfNodeLookupChildValue(conf,
500 "max-open-files");
501 if (file_count_str != NULL) {
502 uint32_t file_count = 0;
503 if (ParseSizeStringU32(file_count_str,
504 &file_count) < 0) {
505 SCLogError(SC_ERR_SIZE_PARSE, "Error parsing "
506 "file-store.max-open-files "
507 "from conf file - %s. Killing engine",
508 file_count_str);
509 exit(EXIT_FAILURE);
510 } else {
511 if (file_count != 0) {
512 FileSetMaxOpenFiles(file_count);
513 SCLogConfig("Filestore (v2) will keep a max of %d "
514 "simultaneously open files", file_count);
515 }
516 }
517 }
518
519 result.ctx = output_ctx;
520 result.ok = true;
521 SCReturnCT(result, "OutputInitResult");
522}
