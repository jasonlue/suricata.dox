98/**
99 * \brief Create a command unix socket on system
100 *
101 * \retval 0 in case of error, 1 in case of success
102 */
103static int UnixNew(UnixCommand * this)
104{
105 struct sockaddr_un addr;
106 int len;
107 int ret;
108 int on = 1;
109 char sockettarget[PATH_MAX];
110 const char *socketname;
111
112 this->start_timestamp = time(NULL);
113 this->socket = -1;
114 this->select_max = 0;
115
116 TAILQ_INIT(&this->commands);
117 TAILQ_INIT(&this->tasks);
118 TAILQ_INIT(&this->clients);
119
120 int check_dir = 0;
121 if (ConfGet("unix-command.filename", &socketname) == 1) {
122 if (PathIsAbsolute(socketname)) {
123 strlcpy(sockettarget, socketname, sizeof(sockettarget));
124 } else {
125 snprintf(sockettarget, sizeof(sockettarget), "%s/%s",
126 SOCKET_PATH, socketname);
127 check_dir = 1;
128 }
129 } else {
130 strlcpy(sockettarget, SOCKET_TARGET, sizeof(sockettarget));
131 check_dir = 1;
132 }
133 SCLogInfo("Using unix socket file '%s'", sockettarget);
134
135 if (check_dir) {
136 struct stat stat_buf;
137 /* coverity[toctou] */
138 if (stat(SOCKET_PATH, &stat_buf) != 0) {
139 /* coverity[toctou] */
140 ret = SCMkDir(SOCKET_PATH, S_IRWXU|S_IXGRP|S_IRGRP);
141 if (ret != 0) {
142 int err = errno;
143 if (err != EEXIST) {
144 SCLogError(SC_ERR_INITIALIZATION,
145 "Cannot create socket directory %s: %s",
146 SOCKET_PATH, strerror(err));
147 return 0;
148 }
149 } else {
150 SCLogInfo("Created socket directory %s",
151 SOCKET_PATH);
152 }
153 }
154 }
155
156 /* Remove socket file */
157 (void) unlink(sockettarget);
158
159 /* set address */
160 addr.sun_family = AF_UNIX;
161 strlcpy(addr.sun_path, sockettarget, sizeof(addr.sun_path));
162 addr.sun_path[sizeof(addr.sun_path) - 1] = 0;
163 len = strlen(addr.sun_path) + sizeof(addr.sun_family) + 1;
164
165 /* create socket */
166 this->socket = socket(AF_UNIX, SOCK_STREAM, 0);
167 if (this->socket == -1) {
168 SCLogWarning(SC_ERR_OPENING_FILE,
169 "Unix Socket: unable to create UNIX socket %s: %s",
170 addr.sun_path, strerror(errno));
171 return 0;
172 }
173 this->select_max = this->socket + 1;
174
175 /* set reuse option */
176 ret = setsockopt(this->socket, SOL_SOCKET, SO_REUSEADDR,
177 (char *) &on, sizeof(on));
178 if ( ret != 0 ) {
179 SCLogWarning(SC_ERR_INITIALIZATION,
180 "Cannot set sockets options: %s.", strerror(errno));
181 }
182
183 /* bind socket */
184 ret = bind(this->socket, (struct sockaddr *) &addr, len);
185 if (ret == -1) {
186 SCLogWarning(SC_ERR_INITIALIZATION,
187 "Unix socket: UNIX socket bind(%s) error: %s",
188 sockettarget, strerror(errno));
189 return 0;
190 }
191
192#if !(defined OS_FREEBSD || defined __OpenBSD__)
193 /* Set file mode: will not fully work on most system, the group
194 * permission is not changed on some Linux. *BSD won't do the
195 * chmod: it returns EINVAL when calling chmod on sockets. */
196 ret = chmod(sockettarget, S_IRUSR|S_IWUSR|S_IRGRP|S_IWGRP);
197 if (ret == -1) {
198 int err = errno;
199 SCLogWarning(SC_ERR_INITIALIZATION,
200 "Unable to change permission on socket: %s (%d)",
201 strerror(err),
202 err);
203 }
204#endif
205
206 /* listen */
207 if (listen(this->socket, 1) == -1) {
208 SCLogWarning(SC_ERR_INITIALIZATION,
209 "Command server: UNIX socket listen() error: %s",
210 strerror(errno));
211 return 0;
212 }
213 return 1;
214}

216static void UnixCommandSetMaxFD(UnixCommand *this)
217{
218 UnixClient *item;
219
220 if (this == NULL) {
221 SCLogError(SC_ERR_INVALID_ARGUMENT, "Unix command is NULL, warn devel");
222 return;
223 }
224
225 this->select_max = this->socket + 1;
226 TAILQ_FOREACH(item, &this->clients, next) {
227 if (item->fd >= this->select_max) {
228 this->select_max = item->fd + 1;
229 }
230 }
231}

233static UnixClient *UnixClientAlloc(void)
234{
235 UnixClient *uclient = SCMalloc(sizeof(UnixClient));
236 if (unlikely(uclient == NULL)) {
237 SCLogError(SC_ERR_MEM_ALLOC, "Can't allocate new client");
238 return NULL;
239 }
240 uclient->mbuf = MemBufferCreateNew(CLIENT_BUFFER_SIZE);
241 if (uclient->mbuf == NULL) {
242 SCLogError(SC_ERR_MEM_ALLOC, "Can't allocate new client send buffer");
243 SCFree(uclient);
244 return NULL;
245 }
246 return uclient;
247}

249static void UnixClientFree(UnixClient *c)
250{
251 if (c != NULL) {
252 MemBufferFree(c->mbuf);
253 SCFree(c);
254 }
255}

257/**
258 * \brief Close the unix socket
259 */
260static void UnixCommandClose(UnixCommand *this, int fd)
261{
262 UnixClient *item;
263 int found = 0;
264
265 TAILQ_FOREACH(item, &this->clients, next) {
266 if (item->fd == fd) {
267 found = 1;
268 break;
269 }
270 }
271
272 if (found == 0) {
273 SCLogError(SC_ERR_INVALID_VALUE, "No fd found in client list");
274 return;
275 }
276
277 TAILQ_REMOVE(&this->clients, item, next);
278
279 close(item->fd);
280 UnixCommandSetMaxFD(this);
281 UnixClientFree(item);
282}

290static int UnixCommandSendJSONToClient(UnixClient *client, json_t *js)
291{
292 MemBufferReset(client->mbuf);
293
294 OutputJSONMemBufferWrapper wrapper = {
295 .buffer = &client->mbuf,
296 .expand_by = CLIENT_BUFFER_SIZE
297 };
298
299 int r = json_dump_callback(js, OutputJSONMemBufferCallback, &wrapper,
300 JSON_PRESERVE_ORDER|JSON_COMPACT|JSON_ENSURE_ASCII|
301 JSON_ESCAPE_SLASH);
302 if (r != 0) {
303 SCLogWarning(SC_ERR_SOCKET, "unable to serialize JSON object");
304 return -1;
305 }
306
307 if (client->version > UNIX_PROTO_V1) {
308 if (MEMBUFFER_OFFSET(client->mbuf) + 1 >= MEMBUFFER_SIZE(client->mbuf)) {
309 MemBufferExpand(&client->mbuf, 1);
310 }
311 MemBufferWriteRaw(client->mbuf, "\n", 1);
312 }
313
314 if (send(client->fd, (const char *)MEMBUFFER_BUFFER(client->mbuf),
315 MEMBUFFER_OFFSET(client->mbuf), MSG_NOSIGNAL) == -1)
316 {
317 SCLogWarning(SC_ERR_SOCKET, "unable to send block of size "
318 "%"PRIuMAX": %s", (uintmax_t)MEMBUFFER_OFFSET(client->mbuf),
319 strerror(errno));
320 return -1;
321 }
322
323 SCLogDebug("sent message of size %"PRIuMAX" to client socket %d",
324 (uintmax_t)MEMBUFFER_OFFSET(client->mbuf), client->fd);
325 return 0;
326}

328/**
329 * \brief Accept a new client on unix socket
330 *
331 * The function is called when a new user is detected
332 * in UnixMain(). It does the initial protocol negotiation
333 * with client.
334 *
335 * \retval 0 in case of error, 1 in case of success
336 */
337static int UnixCommandAccept(UnixCommand *this)
338{
339 char buffer[UNIX_PROTO_VERSION_LENGTH + 1];
340 json_t *client_msg;
341 json_t *server_msg;
342 json_t *version;
343 json_error_t jerror;
344 int client;
345 int client_version;
346 int ret;
347 UnixClient *uclient = NULL;
348
349 /* accept client socket */
350 socklen_t len = sizeof(this->client_addr);
351 client = accept(this->socket, (struct sockaddr *) &this->client_addr,
352 &len);
353 if (client < 0) {
354 SCLogInfo("Unix socket: accept() error: %s",
355 strerror(errno));
356 return 0;
357 }
358 SCLogDebug("Unix socket: client connection");
359
360 /* read client version */
361 buffer[sizeof(buffer)-1] = 0;
362 ret = recv(client, buffer, sizeof(buffer)-1, 0);
363 if (ret < 0) {
364 SCLogInfo("Command server: client doesn't send version");
365 close(client);
366 return 0;
367 }
368 if (ret >= (int)(sizeof(buffer)-1)) {
369 SCLogInfo("Command server: client message is too long, "
370 "disconnect him.");
371 close(client);
372 return 0;
373 }
374 buffer[ret] = 0;
375
376 client_msg = json_loads(buffer, 0, &jerror);
377 if (client_msg == NULL) {
378 SCLogInfo("Invalid command, error on line %d: %s\n", jerror.line, jerror.text);
379 close(client);
380 return 0;
381 }
382
383 version = json_object_get(client_msg, "version");
384 if (!json_is_string(version)) {
385 SCLogInfo("error: version is not a string");
386 close(client);
387 json_decref(client_msg);
388 return 0;
389 }
390
391 /* check client version */
392 if ((strcmp(json_string_value(version), UNIX_PROTO_VERSION) != 0)
393 && (strcmp(json_string_value(version), UNIX_PROTO_VERSION_V1) != 0)) {
394 SCLogInfo("Unix socket: invalid client version: \"%s\"",
395 json_string_value(version));
396 json_decref(client_msg);
397 close(client);
398 return 0;
399 } else {
400 SCLogDebug("Unix socket: client version: \"%s\"",
401 json_string_value(version));
402 if (strcmp(json_string_value(version), UNIX_PROTO_VERSION_V1) == 0) {
403 client_version = UNIX_PROTO_V1;
404 } else {
405 client_version = UNIX_PROTO_V2;
406 }
407 }
408
409 json_decref(client_msg);
410 /* send answer */
411 server_msg = json_object();
412 if (server_msg == NULL) {
413 close(client);
414 return 0;
415 }
416 json_object_set_new(server_msg, "return", json_string("OK"));
417
418 uclient = UnixClientAlloc();
419 if (unlikely(uclient == NULL)) {
420 SCLogError(SC_ERR_MEM_ALLOC, "Can't allocate new client");
421 json_decref(server_msg);
422 close(client);
423 return 0;
424 }
425 uclient->fd = client;
426 uclient->version = client_version;
427
428 if (UnixCommandSendJSONToClient(uclient, server_msg) != 0) {
429 SCLogWarning(SC_ERR_SOCKET, "Unable to send command");
430
431 UnixClientFree(uclient);
432 json_decref(server_msg);
433 close(client);
434 return 0;
435 }
436
437 json_decref(server_msg);
438
439 /* client connected */
440 SCLogDebug("Unix socket: client connected");
441 TAILQ_INSERT_TAIL(&this->clients, uclient, next);
442 UnixCommandSetMaxFD(this);
443 return 1;
444}

460/**
461 * \brief Command dispatcher
462 *
463 * \param this a UnixCommand:: structure
464 * \param command a string containing a json formatted
465 * command
466 *
467 * \retval 0 in case of error, 1 in case of success
468 */
469static int UnixCommandExecute(UnixCommand * this, char *command, UnixClient *client)
470{
471 int ret = 1;
472 json_error_t error;
473 json_t *jsoncmd = NULL;
474 json_t *cmd = NULL;
475 json_t *server_msg = json_object();
476 const char * value;
477 int found = 0;
478 Command *lcmd;
479
480 if (server_msg == NULL) {
481 return 0;
482 }
483
484 jsoncmd = json_loads(command, 0, &error);
485 if (jsoncmd == NULL) {
486 SCLogInfo("Invalid command, error on line %d: %s\n", error.line, error.text);
487 goto error;
488 }
489
490 cmd = json_object_get(jsoncmd, "command");
491 if(!json_is_string(cmd)) {
492 SCLogInfo("error: command is not a string");
493 goto error_cmd;
494 }
495 value = json_string_value(cmd);
496
497 TAILQ_FOREACH(lcmd, &this->commands, next) {
498 if (!strcmp(value, lcmd->name)) {
499 int fret = TM_ECODE_OK;
500 found = 1;
501 if (lcmd->flags & UNIX_CMD_TAKE_ARGS) {
502 cmd = json_object_get(jsoncmd, "arguments");
503 if(!json_is_object(cmd)) {
504 SCLogInfo("error: argument is not an object");
505 goto error_cmd;
506 }
507 }
508 fret = lcmd->Func(cmd, server_msg, lcmd->data);
509 if (fret != TM_ECODE_OK) {
510 ret = 0;
511 }
512 break;
513 }
514 }
515
516 if (found == 0) {
517 json_object_set_new(server_msg, "message", json_string("Unknown command"));
518 ret = 0;
519 }
520
521 switch (ret) {
522 case 0:
523 json_object_set_new(server_msg, "return", json_string("NOK"));
524 break;
525 case 1:
526 json_object_set_new(server_msg, "return", json_string("OK"));
527 break;
528 }
529
530 if (UnixCommandSendJSONToClient(client, server_msg) != 0) {
531 goto error;
532 }
533
534 json_decref(jsoncmd);
535 json_decref(server_msg);
536 return ret;
537
538error_cmd:
539 json_decref(jsoncmd);
540error:
541 json_decref(server_msg);
542 UnixCommandClose(this, client->fd);
543 return 0;
544}

546static void UnixCommandRun(UnixCommand * this, UnixClient *client)
547{
548 char buffer[4096];
549 int ret;
550 if (client->version <= UNIX_PROTO_V1) {
551 ret = recv(client->fd, buffer, sizeof(buffer) - 1, 0);
552 if (ret <= 0) {
553 if (ret == 0) {
554 SCLogDebug("Unix socket: lost connection with client");
555 } else {
556 SCLogError(SC_ERR_SOCKET, "Unix socket: error on recv() from client: %s",
557 strerror(errno));
558 }
559 UnixCommandClose(this, client->fd);
560 return;
561 }
562 if (ret >= (int)(sizeof(buffer)-1)) {
563 SCLogError(SC_ERR_SOCKET, "Command server: client command is too long, "
564 "disconnect him.");
565 UnixCommandClose(this, client->fd);
566 }
567 buffer[ret] = 0;
568 } else {
569 int try = 0;
570 int offset = 0;
571 int cmd_over = 0;
572 ret = recv(client->fd, buffer + offset, sizeof(buffer) - offset - 1, 0);
573 do {
574 if (ret <= 0) {
575 if (ret == 0) {
576 SCLogDebug("Unix socket: lost connection with client");
577 } else {
578 SCLogError(SC_ERR_SOCKET, "Unix socket: error on recv() from client: %s",
579 strerror(errno));
580 }
581 UnixCommandClose(this, client->fd);
582 return;
583 }
584 if (ret >= (int)(sizeof(buffer)- offset - 1)) {
585 SCLogInfo("Command server: client command is too long, "
586 "disconnect him.");
587 UnixCommandClose(this, client->fd);
588 }
589 if (buffer[ret - 1] == '\n') {
590 buffer[ret-1] = 0;
591 cmd_over = 1;
592 } else {
593 struct timeval tv;
594 fd_set select_set;
595 offset += ret;
596 do {
597 FD_ZERO(&select_set);
598 FD_SET(client->fd, &select_set);
599 tv.tv_sec = 0;
600 tv.tv_usec = 200 * 1000;
601 try++;
602 ret = select(client->fd, &select_set, NULL, NULL, &tv);
603 /* catch select() error */
604 if (ret == -1) {
605 /* Signal was caught: just ignore it */
606 if (errno != EINTR) {
607 SCLogInfo("Unix socket: lost connection with client");
608 UnixCommandClose(this, client->fd);
609 return;
610 }
611 }
612 } while (ret == 0 && try < 3);
613 if (ret > 0) {
614 ret = recv(client->fd, buffer + offset,
615 sizeof(buffer) - offset - 1, 0);
616 }
617 }
618 } while (try < 3 && cmd_over == 0);
619
620 if (try == 3 && cmd_over == 0) {
621 SCLogInfo("Unix socket: imcomplete client message, closing connection");
622 UnixCommandClose(this, client->fd);
623 return;
624 }
625 }
626 UnixCommandExecute(this, buffer, client);
627}

629/**
630 * \brief Select function
631 *
632 * \retval 0 in case of error, 1 in case of success
633 */
634static int UnixMain(UnixCommand * this)
635{
636 struct timeval tv;
637 int ret;
638 fd_set select_set;
639 UnixClient *uclient;
640 UnixClient *tclient;
641
642 /* Wait activity on the socket */
643 FD_ZERO(&select_set);
644 FD_SET(this->socket, &select_set);
645 TAILQ_FOREACH(uclient, &this->clients, next) {
646 FD_SET(uclient->fd, &select_set);
647 }
648
649 tv.tv_sec = 0;
650 tv.tv_usec = 200 * 1000;
651 ret = select(this->select_max, &select_set, NULL, NULL, &tv);
652
653 /* catch select() error */
654 if (ret == -1) {
655 /* Signal was caught: just ignore it */
656 if (errno == EINTR) {
657 return 1;
658 }
659 SCLogError(SC_ERR_SOCKET, "Command server: select() fatal error: %s", strerror(errno));
660 return 0;
661 }
662
663 if (suricata_ctl_flags & SURICATA_STOP) {
664 TAILQ_FOREACH_SAFE(uclient, &this->clients, next, tclient) {
665 UnixCommandClose(this, uclient->fd);
666 }
667 return 1;
668 }
669
670 /* timeout: continue */
671 if (ret == 0) {
672 return 1;
673 }
674
675 TAILQ_FOREACH_SAFE(uclient, &this->clients, next, tclient) {
676 if (FD_ISSET(uclient->fd, &select_set)) {
677 UnixCommandRun(this, uclient);
678 }
679 }
680 if (FD_ISSET(this->socket, &select_set)) {
681 if (!UnixCommandAccept(this))
682 return 1;
683 }
684
685 return 1;
686}

733static TmEcode UnixManagerReloadRulesWrapper(json_t *cmd, json_t *server_msg, void *data, int do_wait)
734{
735 SCEnter();
736
737 if (SuriHasSigFile()) {
738 json_object_set_new(server_msg, "message",
739 json_string("Live rule reload not possible if -s "
740 "or -S option used at runtime."));
741 SCReturnInt(TM_ECODE_FAILED);
742 }
743
744 int r = DetectEngineReloadStart();
745
746 if (r == 0 && do_wait) {
747 while (!DetectEngineReloadIsIdle())
748 usleep(100);
749 } else {
750 if (r == -1) {
751 json_object_set_new(server_msg, "message", json_string("Reload already in progress"));
752 SCReturnInt(TM_ECODE_FAILED);
753 }
754 }
755
756 json_object_set_new(server_msg, "message", json_string("done"));
757 SCReturnInt(TM_ECODE_OK);
758}

944/**
945 * \brief Add a command to the list of commands
946 *
947 * This function adds a command to the list of commands available
948 * through the unix socket.
949 *
950 * When a command is received from user through the unix socket, the content
951 * of 'Command' field in the JSON message is match against keyword, then the
952 * Func is called. See UnixSocketAddPcapFile() for an example.
953 *
954 * \param keyword name of the command
955 * \param Func function to run when command is received
956 * \param data a pointer to data that are passed to Func when it is run
957 * \param flags a flag now used to tune the command type
958 * \retval TM_ECODE_OK in case of success, TM_ECODE_FAILED in case of failure
959 */
960TmEcode UnixManagerRegisterCommand(const char * keyword,
961 TmEcode (*Func)(json_t *, json_t *, void *),
962 void *data, int flags)
963{
964 SCEnter();
965 Command *cmd = NULL;
966 Command *lcmd = NULL;
967
968 if (Func == NULL) {
969 SCLogError(SC_ERR_INVALID_ARGUMENT, "Null function");
970 SCReturnInt(TM_ECODE_FAILED);
971 }
972
973 if (keyword == NULL) {
974 SCLogError(SC_ERR_INVALID_ARGUMENT, "Null keyword");
975 SCReturnInt(TM_ECODE_FAILED);
976 }
977
978 TAILQ_FOREACH(lcmd, &command.commands, next) {
979 if (!strcmp(keyword, lcmd->name)) {
980 SCLogError(SC_ERR_INVALID_ARGUMENT, "%s already registered", keyword);
981 SCReturnInt(TM_ECODE_FAILED);
982 }
983 }
984
985 cmd = SCMalloc(sizeof(Command));
986 if (unlikely(cmd == NULL)) {
987 SCLogError(SC_ERR_MEM_ALLOC, "Can't alloc cmd");
988 SCReturnInt(TM_ECODE_FAILED);
989 }
990 cmd->name = SCStrdup(keyword);
991 if (unlikely(cmd->name == NULL)) {
992 SCLogError(SC_ERR_MEM_ALLOC, "Can't alloc cmd name");
993 SCFree(cmd);
994 SCReturnInt(TM_ECODE_FAILED);
995 }
996 cmd->Func = Func;
997 cmd->data = data;
998 cmd->flags = flags;
999 /* Add it to the list */
1000 TAILQ_INSERT_TAIL(&command.commands, cmd, next);
1001
1002 SCReturnInt(TM_ECODE_OK);
1003}

1111static TmEcode UnixManager(ThreadVars *th_v, void *thread_data)
1112{
1113 int ret;
1114
1115 /* set the thread name */
1116 SCLogDebug("%s started...", th_v->name);
1117
1118 StatsSetupPrivate(th_v);
1119
1120 /* Set the threads capability */
1121 th_v->cap_flags = 0;
1122 SCDropCaps(th_v);
1123
1124 TmThreadsSetFlag(th_v, THV_INIT_DONE);
1125 while (1) {
1126 ret = UnixMain(&command);
1127 if (ret == 0) {
1128 SCLogError(SC_ERR_FATAL, "Fatal error on unix socket");
1129 }
1130
1131 if ((ret == 0) || (TmThreadsCheckFlag(th_v, THV_KILL))) {
1132 UnixClient *item;
1133 UnixClient *titem;
1134 TAILQ_FOREACH_SAFE(item, &(&command)->clients, next, titem) {
1135 close(item->fd);
1136 SCFree(item);
1137 }
1138 StatsSyncCounters(th_v);
1139 break;
1140 }
1141
1142 UnixCommandBackgroundTasks(&command);
1143 }
1144 return TM_ECODE_OK;
1145}

1198 *
1199 * \todo Kinda hackish since it uses the tv name to identify unix manager
1200 * thread. We need an all weather identification scheme.

1202void UnixSocketKillSocketThread(void)

1205
1206again:
1207 SCMutexLock(&tv_root_lock);
1208
1209 /* unix manager thread(s) is/are a part of command threads */
1210 tv = tv_root[TVT_CMD];
1211
1212 while (tv != NULL) {
1213 if (strcasecmp(tv->name, "UnixManagerThread") == 0) {

1215 * THV_RUNNING_DONE set, so we can set the correct flag
1216 * and exit.
1217 */

1219 TmThreadsSetFlag(tv, THV_KILL);
1220 TmThreadsSetFlag(tv, THV_DEINIT);
1221 TmThreadsSetFlag(tv, THV_CLOSED);
1222 break;
1223 }
1224 TmThreadsSetFlag(tv, THV_KILL);

1231 }
