125static inline void TmThreadsCleanDecodePQ(PacketQueueNoLock *pq)
126{
127 while (1) {
128 Packet *p = PacketDequeueNoLock(pq);
129 if (unlikely(p == NULL))
130 break;
131 TmqhOutputPacketpool(NULL, p);
132 }
133}

135static inline void TmThreadsSlotProcessPktFail(ThreadVars *tv, TmSlot *s, Packet *p)
136{
137 if (p != NULL) {
138 TmqhOutputPacketpool(tv, p);
139 }
140 TmThreadsCleanDecodePQ(&tv->decode_pq);
141 if (tv->stream_pq_local) {
142 SCMutexLock(&tv->stream_pq_local->mutex_q);
143 TmqhReleasePacketsToPacketPool(tv->stream_pq_local);
144 SCMutexUnlock(&tv->stream_pq_local->mutex_q);
145 }
146 TmThreadsSetFlag(tv, THV_FAILED);
147}

149/**
150 * \brief Handle timeout from the capture layer. Checks
151 * stream_pq which may have been filled by the flow
152 * manager.
153 * \param s pipeline to run on these packets.
154 */
155static inline bool TmThreadsHandleInjectedPackets(ThreadVars *tv)
156{
157 PacketQueue *pq = tv->stream_pq_local;
158 if (pq && pq->len > 0) {
159 while (1) {
160 SCMutexLock(&pq->mutex_q);
161 Packet *extra_p = PacketDequeue(pq);
162 SCMutexUnlock(&pq->mutex_q);
163 if (extra_p == NULL)
164 break;
165 TmEcode r = TmThreadsSlotVarRun(tv, extra_p, tv->tm_flowworker);
166 if (r == TM_ECODE_FAILED) {
167 TmThreadsSlotProcessPktFail(tv, tv->tm_flowworker, extra_p);
168 break;
169 }
170 tv->tmqh_out(tv, extra_p);
171 }
172 return true;
173 } else {
174 return false;
175 }
176}

178/**
179 * \brief Process the rest of the functions (if any) and queue.
180 */
181static inline TmEcode TmThreadsSlotProcessPkt(ThreadVars *tv, TmSlot *s, Packet *p)
182{
183 if (s == NULL) {
184 tv->tmqh_out(tv, p);
185 return TM_ECODE_OK;
186 }
187
188 TmEcode r = TmThreadsSlotVarRun(tv, p, s);
189 if (unlikely(r == TM_ECODE_FAILED)) {
190 TmThreadsSlotProcessPktFail(tv, s, p);
191 return TM_ECODE_FAILED;
192 }
193
194 tv->tmqh_out(tv, p);
195
196 TmThreadsHandleInjectedPackets(tv);
197
198 return TM_ECODE_OK;
199}

201/** \brief inject packet if THV_CAPTURE_INJECT_PKT is set
202 * Allow caller to supply their own packet
203 *
204 * Meant for detect reload process that interupts an sleeping capture thread
205 * to force a packet through the engine to complete a reload */
206static inline void TmThreadsCaptureInjectPacket(ThreadVars *tv, Packet *p)
207{
208 TmThreadsUnsetFlag(tv, THV_CAPTURE_INJECT_PKT);
209 if (p == NULL)
210 p = PacketGetFromQueueOrAlloc();
211 if (p != NULL) {
212 p->flags |= PKT_PSEUDO_STREAM_END;
213 PKT_SET_SRC(p, PKT_SRC_CAPTURE_TIMEOUT);
214 if (TmThreadsSlotProcessPkt(tv, tv->tm_flowworker, p) != TM_ECODE_OK) {
215 TmqhOutputPacketpool(tv, p);
216 }
217 }
218}

220/** \brief handle capture timeout
221 * When a capture method times out we check for house keeping
222 * tasks in the capture thread.
223 *
224 * \param p packet. Capture method may have taken a packet from
225 * the pool prior to the timing out call. We will then
226 * use that packet. Otherwise we can get our own.
227 */
228static inline void TmThreadsCaptureHandleTimeout(ThreadVars *tv, Packet *p)
229{
230 if (TmThreadsCheckFlag(tv, THV_CAPTURE_INJECT_PKT)) {
231 TmThreadsCaptureInjectPacket(tv, p); /* consumes 'p' */
232 return;
233
234 } else {
235 if (TmThreadsHandleInjectedPackets(tv) == false) {
236 /* see if we have to do some house keeping */
237 if (tv->flow_queue && SC_ATOMIC_GET(tv->flow_queue->non_empty) == true) {
238 TmThreadsCaptureInjectPacket(tv, p); /* consumes 'p' */
239 return;
240 }
241 }
242 }
243
244 /* packet could have been passed to us that we won't use
245 * return it to the pool. */
246 if (p != NULL)
247 tv->tmqh_out(tv, p);
248}
