87{
88 sigmatch_table[DETECT_AL_TLS_CERTS].name = "tls.certs";

99 SIG_FLAG_TOCLIENT, TLS_STATE_CERT_READY,
100 DetectEngineInspectTlsCerts, NULL);
101
102 DetectAppLayerMpmRegister2("tls.certs", SIG_FLAG_TOCLIENT, 2,
103 PrefilterMpmTlsCertsRegister, NULL, ALPROTO_TLS,
104 TLS_STATE_CERT_READY);
105
106 DetectBufferTypeSetDescriptionByName("tls.certs", "TLS certificate");
107
108 g_tls_certs_buffer_id = DetectBufferTypeGetByName("tls.certs");
109}

111/**
112 * \brief This function setup the tls.certs modifier keyword
113 *
114 * \param de_ctx Pointer to the Detect Engine Context
115 * \param s Pointer to the Signature to which the keyword belongs
116 * \param str Should hold an empty string always
117 *
118 * \retval 0 On success
119 * \retval -1 On failure
120 */
121static int DetectTlsCertsSetup(DetectEngineCtx *de_ctx, Signature *s,
122 const char *str)
123{
124 if (DetectBufferSetActiveList(s, g_tls_certs_buffer_id) < 0)
125 return -1;

127 if (DetectSignatureSetAppProto(s, ALPROTO_TLS) < 0)
128 return -1;
129
130 return 0;

133static InspectionBuffer *TlsCertsGetData(DetectEngineThreadCtx *det_ctx,
134 const DetectEngineTransforms *transforms, Flow *f,
135 struct TlsCertsGetDataArgs *cbdata, int list_id)
136{
137 SCEnter();
138
139 InspectionBuffer *buffer =
140 InspectionBufferMultipleForListGet(det_ctx, list_id, cbdata->local_id);
141 if (buffer == NULL)
142 return NULL;
143
144 const SSLState *ssl_state = (SSLState *)f->alstate;
145
146 if (TAILQ_EMPTY(&ssl_state->server_connp.certs)) {
147 return NULL;
148 }
149
150 if (cbdata->cert == NULL) {
151 cbdata->cert = TAILQ_FIRST(&ssl_state->server_connp.certs);
152 } else {
153 cbdata->cert = TAILQ_NEXT(cbdata->cert, next);
154 }
155 if (cbdata->cert == NULL) {
156 return NULL;
157 }
158
159 InspectionBufferSetupMulti(buffer, transforms, cbdata->cert->cert_data, cbdata->cert->cert_len);
160
161 SCReturnPtr(buffer, "InspectionBuffer");
162}

164static uint8_t DetectEngineInspectTlsCerts(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
165 const DetectEngineAppInspectionEngine *engine, const Signature *s, Flow *f, uint8_t flags,
166 void *alstate, void *txv, uint64_t tx_id)
167{
168 const DetectEngineTransforms *transforms = NULL;
169 if (!engine->mpm) {
170 transforms = engine->v2.transforms;
171 }
172
173 struct TlsCertsGetDataArgs cbdata = { 0, NULL };
174
175 while (1)
176 {
177 InspectionBuffer *buffer = TlsCertsGetData(det_ctx, transforms, f,
178 &cbdata, engine->sm_list);
179 if (buffer == NULL || buffer->inspect == NULL)
180 break;
181
182 det_ctx->buffer_offset = 0;
183 det_ctx->discontinue_matching = 0;
184 det_ctx->inspection_recursion_counter = 0;
185
186 const int match = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd,
187 NULL, f, (uint8_t *)buffer->inspect,
188 buffer->inspect_len,
189 buffer->inspect_offset, DETECT_CI_FLAGS_SINGLE,
190 DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
191 if (match == 1) {
192 return DETECT_ENGINE_INSPECT_SIG_MATCH;
193 }
194
195 cbdata.local_id++;
196 }
197
198 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
199}

201static void PrefilterTxTlsCerts(DetectEngineThreadCtx *det_ctx,
202 const void *pectx, Packet *p, Flow *f, void *txv,
203 const uint64_t idx, const uint8_t flags)
204{
205 SCEnter();
206
207 const PrefilterMpmTlsCerts *ctx = (const PrefilterMpmTlsCerts *)pectx;
208 const MpmCtx *mpm_ctx = ctx->mpm_ctx;
209 const int list_id = ctx->list_id;
210
211 struct TlsCertsGetDataArgs cbdata = { 0, NULL };
212
213 while (1)
214 {
215 InspectionBuffer *buffer = TlsCertsGetData(det_ctx, ctx->transforms,
216 f, &cbdata, list_id);
217 if (buffer == NULL)
218 break;
219
220 if (buffer->inspect_len >= mpm_ctx->minlen) {
221 (void)mpm_table[mpm_ctx->mpm_type].Search(mpm_ctx,
222 &det_ctx->mtcu, &det_ctx->pmq,
223 buffer->inspect, buffer->inspect_len);
224 PREFILTER_PROFILING_ADD_BYTES(det_ctx, buffer->inspect_len);
225 }
226
227 cbdata.local_id++;
228 }
229}

260/**
261 * \internal
262 * \brief Function to match cert chain length in TLS
263 *
264 * \param t Pointer to thread vars.
265 * \param det_ctx Pointer to the pattern matcher thread.
266 * \param f Pointer to the current flow.
267 * \param flags Flags.
268 * \param state App layer state.
269 * \param s Pointer to the Signature.
270 * \param m Pointer to the sigmatch that we will cast into
271 * DetectU64Data.
272 *
273 * \retval 0 no match.
274 * \retval 1 match.
275 */
276static int DetectTLSCertChainLenMatch(DetectEngineThreadCtx *det_ctx, Flow *f, uint8_t flags,
277 void *state, void *txv, const Signature *s, const SigMatchCtx *ctx)
278{
279 SCEnter();
280
281 SSLState *ssl_state = state;
282 if (flags & STREAM_TOCLIENT) {
283 SSLStateConnp *connp = &ssl_state->server_connp;
284 uint32_t cnt = 0;
285 SSLCertsChain *cert;
286 TAILQ_FOREACH (cert, &connp->certs, next) {
287 cnt++;
288 }
289 SCLogDebug("%u certs in chain", cnt);
290
291 const DetectU32Data *dd = (const DetectU32Data *)ctx;
292 if (DetectU32Match(cnt, dd)) {
293 SCReturnInt(1);
294 }
295 }
296 SCReturnInt(0);
297}

310/**
311 * \brief Function to add the parsed tls cert chain len field into the current signature.
312 *
313 * \param de_ctx Pointer to the Detection Engine Context.
314 * \param s Pointer to the Current Signature.
315 * \param rawstr Pointer to the user provided flags options.
316 * \param type Defines if this is notBefore or notAfter.
317 *
318 * \retval 0 on Success.
319 * \retval -1 on Failure.
320 */
321static int DetectTLSCertChainLenSetup(DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
322{
323 if (DetectSignatureSetAppProto(s, ALPROTO_TLS) != 0)
324 return -1;
325
326 DetectU32Data *dd = DetectU32Parse(rawstr);
327 if (dd == NULL) {
328 SCLogError(SC_ERR_INVALID_ARGUMENT, "Parsing \'%s\' failed for %s", rawstr,
329 sigmatch_table[KEYWORD_ID].name);
330 return -1;
331 }
332
333 SigMatch *sm = SigMatchAlloc();
334 if (sm == NULL) {
335 rs_detect_u32_free(dd);
336 return -1;
337 }
338 sm->type = KEYWORD_ID;
339 sm->ctx = (void *)dd;
340
341 SigMatchAppendSMToList(s, sm, g_tls_cert_buffer_id);
342 return 0;
343}
