86static inline char *SkipWhiteSpaceTill(char *p, char *savep)
87{
88 char *sp = p;
89 if (unlikely(p == NULL)) {
90 return NULL;
91 }
92 while (((*sp == '\t') || (*sp == ' ')) && (sp < savep)) {
93 sp++;
94 }
95 return sp;
96}

98static bool EveEmailJsonArrayFromCommaList(JsonBuilder *js, const uint8_t *val, size_t len)
99{
100 char *savep = NULL;
101 char *p;
102 char *sp;
103 char *to_line = BytesToString((uint8_t *)val, len);
104 if (likely(to_line != NULL)) {
105 p = strtok_r(to_line, ",", &savep);
106 if (p == NULL) {
107 SCFree(to_line);
108 return false;
109 }
110 sp = SkipWhiteSpaceTill(p, savep);
111 jb_append_string(js, sp);
112 while ((p = strtok_r(NULL, ",", &savep)) != NULL) {
113 sp = SkipWhiteSpaceTill(p, savep);
114 jb_append_string(js, sp);
115 }
116 } else {
117 return false;
118 }
119 SCFree(to_line);
120 return true;
121}

123static void EveEmailLogJSONMd5(OutputJsonEmailCtx *email_ctx, JsonBuilder *js, SMTPTransaction *tx)
124{
125 if (email_ctx->flags & LOG_EMAIL_SUBJECT_MD5) {
126 MimeDecEntity *entity = tx->msg_tail;
127 if (entity == NULL) {
128 return;
129 }
130 MimeDecField *field = MimeDecFindField(entity, "subject");
131 if (field != NULL) {
132 char smd5[SC_MD5_HEX_LEN + 1];
133 SCMd5HashBufferToHex((uint8_t *)field->value, field->value_len, smd5, sizeof(smd5));
134 jb_set_string(js, "subject_md5", smd5);
135 }
136 }
137
138 if (email_ctx->flags & LOG_EMAIL_BODY_MD5) {
139 MimeDecParseState *mime_state = tx->mime_state;
140 if (mime_state && mime_state->has_md5 && (mime_state->state_flag == PARSE_DONE)) {
141 jb_set_hex(js, "body_md5", mime_state->md5, (uint32_t)sizeof(mime_state->md5));
142 }
143 }
144}

146static int JsonEmailAddToJsonArray(const uint8_t *val, size_t len, void *data)
147{
148 JsonBuilder *ajs = data;
149
150 if (ajs == NULL)
151 return 0;
152 char *value = BytesToString((uint8_t *)val, len);
153 jb_append_string(ajs, value);
154 SCFree(value);
155 return 1;
156}

158static void EveEmailLogJSONCustom(OutputJsonEmailCtx *email_ctx, JsonBuilder *js, SMTPTransaction *tx)
159{
160 int f = 0;
161 JsonBuilderMark mark = { 0, 0, 0 };
162 MimeDecField *field;
163 MimeDecEntity *entity = tx->msg_tail;
164 if (entity == NULL) {
165 return;
166 }
167
168 while(email_fields[f].config_field) {
169 if (((email_ctx->fields & (1ULL<<f)) != 0)
170 ||
171 ((email_ctx->flags & LOG_EMAIL_EXTENDED) && (email_fields[f].flags & LOG_EMAIL_EXTENDED))
172 ) {
173 if (email_fields[f].flags & LOG_EMAIL_ARRAY) {
174 jb_get_mark(js, &mark);
175 jb_open_array(js, email_fields[f].config_field);
176 int found = MimeDecFindFieldsForEach(entity, email_fields[f].email_field, JsonEmailAddToJsonArray, js);
177 if (found > 0) {
178 jb_close(js);
179 } else {
180 jb_restore_mark(js, &mark);
181 }
182 } else if (email_fields[f].flags & LOG_EMAIL_COMMA) {
183 field = MimeDecFindField(entity, email_fields[f].email_field);
184 if (field) {
185 jb_get_mark(js, &mark);
186 jb_open_array(js, email_fields[f].config_field);
187 if (EveEmailJsonArrayFromCommaList(js, field->value, field->value_len)) {
188 jb_close(js);
189 } else {
190 jb_restore_mark(js, &mark);
191 }
192 }
193 } else {
194 field = MimeDecFindField(entity, email_fields[f].email_field);
195 if (field != NULL) {
196 char *s = BytesToString((uint8_t *)field->value,
197 (size_t)field->value_len);
198 if (likely(s != NULL)) {
199 jb_set_string(js, email_fields[f].config_field, s);
200 SCFree(s);
201 }
202 }
203 }
204
205 }
206 f++;
207 }
208}

210/* JSON format logging */
211static bool EveEmailLogJsonData(const Flow *f, void *state, void *vtx, uint64_t tx_id, JsonBuilder *sjs)
212{
213 SMTPState *smtp_state;
214 MimeDecParseState *mime_state;
215 MimeDecEntity *entity;
216 JsonBuilderMark mark = { 0, 0, 0 };
217
218 /* check if we have SMTP state or not */
219 AppProto proto = FlowGetAppProtocol(f);
220 switch (proto) {
221 case ALPROTO_SMTP:
222 smtp_state = (SMTPState *)state;
223 if (smtp_state == NULL) {
224 SCLogDebug("no smtp state, so no request logging");
225 jb_free(sjs);
226 SCReturnPtr(NULL, "JsonBuilder");
227 }
228 SMTPTransaction *tx = vtx;
229 mime_state = tx->mime_state;
230 entity = tx->msg_tail;
231 SCLogDebug("lets go mime_state %p, entity %p, state_flag %u", mime_state, entity, mime_state ? mime_state->state_flag : 0);
232 break;
233 default:
234 /* don't know how we got here */
235 SCReturnBool(false);
236 }
237 if ((mime_state != NULL)) {
238 if (entity == NULL) {
239 SCReturnBool(false);
240 }
241
242 jb_set_string(sjs, "status", MimeDecParseStateGetStatus(mime_state));
243
244 MimeDecField *field;
245
246 /* From: */
247 field = MimeDecFindField(entity, "from");
248 if (field != NULL) {
249 char *s = BytesToString((uint8_t *)field->value,
250 (size_t)field->value_len);
251 if (likely(s != NULL)) {
252 //printf("From: \"%s\"\n", s);
253 char * sp = SkipWhiteSpaceTill(s, s + strlen(s));
254 jb_set_string(sjs, "from", sp);
255 SCFree(s);
256 }
257 }
258
259 /* To: */
260 field = MimeDecFindField(entity, "to");
261 if (field != NULL) {
262 jb_get_mark(sjs, &mark);
263 jb_open_array(sjs, "to");
264 if (EveEmailJsonArrayFromCommaList(sjs, field->value, field->value_len)) {
265 jb_close(sjs);
266 } else {
267 jb_restore_mark(sjs, &mark);
268 }
269 }
270
271 /* Cc: */
272 field = MimeDecFindField(entity, "cc");
273 if (field != NULL) {
274 jb_get_mark(sjs, &mark);
275 jb_open_array(sjs, "cc");
276 if (EveEmailJsonArrayFromCommaList(sjs, field->value, field->value_len)) {
277 jb_close(sjs);
278 } else {
279 jb_restore_mark(sjs, &mark);
280 }
281 }
282
283 if (mime_state->stack == NULL || mime_state->stack->top == NULL || mime_state->stack->top->data == NULL) {
284 SCReturnBool(false);
285 }
286
287 entity = (MimeDecEntity *)mime_state->stack->top->data;
288 int attch_cnt = 0;
289 int url_cnt = 0;
290 JsonBuilder *js_attch = jb_new_array();
291 JsonBuilder *js_url = jb_new_array();
292 if (entity->url_list != NULL) {
293 MimeDecUrl *url;
294 bool has_ipv6_url = false;
295 bool has_ipv4_url = false;
296 bool has_exe_url = false;
297 for (url = entity->url_list; url != NULL; url = url->next) {
298 char *s = BytesToString((uint8_t *)url->url,
299 (size_t)url->url_len);
300 if (s != NULL) {
301 jb_append_string(js_url, s);
302 if (url->url_flags & URL_IS_EXE)
303 has_exe_url = true;
304 if (url->url_flags & URL_IS_IP6)
305 has_ipv6_url = true;
306 if (url->url_flags & URL_IS_IP4)
307 has_ipv6_url = true;
308 SCFree(s);
309 url_cnt += 1;
310 }
311 }
312 jb_set_bool(sjs, "has_ipv6_url", has_ipv6_url);
313 jb_set_bool(sjs, "has_ipv4_url", has_ipv4_url);
314 jb_set_bool(sjs, "has_exe_url", has_exe_url);
315 }
316 for (entity = entity->child; entity != NULL; entity = entity->next) {
317 if (entity->ctnt_flags & CTNT_IS_ATTACHMENT) {
318
319 char *s = BytesToString((uint8_t *)entity->filename,
320 (size_t)entity->filename_len);
321 jb_append_string(js_attch, s);
322 SCFree(s);
323 attch_cnt += 1;
324 }
325 if (entity->url_list != NULL) {
326 MimeDecUrl *url;
327 for (url = entity->url_list; url != NULL; url = url->next) {
328 char *s = BytesToString((uint8_t *)url->url,
329 (size_t)url->url_len);
330 if (s != NULL) {
331 jb_append_string(js_url, s);
332 SCFree(s);
333 url_cnt += 1;
334 }
335 }
336 }
337 }
338 if (attch_cnt > 0) {
339 jb_close(js_attch);
340 jb_set_object(sjs, "attachment", js_attch);
341 }
342 jb_free(js_attch);
343 if (url_cnt > 0) {
344 jb_close(js_url);
345 jb_set_object(sjs, "url", js_url);
346 }
347 jb_free(js_url);
348 SCReturnBool(true);
349 }
350
351 SCReturnBool(false);
352}

355TmEcode EveEmailLogJson(JsonEmailLogThread *aft, JsonBuilder *js, const Packet *p, Flow *f, void *state, void *vtx, uint64_t tx_id)
356{
357 OutputJsonEmailCtx *email_ctx = aft->emaillog_ctx;
358 SMTPTransaction *tx = (SMTPTransaction *) vtx;
359 JsonBuilderMark mark = { 0, 0, 0 };
360
361 jb_get_mark(js, &mark);
362 jb_open_object(js, "email");
363 if (!EveEmailLogJsonData(f, state, vtx, tx_id, js)) {
364 jb_restore_mark(js, &mark);
365 SCReturnInt(TM_ECODE_FAILED);
366 }
367
368 if ((email_ctx->flags & LOG_EMAIL_EXTENDED) || (email_ctx->fields != 0))
369 EveEmailLogJSONCustom(email_ctx, js, tx);
370
371 if (!g_disable_hashing) {
372 EveEmailLogJSONMd5(email_ctx, js, tx);
373 }
374
375 jb_close(js);
376 SCReturnInt(TM_ECODE_OK);
377}

392void OutputEmailInitConf(ConfNode *conf, OutputJsonEmailCtx *email_ctx)
393{
394 if (conf) {
395 const char *extended = ConfNodeLookupChildValue(conf, "extended");
396
397 if (extended != NULL) {
398 if (ConfValIsTrue(extended)) {
399 email_ctx->flags = LOG_EMAIL_EXTENDED;
400 }
401 }
402
403 email_ctx->fields = 0;
404 ConfNode *custom;
405 if ((custom = ConfNodeLookupChild(conf, "custom")) != NULL) {
406 ConfNode *field;
407 TAILQ_FOREACH(field, &custom->head, next) {
408 if (field != NULL) {
409 int f = 0;
410 while(email_fields[f].config_field) {
411 if ((strcmp(email_fields[f].config_field,
412 field->val) == 0) ||
413 (strcasecmp(email_fields[f].email_field,
414 field->val) == 0))
415 {
416 email_ctx->fields |= (1ULL<<f);
417 break;
418 }
419 f++;
420 }
421 }
422 }
423 }
424
425 email_ctx->flags = 0;
426 ConfNode *md5_conf;
427 if ((md5_conf = ConfNodeLookupChild(conf, "md5")) != NULL) {
428 ConfNode *field;
429 TAILQ_FOREACH(field, &md5_conf->head, next) {
430 if (field != NULL) {
431 if (strcmp("body", field->val) == 0) {
432 SCLogInfo("Going to log the md5 sum of email body");
433 email_ctx->flags |= LOG_EMAIL_BODY_MD5;
434 }
435 if (strcmp("subject", field->val) == 0) {
436 SCLogInfo("Going to log the md5 sum of email subject");
437 email_ctx->flags |= LOG_EMAIL_SUBJECT_MD5;
438 }
439 }
440 }
441 }
442 }
443 return;
