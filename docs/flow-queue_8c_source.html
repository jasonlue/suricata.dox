65void FlowQueuePrivateAppendFlow(FlowQueuePrivate *fqc, Flow *f)
66{
67 if (fqc->top == NULL) {
68 fqc->top = fqc->bot = f;
69 fqc->len = 1;
70 } else {
71 fqc->bot->next = f;
72 fqc->bot = f;
73 fqc->len++;
74 }
75 f->next = NULL;

79{

88void FlowQueuePrivateAppendPrivate(FlowQueuePrivate *dest, FlowQueuePrivate *src)
89{
90 if (src->top == NULL)
91 return;
92
93 if (dest->bot == NULL) {
94 dest->top = src->top;
95 dest->bot = src->bot;
96 dest->len = src->len;
97 } else {
98 dest->bot->next = src->top;
99 dest->bot = src->bot;
100 dest->len += src->len;
101 }
102 src->top = src->bot = NULL;
103 src->len = 0;
104}

119void FlowQueueAppendPrivate(FlowQueue *fq, FlowQueuePrivate *fqc)
120{
121 if (fqc->top == NULL)
122 return;
123
124 FQLOCK_LOCK(fq);
125 if (fq->qbot == NULL) {
126 fq->qtop = fqc->top;
127 fq->qbot = fqc->bot;
128 fq->qlen = fqc->len;
129 } else {
130 fq->qbot->next = fqc->top;
131 fq->qbot = fqc->bot;
132 fq->qlen += fqc->len;
133 }
134 FlowQueueAtomicSetNonEmpty(fq);
135 FQLOCK_UNLOCK(fq);
136 fqc->top = fqc->bot = NULL;
137 fqc->len = 0;
138}

151Flow *FlowQueuePrivateGetFromTop(FlowQueuePrivate *fqc)
152{
153 Flow *f = fqc->top;
154 if (f == NULL) {
155 return NULL;
156 }
157
158 fqc->top = f->next;
159 f->next = NULL;
160 fqc->len--;
161 if (fqc->top == NULL) {
162 fqc->bot = NULL;
163 }
164 return f;
165}

167/**
168 * \brief add a flow to a queue
169 *
170 * \param q queue
171 * \param f flow
172 */
173void FlowEnqueue (FlowQueue *q, Flow *f)
174{
175#ifdef DEBUG
176 BUG_ON(q == NULL || f == NULL);
177#endif
178 FQLOCK_LOCK(q);
179 FlowQueuePrivateAppendFlow(&q->priv, f);
180 FlowQueueAtomicSetNonEmpty(q);
181 FQLOCK_UNLOCK(q);
182}

184/**
185 * \brief remove a flow from the queue

187 * \param q queue
188 *
189 * \retval f flow or NULL if empty list.
190 */
191Flow *FlowDequeue (FlowQueue *q)
192{
193 FQLOCK_LOCK(q);
194 Flow *f = FlowQueuePrivateGetFromTop(&q->priv);
195 if (f == NULL)
196 FlowQueueAtomicSetEmpty(q);
197 FQLOCK_UNLOCK(q);
198 return f;
