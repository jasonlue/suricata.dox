45void SigCleanSignatures(DetectEngineCtx *de_ctx)
46{
47 if (de_ctx == NULL)
48 return;
49
50 for (Signature *s = de_ctx->sig_list; s != NULL;) {
51 Signature *ns = s->next;
52 SigFree(de_ctx, s);
53 s = ns;
54 }
55 de_ctx->sig_list = NULL;
56
57 DetectEngineResetMaxSigId(de_ctx);
58 de_ctx->sig_list = NULL;
59}

61/** \brief Find a specific signature by sid and gid
62 * \param de_ctx detection engine ctx
63 * \param sid the signature id
64 * \param gid the signature group id
65 *
66 * \retval s sig found
67 * \retval NULL sig not found
68 */
69Signature *SigFindSignatureBySidGid(DetectEngineCtx *de_ctx, uint32_t sid, uint32_t gid)
70{
71 if (de_ctx == NULL)
72 return NULL;
73
74 for (Signature *s = de_ctx->sig_list; s != NULL; s = s->next) {
75 if (s->id == sid && s->gid == gid)
76 return s;
77 }
78
79 return NULL;
80}

187/** \brief Test is a initialized signature is IP only
188 * \param de_ctx detection engine ctx
189 * \param s the signature
190 * \retval 1 sig is ip only
191 * \retval 2 sig is like ip only
192 * \retval 0 sig is not ip only
193 */
194int SignatureIsIPOnly(DetectEngineCtx *de_ctx, const Signature *s)
195{
196 if (s->alproto != ALPROTO_UNKNOWN)
197 return 0;
198
199 if (s->init_data->smlists[DETECT_SM_LIST_PMATCH] != NULL)
200 return 0;
201
202 // may happen for 'config' keyword, postmatch
203 if (s->flags & SIG_FLAG_APPLAYER)
204 return 0;
205
206 /* if flow dir is set we can't process it in ip-only */
207 if (!(((s->flags & (SIG_FLAG_TOSERVER|SIG_FLAG_TOCLIENT)) == 0) ||
208 (s->flags & (SIG_FLAG_TOSERVER|SIG_FLAG_TOCLIENT)) ==
209 (SIG_FLAG_TOSERVER|SIG_FLAG_TOCLIENT)))
210 return 0;
211
212 /* for now assume that all registered buffer types are incompatible */
213 const int nlists = s->init_data->smlists_array_size;
214 for (int i = 0; i < nlists; i++) {
215 if (s->init_data->smlists[i] == NULL)
216 continue;
217 if (!(DetectEngineBufferTypeGetNameById(de_ctx, i)))
218 continue;
219
220 SCReturnInt(0);
221 }
222
223 /* TMATCH list can be ignored, it contains TAGs and
224 * tags are compatible to IP-only. */
225
226 SigMatch *sm = s->init_data->smlists[DETECT_SM_LIST_MATCH];
227 for (; sm != NULL; sm = sm->next) {
228 if (!(sigmatch_table[sm->type].flags & SIGMATCH_IPONLY_COMPAT))
229 return 0;
230 /* we have enabled flowbits to be compatible with ip only sigs, as long
231 * as the sig only has a "set" flowbits */
232 if (sm->type == DETECT_FLOWBITS &&
233 (((DetectFlowbitsData *)sm->ctx)->cmd != DETECT_FLOWBITS_CMD_SET)) {
234 return 0;
235 }
236 }
237 sm = s->init_data->smlists[DETECT_SM_LIST_POSTMATCH];
238 for ( ; sm != NULL; sm = sm->next) {
239 if ( !(sigmatch_table[sm->type].flags & SIGMATCH_IPONLY_COMPAT))
240 return 0;
241 /* we have enabled flowbits to be compatible with ip only sigs, as long
242 * as the sig only has a "set" flowbits */
243 if (sm->type == DETECT_FLOWBITS &&
244 (((DetectFlowbitsData *)sm->ctx)->cmd != DETECT_FLOWBITS_CMD_SET) ) {
245 return 0;
246 }
247 }
248
249 if (s->init_data->src_contains_negation || s->init_data->dst_contains_negation) {
250 /* Rule is IP only, but contains negated addresses. */
251 return 2;
252 }
253 if (!(de_ctx->flags & DE_QUIET)) {
254 SCLogDebug("IP-ONLY (%" PRIu32 "): source %s, dest %s", s->id,
255 s->flags & SIG_FLAG_SRC_ANY ? "ANY" : "SET",
256 s->flags & SIG_FLAG_DST_ANY ? "ANY" : "SET");
257 }
258 return 1;
259}

261/** \internal
262 * \brief Test is a initialized signature is inspecting protocol detection only
263 * \param de_ctx detection engine ctx
264 * \param s the signature
265 * \retval 1 sig is dp only
266 * \retval 0 sig is not dp only
267 */
268static int SignatureIsPDOnly(const DetectEngineCtx *de_ctx, const Signature *s)
269{
270 if (s->alproto != ALPROTO_UNKNOWN)
271 return 0;
272
273 if (s->init_data->smlists[DETECT_SM_LIST_PMATCH] != NULL)
274 return 0;
275
276 /* for now assume that all registered buffer types are incompatible */
277 const int nlists = s->init_data->smlists_array_size;
278 for (int i = 0; i < nlists; i++) {
279 if (s->init_data->smlists[i] == NULL)
280 continue;
281 if (!(DetectEngineBufferTypeGetNameById(de_ctx, i)))
282 continue;
283
284 SCReturnInt(0);
285 }
286
287 /* TMATCH list can be ignored, it contains TAGs and
288 * tags are compatible to DP-only. */
289
290 /* match list matches may be compatible to DP only. We follow the same
291 * logic as IP-only so we can use that flag */
292
293 SigMatch *sm = s->init_data->smlists[DETECT_SM_LIST_MATCH];
294 if (sm == NULL)
295 return 0;
296
297 int pd = 0;
298 for ( ; sm != NULL; sm = sm->next) {
299 if (sm->type == DETECT_AL_APP_LAYER_PROTOCOL) {
300 pd = 1;
301 } else {
302 /* flowbits are supported for dp only sigs, as long
303 * as the sig only has a "set" flowbits */
304 if (sm->type == DETECT_FLOWBITS) {
305 if ((((DetectFlowbitsData *)sm->ctx)->cmd != DETECT_FLOWBITS_CMD_SET) ) {
306 SCLogDebug("%u: not PD-only: flowbit settings other than 'set'", s->id);
307 return 0;
308 }
309 } else if (sm->type == DETECT_FLOW) {
310 if (((DetectFlowData *)sm->ctx)->flags & ~(DETECT_FLOW_FLAG_TOSERVER|DETECT_FLOW_FLAG_TOCLIENT)) {
311 SCLogDebug("%u: not PD-only: flow settings other than toserver/toclient", s->id);
312 return 0;
313 }
314 } else if ( !(sigmatch_table[sm->type].flags & SIGMATCH_IPONLY_COMPAT)) {
315 SCLogDebug("%u: not PD-only: %s not PD/IP-only compat", s->id, sigmatch_table[sm->type].name);
316 return 0;
317 }
318 }
319 }
320
321 if (pd) {
322 SCLogDebug("PD-ONLY (%" PRIu32 ")", s->id);
323 }
324 return pd;
325}

344/**
345 * \internal
346 * \brief check if a signature is decoder event matching only
347 * \param de_ctx detection engine
348 * \param s the signature to test
349 * \retval 0 not a DEOnly sig
350 * \retval 1 DEOnly sig
351 */
352static int SignatureIsDEOnly(DetectEngineCtx *de_ctx, const Signature *s)
353{
354 if (s->alproto != ALPROTO_UNKNOWN) {
355 SCReturnInt(0);
356 }
357
358 if (s->init_data->smlists[DETECT_SM_LIST_PMATCH] != NULL)
359 {
360 SCReturnInt(0);
361 }
362
363 /* for now assume that all registered buffer types are incompatible */
364 const int nlists = s->init_data->smlists_array_size;
365 for (int i = 0; i < nlists; i++) {
366 if (s->init_data->smlists[i] == NULL)
367 continue;
368 if (!(DetectEngineBufferTypeGetNameById(de_ctx, i)))
369 continue;
370
371 SCReturnInt(0);
372 }
373
374 /* check for conflicting keywords */
375 SigMatch *sm = s->init_data->smlists[DETECT_SM_LIST_MATCH];
376 for ( ;sm != NULL; sm = sm->next) {
377 if ( !(sigmatch_table[sm->type].flags & SIGMATCH_DEONLY_COMPAT))
378 SCReturnInt(0);
379 }
380
381 /* need at least one decode event keyword to be considered decode event. */
382 sm = s->init_data->smlists[DETECT_SM_LIST_MATCH];
383 for ( ;sm != NULL; sm = sm->next) {
384 if (sm->type == DETECT_DECODE_EVENT)
385 goto deonly;
386 if (sm->type == DETECT_ENGINE_EVENT)
387 goto deonly;
388 if (sm->type == DETECT_STREAM_EVENT)
389 goto deonly;
390 }
391
392 SCReturnInt(0);
393
394deonly:
395 if (!(de_ctx->flags & DE_QUIET)) {
396 SCLogDebug("DE-ONLY (%" PRIu32 "): source %s, dest %s", s->id,
397 s->flags & SIG_FLAG_SRC_ANY ? "ANY" : "SET",
398 s->flags & SIG_FLAG_DST_ANY ? "ANY" : "SET");
399 }
400
401 SCReturnInt(1);
402}

407/* Create mask for this packet + it's flow if it has one
408 */
409void
410PacketCreateMask(Packet *p, SignatureMask *mask, AppProto alproto,
411 bool app_decoder_events)
412{
413 if (!(p->flags & PKT_NOPAYLOAD_INSPECTION) && p->payload_len > 0) {
414 SCLogDebug("packet has payload");
415 (*mask) |= SIG_MASK_REQUIRE_PAYLOAD;
416 } else if (p->flags & PKT_DETECT_HAS_STREAMDATA) {
417 SCLogDebug("stream data available");
418 (*mask) |= SIG_MASK_REQUIRE_PAYLOAD;
419 } else {
420 SCLogDebug("packet has no payload");
421 (*mask) |= SIG_MASK_REQUIRE_NO_PAYLOAD;
422 }
423
424 if (p->events.cnt > 0 || app_decoder_events != 0 || p->app_layer_events != NULL) {
425 SCLogDebug("packet/flow has events set");
426 (*mask) |= SIG_MASK_REQUIRE_ENGINE_EVENT;
427 }
428
429 if (!(PKT_IS_PSEUDOPKT(p)) && PKT_IS_TCP(p)) {
430 if ((p->tcph->th_flags & MASK_TCP_INITDEINIT_FLAGS) != 0) {
431 (*mask) |= SIG_MASK_REQUIRE_FLAGS_INITDEINIT;
432 }
433 if ((p->tcph->th_flags & MASK_TCP_UNUSUAL_FLAGS) != 0) {
434 (*mask) |= SIG_MASK_REQUIRE_FLAGS_UNUSUAL;
435 }
436 }
437
438 if (p->flags & PKT_HAS_FLOW) {
439 SCLogDebug("packet has flow");
440 (*mask) |= SIG_MASK_REQUIRE_FLOW;
441 }
442
443 if (alproto == ALPROTO_SMB || alproto == ALPROTO_DCERPC) {
444 SCLogDebug("packet will be inspected for DCERPC");
445 (*mask) |= SIG_MASK_REQUIRE_DCERPC;
446 }
447}

476static int SignatureCreateMask(Signature *s)
477{
478 SCEnter();
479
480 if (SignatureNeedsDCERPCMask(s)) {
481 s->mask |= SIG_MASK_REQUIRE_DCERPC;
482 SCLogDebug("sig requires DCERPC");
483 }
484
485 if (s->init_data->smlists[DETECT_SM_LIST_PMATCH] != NULL) {
486 s->mask |= SIG_MASK_REQUIRE_PAYLOAD;
487 SCLogDebug("sig requires payload");
488 }
489
490 SigMatch *sm;
491 for (sm = s->init_data->smlists[DETECT_SM_LIST_MATCH] ; sm != NULL; sm = sm->next) {
492 switch(sm->type) {
493 case DETECT_FLOWBITS:
494 {
495 /* figure out what flowbit action */
496 DetectFlowbitsData *fb = (DetectFlowbitsData *)sm->ctx;
497 if (fb->cmd == DETECT_FLOWBITS_CMD_ISSET) {
498 /* not a mask flag, but still set it here */
499 s->flags |= SIG_FLAG_REQUIRE_FLOWVAR;
500
501 SCLogDebug("SIG_FLAG_REQUIRE_FLOWVAR set as sig has "
502 "flowbit isset option.");
503 }
504
505 /* flow is required for any flowbit manipulation */
506 s->mask |= SIG_MASK_REQUIRE_FLOW;
507 SCLogDebug("sig requires flow to be able to manipulate "
508 "flowbit(s)");
509 break;
510 }
511 case DETECT_FLOWINT:
512 /* flow is required for any flowint manipulation */
513 s->mask |= SIG_MASK_REQUIRE_FLOW;
514 SCLogDebug("sig requires flow to be able to manipulate "
515 "flowint(s)");
516 break;
517 case DETECT_FLAGS:
518 {
519 DetectFlagsData *fl = (DetectFlagsData *)sm->ctx;
520
521 if (fl->flags & TH_SYN) {
522 s->mask |= SIG_MASK_REQUIRE_FLAGS_INITDEINIT;
523 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_INITDEINIT");
524 }
525 if (fl->flags & TH_RST) {
526 s->mask |= SIG_MASK_REQUIRE_FLAGS_INITDEINIT;
527 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_INITDEINIT");
528 }
529 if (fl->flags & TH_FIN) {
530 s->mask |= SIG_MASK_REQUIRE_FLAGS_INITDEINIT;
531 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_INITDEINIT");
532 }
533 if (fl->flags & TH_URG) {
534 s->mask |= SIG_MASK_REQUIRE_FLAGS_UNUSUAL;
535 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_UNUSUAL");
536 }
537 if (fl->flags & TH_ECN) {
538 s->mask |= SIG_MASK_REQUIRE_FLAGS_UNUSUAL;
539 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_UNUSUAL");
540 }
541 if (fl->flags & TH_CWR) {
542 s->mask |= SIG_MASK_REQUIRE_FLAGS_UNUSUAL;
543 SCLogDebug("sig requires SIG_MASK_REQUIRE_FLAGS_UNUSUAL");
544 }
545 break;
546 }
547 case DETECT_DSIZE:
548 {
549 DetectU16Data *ds = (DetectU16Data *)sm->ctx;
550 /* LT will include 0, so no payload.
551 * if GT is used in the same rule the
552 * flag will be set anyway. */
553 if (ds->mode == DETECT_UINT_RA || ds->mode == DETECT_UINT_GT ||
554 ds->mode == DETECT_UINT_NE || ds->mode == DETECT_UINT_GTE) {
555
556 s->mask |= SIG_MASK_REQUIRE_PAYLOAD;
557 SCLogDebug("sig requires payload");
558
559 } else if (ds->mode == DETECT_UINT_EQ) {
560 if (ds->arg1 > 0) {
561 s->mask |= SIG_MASK_REQUIRE_PAYLOAD;
562 SCLogDebug("sig requires payload");
563 } else {
564 s->mask |= SIG_MASK_REQUIRE_NO_PAYLOAD;
565 SCLogDebug("sig requires no payload");
566 }
567 }
568 break;
569 }
570 case DETECT_AL_APP_LAYER_EVENT:
571 s->mask |= SIG_MASK_REQUIRE_ENGINE_EVENT;
572 break;
573 case DETECT_ENGINE_EVENT:
574 s->mask |= SIG_MASK_REQUIRE_ENGINE_EVENT;
575 break;
576 }
577 }
578
579 for (sm = s->init_data->smlists[DETECT_SM_LIST_POSTMATCH]; sm != NULL; sm = sm->next) {
580 switch (sm->type) {
581 case DETECT_CONFIG: {
582 DetectConfigData *fd = (DetectConfigData *)sm->ctx;
583 if (fd->scope == CONFIG_SCOPE_FLOW) {
584 s->mask |= SIG_MASK_REQUIRE_FLOW;
585 }
586 break;
587 }
588 }
589 }
590
591 if (s->init_data->init_flags & SIG_FLAG_INIT_FLOW) {
592 s->mask |= SIG_MASK_REQUIRE_FLOW;
593 SCLogDebug("sig requires flow");
594 }
595
596 if (s->flags & SIG_FLAG_APPLAYER) {
597 s->mask |= SIG_MASK_REQUIRE_FLOW;
598 SCLogDebug("sig requires flow");
599 }
600
601 SCLogDebug("mask %02X", s->mask);
602 SCReturnInt(0);
603}

644static json_t *RulesGroupPrintSghStats(const DetectEngineCtx *de_ctx, const SigGroupHead *sgh,
645 const int add_rules, const int add_mpm_stats)
646{
647 uint32_t prefilter_cnt = 0;
648 uint32_t mpm_cnt = 0;
649 uint32_t nonmpm_cnt = 0;
650 uint32_t mpm_depth_cnt = 0;
651 uint32_t mpm_endswith_cnt = 0;
652 uint32_t negmpm_cnt = 0;
653 uint32_t any5_cnt = 0;
654 uint32_t payload_no_mpm_cnt = 0;
655 uint32_t syn_cnt = 0;
656
657 uint32_t mpms_min = 0;
658 uint32_t mpms_max = 0;
659
660 int max_buffer_type_id = DetectBufferTypeMaxId() + 1;
661
662 struct {
663 uint32_t total;
664 uint32_t cnt;
665 uint32_t min;
666 uint32_t max;
667 } mpm_stats[max_buffer_type_id];
668 memset(mpm_stats, 0x00, sizeof(mpm_stats));
669
670 uint32_t alstats[ALPROTO_MAX] = {0};
671 uint32_t mpm_sizes[max_buffer_type_id][256];
672 memset(mpm_sizes, 0, sizeof(mpm_sizes));
673 uint32_t alproto_mpm_bufs[ALPROTO_MAX][max_buffer_type_id];
674 memset(alproto_mpm_bufs, 0, sizeof(alproto_mpm_bufs));
675
676 DEBUG_VALIDATE_BUG_ON(sgh->init == NULL);
677 if (sgh->init == NULL)
678 return NULL;
679
680 json_t *js = json_object();
681 if (unlikely(js == NULL))
682 return NULL;
683
684 json_object_set_new(js, "id", json_integer(sgh->id));
685
686 json_t *js_array = json_array();
687
688 const Signature *s;
689 uint32_t x;
690 for (x = 0; x < sgh->init->sig_cnt; x++) {
691 s = sgh->init->match_array[x];
692 if (s == NULL)
693 continue;
694
695 int any = 0;
696 if (s->proto.flags & DETECT_PROTO_ANY) {
697 any++;
698 }
699 if (s->flags & SIG_FLAG_DST_ANY) {
700 any++;
701 }
702 if (s->flags & SIG_FLAG_SRC_ANY) {
703 any++;
704 }
705 if (s->flags & SIG_FLAG_DP_ANY) {

1185{
1186 /* step 1: create a hash of 'DetectPort' objects based on all the
1187 * rules. Each object will have a SGH with the sigs added
1188 * that belong to the SGH. */
1189 DetectPortHashInit(de_ctx);
1190
1191 uint32_t max_idx = 0;
1192 const Signature *s = de_ctx->sig_list;
1193 DetectPort *list = NULL;
1194 while (s) {
1195 /* IP Only rules are handled separately */
1196 if (s->flags & SIG_FLAG_IPONLY)
1197 goto next;
1198 if (!(s->proto.proto[ipproto / 8] & (1<<(ipproto % 8)) || (s->proto.flags & DETECT_PROTO_ANY)))
1199 goto next;
1200 if (direction == SIG_FLAG_TOSERVER) {
1201 if (!(s->flags & SIG_FLAG_TOSERVER))
1202 goto next;
1203 } else if (direction == SIG_FLAG_TOCLIENT) {
1204 if (!(s->flags & SIG_FLAG_TOCLIENT))
1205 goto next;
1206 }
1207
1208 DetectPort *p = NULL;
1209 if (direction == SIG_FLAG_TOSERVER)
1210 p = s->dp;
1211 else if (direction == SIG_FLAG_TOCLIENT)
1212 p = s->sp;
1213 else
1214 BUG_ON(1);
1215
1216 /* see if we want to exclude directionless sigs that really care only for
1217 * to_server syn scans/floods */
1218 if ((direction == SIG_FLAG_TOCLIENT) &&
1219 DetectFlagsSignatureNeedsSynPackets(s) &&
1220 DetectFlagsSignatureNeedsSynOnlyPackets(s) &&
1221 ((s->flags & (SIG_FLAG_TOSERVER|SIG_FLAG_TOCLIENT)) == (SIG_FLAG_TOSERVER|SIG_FLAG_TOCLIENT)) &&
1222 (!(s->dp->port == 0 && s->dp->port2 == 65535)))
1223 {
1224 SCLogWarning(SC_WARN_POOR_RULE, "rule %u: SYN-only to port(s) %u:%u "
1225 "w/o direction specified, disabling for toclient direction",
1226 s->id, s->dp->port, s->dp->port2);
1227 goto next;
1228 }
1229
1230 int wl = s->init_data->whitelist;
1231 while (p) {
1232 int pwl = PortIsWhitelisted(de_ctx, p, ipproto) ? 111 : 0;
1233 pwl = MAX(wl,pwl);
1234
1235 DetectPort *lookup = DetectPortHashLookup(de_ctx, p);
1236 if (lookup) {
1237 SigGroupHeadAppendSig(de_ctx, &lookup->sh, s);
1238 lookup->sh->init->whitelist = MAX(lookup->sh->init->whitelist, pwl);
1239 } else {
1240 DetectPort *tmp2 = DetectPortCopySingle(de_ctx, p);
1241 BUG_ON(tmp2 == NULL);
1242 SigGroupHeadAppendSig(de_ctx, &tmp2->sh, s);
1243 tmp2->sh->init->whitelist = pwl;
1244 DetectPortHashAdd(de_ctx, tmp2);
1245 }
1246
1247 p = p->next;
1248 }
1249 max_idx = s->num;
1250 next:
1251 s = s->next;
1252 }
1253
1254 /* step 2: create a list of DetectPort objects */
1255 HashListTableBucket *htb = NULL;
1256 for (htb = HashListTableGetListHead(de_ctx->dport_hash_table);
1257 htb != NULL;
1258 htb = HashListTableGetListNext(htb))
1259 {
1260 DetectPort *p = HashListTableGetListData(htb);
1261 DetectPort *tmp = DetectPortCopySingle(de_ctx, p);
1262 BUG_ON(tmp == NULL);
1263 int r = DetectPortInsert(de_ctx, &list , tmp);
1264 BUG_ON(r == -1);
1265 }
1266 DetectPortHashFree(de_ctx);
1267 de_ctx->dport_hash_table = NULL;
1268
1269 SCLogDebug("rules analyzed");
1270
1271 /* step 3: group the list and shrink it if necessary */
1272 DetectPort *newlist = NULL;
1273 uint16_t groupmax = (direction == SIG_FLAG_TOCLIENT) ? de_ctx->max_uniq_toclient_groups :
1274 de_ctx->max_uniq_toserver_groups;
1275 CreateGroupedPortList(de_ctx, list, &newlist, groupmax, CreateGroupedPortListCmpCnt, max_idx);
1276 list = newlist;
1277
1278 /* step 4: deduplicate the SGH's */
1279 SigGroupHeadHashFree(de_ctx);
1280 SigGroupHeadHashInit(de_ctx);
1281
1282 uint32_t cnt = 0;
1283 uint32_t own = 0;
1284 uint32_t ref = 0;
1285 DetectPort *iter;
1286 for (iter = list ; iter != NULL; iter = iter->next) {
1287 BUG_ON (iter->sh == NULL);
1288 cnt++;
1289
1290 SigGroupHead *lookup_sgh = SigGroupHeadHashLookup(de_ctx, iter->sh);
1291 if (lookup_sgh == NULL) {
1292 SCLogDebug("port group %p sgh %p is the original", iter, iter->sh);

1535/** \internal
1536 * \brief Create a list of DetectPort objects sorted based on CompareFunc's
1537 * logic.
1538 *
1539 * List can limit the number of groups. In this case an extra "join" group
1540 * is created that contains the sigs belonging to that. It's *appended* to
1541 * the list, meaning that if the list is walked linearly it's found last.
1542 * The joingr is meant to be a catch all.
1543 *
1544 */
1545int CreateGroupedPortList(DetectEngineCtx *de_ctx, DetectPort *port_list, DetectPort **newhead, uint32_t unique_groups, int (*CompareFunc)(DetectPort *, DetectPort *), uint32_t max_idx)
1546{
1547 DetectPort *tmplist = NULL, *joingr = NULL;
1548 char insert = 0;
1549 uint32_t groups = 0;
1550 DetectPort *list;
1551
1552 /* insert the addresses into the tmplist, where it will
1553 * be sorted descending on 'cnt' and on wehther a group
1554 * is whitelisted. */
1555
1556 DetectPort *oldhead = port_list;
1557 while (oldhead) {
1558 /* take the top of the list */
1559 list = oldhead;
1560 oldhead = oldhead->next;
1561 list->next = NULL;
1562
1563 groups++;
1564
1565 SigGroupHeadSetSigCnt(list->sh, max_idx);
1566
1567 /* insert it */
1568 DetectPort *tmpgr = tmplist, *prevtmpgr = NULL;
1569 if (tmplist == NULL) {
1570 /* empty list, set head */
1571 tmplist = list;
1572 } else {
1573 /* look for the place to insert */
1574 for ( ; tmpgr != NULL && !insert; tmpgr = tmpgr->next) {
1575 if (CompareFunc(list, tmpgr) == 1) {
1576 if (tmpgr == tmplist) {
1577 list->next = tmplist;
1578 tmplist = list;
1579 SCLogDebug("new list top: %u:%u", tmplist->port, tmplist->port2);
1580 } else {
1581 list->next = prevtmpgr->next;
1582 prevtmpgr->next = list;
1583 }
1584 insert = 1;
1585 break;
1586 }
1587 prevtmpgr = tmpgr;
1588 }
1589 if (insert == 0) {
1590 list->next = NULL;
1591 prevtmpgr->next = list;
1592 }
1593 insert = 0;
1594 }
1595 }
1596
1597 uint32_t left = unique_groups;
1598 if (left == 0)
1599 left = groups;
1600
1601 /* create another list: take the port groups from above
1602 * and add them to the 2nd list until we have met our
1603 * count. The rest is added to the 'join' group. */
1604 DetectPort *tmplist2 = NULL, *tmplist2_tail = NULL;
1605 DetectPort *gr, *next_gr;
1606 for (gr = tmplist; gr != NULL; ) {
1607 next_gr = gr->next;
1608
1609 SCLogDebug("temp list gr %p %u:%u", gr, gr->port, gr->port2);
1610 DetectPortPrint(gr);
1611
1612 /* if we've set up all the unique groups, add the rest to the
1613 * catch-all joingr */
1614 if (left == 0) {
1615 if (joingr == NULL) {
1616 DetectPortParse(de_ctx, &joingr, "0:65535");
1617 if (joingr == NULL) {
1618 goto error;
1619 }
1620 SCLogDebug("joingr => %u-%u", joingr->port, joingr->port2);
1621 joingr->next = NULL;
1622 }
1623 SigGroupHeadCopySigs(de_ctx,gr->sh,&joingr->sh);
1624
1625 /* when a group's sigs are added to the joingr, we can free it */
1626 gr->next = NULL;
1627 DetectPortFree(de_ctx, gr);
1628 gr = NULL;
1629
1630 /* append */
1631 } else {
1632 gr->next = NULL;
1633
1634 if (tmplist2 == NULL) {
1635 tmplist2 = gr;
1636 tmplist2_tail = gr;
1637 } else {
1638 tmplist2_tail->next = gr;
1639 tmplist2_tail = gr;
1640 }
1641 }
1642
1643 if (left > 0)
1644 left--;
1645
1646 gr = next_gr;
1647 }
1648
1649 /* if present, append the joingr that covers the rest */
1650 if (joingr != NULL) {
1651 SCLogDebug("appending joingr %p %u:%u", joingr, joingr->port, joingr->port2);
1652
1653 if (tmplist2 == NULL) {
1654 tmplist2 = joingr;
1655 //tmplist2_tail = joingr;
1656 } else {
1657 tmplist2_tail->next = joingr;
1658 //tmplist2_tail = joingr;
1659 }
1660 } else {
1661 SCLogDebug("no joingr");
1662 }
1663
1664 /* pass back our new list to the caller */
1665 *newhead = tmplist2;
1666 DetectPortPrintList(*newhead);
1667
1668 return 0;
1669error:
1670 return -1;
1671}
