48#define VXLAN_MAX_PORTS 4
49#define VXLAN_UNSET_PORT -1
50#define VXLAN_DEFAULT_PORT 4789
51#define VXLAN_DEFAULT_PORT_S "4789"

53static bool g_vxlan_enabled = true;
54static int g_vxlan_ports_idx = 0;
55static int g_vxlan_ports[VXLAN_MAX_PORTS] = { VXLAN_DEFAULT_PORT, VXLAN_UNSET_PORT,
56 VXLAN_UNSET_PORT, VXLAN_UNSET_PORT };

65bool DecodeVXLANEnabledForPort(const uint16_t sp, const uint16_t dp)
66{
67 SCLogDebug("ports %u->%u ports %d %d %d %d", sp, dp, g_vxlan_ports[0], g_vxlan_ports[1],
68 g_vxlan_ports[2], g_vxlan_ports[3]);
69
70 if (g_vxlan_enabled) {
71 for (int i = 0; i < g_vxlan_ports_idx; i++) {
72 if (g_vxlan_ports[i] == VXLAN_UNSET_PORT)
73 return false;
74 const int port = g_vxlan_ports[i];
75 if (port == (const int)sp || port == (const int)dp)
76 return true;
77 }
78 }
79 return false;
80}

82static void DecodeVXLANConfigPorts(const char *pstr)
83{
84 SCLogDebug("parsing \'%s\'", pstr);
85
86 DetectPort *head = NULL;
87 DetectPortParse(NULL, &head, pstr);
88
89 g_vxlan_ports_idx = 0;
90 for (DetectPort *p = head; p != NULL; p = p->next) {
91 if (g_vxlan_ports_idx >= VXLAN_MAX_PORTS) {
92 SCLogWarning(SC_ERR_INVALID_YAML_CONF_ENTRY, "more than %d VXLAN ports defined",
93 VXLAN_MAX_PORTS);
94 break;
95 }
96 g_vxlan_ports[g_vxlan_ports_idx++] = (int)p->port;
97 }
98
99 DetectPortCleanupList(NULL, head);
100}

102void DecodeVXLANConfig(void)
103{
104 int enabled = 0;
105 if (ConfGetBool("decoder.vxlan.enabled", &enabled) == 1) {
106 if (enabled) {
107 g_vxlan_enabled = true;
108 } else {
109 g_vxlan_enabled = false;
110 }
111 }
112
113 if (g_vxlan_enabled) {
114 ConfNode *node = ConfGetNode("decoder.vxlan.ports");
115 if (node && node->val) {
116 DecodeVXLANConfigPorts(node->val);
117 } else {
118 DecodeVXLANConfigPorts(VXLAN_DEFAULT_PORT_S);
119 }
120 }
121}

123/** \param pkt payload data directly above UDP header
124 * \param len length in bytes of pkt
125 */
126int DecodeVXLAN(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p,
127 const uint8_t *pkt, uint32_t len)
128{
129 DEBUG_VALIDATE_BUG_ON(pkt == NULL);
130
131 /* Initial packet validation */
132 if (unlikely(!g_vxlan_enabled))
133 return TM_ECODE_FAILED;
134
135 if (len < (VXLAN_HEADER_LEN + sizeof(EthernetHdr)))
136 return TM_ECODE_FAILED;
137 if (!PacketIncreaseCheckLayers(p)) {
138 return TM_ECODE_FAILED;
139 }
140
141 const VXLANHeader *vxlanh = (const VXLANHeader *)pkt;
142 if ((vxlanh->flags[0] & 0x08) == 0 || vxlanh->res != 0)
143 return TM_ECODE_FAILED;
144
145#if DEBUG
146 uint32_t vni = (vxlanh->vni[0] << 16) + (vxlanh->vni[1] << 8) + (vxlanh->vni[2]);
147 SCLogDebug("VXLAN vni %u", vni);
148#endif
149
150 /* Increment stats counter for VXLAN packets */
151 StatsIncr(tv, dtv->counter_vxlan);
152
153 EthernetHdr *ethh = (EthernetHdr *)(pkt + VXLAN_HEADER_LEN);
154 int decode_tunnel_proto = DECODE_TUNNEL_UNSET;
155
156 /* Look at encapsulated Ethernet frame to get next protocol */
157 uint16_t eth_type = SCNtohs(ethh->eth_type);
158 SCLogDebug("VXLAN ethertype 0x%04x", eth_type);
159
160 switch (eth_type) {
161 case ETHERNET_TYPE_ARP:
162 SCLogDebug("VXLAN found ARP");
163 break;
164 case ETHERNET_TYPE_IP:
165 SCLogDebug("VXLAN found IPv4");
166 decode_tunnel_proto = DECODE_TUNNEL_IPV4;
167 break;
168 case ETHERNET_TYPE_IPV6:
169 SCLogDebug("VXLAN found IPv6");
170 decode_tunnel_proto = DECODE_TUNNEL_IPV6;
171 break;
172 case ETHERNET_TYPE_VLAN:
173 case ETHERNET_TYPE_8021AD:
174 case ETHERNET_TYPE_8021QINQ:
175 SCLogDebug("VXLAN found VLAN");
176 decode_tunnel_proto = DECODE_TUNNEL_VLAN;
177 break;
178 default:
179 SCLogDebug("VXLAN found unsupported Ethertype - expected IPv4, IPv6, VLAN, or ARP");
180 ENGINE_SET_INVALID_EVENT(p, VXLAN_UNKNOWN_PAYLOAD_TYPE);
181 }
182
183 /* Set-up and process inner packet if it is a supported ethertype */
184 if (decode_tunnel_proto != DECODE_TUNNEL_UNSET) {
185 Packet *tp = PacketTunnelPktSetup(tv, dtv, p, pkt + VXLAN_HEADER_LEN + ETHERNET_HEADER_LEN,
186 len - (VXLAN_HEADER_LEN + ETHERNET_HEADER_LEN), decode_tunnel_proto);
187 if (tp != NULL) {
188 PKT_SET_SRC(tp, PKT_SRC_DECODER_VXLAN);
189 PacketEnqueueNoLock(&tv->decode_pq, tp);
190 }
191 }
192
193 return TM_ECODE_OK;
194}

198/**
199 * \test DecodeVXLANTest01 test a good vxlan header.
200 * Contains a DNS request packet.
201 */
202static int DecodeVXLANtest01 (void)
203{
204 uint8_t raw_vxlan[] = {
205 0x12, 0xb5, 0x12, 0xb5, 0x00, 0x3a, 0x87, 0x51, /* UDP header */
206 0x08, 0x00, 0x00, 0x00, 0x00, 0x00, 0x25, 0x00, /* VXLAN header */
207 0x10, 0x00, 0x00, 0x0c, 0x01, 0x00, /* inner destination MAC */
208 0x00, 0x51, 0x52, 0xb3, 0x54, 0xe5, /* inner source MAC */
209 0x08, 0x00, /* another IPv4 0x0800 */
210 0x45, 0x00, 0x00, 0x1c, 0x00, 0x01, 0x00, 0x00, 0x40, 0x11,
211 0x44, 0x45, 0x0a, 0x60, 0x00, 0x0a, 0xb9, 0x1b, 0x73, 0x06, /* IPv4 hdr */
212 0x00, 0x35, 0x30, 0x39, 0x00, 0x08, 0x98, 0xe4 /* UDP probe src port 53 */
213 };
214 Packet *p = PacketGetFromAlloc();
215 FAIL_IF_NULL(p);
216 ThreadVars tv;
217 DecodeThreadVars dtv;
218
219 DecodeVXLANConfigPorts(VXLAN_DEFAULT_PORT_S);
220
221 memset(&tv, 0, sizeof(ThreadVars));
222 memset(p, 0, SIZE_OF_PACKET);
223 memset(&dtv, 0, sizeof(DecodeThreadVars));
224
225 FlowInitConfig(FLOW_QUIET);
226 DecodeUDP(&tv, &dtv, p, raw_vxlan, sizeof(raw_vxlan));
227
228 FAIL_IF(p->udph == NULL);
229 FAIL_IF(tv.decode_pq.top == NULL);
230
231 Packet *tp = PacketDequeueNoLock(&tv.decode_pq);
232 FAIL_IF(tp->udph == NULL);
233 FAIL_IF_NOT(tp->sp == 53);
234
235 FlowShutdown();
236 PacketFree(p);
237 PacketFree(tp);
238 PASS;
239}
