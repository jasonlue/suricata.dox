120/**
121 * \brief This function is used to parse template options passed via template: keyword
122 *
123 * \param templatestr Pointer to the user provided template options
124 *
125 * \retval templated pointer to DetectTemplateData on success
126 * \retval NULL on failure
127 */
128static DetectTemplateData *DetectTemplateParse (const char *templatestr)
129{
130 char arg1[4] = "";
131 char arg2[4] = "";
132 size_t pcre2len;
133
134 int ret = DetectParsePcreExec(&parse_regex, templatestr, 0, 0);
135 if (ret != 3) {
136 SCLogError(SC_ERR_PCRE_MATCH, "parse error, ret %" PRId32 "", ret);
137 return NULL;
138 }
139
140 pcre2len = sizeof(arg1);
141 ret = pcre2_substring_copy_bynumber(parse_regex.match, 1, (PCRE2_UCHAR8 *)arg1, &pcre2len);
142 if (ret < 0) {
143 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
144 return NULL;
145 }
146 SCLogDebug("Arg1 \"%s\"", arg1);
147
148 pcre2len = sizeof(arg2);
149 ret = pcre2_substring_copy_bynumber(parse_regex.match, 2, (PCRE2_UCHAR8 *)arg2, &pcre2len);
150 if (ret < 0) {
151 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
152 return NULL;
153 }
154 SCLogDebug("Arg2 \"%s\"", arg2);
155
156 DetectTemplateData *templated = SCMalloc(sizeof (DetectTemplateData));
157 if (unlikely(templated == NULL))
158 return NULL;
159
160 if (ByteExtractStringUint8(&templated->arg1, 10, 0, (const char *)arg1) < 0) {
161 SCFree(templated);
162 return NULL;
163 }
164 if (ByteExtractStringUint8(&templated->arg2, 10, 0, (const char *)arg2) < 0) {
165 SCFree(templated);
166 return NULL;
167 }
168 return templated;
169}

171/**
172 * \brief parse the options from the 'template' keyword in the rule into
173 * the Signature data structure.
174 *
175 * \param de_ctx pointer to the Detection Engine Context
176 * \param s pointer to the Current Signature
177 * \param templatestr pointer to the user provided template options
178 *
179 * \retval 0 on Success
180 * \retval -1 on Failure
181 */
182static int DetectTemplateSetup (DetectEngineCtx *de_ctx, Signature *s, const char *templatestr)
183{
184 DetectTemplateData *templated = DetectTemplateParse(templatestr);
185 if (templated == NULL)
186 return -1;
187
188 SigMatch *sm = SigMatchAlloc();
189 if (sm == NULL) {
190 DetectTemplateFree(de_ctx, templated);
191 return -1;
192 }
193
194 sm->type = DETECT_TEMPLATE;
195 sm->ctx = (void *)templated;
196
197 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
198 s->flags |= SIG_FLAG_REQUIRE_PACKET;
199
200 return 0;
201}
