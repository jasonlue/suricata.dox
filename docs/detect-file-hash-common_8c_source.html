35/**
36 * \brief Read the bytes of a hash from an hexadecimal string
37 *
38 * \param hash buffer to store the resulting bytes
39 * \param string hexadecimal string representing the hash
40 * \param filename file name from where the string was read
41 * \param line_no file line number from where the string was read
42 * \param expected_len the expected length of the string that was read
43 *
44 * \retval -1 the hexadecimal string is invalid
45 * \retval 1 the hexadecimal string was read successfully
46 */
47int ReadHashString(uint8_t *hash, const char *string, const char *filename, int line_no,
48 uint16_t expected_len)
49{
50 if (strlen(string) != expected_len) {
51 SCLogError(SC_ERR_INVALID_HASH, "%s:%d hash string not %d characters",
52 filename, line_no, expected_len);
53 return -1;
54 }
55
56 int i, x;
57 for (x = 0, i = 0; i < expected_len; i+=2, x++) {
58 char buf[3] = { 0, 0, 0 };
59 buf[0] = string[i];
60 buf[1] = string[i+1];
61
62 long value = strtol(buf, NULL, 16);
63 if (value >= 0 && value <= 255)
64 hash[x] = (uint8_t)value;
65 else {
66 SCLogError(SC_ERR_INVALID_HASH, "%s:%d hash byte out of range %ld",
67 filename, line_no, value);
68 return -1;
69 }
70 }
71
72 return 1;
73}

75/**
76 * \brief Store a hash into the hash table
77 *
78 * \param hash_table hash table that will hold the hash
79 * \param string hexadecimal string representing the hash
80 * \param filename file name from where the string was read
81 * \param line_no file line number from where the string was read
82 * \param type the hash algorithm
83 *
84 * \retval -1 failed to load the hash into the hash table
85 * \retval 1 successfully loaded the has into the hash table
86 */
87int LoadHashTable(ROHashTable *hash_table, const char *string, const char *filename,
88 int line_no, uint32_t type)
89{
90 /* allocate the maximum size a hash can have (in this case is SHA256, 32 bytes) */
91 uint8_t hash[32];
92 /* specify the actual size that should be read depending on the hash algorithm */
93 uint16_t size = 32;
94
95 if (type == DETECT_FILEMD5) {
96 size = 16;
97 }
98 else if (type == DETECT_FILESHA1) {
99 size = 20;
100 }
101
102 /* every byte represented with hexadecimal digits is two characters */
103 uint16_t expected_len = (size * 2);
104
105 if (ReadHashString(hash, string, filename, line_no, expected_len) == 1) {
106 if (ROHashInitQueueValue(hash_table, &hash, size) != 1)
107 return -1;
108 }
109
110 return 1;
111}

113/**
114 * \brief Match a hash stored in a hash table
115 *
116 * \param hash_table hash table that will hold the hash
117 * \param hash buffer containing the bytes of the has
118 * \param hash_len length of the hash buffer
119 *
120 * \retval 0 didn't find the specified hash
121 * \retval 1 the hash matched a stored value
122 */
123static int HashMatchHashTable(ROHashTable *hash_table, uint8_t *hash,
124 size_t hash_len)
125{
126 void *ptr = ROHashLookup(hash_table, hash, (uint16_t)hash_len);
127 if (ptr == NULL)
128 return 0;
129 else
130 return 1;
131}

133/**
134 * \brief Match the specified file hash
135 *
136 * \param det_ctx pattern matcher thread local data
137 * \param f *LOCKED* flow
138 * \param flags direction flags
139 * \param file file being inspected
140 * \param s signature being inspected
141 * \param m sigmatch that we will cast into DetectFileHashData
142 *
143 * \retval 0 no match
144 * \retval 1 match
145 */
146int DetectFileHashMatch (DetectEngineThreadCtx *det_ctx,
147 Flow *f, uint8_t flags, File *file, const Signature *s, const SigMatchCtx *m)
148{
149 SCEnter();
150 int ret = 0;
151 DetectFileHashData *filehash = (DetectFileHashData *)m;
152
153 if (file->state != FILE_STATE_CLOSED) {
154 SCReturnInt(0);
155 }
156
157 int match = -1;
158
159 if (s->file_flags & FILE_SIG_NEED_MD5 && file->flags & FILE_MD5) {
160 match = HashMatchHashTable(filehash->hash, file->md5, sizeof(file->md5));
161 }
162 else if (s->file_flags & FILE_SIG_NEED_SHA1 && file->flags & FILE_SHA1) {
163 match = HashMatchHashTable(filehash->hash, file->sha1, sizeof(file->sha1));
164 }
165 else if (s->file_flags & FILE_SIG_NEED_SHA256 && file->flags & FILE_SHA256) {
166 match = HashMatchHashTable(filehash->hash, file->sha256, sizeof(file->sha256));
167 }
168
169 if (match == 1) {
170 if (filehash->negated == 0)
171 ret = 1;
172 else
173 ret = 0;
174 }
175 else if (match == 0) {
176 if (filehash->negated == 0)
177 ret = 0;
178 else
179 ret = 1;
180 }
181
182 SCReturnInt(ret);
183}

187/**
188 * \brief Parse the filemd5, filesha1 or filesha256 keyword
189 *
190 * \param det_ctx pattern matcher thread local data
191 * \param str Pointer to the user provided option
192 * \param type the hash algorithm
193 *
194 * \retval hash pointer to DetectFileHashData on success
195 * \retval NULL on failure
196 */
197static DetectFileHashData *DetectFileHashParse (const DetectEngineCtx *de_ctx,
198 const char *str, uint32_t type)
199{
200 DetectFileHashData *filehash = NULL;
201 FILE *fp = NULL;
202 char *filename = NULL;
203 char *rule_filename = NULL;
204
205 /* We have a correct hash algorithm option */
206 filehash = SCMalloc(sizeof(DetectFileHashData));
207 if (unlikely(filehash == NULL))
208 goto error;
209
210 memset(filehash, 0x00, sizeof(DetectFileHashData));
211
212 if (strlen(str) && str[0] == '!') {
213 filehash->negated = 1;
214 str++;
215 }
216
217 if (type == DETECT_FILEMD5) {
218 filehash->hash = ROHashInit(18, 16);
219 }
220 else if (type == DETECT_FILESHA1) {
221 filehash->hash = ROHashInit(18, 20);
222 }
223 else if (type == DETECT_FILESHA256) {
224 filehash->hash = ROHashInit(18, 32);
225 }
226
227 if (filehash->hash == NULL) {
228 goto error;
229 }
230
231 /* get full filename */
232 filename = DetectLoadCompleteSigPath(de_ctx, str);
233 if (filename == NULL) {
234 goto error;
235 }
236
237 rule_filename = SCStrdup(de_ctx->rule_file);
238 if (rule_filename == NULL) {
239 goto error;
240 }
241
242 char line[8192] = "";
243 fp = fopen(filename, "r");
244 if (fp == NULL) {
245#ifdef HAVE_LIBGEN_H
246 if (de_ctx->rule_file != NULL) {
247 char *dir = dirname(rule_filename);
248 if (dir != NULL) {
249 char path[PATH_MAX];
250 snprintf(path, sizeof(path), "%s/%s", dir, str);
251 fp = fopen(path, "r");
252 if (fp == NULL) {
253 SCLogError(SC_ERR_OPENING_RULE_FILE,
254 "opening hash file %s: %s", path, strerror(errno));
255 goto error;
256 }
257 }
258 }
259 if (fp == NULL) {
260#endif
261 SCLogError(SC_ERR_OPENING_RULE_FILE, "opening hash file %s: %s", filename, strerror(errno));
262 goto error;
263#ifdef HAVE_LIBGEN_H
264 }
265#endif
266 }
267
268 int line_no = 0;
269 while(fgets(line, (int)sizeof(line), fp) != NULL) {
270 size_t valid = 0, len = strlen(line);
271 line_no++;
272
273 while (strchr(hexcodes, line[valid]) != NULL && valid++ < len);
274
275 /* lines that do not contain sequentially any valid character are ignored */
276 if (valid == 0)
277 continue;
278
279 /* ignore anything after the sequence of valid characters */
280 line[valid] = '\0';
281
282 if (LoadHashTable(filehash->hash, line, filename, line_no, type) != 1) {
283 goto error;
284 }
285 }
286 fclose(fp);
287 fp = NULL;
288
289 if (ROHashInitFinalize(filehash->hash) != 1) {
290 goto error;
291 }
292 SCLogInfo("Hash hash table size %u bytes%s", ROHashMemorySize(filehash->hash), filehash->negated ? ", negated match" : "");
293
294 SCFree(rule_filename);
295 SCFree(filename);
296 return filehash;
297
298error:
299 if (filehash != NULL)
300 DetectFileHashFree((DetectEngineCtx *) de_ctx, filehash);
301 if (fp != NULL)
302 fclose(fp);
303 if (filename != NULL)
304 SCFree(filename);
305 if (rule_filename != NULL) {
306 SCFree(rule_filename);
307 }
308 return NULL;
309}

311/**
312 * \brief this function is used to parse filemd5, filesha1 and filesha256 options
313 * \brief into the current signature
314 *
315 * \param de_ctx pointer to the Detection Engine Context
316 * \param s pointer to the Current Signature
317 * \param str pointer to the user provided "filemd5", "filesha1" or "filesha256" option
318 * \param type type of file hash to setup
319 *
320 * \retval 0 on Success
321 * \retval -1 on Failure
322 */
323int DetectFileHashSetup(
324 DetectEngineCtx *de_ctx, Signature *s, const char *str, uint16_t type, int list)
325{
326 DetectFileHashData *filehash = NULL;
327 SigMatch *sm = NULL;
328
329 filehash = DetectFileHashParse(de_ctx, str, type);
330 if (filehash == NULL)
331 goto error;
332
333 /* Okay so far so good, lets get this into a SigMatch
334 * and put it in the Signature. */
335 sm = SigMatchAlloc();
336 if (sm == NULL)
337 goto error;
338
339 sm->type = type;
340 sm->ctx = (void *)filehash;
341
342 SigMatchAppendSMToList(s, sm, list);
343
344 s->file_flags |= FILE_SIG_NEED_FILE;
345
346 // Setup the file flags depending on the hashing algorithm
347 if (type == DETECT_FILEMD5) {
348 s->file_flags |= FILE_SIG_NEED_MD5;
349 }
350 if (type == DETECT_FILESHA1) {
351 s->file_flags |= FILE_SIG_NEED_SHA1;
352 }
353 if (type == DETECT_FILESHA256) {
354 s->file_flags |= FILE_SIG_NEED_SHA256;
355 }
356 return 0;
357
358error:
359 if (filehash != NULL)
360 DetectFileHashFree(de_ctx, filehash);
361 if (sm != NULL)
362 SCFree(sm);
363 return -1;
364}
