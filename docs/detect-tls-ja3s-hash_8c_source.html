71/**
72 * \brief Registration function for keyword: ja3s.hash
73 */
74void DetectTlsJa3SHashRegister(void)
75{
76 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].name = "ja3s.hash";
77 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].desc = "sticky buffer to match the JA3S hash buffer";
78 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].url = "/rules/ja3-keywords.html#ja3s-hash";
79 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].Setup = DetectTlsJa3SHashSetup;
80 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].flags |= SIGMATCH_NOOPT;
81 sigmatch_table[DETECT_AL_TLS_JA3S_HASH].flags |= SIGMATCH_INFO_STICKY_BUFFER;
82
83 DetectAppLayerInspectEngineRegister2("ja3s.hash", ALPROTO_TLS, SIG_FLAG_TOCLIENT, 0,
84 DetectEngineInspectBufferGeneric, GetData);
85
86 DetectAppLayerMpmRegister2("ja3s.hash", SIG_FLAG_TOCLIENT, 2,
87 PrefilterGenericMpmRegister, GetData, ALPROTO_TLS, 0);
88
89 DetectBufferTypeSetDescriptionByName("ja3s.hash", "TLS JA3S hash");
90
91 DetectBufferTypeRegisterSetupCallback("ja3s.hash",
92 DetectTlsJa3SHashSetupCallback);
93
94 DetectBufferTypeRegisterValidateCallback("ja3s.hash",
95 DetectTlsJa3SHashValidateCallback);
96
97 g_tls_ja3s_hash_buffer_id = DetectBufferTypeGetByName("ja3s.hash");
98}

100/**
101 * \brief this function setup the ja3s.hash modifier keyword used in the rule
102 *
103 * \param de_ctx Pointer to the Detection Engine Context
104 * \param s Pointer to the Signature to which the current keyword belongs
105 * \param str Should hold an empty string always
106 *
107 * \retval 0 On success
108 * \retval -1 On failure
109 */
110static int DetectTlsJa3SHashSetup(DetectEngineCtx *de_ctx, Signature *s, const char *str)
111{
112 if (DetectBufferSetActiveList(s, g_tls_ja3s_hash_buffer_id) < 0)
113 return -1;
114
115 if (DetectSignatureSetAppProto(s, ALPROTO_TLS) < 0)
116 return -1;
117
118 /* try to enable JA3 */
119 SSLEnableJA3();
120
121 /* Check if JA3 is disabled */
122 if (!RunmodeIsUnittests() && Ja3IsDisabled("rule")) {
123 if (!SigMatchSilentErrorEnabled(de_ctx, DETECT_AL_TLS_JA3S_HASH)) {
124 SCLogError(SC_WARN_JA3_DISABLED, "ja3(s) support is not enabled");
125 }
126 return -2;
127 }
128
129 return 0;
130}

132static InspectionBuffer *GetData(DetectEngineThreadCtx *det_ctx,
133 const DetectEngineTransforms *transforms, Flow *f,
134 const uint8_t flow_flags, void *txv, const int list_id)
135{
136 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
137 if (buffer->inspect == NULL) {
138 const SSLState *ssl_state = (SSLState *)f->alstate;
139
140 if (ssl_state->server_connp.ja3_hash == NULL) {
141 return NULL;
142 }
143
144 const uint32_t data_len = strlen(ssl_state->server_connp.ja3_hash);
145 const uint8_t *data = (uint8_t *)ssl_state->server_connp.ja3_hash;
146
147 InspectionBufferSetup(det_ctx, list_id, buffer, data, data_len);
148 InspectionBufferApplyTransforms(buffer, transforms);
149 }
150
151 return buffer;
152}

154static bool DetectTlsJa3SHashValidateCallback(const Signature *s,
155 const char **sigerror)
156{
157 const SigMatch *sm = s->init_data->smlists[g_tls_ja3s_hash_buffer_id];
158 for ( ; sm != NULL; sm = sm->next)
159 {
160 if (sm->type != DETECT_CONTENT)
161 continue;
162
163 const DetectContentData *cd = (DetectContentData *)sm->ctx;
164
165 if (cd->flags & DETECT_CONTENT_NOCASE) {
166 *sigerror = "ja3s.hash should not be used together with "
167 "nocase, since the rule is automatically "
168 "lowercased anyway which makes nocase redundant.";
169 SCLogWarning(SC_WARN_POOR_RULE, "rule %u: %s", s->id, *sigerror);
170 }
171
172 if (cd->content_len == 32)
173 return true;
174
175 *sigerror = "Invalid length of the specified JA3S hash (should "
176 "be 32 characters long). This rule will therefore "
177 "never match.";
178 SCLogError(SC_ERR_INVALID_RULE_ARGUMENT, "rule %u: %s", s->id, *sigerror);
179 return false;
180 }
181
182 return true;
183}

185static void DetectTlsJa3SHashSetupCallback(const DetectEngineCtx *de_ctx,
186 Signature *s)
187{
188 SigMatch *sm = s->init_data->smlists[g_tls_ja3s_hash_buffer_id];
189 for ( ; sm != NULL; sm = sm->next)
190 {
191 if (sm->type != DETECT_CONTENT)
192 continue;
193
194 DetectContentData *cd = (DetectContentData *)sm->ctx;
195
196 bool changed = false;
197 uint32_t u;
198 for (u = 0; u < cd->content_len; u++)
199 {
200 if (isupper(cd->content[u])) {
201 cd->content[u] = u8_tolower(cd->content[u]);
202 changed = true;
203 }
204 }
205
206 /* recreate the context if changes were made */
207 if (changed) {
208 SpmDestroyCtx(cd->spm_ctx);
209 cd->spm_ctx = SpmInitCtx(cd->content, cd->content_len, 1,
210 de_ctx->spm_global_thread_ctx);
211 }
212 }
