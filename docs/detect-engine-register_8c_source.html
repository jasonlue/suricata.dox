310static void PrintFeatureList(const SigTableElmt *e, char sep)
311{
312 const uint16_t flags = e->flags;
313
314 int prev = 0;
315 if (flags & SIGMATCH_NOOPT) {
316 printf("No option");
317 prev = 1;
318 }
319 if (flags & SIGMATCH_IPONLY_COMPAT) {
320 if (prev == 1)
321 printf("%c", sep);
322 printf("compatible with IP only rule");
323 prev = 1;
324 }
325 if (flags & SIGMATCH_DEONLY_COMPAT) {
326 if (prev == 1)
327 printf("%c", sep);
328 printf("compatible with decoder event only rule");
329 prev = 1;
330 }
331 if (flags & SIGMATCH_INFO_CONTENT_MODIFIER) {
332 if (prev == 1)
333 printf("%c", sep);
334 printf("content modifier");
335 prev = 1;
336 }
337 if (flags & SIGMATCH_INFO_STICKY_BUFFER) {
338 if (prev == 1)
339 printf("%c", sep);
340 printf("sticky buffer");
341 prev = 1;
342 }
343 if (e->Transform) {
344 if (prev == 1)
345 printf("%c", sep);
346 printf("transform");
347 prev = 1;
348 }
349 if (e->SupportsPrefilter) {
350 if (prev == 1)
351 printf("%c", sep);
352 printf("prefilter");
353 prev = 1;
354 }
355 if (prev == 0) {
356 printf("none");
357 }
358}

360static void SigMultilinePrint(int i, const char *prefix)
361{
362 if (sigmatch_table[i].desc) {
363 printf("%sDescription: %s\n", prefix, sigmatch_table[i].desc);
364 }
365 printf("%sFeatures: ", prefix);
366 PrintFeatureList(&sigmatch_table[i], ',');
367 if (sigmatch_table[i].url) {
368 printf("\n%sDocumentation: %s%s", prefix, GetDocURL(), sigmatch_table[i].url);
369 }
370 if (sigmatch_table[i].alternative) {
371 printf("\n%sReplaced by: %s", prefix, sigmatch_table[sigmatch_table[i].alternative].name);
372 }
373 printf("\n");
374}

376int SigTableList(const char *keyword)
377{
378 size_t size = sizeof(sigmatch_table) / sizeof(SigTableElmt);
379 size_t i;
380
381 if (keyword == NULL) {
382 printf("=====Supported keywords=====\n");
383 for (i = 0; i < size; i++) {
384 const char *name = sigmatch_table[i].name;
385 if (name != NULL && strlen(name) > 0) {
386 if (name[0] == '_' || strcmp(name, "template") == 0)
387 continue;
388
389 if (sigmatch_table[i].flags & SIGMATCH_NOT_BUILT) {
390 printf("- %s (not built-in)\n", name);
391 } else {
392 printf("- %s\n", name);
393 }
394 }
395 }
396 } else if (strcmp("csv", keyword) == 0) {
397 printf("name;description;app layer;features;documentation\n");
398 for (i = 0; i < size; i++) {
399 const char *name = sigmatch_table[i].name;
400 if (name != NULL && strlen(name) > 0) {
401 if (sigmatch_table[i].flags & SIGMATCH_NOT_BUILT) {
402 continue;
403 }
404 if (name[0] == '_' || strcmp(name, "template") == 0)
405 continue;
406
407 printf("%s;", name);
408 if (sigmatch_table[i].desc) {
409 printf("%s", sigmatch_table[i].desc);
410 }
411 /* Build feature */
412 printf(";Unset;"); // this used to be alproto
413 PrintFeatureList(&sigmatch_table[i], ':');
414 printf(";");
415 if (sigmatch_table[i].url) {
416 printf("%s%s", GetDocURL(), sigmatch_table[i].url);
417 }
418 printf(";");
419 printf("\n");
420 }
421 }
422 } else if (strcmp("all", keyword) == 0) {
423 for (i = 0; i < size; i++) {
424 const char *name = sigmatch_table[i].name;
425 if (name != NULL && strlen(name) > 0) {
426 if (name[0] == '_' || strcmp(name, "template") == 0)
427 continue;
428 printf("%s:\n", sigmatch_table[i].name);
429 SigMultilinePrint(i, "\t");
430 }
431 }
432 } else {
433 for (i = 0; i < size; i++) {
434 if ((sigmatch_table[i].name != NULL) &&
435 strcmp(sigmatch_table[i].name, keyword) == 0) {
436 printf("= %s =\n", sigmatch_table[i].name);
437 if (sigmatch_table[i].flags & SIGMATCH_NOT_BUILT) {
438 printf("Not built-in\n");
439 return TM_ECODE_FAILED;
440 }
441 SigMultilinePrint(i, "");
442 return TM_ECODE_DONE;
443 }
444 }
445 printf("Non existing keyword\n");
446 return TM_ECODE_FAILED;
447 }
448 return TM_ECODE_DONE;
449}
