238static int PcapLogCondition(ThreadVars *tv, void *thread_data, const Packet *p)
239{
240 PcapLogThreadData *ptd = (PcapLogThreadData *)thread_data;
241
242 /* Log alerted flow or tagged flow */
243 switch (ptd->pcap_log->conditional) {
244 case LOGMODE_COND_ALL:
245 break;
246 case LOGMODE_COND_ALERTS:
247 if (p->alerts.cnt || (p->flow && FlowHasAlerts(p->flow))) {
248 return TRUE;
249 } else {
250 return FALSE;
251 }
252 break;
253 case LOGMODE_COND_TAG:
254 if (p->flags & (PKT_HAS_TAG | PKT_FIRST_TAG)) {
255 return TRUE;
256 } else {
257 return FALSE;
258 }
259 break;
260 }
261
262 if (p->flags & PKT_PSEUDO_STREAM_END) {
263 return FALSE;
264 }
265
266 if (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) {
267 return FALSE;
268 }
269 return TRUE;
270}

272/**
273 * \brief Function to close pcaplog file
274 *
275 * \param t Thread Variable containing input/output queue, cpu affinity etc.
276 * \param pl PcapLog thread variable.
277 */
278static int PcapLogCloseFile(ThreadVars *t, PcapLogData *pl)
279{
280 if (pl != NULL) {
281 PCAPLOG_PROFILE_START;
282
283 if (pl->pcap_dumper != NULL) {
284 pcap_dump_close(pl->pcap_dumper);
285#ifdef HAVE_LIBLZ4
286 PcapLogCompressionData *comp = &pl->compression;
287 if (comp->format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
288 /* pcap_dump_close() has closed its output ``file'',
289 * so we need to call fmemopen again. */
290
291 comp->pcap_buf_wrapper = SCFmemopen(comp->pcap_buf,
292 comp->pcap_buf_size, "w");
293 if (comp->pcap_buf_wrapper == NULL) {
294 SCLogError(SC_ERR_FOPEN, "SCFmemopen failed: %s",
295 strerror(errno));
296 return TM_ECODE_FAILED;
297 }
298 }
299#endif /* HAVE_LIBLZ4 */
300 }
301 pl->size_current = 0;
302 pl->pcap_dumper = NULL;
303
304 if (pl->pcap_dead_handle != NULL)
305 pcap_close(pl->pcap_dead_handle);
306 pl->pcap_dead_handle = NULL;
307
308#ifdef HAVE_LIBLZ4
309 PcapLogCompressionData *comp = &pl->compression;
310 if (comp->format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
311 /* pcap_dump_close did not write any data because we call
312 * pcap_dump_flush() after every write when writing
313 * compressed output. */
314 uint64_t bytes_written = LZ4F_compressEnd(comp->lz4f_context,
315 comp->buffer, comp->buffer_size, NULL);
316 if (LZ4F_isError(bytes_written)) {
317 SCLogError(SC_ERR_PCAP_LOG_COMPRESS, "LZ4F_compressEnd: %s",
318 LZ4F_getErrorName(bytes_written));
319 return TM_ECODE_FAILED;
320 }
321 if (fwrite(comp->buffer, 1, bytes_written, comp->file) < bytes_written) {
322 SCLogError(SC_ERR_FWRITE, "fwrite failed: %s", strerror(errno));
323 return TM_ECODE_FAILED;
324 }
325 fclose(comp->file);
326 comp->bytes_in_block = 0;
327 }
328#endif /* HAVE_LIBLZ4 */
329
330 PCAPLOG_PROFILE_END(pl->profile_close);
331 }
332
333 return 0;
334}

351/**
352 * \brief Function to rotate pcaplog file
353 *
354 * \param t Thread Variable containing input/output queue, cpu affinity etc.
355 * \param pl PcapLog thread variable.
356 *
357 * \retval 0 on succces
358 * \retval -1 on failure
359 */
360static int PcapLogRotateFile(ThreadVars *t, PcapLogData *pl)
361{
362 PcapFileName *pf;
363 PcapFileName *pfnext;
364
365 PCAPLOG_PROFILE_START;
366
367 if (PcapLogCloseFile(t,pl) < 0) {
368 SCLogDebug("PcapLogCloseFile failed");
369 return -1;
370 }
371
372 if (pl->use_ringbuffer == RING_BUFFER_MODE_ENABLED && pl->file_cnt >= pl->max_files) {
373 pf = TAILQ_FIRST(&pl->pcap_file_list);
374 SCLogDebug("Removing pcap file %s", pf->filename);
375
376 if (remove(pf->filename) != 0) {
377 // VJ remove can fail because file is already gone
378 //LogWarning(SC_ERR_PCAP_FILE_DELETE_FAILED,
379 // "failed to remove log file %s: %s",
380 // pf->filename, strerror( errno ));
381 }
382
383 /* Remove directory if Sguil mode and no files left in sguil dir */
384 if (pl->mode == LOGMODE_SGUIL) {
385 pfnext = TAILQ_NEXT(pf,next);
386
387 if (strcmp(pf->dirname, pfnext->dirname) == 0) {
388 SCLogDebug("Current entry dir %s and next entry %s "
389 "are equal: not removing dir",
390 pf->dirname, pfnext->dirname);
391 } else {
392 SCLogDebug("current entry %s and %s are "
393 "not equal: removing dir",
394 pf->dirname, pfnext->dirname);
395
396 if (remove(pf->dirname) != 0) {
397 SCLogWarning(SC_ERR_PCAP_FILE_DELETE_FAILED,
398 "failed to remove sguil log %s: %s",
399 pf->dirname, strerror( errno ));
400 }
401 }
402 }
403
404 TAILQ_REMOVE(&pl->pcap_file_list, pf, next);
405 PcapFileNameFree(pf);
406 pl->file_cnt--;
407 }
408
409 if (PcapLogOpenFileCtx(pl) < 0) {
410 SCLogError(SC_ERR_FOPEN, "opening new pcap log file failed");
411 return -1;
412 }
413 pl->file_cnt++;
414 SCLogDebug("file_cnt %u", pl->file_cnt);
415
416 PCAPLOG_PROFILE_END(pl->profile_rotate);
417 return 0;
418}

420static int PcapLogOpenHandles(PcapLogData *pl, const Packet *p)
421{
422 PCAPLOG_PROFILE_START;
423
424 int datalink = p->datalink;
425 if (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) {
426 Packet *real_p = p->root;
427 datalink = real_p->datalink;
428 }
429 if (pl->pcap_dead_handle == NULL) {
430 SCLogDebug("Setting pcap-log link type to %u", datalink);
431 if ((pl->pcap_dead_handle = pcap_open_dead(datalink, PCAP_SNAPLEN)) == NULL) {
432 SCLogDebug("Error opening dead pcap handle");
433 return TM_ECODE_FAILED;
434 }
435 }
436
437 if (pl->pcap_dumper == NULL) {
438 if (pl->compression.format == PCAP_LOG_COMPRESSION_FORMAT_NONE) {
439 if ((pl->pcap_dumper = pcap_dump_open(pl->pcap_dead_handle,
440 pl->filename)) == NULL) {
441 SCLogInfo("Error opening dump file %s", pcap_geterr(pl->pcap_dead_handle));
442 return TM_ECODE_FAILED;
443 }
444 }
445#ifdef HAVE_LIBLZ4
446 else if (pl->compression.format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
447 PcapLogCompressionData *comp = &pl->compression;
448 if ((pl->pcap_dumper = pcap_dump_fopen(pl->pcap_dead_handle,
449 comp->pcap_buf_wrapper)) == NULL) {
450 SCLogError(SC_ERR_OPENING_FILE, "Error opening dump file %s",
451 pcap_geterr(pl->pcap_dead_handle));
452 return TM_ECODE_FAILED;
453 }
454 comp->file = fopen(pl->filename, "w");
455 if (comp->file == NULL) {
456 SCLogError(SC_ERR_OPENING_FILE,
457 "Error opening file for compressed output: %s",
458 strerror(errno));
459 return TM_ECODE_FAILED;
460 }
461
462 uint64_t bytes_written = LZ4F_compressBegin(comp->lz4f_context,
463 comp->buffer, comp->buffer_size, NULL);
464 if (LZ4F_isError(bytes_written)) {
465 SCLogError(SC_ERR_PCAP_LOG_COMPRESS, "LZ4F_compressBegin: %s",
466 LZ4F_getErrorName(bytes_written));
467 return TM_ECODE_FAILED;
468 }
469 if (fwrite(comp->buffer, 1, bytes_written, comp->file) < bytes_written) {
470 SCLogError(SC_ERR_FWRITE, "fwrite failed: %s", strerror(errno));
471 return TM_ECODE_FAILED;
472 }
473 }
474#endif /* HAVE_LIBLZ4 */
475 }
476
477 PCAPLOG_PROFILE_END(pl->profile_handles);
478 return TM_ECODE_OK;
479}

507static inline int PcapWrite(
508 PcapLogData *pl, PcapLogCompressionData *comp, uint8_t *data, size_t len)
509{
510 struct timeval current_dump;
511 gettimeofday(&current_dump, NULL);
512 pcap_dump((u_char *)pl->pcap_dumper, pl->h, data);
513 if (pl->compression.format == PCAP_LOG_COMPRESSION_FORMAT_NONE) {
514 pl->size_current += len;
515 }
516#ifdef HAVE_LIBLZ4
517 else if (pl->compression.format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
518 pcap_dump_flush(pl->pcap_dumper);
519 long in_size = ftell(comp->pcap_buf_wrapper);
520 if (in_size < 0) {
521 SCLogError(SC_ERR_PCAP_LOG_COMPRESS, "ftell failed with: %s", strerror(errno));
522 return TM_ECODE_FAILED;
523 }
524 uint64_t out_size = LZ4F_compressUpdate(comp->lz4f_context, comp->buffer, comp->buffer_size,
525 comp->pcap_buf, (uint64_t)in_size, NULL);
526 if (LZ4F_isError(len)) {
527 SCLogError(SC_ERR_PCAP_LOG_COMPRESS, "LZ4F_compressUpdate: %s", LZ4F_getErrorName(len));
528 return TM_ECODE_FAILED;
529 }
530 if (fseek(pl->compression.pcap_buf_wrapper, 0, SEEK_SET) != 0) {
531 SCLogError(SC_ERR_FSEEK, "fseek failed: %s", strerror(errno));
532 return TM_ECODE_FAILED;
533 }
534 if (fwrite(comp->buffer, 1, out_size, comp->file) < out_size) {
535 SCLogError(SC_ERR_FWRITE, "fwrite failed: %s", strerror(errno));
536 return TM_ECODE_FAILED;
537 }
538 if (out_size > 0) {
539 pl->size_current += out_size;
540 comp->bytes_in_block = len;
541 } else {
542 comp->bytes_in_block += len;
543 }
544 }
545#endif /* HAVE_LIBLZ4 */
546 if (TimeDifferenceMicros(pl->last_pcap_dump, current_dump) >= PCAP_BUFFER_TIMEOUT) {
547 pcap_dump_flush(pl->pcap_dumper);
548 }
549 pl->last_pcap_dump = current_dump;
550 return TM_ECODE_OK;
551}

559static int PcapLogSegmentCallback(
560 const Packet *p, TcpSegment *seg, void *data, const uint8_t *buf, uint32_t buflen)
561{
562 struct PcapLogCallbackContext *pctx = (struct PcapLogCallbackContext *)data;
563
564 if (seg->pcap_hdr_storage->pktlen) {
565 pctx->pl->h->ts.tv_sec = seg->pcap_hdr_storage->ts.tv_sec;
566 pctx->pl->h->ts.tv_usec = seg->pcap_hdr_storage->ts.tv_usec;
567 pctx->pl->h->len = seg->pcap_hdr_storage->pktlen + buflen;
568 pctx->pl->h->caplen = seg->pcap_hdr_storage->pktlen + buflen;
569 MemBufferReset(pctx->buf);
570 MemBufferWriteRaw(pctx->buf, seg->pcap_hdr_storage->pkt_hdr, seg->pcap_hdr_storage->pktlen);
571 MemBufferWriteRaw(pctx->buf, buf, buflen);
572
573 PcapWrite(pctx->pl, pctx->connp, (uint8_t *)pctx->buf->buffer, pctx->pl->h->len);
574 }
575 return 1;
576}

589/**
590 * \brief Pcap logging main function
591 *
592 * \param t threadvar
593 * \param p packet
594 * \param thread_data thread module specific data
595 *
596 * \retval TM_ECODE_OK on succes
597 * \retval TM_ECODE_FAILED on serious error
598 */
599static int PcapLog (ThreadVars *t, void *thread_data, const Packet *p)
600{
601 size_t len;
602 int rotate = 0;
603 int ret = 0;
604 Packet *rp = NULL;
605
606 PcapLogThreadData *td = (PcapLogThreadData *)thread_data;
607 PcapLogData *pl = td->pcap_log;
608
609 if (((p->flags & PKT_STREAM_NOPCAPLOG) && (pl->use_stream_depth == USE_STREAM_DEPTH_ENABLED)) ||
610 (pl->honor_pass_rules && (p->flags & PKT_NOPACKET_INSPECTION))) {
611 return TM_ECODE_OK;
612 }
613
614 PcapLogLock(pl);
615
616 pl->pkt_cnt++;
617 pl->h->ts.tv_sec = p->ts.tv_sec;
618 pl->h->ts.tv_usec = p->ts.tv_usec;
619 if (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) {
620 rp = p->root;
621 pl->h->caplen = GET_PKT_LEN(rp);
622 pl->h->len = GET_PKT_LEN(rp);
623 len = sizeof(*pl->h) + GET_PKT_LEN(rp);
624 } else {
625 pl->h->caplen = GET_PKT_LEN(p);
626 pl->h->len = GET_PKT_LEN(p);
627 len = sizeof(*pl->h) + GET_PKT_LEN(p);
628 }
629
630 if (pl->filename == NULL) {
631 ret = PcapLogOpenFileCtx(pl);
632 if (ret < 0) {
633 PcapLogUnlock(pl);
634 return TM_ECODE_FAILED;
635 }
636 SCLogDebug("Opening PCAP log file %s", pl->filename);
637 }
638
639 if (pl->mode == LOGMODE_SGUIL) {
640 struct tm local_tm;
641 struct tm *tms = SCLocalTime(p->ts.tv_sec, &local_tm);
642 if (tms->tm_mday != pl->prev_day) {
643 rotate = 1;
644 pl->prev_day = tms->tm_mday;
645 }
646 }
647
648 PcapLogCompressionData *comp = &pl->compression;
649 if (comp->format == PCAP_LOG_COMPRESSION_FORMAT_NONE) {
650 if ((pl->size_current + len) > pl->size_limit || rotate) {
651 if (PcapLogRotateFile(t,pl) < 0) {
652 PcapLogUnlock(pl);
653 SCLogDebug("rotation of pcap failed");
654 return TM_ECODE_FAILED;
655 }
656 }
657 }
658#ifdef HAVE_LIBLZ4
659 else if (comp->format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
660 /* When writing compressed pcap logs, we have no way of knowing
661 * for sure whether adding this packet would cause the current
662 * file to exceed the size limit. Thus, we record the number of
663 * bytes that have been fed into lz4 since the last write, and
664 * act as if they would be written uncompressed. */
665
666 if ((pl->size_current + comp->bytes_in_block + len) > pl->size_limit ||
667 rotate) {
668 if (PcapLogRotateFile(t,pl) < 0) {
669 PcapLogUnlock(pl);
670 SCLogDebug("rotation of pcap failed");
671 return TM_ECODE_FAILED;
672 }
673 }
674 }
675#endif /* HAVE_LIBLZ4 */
676
677 /* XXX pcap handles, nfq, pfring, can only have one link type ipfw? we do
678 * this here as we don't know the link type until we get our first packet */
679 if (pl->pcap_dead_handle == NULL || pl->pcap_dumper == NULL) {
680 if (PcapLogOpenHandles(pl, p) != TM_ECODE_OK) {
681 PcapLogUnlock(pl);
682 return TM_ECODE_FAILED;
683 }
684 }
685
686 PCAPLOG_PROFILE_START;
687
688 /* if we are using alerted logging and if packet is first one with alert in flow
689 * then we need to dump in the pcap the stream acked by the packet */
690 if ((p->flags & PKT_FIRST_ALERTS) && (td->pcap_log->conditional != LOGMODE_COND_ALL)) {
691 if (PKT_IS_TCP(p)) {
692 /* dump fake packets for all segments we have on acked by packet */
693#ifdef HAVE_LIBLZ4
694 PcapLogDumpSegments(td, comp, p);
695#else
696 PcapLogDumpSegments(td, NULL, p);
697#endif
698 if (p->flags & PKT_PSEUDO_STREAM_END) {
699 PcapLogUnlock(pl);
700 return TM_ECODE_OK;
701 }
702
703 /* PcapLogDumpSegment has writtens over the PcapLogData variables so need to update */
704 pl->h->ts.tv_sec = p->ts.tv_sec;
705 pl->h->ts.tv_usec = p->ts.tv_usec;
706 if (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) {
707 rp = p->root;
708 pl->h->caplen = GET_PKT_LEN(rp);
709 pl->h->len = GET_PKT_LEN(rp);
710 len = sizeof(*pl->h) + GET_PKT_LEN(rp);
711 } else {
712 pl->h->caplen = GET_PKT_LEN(p);
713 pl->h->len = GET_PKT_LEN(p);
714 len = sizeof(*pl->h) + GET_PKT_LEN(p);
715 }
716 }
717 }
718
719 if (IS_TUNNEL_PKT(p) && !IS_TUNNEL_ROOT_PKT(p)) {
720 rp = p->root;
721#ifdef HAVE_LIBLZ4
722 ret = PcapWrite(pl, comp, GET_PKT_DATA(rp), len);
723#else
724 ret = PcapWrite(pl, NULL, GET_PKT_DATA(rp), len);
725#endif
726 } else {
727#ifdef HAVE_LIBLZ4
728 ret = PcapWrite(pl, comp, GET_PKT_DATA(p), len);
729#else
730 ret = PcapWrite(pl, NULL, GET_PKT_DATA(p), len);
731#endif
732 }
733 if (ret != TM_ECODE_OK) {
734 PCAPLOG_PROFILE_END(pl->profile_write);
735 PcapLogUnlock(pl);
736 return ret;
737 }
738
739 PCAPLOG_PROFILE_END(pl->profile_write);
740 pl->profile_data_size += len;
741
742 SCLogDebug("pl->size_current %"PRIu64", pl->size_limit %"PRIu64,
743 pl->size_current, pl->size_limit);
744
745 PcapLogUnlock(pl);
746 return TM_ECODE_OK;
747}

749static PcapLogData *PcapLogDataCopy(const PcapLogData *pl)
750{
751 BUG_ON(pl->mode != LOGMODE_MULTI);
752 PcapLogData *copy = SCCalloc(1, sizeof(*copy));
753 if (unlikely(copy == NULL)) {
754 return NULL;
755 }
756
757 copy->h = SCCalloc(1, sizeof(*copy->h));
758 if (unlikely(copy->h == NULL)) {
759 SCFree(copy);
760 return NULL;
761 }
762
763 copy->prefix = SCStrdup(pl->prefix);
764 if (unlikely(copy->prefix == NULL)) {
765 SCFree(copy->h);
766 SCFree(copy);
767 return NULL;
768 }
769
770 copy->suffix = pl->suffix;
771
772 /* settings TODO move to global cfg struct */
773 copy->is_private = TRUE;
774 copy->mode = pl->mode;
775 copy->max_files = pl->max_files;
776 copy->use_ringbuffer = pl->use_ringbuffer;
777 copy->timestamp_format = pl->timestamp_format;
778 copy->use_stream_depth = pl->use_stream_depth;
779 copy->size_limit = pl->size_limit;
780 copy->conditional = pl->conditional;
781
782 const PcapLogCompressionData *comp = &pl->compression;
783 PcapLogCompressionData *copy_comp = &copy->compression;
784 copy_comp->format = comp->format;
785#ifdef HAVE_LIBLZ4
786 if (comp->format == PCAP_LOG_COMPRESSION_FORMAT_LZ4) {
787 /* We need to allocate a new compression context and buffers for
788 * the copy. First copy the things that can simply be copied. */
789
790 copy_comp->buffer_size = comp->buffer_size;
791 copy_comp->pcap_buf_size = comp->pcap_buf_size;
792 copy_comp->lz4f_prefs = comp->lz4f_prefs;
793
794 /* Allocate the buffers. */
795
796 copy_comp->buffer = SCMalloc(copy_comp->buffer_size);
797 if (copy_comp->buffer == NULL) {
798 SCLogError(SC_ERR_MEM_ALLOC, "SCMalloc failed: %s",
799 strerror(errno));
800 SCFree(copy->h);
801 SCFree(copy);
802 return NULL;
803 }
804 copy_comp->pcap_buf = SCMalloc(copy_comp->pcap_buf_size);
805 if (copy_comp->pcap_buf == NULL) {
806 SCLogError(SC_ERR_MEM_ALLOC, "SCMalloc failed: %s",
807 strerror(errno));
808 SCFree(copy_comp->buffer);
809 SCFree(copy->h);
810 SCFree(copy);
811 return NULL;
812 }
813 copy_comp->pcap_buf_wrapper = SCFmemopen(copy_comp->pcap_buf,
814 copy_comp->pcap_buf_size, "w");
815 if (copy_comp->pcap_buf_wrapper == NULL) {
816 SCLogError(SC_ERR_FOPEN, "SCFmemopen failed: %s", strerror(errno));
817 SCFree(copy_comp->buffer);
818 SCFree(copy_comp->pcap_buf);
819 SCFree(copy->h);
820 SCFree(copy);
821 return NULL;
822 }
823
824 /* Initialize a new compression context. */
825
826 LZ4F_errorCode_t errcode =
827 LZ4F_createCompressionContext(&copy_comp->lz4f_context, 1);
828 if (LZ4F_isError(errcode)) {
829 SCLogError(SC_ERR_PCAP_LOG_COMPRESS,
830 "LZ4F_createCompressionContext failed: %s",
831 LZ4F_getErrorName(errcode));
832 fclose(copy_comp->pcap_buf_wrapper);
833 SCFree(copy_comp->buffer);
834 SCFree(copy_comp->pcap_buf);
835 SCFree(copy->h);
836 SCFree(copy);
837 return NULL;
838 }
839
840 /* Initialize the rest. */
841
842 copy_comp->file = NULL;
843 copy_comp->bytes_in_block = 0;
844 }
845#endif /* HAVE_LIBLZ4 */
846
847 TAILQ_INIT(&copy->pcap_file_list);
848 SCMutexInit(&copy->plog_lock, NULL);
849
850 strlcpy(copy->dir, pl->dir, sizeof(copy->dir));
851
852 int i;
853 for (i = 0; i < pl->filename_part_cnt && i < MAX_TOKS; i++)
854 copy->filename_parts[i] = pl->filename_parts[i];
855 copy->filename_part_cnt = pl->filename_part_cnt;
856
857 /* set thread number, first thread is 1 */
858 copy->thread_number = SC_ATOMIC_ADD(thread_cnt, 1);
859
860 SCLogDebug("copied, returning %p", copy);
861 return copy;
862}

865static int PcapLogGetTimeOfFile(const char *filename, uint64_t *secs,
866 uint32_t *usecs)
867{
868 char buf[PATH_MAX];
869 size_t copylen;
870
871 int n = pcre2_match(pcre_timestamp_code, (PCRE2_SPTR8)filename, strlen(filename), 0, 0,
872 pcre_timestamp_match, NULL);
873 if (n != 2 && n != 4) {
874 /* No match. */
875 return 0;
876 }
877
878 if (n >= 2) {
879 /* Extract seconds. */
880 copylen = sizeof(buf);
881 if (pcre2_substring_copy_bynumber(pcre_timestamp_match, 1, (PCRE2_UCHAR8 *)buf, &copylen) <
882 0) {
883 return 0;
884 }
885 if (StringParseUint64(secs, 10, 0, buf) < 0) {
886 return 0;
887 }
888 }
889 if (n == 4) {
890 /* Extract microseconds. */
891 copylen = sizeof(buf);
892 if (pcre2_substring_copy_bynumber(pcre_timestamp_match, 3, (PCRE2_UCHAR8 *)buf, &copylen) <
893 0) {
894 return 0;
895 }
896 if (StringParseUint32(usecs, 10, 0, buf) < 0) {
897 return 0;
898 }
899 }
900
901 return 1;
902}

904static TmEcode PcapLogInitRingBuffer(PcapLogData *pl)
905{
906 char pattern[PATH_MAX];
907
908 SCLogInfo("Initializing PCAP ring buffer for %s/%s.",
909 pl->dir, pl->prefix);
910
911 strlcpy(pattern, pl->dir, PATH_MAX);
912 if (pattern[strlen(pattern) - 1] != '/') {
913 strlcat(pattern, "/", PATH_MAX);
914 }
915 if (pl->mode == LOGMODE_MULTI) {
916 for (int i = 0; i < pl->filename_part_cnt; i++) {
917 char *part = pl->filename_parts[i];
918 if (part == NULL || strlen(part) == 0) {
919 continue;
920 }
921 if (part[0] != '%' || strlen(part) < 2) {
922 strlcat(pattern, part, PATH_MAX);
923 continue;
924 }
925 switch (part[1]) {
926 case 'i':
927 SCLogError(
928 SC_ERR_INVALID_ARGUMENT, "Thread ID not allowed in ring buffer mode.");
929 return TM_ECODE_FAILED;
930 case 'n': {
931 char tmp[PATH_MAX];
932 snprintf(tmp, PATH_MAX, "%"PRIu32, pl->thread_number);
933 strlcat(pattern, tmp, PATH_MAX);
934 break;
935 }
936 case 't':
937 strlcat(pattern, "*", PATH_MAX);
938 break;
939 default:
940 SCLogError(SC_ERR_INVALID_ARGUMENT,
941 "Unsupported format character: %%%s", part);
942 return TM_ECODE_FAILED;
943 }
944 }
945 } else {
946 strlcat(pattern, pl->prefix, PATH_MAX);
947 strlcat(pattern, ".*", PATH_MAX);
948 }
949 strlcat(pattern, pl->suffix, PATH_MAX);
950
951 char *basename = strrchr(pattern, '/');
952 *basename++ = '\0';
953
954 /* Pattern is now just the directory name. */
955 DIR *dir = opendir(pattern);
956 if (dir == NULL) {
957 SCLogWarning(SC_ERR_DIR_OPEN, "Failed to open directory %s: %s",
958 pattern, strerror(errno));
959 return TM_ECODE_FAILED;
960 }
961
962 for (;;) {
963 struct dirent *entry = readdir(dir);
964 if (entry == NULL) {
965 break;
966 }
967 if (fnmatch(basename, entry->d_name, 0) != 0) {
968 continue;
969 }
970
971 uint64_t secs = 0;
972 uint32_t usecs = 0;
973
974 if (!PcapLogGetTimeOfFile(entry->d_name, &secs, &usecs)) {
975 /* Failed to get time stamp out of file name. Not necessarily a
976 * failure as the file might just not be a pcap log file. */
977 continue;
978 }
979
980 PcapFileName *pf = SCCalloc(sizeof(*pf), 1);
981 if (unlikely(pf == NULL)) {
982 goto fail;
983 }
984 char path[PATH_MAX];
985 if (snprintf(path, PATH_MAX, "%s/%s", pattern, entry->d_name) == PATH_MAX)
986 goto fail;
987
988 if ((pf->filename = SCStrdup(path)) == NULL) {
989 goto fail;
990 }
991 if ((pf->dirname = SCStrdup(pattern)) == NULL) {
992 goto fail;
993 }
994 pf->secs = secs;
995 pf->usecs = usecs;
996
997 if (TAILQ_EMPTY(&pl->pcap_file_list)) {
998 TAILQ_INSERT_TAIL(&pl->pcap_file_list, pf, next);
999 } else {
1000 /* Ordered insert. */
1001 PcapFileName *it = NULL;
1002 TAILQ_FOREACH(it, &pl->pcap_file_list, next) {
1003 if (pf->secs < it->secs) {
1004 break;
1005 } else if (pf->secs == it->secs && pf->usecs < it->usecs) {
1006 break;
1007 }
1008 }
1009 if (it == NULL) {
1010 TAILQ_INSERT_TAIL(&pl->pcap_file_list, pf, next);
1011 } else {
1012 TAILQ_INSERT_BEFORE(it, pf, next);
1013 }
1014 }
1015 pl->file_cnt++;
1016 continue;
1017
1018 fail:
1019 if (pf != NULL) {
1020 if (pf->filename != NULL) {
1021 SCFree(pf->filename);
1022 }
1023 if (pf->dirname != NULL) {
1024 SCFree(pf->dirname);
1025 }
1026 SCFree(pf);
1027 }
1028 break;
1029 }
1030
1031 if (pl->file_cnt > pl->max_files) {
1032 PcapFileName *pf = TAILQ_FIRST(&pl->pcap_file_list);
1033 while (pf != NULL && pl->file_cnt > pl->max_files) {
1034 SCLogDebug("Removing PCAP file %s", pf->filename);
1035 if (remove(pf->filename) != 0) {
1036 SCLogWarning(SC_WARN_REMOVE_FILE,
1037 "Failed to remove PCAP file %s: %s", pf->filename,
1038 strerror(errno));
1039 }
1040 TAILQ_REMOVE(&pl->pcap_file_list, pf, next);
1041 PcapFileNameFree(pf);
1042 pf = TAILQ_FIRST(&pl->pcap_file_list);
1043 pl->file_cnt--;
1044 }
1045 }
1046
1047 closedir(dir);
1048
1049 /* For some reason file count is initialized at one, instead of 0. */
1050 SCLogNotice("Ring buffer initialized with %d files.", pl->file_cnt - 1);
1051
1052 return TM_ECODE_OK;
1053}
1054#endif /* INIT_RING_BUFFER */

1056static TmEcode PcapLogDataInit(ThreadVars *t, const void *initdata, void **data)
1057{
1058 if (initdata == NULL) {
1059 SCLogDebug("Error getting context for LogPcap. \"initdata\" argument NULL");
1060 return TM_ECODE_FAILED;
1061 }
1062
1063 PcapLogData *pl = ((OutputCtx *)initdata)->data;
1064
1065 PcapLogThreadData *td = SCCalloc(1, sizeof(*td));
1066 if (unlikely(td == NULL))
1067 return TM_ECODE_FAILED;
1068
1069 if (pl->mode == LOGMODE_MULTI)
1070 td->pcap_log = PcapLogDataCopy(pl);
1071 else
1072 td->pcap_log = pl;
1073 BUG_ON(td->pcap_log == NULL);
1074
1075 if (DatalinkHasMultipleValues()) {
1076 if (pl->mode != LOGMODE_MULTI) {
1077 FatalError(SC_ERR_PCAP_MULTI_DEV_NO_SUPPORT,
1078 "Pcap logging with multiple link type is not supported.");
1079 } else {
1080 /* In multi mode, only pcap conditional is not supported as a flow timeout
1081 * will trigger packet logging with potentially invalid datalink. In regular
1082 * pcap logging, the logging should be done in the same thread if we
1083 * have a proper load balancing. So no mix of datalink should occur. But we need a
1084 * proper load balancing so this needs at least a warning.
1085 */
1086 switch (pl->conditional) {
1087 case LOGMODE_COND_ALERTS:
1088 case LOGMODE_COND_TAG:
1089 FatalError(SC_ERR_PCAP_MULTI_DEV_NO_SUPPORT,
1090 "Can't have multiple link types in pcap conditional mode.");
1091 break;
1092 default:
1093 SCLogWarning(SC_WARN_COMPATIBILITY,
1094 "Using multiple link types can result in invalid pcap output");
1095 }
1096 }
1097 }
1098
1099 PcapLogLock(td->pcap_log);
1100
1101 /** Use the Ouptut Context (file pointer and mutex) */
1102 td->pcap_log->pkt_cnt = 0;
1103 td->pcap_log->pcap_dead_handle = NULL;
1104 td->pcap_log->pcap_dumper = NULL;
1105 if (td->pcap_log->file_cnt < 1) {
1106 td->pcap_log->file_cnt = 1;
1107 }
1108
1109 struct timeval ts;
1110 memset(&ts, 0x00, sizeof(struct timeval));
1111 TimeGet(&ts);
1112 struct tm local_tm;
1113 struct tm *tms = SCLocalTime(ts.tv_sec, &local_tm);
1114 td->pcap_log->prev_day = tms->tm_mday;
1115
1116 PcapLogUnlock(td->pcap_log);
1117
1118 /* count threads in the global structure */
1119 SCMutexLock(&pl->plog_lock);
1120 pl->threads++;
1121 SCMutexUnlock(&pl->plog_lock);
1122
1123 *data = (void *)td;
1124
1125 if (IsTcpSessionDumpingEnabled()) {
1126 td->buf = MemBufferCreateNew(PCAP_OUTPUT_BUFFER_SIZE);
1127 } else {
1128 td->buf = NULL;
1129 }
1130
1131 if (pl->max_files && (pl->mode == LOGMODE_MULTI || pl->threads == 1)) {
1132#ifdef INIT_RING_BUFFER
1133 if (PcapLogInitRingBuffer(td->pcap_log) == TM_ECODE_FAILED) {
1134 return TM_ECODE_FAILED;
1135 }
1136#else
1137 SCLogInfo("Unable to initialize ring buffer on this platform.");
1138#endif /* INIT_RING_BUFFER */
1139 }
1140
1141 if (pl->mode == LOGMODE_MULTI) {
1142 PcapLogOpenFileCtx(td->pcap_log);
1143 } else {
1144 if (pl->filename == NULL) {
1145 PcapLogOpenFileCtx(pl);
1146 }
1147 }
1148
1149 return TM_ECODE_OK;
1150}

1257static int ParseFilename(PcapLogData *pl, const char *filename)
1258{
1259 char *toks[MAX_TOKS] = { NULL };
1260 int tok = 0;
1261 char str[MAX_FILENAMELEN] = "";
1262 int s = 0;
1263 int i, x;
1264 char *p = NULL;
1265 size_t filename_len = 0;
1266
1267 if (filename) {
1268 filename_len = strlen(filename);
1269 if (filename_len > (MAX_FILENAMELEN-1)) {
1270 SCLogError(SC_ERR_INVALID_ARGUMENT, "invalid filename option. Max filename-length: %d",MAX_FILENAMELEN-1);
1271 goto error;
1272 }
1273
1274 for (i = 0; i < (int)strlen(filename); i++) {
1275 if (tok >= MAX_TOKS) {
1276 SCLogError(SC_ERR_INVALID_ARGUMENT,
1277 "invalid filename option. Max 2 %%-sign options");
1278 goto error;
1279 }
1280
1281 str[s++] = filename[i];
1282
1283 if (filename[i] == '%') {
1284 str[s-1] = '\0';
1285 SCLogDebug("filename with %%-sign: %s", str);
1286
1287 p = SCStrdup(str);
1288 if (p == NULL)
1289 goto error;
1290 toks[tok++] = p;
1291
1292 s = 0;
1293
1294 if (i+1 < (int)strlen(filename)) {
1295 if (tok >= MAX_TOKS) {
1296 SCLogError(SC_ERR_INVALID_ARGUMENT,
1297 "invalid filename option. Max 2 %%-sign options");
1298 goto error;
1299 }
1300
1301 if (filename[i+1] != 'n' && filename[i+1] != 't' && filename[i+1] != 'i') {
1302 SCLogError(SC_ERR_INVALID_ARGUMENT,
1303 "invalid filename option. Valid %%-sign options: %%n, %%i and %%t");
1304 goto error;
1305 }
1306 str[0] = '%';
1307 str[1] = filename[i+1];
1308 str[2] = '\0';
1309 p = SCStrdup(str);
1310 if (p == NULL)
1311 goto error;
1312 toks[tok++] = p;
1313 i++;
1314 }
1315 }
1316 }
1317
1318 if ((tok == 0) && (pl->mode == LOGMODE_MULTI)) {
1319 SCLogError(SC_ERR_INVALID_ARGUMENT,
1320 "Invalid filename for multimode. Need at list one %%-sign option");
1321 goto error;
1322 }
1323
1324 if (s) {
1325 if (tok >= MAX_TOKS) {
1326 SCLogError(SC_ERR_INVALID_ARGUMENT,
1327 "invalid filename option. Max 3 %%-sign options");
1328 goto error;
1329
1330 }
1331 str[s++] = '\0';
1332 p = SCStrdup(str);
1333 if (p == NULL)
1334 goto error;
1335 toks[tok++] = p;
1336 }
1337
1338 /* finally, store tokens in the pl */
1339 for (i = 0; i < tok; i++) {
1340 if (toks[i] == NULL)
1341 goto error;
1342
1343 SCLogDebug("toks[%d] %s", i, toks[i]);
1344 pl->filename_parts[i] = toks[i];
1345 }
1346 pl->filename_part_cnt = tok;
1347 }
1348 return 0;
1349error:
1350 for (x = 0; x < MAX_TOKS; x++) {
1351 if (toks[x] != NULL)
1352 SCFree(toks[x]);
1353 }
1354 return -1;
1355}

1738/**
1739 * \brief Read the config set the file pointer, open the file
1740 *
1741 * \param PcapLogData.
1742 *
1743 * \retval -1 if failure
1744 * \retval 0 if succesful
1745 */
1746static int PcapLogOpenFileCtx(PcapLogData *pl)
1747{
1748 char *filename = NULL;
1749
1750 PCAPLOG_PROFILE_START;
1751
1752 if (pl->filename != NULL)
1753 filename = pl->filename;
1754 else {
1755 filename = SCMalloc(PATH_MAX);
1756 if (unlikely(filename == NULL)) {
1757 return -1;
1758 }
1759 pl->filename = filename;
1760 }
1761
1762 /** get the time so we can have a filename with seconds since epoch */
1763 struct timeval ts;
1764 memset(&ts, 0x00, sizeof(struct timeval));
1765 TimeGet(&ts);
1766
1767 /* Place to store the name of our PCAP file */
1768 PcapFileName *pf = SCMalloc(sizeof(PcapFileName));
1769 if (unlikely(pf == NULL)) {
1770 return -1;
1771 }
1772 memset(pf, 0, sizeof(PcapFileName));
1773
1774 if (pl->mode == LOGMODE_SGUIL) {
1775 struct tm local_tm;
1776 struct tm *tms = SCLocalTime(ts.tv_sec, &local_tm);
1777
1778 char dirname[32], dirfull[PATH_MAX] = "";
1779
1780 snprintf(dirname, sizeof(dirname), "%04d-%02d-%02d",
1781 tms->tm_year + 1900, tms->tm_mon + 1, tms->tm_mday);
1782
1783 /* create the filename to use */
1784 int ret = snprintf(dirfull, sizeof(dirfull), "%s/%s", pl->dir, dirname);
1785 if (ret < 0 || (size_t)ret >= sizeof(dirfull)) {
1786 SCLogError(SC_ERR_SPRINTF,"failed to construct path");
1787 goto error;
1788 }
1789
1790 /* if mkdir fails file open will fail, so deal with errors there */
1791 (void)SCMkDir(dirfull, 0700);
1792
1793 if ((pf->dirname = SCStrdup(dirfull)) == NULL) {
1794 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory for "
1795 "directory name");
1796 goto error;
1797 }
1798
1799 int written;
1800 if (pl->timestamp_format == TS_FORMAT_SEC) {
1801 written = snprintf(filename, PATH_MAX, "%s/%s.%" PRIu32 "%s",
1802 dirfull, pl->prefix, (uint32_t)ts.tv_sec, pl->suffix);
1803 } else {
1804 written = snprintf(filename, PATH_MAX, "%s/%s.%" PRIu32 ".%" PRIu32 "%s",
1805 dirfull, pl->prefix, (uint32_t)ts.tv_sec,
1806 (uint32_t)ts.tv_usec, pl->suffix);
1807 }
1808 if (written == PATH_MAX) {
1809 SCLogError(SC_ERR_SPRINTF,"log-pcap path overflow");
1810 goto error;
1811 }
1812 } else if (pl->mode == LOGMODE_NORMAL) {
1813 int ret;
1814 /* create the filename to use */
1815 if (pl->timestamp_format == TS_FORMAT_SEC) {
1816 ret = snprintf(filename, PATH_MAX, "%s/%s.%" PRIu32 "%s", pl->dir,
1817 pl->prefix, (uint32_t)ts.tv_sec, pl->suffix);
1818 } else {
1819 ret = snprintf(filename, PATH_MAX,
1820 "%s/%s.%" PRIu32 ".%" PRIu32 "%s", pl->dir, pl->prefix,
1821 (uint32_t)ts.tv_sec, (uint32_t)ts.tv_usec, pl->suffix);
1822 }
1823 if (ret < 0 || (size_t)ret >= PATH_MAX) {
1824 SCLogError(SC_ERR_SPRINTF,"failed to construct path");
1825 goto error;
1826 }
1827 } else if (pl->mode == LOGMODE_MULTI) {
1828 if (pl->filename_part_cnt > 0) {
1829 /* assemble filename from stored tokens */
1830
1831 strlcpy(filename, pl->dir, PATH_MAX);
1832 strlcat(filename, "/", PATH_MAX);
1833
1834 int i;
1835 for (i = 0; i < pl->filename_part_cnt; i++) {
1836 if (pl->filename_parts[i] == NULL ||strlen(pl->filename_parts[i]) == 0)
1837 continue;
1838
1839 /* handle variables */
1840 if (pl->filename_parts[i][0] == '%') {
1841 char str[64] = "";
1842 if (strlen(pl->filename_parts[i]) < 2)
1843 continue;
1844
1845 switch(pl->filename_parts[i][1]) {
1846 case 'n':
1847 snprintf(str, sizeof(str), "%u", pl->thread_number);
1848 break;
1849 case 'i':
1850 {
1851 long thread_id = SCGetThreadIdLong();
1852 snprintf(str, sizeof(str), "%"PRIu64, (uint64_t)thread_id);
1853 break;
1854 }
1855 case 't':
1856 /* create the filename to use */
1857 if (pl->timestamp_format == TS_FORMAT_SEC) {
1858 snprintf(str, sizeof(str), "%"PRIu32, (uint32_t)ts.tv_sec);
1859 } else {
1860 snprintf(str, sizeof(str), "%"PRIu32".%"PRIu32,
1861 (uint32_t)ts.tv_sec, (uint32_t)ts.tv_usec);
1862 }
1863 }
1864 strlcat(filename, str, PATH_MAX);
1865
1866 /* copy the rest over */
1867 } else {
1868 strlcat(filename, pl->filename_parts[i], PATH_MAX);
1869 }
1870 }
1871 strlcat(filename, pl->suffix, PATH_MAX);
1872 } else {
1873 int ret;
1874 /* create the filename to use */
1875 if (pl->timestamp_format == TS_FORMAT_SEC) {
1876 ret = snprintf(filename, PATH_MAX, "%s/%s.%u.%" PRIu32 "%s",
1877 pl->dir, pl->prefix, pl->thread_number,
1878 (uint32_t)ts.tv_sec, pl->suffix);
1879 } else {
1880 ret = snprintf(filename, PATH_MAX,
1881 "%s/%s.%u.%" PRIu32 ".%" PRIu32 "%s", pl->dir,
1882 pl->prefix, pl->thread_number, (uint32_t)ts.tv_sec,
1883 (uint32_t)ts.tv_usec, pl->suffix);
1884 }
1885 if (ret < 0 || (size_t)ret >= PATH_MAX) {
1886 SCLogError(SC_ERR_SPRINTF,"failed to construct path");
1887 goto error;
1888 }
1889 }
1890 SCLogDebug("multi-mode: filename %s", filename);
1891 }
1892
1893 if ((pf->filename = SCStrdup(pl->filename)) == NULL) {
1894 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory. For filename");
1895 goto error;
1896 }
1897 SCLogDebug("Opening pcap file log %s", pf->filename);
1898 TAILQ_INSERT_TAIL(&pl->pcap_file_list, pf, next);
1899
1900 if (pl->mode == LOGMODE_MULTI || pl->mode == LOGMODE_NORMAL) {
1901 pcap_file_thread = pl->filename;
1902 }
1903 PCAPLOG_PROFILE_END(pl->profile_open);
1904 return 0;
1905
1906error:
1907 PcapFileNameFree(pf);
1908 return -1;
1909}

1974static void PcapLogProfilingDump(PcapLogData *pl)
1975{
1976 FILE *fp = NULL;
1977
1978 if (profiling_pcaplog_enabled == 0)
1979 return;
1980
1981 if (profiling_pcaplog_output_to_file == 1) {
1982 fp = fopen(profiling_pcaplog_file_name, profiling_pcaplog_file_mode);
1983 if (fp == NULL) {
1984 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s",
1985 profiling_pcaplog_file_name, strerror(errno));
1986 return;
1987 }
1988 } else {
1989 fp = stdout;
1990 }
1991
1992 /* counters */
1993 fprintf(fp, "\n\nOperation Cnt Avg ticks Total ticks\n");
1994 fprintf(fp, "---------------------------- ---------- ---------- -----------\n");
1995
1996 ProfileReport(fp, pl);
1997 uint64_t total = pl->profile_write.total + pl->profile_rotate.total +
1998 pl->profile_handles.total + pl->profile_open.total +
1999 pl->profile_close.total + pl->profile_lock.total +
2000 pl->profile_unlock.total;
2001
2002 /* overall stats */
2003 fprintf(fp, "\nOverall: %"PRIu64" bytes written, average %d bytes per write.\n",
2004 pl->profile_data_size, pl->profile_write.cnt ?
2005 (int)(pl->profile_data_size / pl->profile_write.cnt) : 0);
2006 fprintf(fp, " PCAP data structure overhead: %"PRIuMAX" per write.\n",
2007 (uintmax_t)sizeof(struct pcap_pkthdr));
2008
2009 /* print total bytes written */
2010 char bytes_str[32];
2011 FormatBytes(pl->profile_data_size, bytes_str, sizeof(bytes_str));
2012 fprintf(fp, " Size written: %s\n", bytes_str);
2013
2014 /* ticks per MiB and GiB */
2015 uint64_t ticks_per_mib = 0, ticks_per_gib = 0;
2016 uint64_t mib = pl->profile_data_size/(1024*1024);
2017 if (mib)
2018 ticks_per_mib = total/mib;
2019 char ticks_per_mib_str[32] = "n/a";
2020 if (ticks_per_mib > 0)
2021 FormatNumber(ticks_per_mib, ticks_per_mib_str, sizeof(ticks_per_mib_str));
2022 fprintf(fp, " Ticks per MiB: %s\n", ticks_per_mib_str);
2023
2024 uint64_t gib = pl->profile_data_size/(1024*1024*1024);
2025 if (gib)
2026 ticks_per_gib = total/gib;
2027 char ticks_per_gib_str[32] = "n/a";
2028 if (ticks_per_gib > 0)
2029 FormatNumber(ticks_per_gib, ticks_per_gib_str, sizeof(ticks_per_gib_str));
2030 fprintf(fp, " Ticks per GiB: %s\n", ticks_per_gib_str);
2031
2032 if (fp != stdout)
2033 fclose(fp);
