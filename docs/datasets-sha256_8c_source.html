49uint32_t Sha256StrHash(void *s)
50{
51 Sha256Type *str = s;
52 uint32_t hash = 5381;
53
54 for (int i = 0; i < (int)sizeof(str->sha256); i++) {
55 hash = ((hash << 5) + hash) + str->sha256[i]; /* hash * 33 + c */
56 }
57 return hash;
58}
