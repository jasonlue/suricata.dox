100int AlertFastLogger(ThreadVars *tv, void *data, const Packet *p)
101{
102 AlertFastLogThread *aft = (AlertFastLogThread *)data;
103 int i;
104 char timebuf[64];
105 int decoder_event = 0;
106
107 CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
108
109 char srcip[46], dstip[46];
110 if (PKT_IS_IPV4(p)) {
111 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
112 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
113 } else if (PKT_IS_IPV6(p)) {
114 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p), srcip, sizeof(srcip));
115 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p), dstip, sizeof(dstip));
116 } else {
117 decoder_event = 1;
118 }
119
120 /* Buffer to store the generated alert strings. The buffer is
121 * filled with alert strings until it doesn't have room to store
122 * another full alert, only then is the buffer written. This is
123 * more efficient for multiple alerts and only slightly slower for
124 * single alerts.
125 */
126 char alert_buffer[MAX_FASTLOG_BUFFER_SIZE];
127
128 char proto[16] = "";
129 const char *protoptr;
130 if (SCProtoNameValid(IP_GET_IPPROTO(p))) {
131 protoptr = known_proto[IP_GET_IPPROTO(p)];
132 } else {
133 snprintf(proto, sizeof(proto), "PROTO:%03" PRIu32, IP_GET_IPPROTO(p));
134 protoptr = proto;
135 }
136 uint16_t src_port_or_icmp = p->sp;
137 uint16_t dst_port_or_icmp = p->dp;
138 if (IP_GET_IPPROTO(p) == IPPROTO_ICMP || IP_GET_IPPROTO(p) == IPPROTO_ICMPV6) {
139 src_port_or_icmp = p->icmp_s.type;
140 dst_port_or_icmp = p->icmp_s.code;
141 }
142 for (i = 0; i < p->alerts.cnt; i++) {
143 const PacketAlert *pa = &p->alerts.alerts[i];
144 if (unlikely(pa->s == NULL)) {
145 continue;
146 }
147
148 const char *action = "";
149 if ((pa->action & ACTION_DROP) && EngineModeIsIPS()) {
150 action = "[Drop] ";
151 } else if (pa->action & ACTION_DROP) {
152 action = "[wDrop] ";
153 }
154
155 /* Create the alert string without locking. */
156 int size = 0;
157 if (likely(decoder_event == 0)) {
158 PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
159 "%s %s[**] [%" PRIu32 ":%" PRIu32 ":%"
160 PRIu32 "] %s [**] [Classification: %s] [Priority: %"PRIu32"]"
161 " {%s} %s:%" PRIu32 " -> %s:%" PRIu32 "\n", timebuf, action,
162 pa->s->gid, pa->s->id, pa->s->rev, pa->s->msg, pa->s->class_msg, pa->s->prio,
163 protoptr, srcip, src_port_or_icmp, dstip, dst_port_or_icmp);
164 } else {
165 PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
166 "%s %s[**] [%" PRIu32 ":%" PRIu32
167 ":%" PRIu32 "] %s [**] [Classification: %s] [Priority: "
168 "%" PRIu32 "] [**] [Raw pkt: ", timebuf, action, pa->s->gid,
169 pa->s->id, pa->s->rev, pa->s->msg, pa->s->class_msg, pa->s->prio);
170 PrintBufferRawLineHex(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
171 GET_PKT_DATA(p), GET_PKT_LEN(p) < 32 ? GET_PKT_LEN(p) : 32);
172 if (p->pcap_cnt != 0) {
173 PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE,
174 "] [pcap file packet: %"PRIu64"]\n", p->pcap_cnt);
175 } else {
176 PrintBufferData(alert_buffer, &size, MAX_FASTLOG_ALERT_SIZE, "]\n");
177 }
178 }
179
180 /* Write the alert to output file */
181 AlertFastLogOutputAlert(aft, alert_buffer, size);
182 }
183
184 return TM_ECODE_OK;
185}

187TmEcode AlertFastLogThreadInit(ThreadVars *t, const void *initdata, void **data)
188{
189 AlertFastLogThread *aft = SCMalloc(sizeof(AlertFastLogThread));
190 if (unlikely(aft == NULL))
191 return TM_ECODE_FAILED;
192 memset(aft, 0, sizeof(AlertFastLogThread));
193 if(initdata == NULL)
194 {
195 SCLogDebug("Error getting context for AlertFastLog. \"initdata\" argument NULL");
196 SCFree(aft);
197 return TM_ECODE_FAILED;
198 }
199 /** Use the Ouptut Context (file pointer and mutex) */
200 aft->file_ctx = ((OutputCtx *)initdata)->data;
201
202 *data = (void *)aft;
203 return TM_ECODE_OK;
204}

206TmEcode AlertFastLogThreadDeinit(ThreadVars *t, void *data)
207{
208 AlertFastLogThread *aft = (AlertFastLogThread *)data;
209 if (aft == NULL) {
210 return TM_ECODE_OK;
211 }
212
213 /* clear memory */
214 memset(aft, 0, sizeof(AlertFastLogThread));
215
216 SCFree(aft);
217 return TM_ECODE_OK;
218}

220/**
221 * \brief Create a new LogFileCtx for "fast" output style.
222 * \param conf The configuration node for this output.
223 * \return A LogFileCtx pointer on success, NULL on failure.
224 */
225OutputInitResult AlertFastLogInitCtx(ConfNode *conf)
226{
227 OutputInitResult result = { NULL, false };
228 LogFileCtx *logfile_ctx = LogFileNewCtx();
229 if (logfile_ctx == NULL) {
230 SCLogDebug("AlertFastLogInitCtx2: Could not create new LogFileCtx");
231 return result;
232 }
233
234 if (SCConfLogOpenGeneric(conf, logfile_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
235 LogFileFreeCtx(logfile_ctx);
236 return result;
237 }
238
239 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
240 if (unlikely(output_ctx == NULL)) {
241 LogFileFreeCtx(logfile_ctx);
242 return result;
243 }
244
245 output_ctx->data = logfile_ctx;
246 output_ctx->DeInit = AlertFastLogDeInitCtx;
247
248 result.ctx = output_ctx;
249 result.ok = true;
250 return result;
251}

253static void AlertFastLogDeInitCtx(OutputCtx *output_ctx)
254{
255 LogFileCtx *logfile_ctx = (LogFileCtx *)output_ctx->data;
256 LogFileFreeCtx(logfile_ctx);
257 SCFree(output_ctx);
258}

264static int AlertFastLogTest01(void)
265{
266 uint8_t *buf = (uint8_t *) "GET /one/ HTTP/1.1\r\n"
267 "Host: one.example.org\r\n";
268
269 uint16_t buflen = strlen((char *)buf);
270 Packet *p = NULL;
271 ThreadVars th_v;
272 DetectEngineThreadCtx *det_ctx;
273
274 memset(&th_v, 0, sizeof(th_v));
275 p = UTHBuildPacket(buf, buflen, IPPROTO_TCP);
276
277 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
278 FAIL_IF(de_ctx == NULL);
279
280 de_ctx->flags |= DE_QUIET;
281
282 FILE *fd = SCClassConfGenerateValidDummyClassConfigFD01();
283 SCClassConfLoadClassficationConfigFile(de_ctx, fd);
284
285 de_ctx->sig_list = SigInit(de_ctx, "alert tcp any any -> any any "
286 "(msg:\"FastLog test\"; content:\"GET\"; "
287 "Classtype:unknown; sid:1;)");
288
289 SigGroupBuild(de_ctx);
290 DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
291
292 SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
293 FAIL_IF_NOT(p->alerts.cnt == 1);
294 FAIL_IF_NOT(strcmp(p->alerts.alerts[0].s->class_msg, "Unknown are we") == 0);
295
296 SigGroupCleanup(de_ctx);
297 SigCleanSignatures(de_ctx);
298 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
299 DetectEngineCtxFree(de_ctx);
300
301 UTHFreePackets(&p, 1);
302 PASS;
303}
