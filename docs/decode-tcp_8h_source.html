180/**
181 * \brief Calculate or validate the checksum for the TCP packet
182 *
183 * \param shdr Pointer to source address field from the IP packet. Used as a
184 * part of the pseudoheader for computing the checksum
185 * \param pkt Pointer to the start of the TCP packet
186 * \param tlen Total length of the TCP packet(header + payload)
187 * \param init The current checksum if validating, 0 if generating.
188 *
189 * \retval csum For validation 0 will be returned for success, for calculation
190 * this will be the checksum.
191 */
192static inline uint16_t TCPChecksum(
193 const uint16_t *shdr, const uint16_t *pkt, uint16_t tlen, uint16_t init)
194{
195 uint16_t pad = 0;
196 uint32_t csum = init;
197
198 csum += shdr[0] + shdr[1] + shdr[2] + shdr[3] + htons(6) + htons(tlen);
199
200 csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
201 pkt[7] + pkt[9];
202
203 tlen -= 20;
204 pkt += 10;
205
206 while (tlen >= 32) {
207 csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
208 pkt[7] +
209 pkt[8] +
210 pkt[9] + pkt[10] + pkt[11] + pkt[12] + pkt[13] +
211 pkt[14] + pkt[15];
212 tlen -= 32;
213 pkt += 16;
214 }
215
216 while(tlen >= 8) {
217 csum += pkt[0] + pkt[1] + pkt[2] + pkt[3];
218 tlen -= 8;
219 pkt += 4;
220 }
221
222 while(tlen >= 4) {
223 csum += pkt[0] + pkt[1];
224 tlen -= 4;
225 pkt += 2;
226 }
227
228 while (tlen > 1) {
229 csum += pkt[0];
230 pkt += 1;
231 tlen -= 2;
232 }
233
234 if (tlen == 1) {
235 *(uint8_t *)(&pad) = (*(uint8_t *)pkt);
236 csum += pad;
237 }
238
239 csum = (csum >> 16) + (csum & 0x0000FFFF);
240 csum += (csum >> 16);
241
242 return (uint16_t)~csum;
243}

247 *
248 * \param shdr Pointer to source address field from the IPV6 packet. Used as a
249 * part of the psuedoheader for computing the checksum
250 * \param pkt Pointer to the start of the TCP packet
251 * \param tlen Total length of the TCP packet(header + payload)
252 * \param init The current checksum if validating, 0 if generating.
253 *
254 * \retval csum For validation 0 will be returned for success, for calculation
255 * this will be the checksum.

259{

262
263 csum += shdr[0] + shdr[1] + shdr[2] + shdr[3] + shdr[4] + shdr[5] +

267 csum += pkt[0] + pkt[1] + pkt[2] + pkt[3] + pkt[4] + pkt[5] + pkt[6] +
268 pkt[7] + pkt[9];
