102json_t *SCJsonString(const char *val)
103{
104 if (val == NULL){
105 return NULL;
106 }
107 json_t * retval = json_string(val);
108 char retbuf[MAX_JSON_SIZE] = {0};
109 if (retval == NULL) {
110 uint32_t u = 0;
111 uint32_t offset = 0;
112 for (u = 0; u < strlen(val); u++) {
113 if (isprint(val[u])) {
114 PrintBufferData(retbuf, &offset, MAX_JSON_SIZE-1, "%c",
115 val[u]);
116 } else {
117 PrintBufferData(retbuf, &offset, MAX_JSON_SIZE-1,
118 "\\x%02X", val[u]);
119 }
120 }
121 retbuf[offset] = '\0';
122 retval = json_string(retbuf);
123 }
124 return retval;
125}

130void EveFileInfo(JsonBuilder *jb, const File *ff, const bool stored)
131{
132 jb_set_string_from_bytes(jb, "filename", ff->name, ff->name_len);
133
134 if (ff->sid_cnt > 0) {
135 jb_open_array(jb, "sid");
136 for (uint32_t i = 0; ff->sid != NULL && i < ff->sid_cnt; i++) {
137 jb_append_uint(jb, ff->sid[i]);
138 }
139 jb_close(jb);
140 }
141
142#ifdef HAVE_MAGIC
143 if (ff->magic)
144 jb_set_string(jb, "magic", (char *)ff->magic);
145#endif
146 jb_set_bool(jb, "gaps", ff->flags & FILE_HAS_GAPS);
147 switch (ff->state) {
148 case FILE_STATE_CLOSED:
149 JB_SET_STRING(jb, "state", "CLOSED");
150 if (ff->flags & FILE_MD5) {
151 jb_set_hex(jb, "md5", (uint8_t *)ff->md5, (uint32_t)sizeof(ff->md5));
152 }
153 if (ff->flags & FILE_SHA1) {
154 jb_set_hex(jb, "sha1", (uint8_t *)ff->sha1, (uint32_t)sizeof(ff->sha1));
155 }
156 break;
157 case FILE_STATE_TRUNCATED:
158 JB_SET_STRING(jb, "state", "TRUNCATED");
159 break;
160 case FILE_STATE_ERROR:
161 JB_SET_STRING(jb, "state", "ERROR");
162 break;
163 default:
164 JB_SET_STRING(jb, "state", "UNKNOWN");
165 break;
166 }
167
168 if (ff->flags & FILE_SHA256) {
169 jb_set_hex(jb, "sha256", (uint8_t *)ff->sha256, (uint32_t)sizeof(ff->sha256));
170 }
171
172 if (stored) {
173 JB_SET_TRUE(jb, "stored");
174 jb_set_uint(jb, "file_id", ff->file_store_id);
175 } else {
176 JB_SET_FALSE(jb, "stored");
177 }
178
179 jb_set_uint(jb, "size", FileTrackedSize(ff));
180 if (ff->end > 0) {
181 jb_set_uint(jb, "start", ff->start);
182 jb_set_uint(jb, "end", ff->end);
183 }
184 jb_set_uint(jb, "tx_id", ff->txid);
185}

187static void EveAddPacketVars(const Packet *p, JsonBuilder *js_vars)
188{
189 if (p == NULL || p->pktvar == NULL) {
190 return;
191 }
192 PktVar *pv = p->pktvar;
193 bool open = false;
194 while (pv != NULL) {
195 if (pv->key || pv->id > 0) {
196 if (!open) {
197 jb_open_array(js_vars, "pktvars");
198 open = true;
199 }
200 jb_start_object(js_vars);
201
202 if (pv->key != NULL) {
203 uint32_t offset = 0;
204 uint8_t keybuf[pv->key_len + 1];
205 PrintStringsToBuffer(keybuf, &offset,
206 sizeof(keybuf),
207 pv->key, pv->key_len);
208 uint32_t len = pv->value_len;
209 uint8_t printable_buf[len + 1];
210 offset = 0;
211 PrintStringsToBuffer(printable_buf, &offset,
212 sizeof(printable_buf),
213 pv->value, pv->value_len);
214 jb_set_string(js_vars, (char *)keybuf, (char *)printable_buf);
215 } else {
216 const char *varname = VarNameStoreLookupById(pv->id, VAR_TYPE_PKT_VAR);
217 uint32_t len = pv->value_len;
218 uint8_t printable_buf[len + 1];
219 uint32_t offset = 0;
220 PrintStringsToBuffer(printable_buf, &offset,
221 sizeof(printable_buf),
222 pv->value, pv->value_len);
223 jb_set_string(js_vars, varname, (char *)printable_buf);
224 }
225 jb_close(js_vars);
226 }
227 pv = pv->next;
228 }
229 if (open) {
230 jb_close(js_vars);
231 }
232}

250static void EveAddFlowVars(const Flow *f, JsonBuilder *js_root, JsonBuilder **js_traffic)
251{
252 if (f == NULL || f->flowvar == NULL) {
253 return;
254 }
255 JsonBuilder *js_flowvars = NULL;
256 JsonBuilder *js_traffic_id = NULL;
257 JsonBuilder *js_traffic_label = NULL;
258 JsonBuilder *js_flowints = NULL;
259 JsonBuilder *js_flowbits = NULL;
260 GenericVar *gv = f->flowvar;
261 while (gv != NULL) {
262 if (gv->type == DETECT_FLOWVAR || gv->type == DETECT_FLOWINT) {
263 FlowVar *fv = (FlowVar *)gv;
264 if (fv->datatype == FLOWVAR_TYPE_STR && fv->key == NULL) {
265 const char *varname = VarNameStoreLookupById(fv->idx,
266 VAR_TYPE_FLOW_VAR);
267 if (varname) {
268 if (js_flowvars == NULL) {
269 js_flowvars = jb_new_array();
270 if (js_flowvars == NULL)
271 break;
272 }
273
274 uint32_t len = fv->data.fv_str.value_len;
275 uint8_t printable_buf[len + 1];
276 uint32_t offset = 0;
277 PrintStringsToBuffer(printable_buf, &offset,
278 sizeof(printable_buf),
279 fv->data.fv_str.value, fv->data.fv_str.value_len);
280
281 jb_start_object(js_flowvars);
282 jb_set_string(js_flowvars, varname, (char *)printable_buf);
283 jb_close(js_flowvars);
284 }
285 } else if (fv->datatype == FLOWVAR_TYPE_STR && fv->key != NULL) {
286 if (js_flowvars == NULL) {
287 js_flowvars = jb_new_array();
288 if (js_flowvars == NULL)
289 break;
290 }
291
292 uint8_t keybuf[fv->keylen + 1];
293 uint32_t offset = 0;
294 PrintStringsToBuffer(keybuf, &offset,
295 sizeof(keybuf),
296 fv->key, fv->keylen);
297
298 uint32_t len = fv->data.fv_str.value_len;
299 uint8_t printable_buf[len + 1];
300 offset = 0;
301 PrintStringsToBuffer(printable_buf, &offset,
302 sizeof(printable_buf),
303 fv->data.fv_str.value, fv->data.fv_str.value_len);
304
305 jb_start_object(js_flowvars);
306 jb_set_string(js_flowvars, (const char *)keybuf, (char *)printable_buf);
307 jb_close(js_flowvars);
308 } else if (fv->datatype == FLOWVAR_TYPE_INT) {
309 const char *varname = VarNameStoreLookupById(fv->idx,
310 VAR_TYPE_FLOW_INT);
311 if (varname) {
312 if (js_flowints == NULL) {
313 js_flowints = jb_new_object();
314 if (js_flowints == NULL)
315 break;
316 }
317 jb_set_uint(js_flowints, varname, fv->data.fv_int.value);
318 }
319
320 }
321 } else if (gv->type == DETECT_FLOWBITS) {
322 FlowBit *fb = (FlowBit *)gv;
323 const char *varname = VarNameStoreLookupById(fb->idx,
324 VAR_TYPE_FLOW_BIT);
325 if (varname) {
326 if (SCStringHasPrefix(varname, TRAFFIC_ID_PREFIX)) {
327 if (js_traffic_id == NULL) {
328 js_traffic_id = jb_new_array();
329 if (unlikely(js_traffic_id == NULL)) {
330 break;
331 }
332 }
333 jb_append_string(js_traffic_id, &varname[traffic_id_prefix_len]);
334 } else if (SCStringHasPrefix(varname, TRAFFIC_LABEL_PREFIX)) {
335 if (js_traffic_label == NULL) {
336 js_traffic_label = jb_new_array();
337 if (unlikely(js_traffic_label == NULL)) {
338 break;
339 }
340 }
341 jb_append_string(js_traffic_label, &varname[traffic_label_prefix_len]);
342 } else {
343 if (js_flowbits == NULL) {
344 js_flowbits = jb_new_array();
345 if (unlikely(js_flowbits == NULL))
346 break;
347 }
348 jb_append_string(js_flowbits, varname);
349 }
350 }
351 }
352 gv = gv->next;
353 }
354 if (js_flowbits) {
355 jb_close(js_flowbits);
356 jb_set_object(js_root, "flowbits", js_flowbits);
357 jb_free(js_flowbits);
358 }
359 if (js_flowints) {
360 jb_close(js_flowints);
361 jb_set_object(js_root, "flowints", js_flowints);
362 jb_free(js_flowints);
363 }
364 if (js_flowvars) {
365 jb_close(js_flowvars);
366 jb_set_object(js_root, "flowvars", js_flowvars);
367 jb_free(js_flowvars);
368 }
369
370 if (js_traffic_id != NULL || js_traffic_label != NULL) {
371 *js_traffic = jb_new_object();
372 if (likely(*js_traffic != NULL)) {
373 if (js_traffic_id != NULL) {
374 jb_close(js_traffic_id);
375 jb_set_object(*js_traffic, "id", js_traffic_id);
376 jb_free(js_traffic_id);
377 }
378 if (js_traffic_label != NULL) {
379 jb_close(js_traffic_label);
380 jb_set_object(*js_traffic, "label", js_traffic_label);
381 jb_free(js_traffic_label);
382 }
383 jb_close(*js_traffic);
384 }
385 }

474void JsonAddrInfoInit(const Packet *p, enum OutputJsonLogDirection dir, JsonAddrInfo *addr)
475{
476 char srcip[46] = {0}, dstip[46] = {0};
477 Port sp, dp;
478
479 switch (dir) {
480 case LOG_DIR_PACKET:
481 if (PKT_IS_IPV4(p)) {
482 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p),
483 srcip, sizeof(srcip));
484 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p),
485 dstip, sizeof(dstip));
486 } else if (PKT_IS_IPV6(p)) {
487 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p),
488 srcip, sizeof(srcip));
489 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p),
490 dstip, sizeof(dstip));
491 } else {
492 /* Not an IP packet so don't do anything */
493 return;
494 }
495 sp = p->sp;
496 dp = p->dp;
497 break;
498 case LOG_DIR_FLOW:
499 case LOG_DIR_FLOW_TOSERVER:
500 if ((PKT_IS_TOSERVER(p))) {
501 if (PKT_IS_IPV4(p)) {
502 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p),
503 srcip, sizeof(srcip));
504 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p),
505 dstip, sizeof(dstip));
506 } else if (PKT_IS_IPV6(p)) {
507 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p),
508 srcip, sizeof(srcip));
509 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p),
510 dstip, sizeof(dstip));
511 }
512 sp = p->sp;
513 dp = p->dp;
514 } else {
515 if (PKT_IS_IPV4(p)) {
516 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p),
517 srcip, sizeof(srcip));
518 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p),
519 dstip, sizeof(dstip));
520 } else if (PKT_IS_IPV6(p)) {
521 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p),
522 srcip, sizeof(srcip));
523 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p),
524 dstip, sizeof(dstip));
525 }
526 sp = p->dp;
527 dp = p->sp;
528 }
529 break;
530 case LOG_DIR_FLOW_TOCLIENT:
531 if ((PKT_IS_TOCLIENT(p))) {
532 if (PKT_IS_IPV4(p)) {
533 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p),
534 srcip, sizeof(srcip));
535 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p),
536 dstip, sizeof(dstip));
537 } else if (PKT_IS_IPV6(p)) {
538 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p),
539 srcip, sizeof(srcip));
540 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p),
541 dstip, sizeof(dstip));
542 }
543 sp = p->sp;
544 dp = p->dp;
545 } else {
546 if (PKT_IS_IPV4(p)) {
547 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p),
548 srcip, sizeof(srcip));
549 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p),
550 dstip, sizeof(dstip));
551 } else if (PKT_IS_IPV6(p)) {
552 PrintInet(AF_INET6, (const void *)GET_IPV6_DST_ADDR(p),
553 srcip, sizeof(srcip));
554 PrintInet(AF_INET6, (const void *)GET_IPV6_SRC_ADDR(p),
555 dstip, sizeof(dstip));
556 }
557 sp = p->dp;
558 dp = p->sp;
559 }
560 break;
561 default:
562 DEBUG_VALIDATE_BUG_ON(1);
563 return;
564 }
565
566
567 strlcpy(addr->src_ip, srcip, JSON_ADDR_LEN);
568
569 switch(p->proto) {
570 case IPPROTO_ICMP:
571 break;
572 case IPPROTO_UDP:
573 case IPPROTO_TCP:
574 case IPPROTO_SCTP:
575 addr->sp = sp;
576 break;
577 }
578
579 strlcpy(addr->dst_ip, dstip, JSON_ADDR_LEN);
580
581 switch(p->proto) {
582 case IPPROTO_ICMP:
583 break;
584 case IPPROTO_UDP:
585 case IPPROTO_TCP:
586 case IPPROTO_SCTP:
587 addr->dp = dp;
588 break;
589 }
590
591 if (SCProtoNameValid(IP_GET_IPPROTO(p))) {
592 strlcpy(addr->proto, known_proto[IP_GET_IPPROTO(p)], sizeof(addr->proto));
593 } else {
594 snprintf(addr->proto, sizeof(addr->proto), "%" PRIu32, IP_GET_IPPROTO(p));
595 }
596}

600static bool CalculateCommunityFlowIdv4(const Flow *f,
601 const uint16_t seed, unsigned char *base64buf)
602{
603 struct {
604 uint16_t seed;
605 uint32_t src;
606 uint32_t dst;
607 uint8_t proto;
608 uint8_t pad0;
609 uint16_t sp;
610 uint16_t dp;
611 } __attribute__((__packed__)) ipv4;
612
613 uint32_t src = f->src.addr_data32[0];
614 uint32_t dst = f->dst.addr_data32[0];
615 uint16_t sp = f->sp;
616 if (f->proto == IPPROTO_ICMP)
617 sp = f->icmp_s.type;
618 sp = htons(sp);
619 uint16_t dp = f->dp;
620 if (f->proto == IPPROTO_ICMP)
621 dp = f->icmp_d.type;
622 dp = htons(dp);
623
624 ipv4.seed = htons(seed);
625 if (ntohl(src) < ntohl(dst) || (src == dst && sp < dp)) {
626 ipv4.src = src;
627 ipv4.dst = dst;
628 ipv4.sp = sp;
629 ipv4.dp = dp;
630 } else {
631 ipv4.src = dst;
632 ipv4.dst = src;
633 ipv4.sp = dp;
634 ipv4.dp = sp;
635 }
636 ipv4.proto = f->proto;
637 ipv4.pad0 = 0;
638
639 uint8_t hash[20];
640 if (SCSha1HashBuffer((const uint8_t *)&ipv4, sizeof(ipv4), hash, sizeof(hash)) == 1) {
641 strlcpy((char *)base64buf, "1:", COMMUNITY_ID_BUF_SIZE);
642 unsigned long out_len = COMMUNITY_ID_BUF_SIZE - 2;
643 if (Base64Encode(hash, sizeof(hash), base64buf+2, &out_len) == SC_BASE64_OK) {
644 return true;
645 }
646 }
647 return false;
648}

662static bool CalculateCommunityFlowIdv6(const Flow *f,
663 const uint16_t seed, unsigned char *base64buf)
664{
665 struct {
666 uint16_t seed;
667 uint32_t src[4];
668 uint32_t dst[4];
669 uint8_t proto;
670 uint8_t pad0;
671 uint16_t sp;
672 uint16_t dp;
673 } __attribute__((__packed__)) ipv6;
674
675 uint16_t sp = f->sp;
676 if (f->proto == IPPROTO_ICMPV6)
677 sp = f->icmp_s.type;

687 {

689 memcpy(&ipv6.dst, &f->dst.addr_data32, 16);
690 ipv6.sp = sp;
691 ipv6.dp = dp;
692 } else {
693 memcpy(&ipv6.src, &f->dst.addr_data32, 16);
694 memcpy(&ipv6.dst, &f->src.addr_data32, 16);
695 ipv6.sp = dp;
696 ipv6.dp = sp;
697 }
698 ipv6.proto = f->proto;
699 ipv6.pad0 = 0;
700
701 uint8_t hash[20];
702 if (SCSha1HashBuffer((const uint8_t *)&ipv6, sizeof(ipv6), hash, sizeof(hash)) == 1) {
703 strlcpy((char *)base64buf, "1:", COMMUNITY_ID_BUF_SIZE);
704 unsigned long out_len = COMMUNITY_ID_BUF_SIZE - 2;
705 if (Base64Encode(hash, sizeof(hash), base64buf+2, &out_len) == SC_BASE64_OK) {
706 return true;
707 }
708 }
709 return false;
710}

821JsonBuilder *CreateEveHeader(const Packet *p, enum OutputJsonLogDirection dir,
822 const char *event_type, JsonAddrInfo *addr, OutputJsonCtx *eve_ctx)
823{
824 char timebuf[64];
825 const Flow *f = (const Flow *)p->flow;
826
827 JsonBuilder *js = jb_new_object();
828 if (unlikely(js == NULL)) {
829 return NULL;
830 }
831
832 CreateIsoTimeString(&p->ts, timebuf, sizeof(timebuf));
833
834 jb_set_string(js, "timestamp", timebuf);
835
836 CreateEveFlowId(js, f);
837
838 /* sensor id */
839 if (sensor_id >= 0) {
840 jb_set_uint(js, "sensor_id", sensor_id);
841 }
842
843 /* input interface */
844 if (p->livedev) {
845 jb_set_string(js, "in_iface", p->livedev->dev);
846 }
847
848 /* pcap_cnt */
849 if (p->pcap_cnt != 0) {
850 jb_set_uint(js, "pcap_cnt", p->pcap_cnt);
851 }
852
853 if (event_type) {
854 jb_set_string(js, "event_type", event_type);
855 }
856
857 /* vlan */
858 if (p->vlan_idx > 0) {
859 jb_open_array(js, "vlan");
860 jb_append_uint(js, p->vlan_id[0]);
861 if (p->vlan_idx > 1) {
862 jb_append_uint(js, p->vlan_id[1]);
863 }
864 jb_close(js);
865 }
866
867 /* 5-tuple */
868 JsonAddrInfo addr_info = json_addr_info_zero;
869 if (addr == NULL) {
870 JsonAddrInfoInit(p, dir, &addr_info);
871 addr = &addr_info;
872 }
873 jb_set_string(js, "src_ip", addr->src_ip);
874 jb_set_uint(js, "src_port", addr->sp);
875 jb_set_string(js, "dest_ip", addr->dst_ip);
876 jb_set_uint(js, "dest_port", addr->dp);
877 jb_set_string(js, "proto", addr->proto);
878
879 /* icmp */
880 switch (p->proto) {
881 case IPPROTO_ICMP:
882 if (p->icmpv4h) {
883 jb_set_uint(js, "icmp_type", p->icmpv4h->type);
884 jb_set_uint(js, "icmp_code", p->icmpv4h->code);
885 }
886 break;
887 case IPPROTO_ICMPV6:
888 if (p->icmpv6h) {
889 jb_set_uint(js, "icmp_type", p->icmpv6h->type);
890 jb_set_uint(js, "icmp_code", p->icmpv6h->code);
891 }
892 break;
893 }
894
895 jb_set_string(js, "pkt_src", PktSrcToString(p->pkt_src));
896
897 if (eve_ctx != NULL) {
898 EveAddCommonOptions(&eve_ctx->cfg, p, f, js);
899 }
900
901 return js;
902}
