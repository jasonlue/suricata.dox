189static void AlertJsonDnp3(const Flow *f, const uint64_t tx_id, JsonBuilder *js)
190{
191 DNP3State *dnp3_state = (DNP3State *)FlowGetAppState(f);
192 if (dnp3_state) {
193 DNP3Transaction *tx = AppLayerParserGetTx(IPPROTO_TCP, ALPROTO_DNP3,
194 dnp3_state, tx_id);
195 if (tx) {
196 JsonBuilderMark mark = { 0, 0, 0 };
197 jb_get_mark(js, &mark);
198 bool logged = false;
199 jb_open_object(js, "dnp3");
200 if (tx->has_request && tx->request_done) {
201 jb_open_object(js, "request");
202 JsonDNP3LogRequest(js, tx);
203 jb_close(js);
204 logged = true;
205 }
206 if (tx->has_response && tx->response_done) {
207 jb_open_object(js, "response");
208 JsonDNP3LogResponse(js, tx);
209 jb_close(js);
210 logged = true;
211 }
212 if (logged) {
213 /* Close dnp3 object. */
214 jb_close(js);
215 } else {
216 jb_restore_mark(js, &mark);
217 }
218 }
219 }

276static void AlertJsonSourceTarget(const Packet *p, const PacketAlert *pa,
277 JsonBuilder *js, JsonAddrInfo *addr)
278{
279 jb_open_object(js, "source");
280 if (pa->s->flags & SIG_FLAG_DEST_IS_TARGET) {
281 jb_set_string(js, "ip", addr->src_ip);
282 switch (p->proto) {
283 case IPPROTO_ICMP:
284 case IPPROTO_ICMPV6:
285 break;
286 case IPPROTO_UDP:
287 case IPPROTO_TCP:
288 case IPPROTO_SCTP:
289 jb_set_uint(js, "port", addr->sp);
290 break;
291 }
292 } else if (pa->s->flags & SIG_FLAG_SRC_IS_TARGET) {
293 jb_set_string(js, "ip", addr->dst_ip);
294 switch (p->proto) {
295 case IPPROTO_ICMP:
296 case IPPROTO_ICMPV6:
297 break;
298 case IPPROTO_UDP:
299 case IPPROTO_TCP:
300 case IPPROTO_SCTP:
301 jb_set_uint(js, "port", addr->dp);
302 break;
303 }
304 }
305 jb_close(js);
306
307 jb_open_object(js, "target");
308 if (pa->s->flags & SIG_FLAG_DEST_IS_TARGET) {
309 jb_set_string(js, "ip", addr->dst_ip);
310 switch (p->proto) {
311 case IPPROTO_ICMP:
312 case IPPROTO_ICMPV6:
313 break;
314 case IPPROTO_UDP:
315 case IPPROTO_TCP:
316 case IPPROTO_SCTP:
317 jb_set_uint(js, "port", addr->dp);
318 break;
319 }
320 } else if (pa->s->flags & SIG_FLAG_SRC_IS_TARGET) {
321 jb_set_string(js, "ip", addr->src_ip);
322 switch (p->proto) {
323 case IPPROTO_ICMP:
324 case IPPROTO_ICMPV6:
325 break;
326 case IPPROTO_UDP:
327 case IPPROTO_TCP:
328 case IPPROTO_SCTP:
329 jb_set_uint(js, "port", addr->sp);
330 break;
331 }
332 }
333 jb_close(js);
334}

344void AlertJsonHeader(void *ctx, const Packet *p, const PacketAlert *pa, JsonBuilder *js,
345 uint16_t flags, JsonAddrInfo *addr, char *xff_buffer)
346{
347 AlertJsonOutputCtx *json_output_ctx = (AlertJsonOutputCtx *)ctx;
348 const char *action = "allowed";
349 /* use packet action if rate_filter modified the action */
350 if (unlikely(pa->flags & PACKET_ALERT_RATE_FILTER_MODIFIED)) {
351 if (PacketTestAction(
352 p, (ACTION_DROP | ACTION_REJECT | ACTION_REJECT_DST | ACTION_REJECT_BOTH))) {
353 action = "blocked";
354 }
355 } else {
356 if (pa->action & (ACTION_REJECT|ACTION_REJECT_DST|ACTION_REJECT_BOTH)) {
357 action = "blocked";
358 } else if ((pa->action & ACTION_DROP) && EngineModeIsIPS()) {
359 action = "blocked";
360 }
361 }
362
363 /* Add tx_id to root element for correlation with other events. */
364 /* json_object_del(js, "tx_id"); */
365 if (pa->flags & PACKET_ALERT_FLAG_TX) {
366 jb_set_uint(js, "tx_id", pa->tx_id);
367 }
368
369 jb_open_object(js, "alert");
370
371 jb_set_string(js, "action", action);
372 jb_set_uint(js, "gid", pa->s->gid);
373 jb_set_uint(js, "signature_id", pa->s->id);
374 jb_set_uint(js, "rev", pa->s->rev);
375 /* TODO: JsonBuilder should handle unprintable characters like
376 * SCJsonString. */
377 jb_set_string(js, "signature", pa->s->msg ? pa->s->msg: "");
378 jb_set_string(js, "category", pa->s->class_msg ? pa->s->class_msg: "");
379 jb_set_uint(js, "severity", pa->s->prio);
380
381 if (p->tenant_id > 0) {
382 jb_set_uint(js, "tenant_id", p->tenant_id);
383 }
384
385 if (addr && pa->s->flags & SIG_FLAG_HAS_TARGET) {
386 AlertJsonSourceTarget(p, pa, js, addr);
387 }
388
389 if ((json_output_ctx != NULL) && (flags & LOG_JSON_RULE_METADATA)) {
390 AlertJsonMetadata(json_output_ctx, pa, js);
391 }
392
393 if (flags & LOG_JSON_RULE) {
394 jb_set_string(js, "rule", pa->s->sig_str);
395 }
396 if (xff_buffer && xff_buffer[0]) {
397 jb_set_string(js, "xff", xff_buffer);
398 }
399
400 jb_close(js);
401}

403static void AlertJsonTunnel(const Packet *p, JsonBuilder *js)
404{
405 if (p->root == NULL) {
406 return;
407 }
408
409 jb_open_object(js, "tunnel");
410
411 /* get a lock to access root packet fields */
412 SCMutex *m = &p->root->tunnel_mutex;
413
414 enum PktSrcEnum pkt_src;
415 uint64_t pcap_cnt;
416 JsonAddrInfo addr = json_addr_info_zero;
417 SCMutexLock(m);
418 JsonAddrInfoInit(p->root, 0, &addr);
419 pcap_cnt = p->root->pcap_cnt;
420 pkt_src = p->root->pkt_src;
421 SCMutexUnlock(m);
422
423 jb_set_string(js, "src_ip", addr.src_ip);
424 jb_set_uint(js, "src_port", addr.sp);
425 jb_set_string(js, "dest_ip", addr.dst_ip);
426 jb_set_uint(js, "dest_port", addr.dp);
427 jb_set_string(js, "proto", addr.proto);
428
429 jb_set_uint(js, "depth", p->recursion_level);
430 if (pcap_cnt != 0) {
431 jb_set_uint(js, "pcap_cnt", pcap_cnt);
432 }
433 jb_set_string(js, "pkt_src", PktSrcToString(pkt_src));
434 jb_close(js);
435}

454static void AlertAddAppLayer(const Packet *p, JsonBuilder *jb,
455 const uint64_t tx_id, const uint16_t option_flags)
456{
457 const AppProto proto = FlowGetAppProtocol(p->flow);
458 JsonBuilderMark mark = { 0, 0, 0 };
459 switch (proto) {
460 case ALPROTO_HTTP1:
461 // TODO: Could result in an empty http object being logged.
462 jb_open_object(jb, "http");
463 if (EveHttpAddMetadata(p->flow, tx_id, jb)) {
464 if (option_flags & LOG_JSON_HTTP_BODY) {
465 EveHttpLogJSONBodyPrintable(jb, p->flow, tx_id);
466 }
467 if (option_flags & LOG_JSON_HTTP_BODY_BASE64) {
468 EveHttpLogJSONBodyBase64(jb, p->flow, tx_id);
469 }
470 }
471 jb_close(jb);
472 break;
473 case ALPROTO_TLS:
474 AlertJsonTls(p->flow, jb);
475 break;
476 case ALPROTO_SSH:
477 AlertJsonSsh(p->flow, jb);
478 break;
479 case ALPROTO_SMTP:
480 jb_get_mark(jb, &mark);
481 jb_open_object(jb, "smtp");
482 if (EveSMTPAddMetadata(p->flow, tx_id, jb)) {
483 jb_close(jb);
484 } else {
485 jb_restore_mark(jb, &mark);
486 }
487 jb_get_mark(jb, &mark);
488 jb_open_object(jb, "email");
489 if (EveEmailAddMetadata(p->flow, tx_id, jb)) {
490 jb_close(jb);
491 } else {
492 jb_restore_mark(jb, &mark);
493 }
494 break;
495 case ALPROTO_NFS:
496 /* rpc */
497 jb_get_mark(jb, &mark);
498 jb_open_object(jb, "rpc");
499 if (EveNFSAddMetadataRPC(p->flow, tx_id, jb)) {
500 jb_close(jb);
501 } else {
502 jb_restore_mark(jb, &mark);
503 }
504 /* nfs */
505 jb_get_mark(jb, &mark);
506 jb_open_object(jb, "nfs");
507 if (EveNFSAddMetadata(p->flow, tx_id, jb)) {
508 jb_close(jb);
509 } else {
510 jb_restore_mark(jb, &mark);
511 }
512 break;
513 case ALPROTO_SMB:
514 jb_get_mark(jb, &mark);
515 jb_open_object(jb, "smb");
516 if (EveSMBAddMetadata(p->flow, tx_id, jb)) {
517 jb_close(jb);
518 } else {
519 jb_restore_mark(jb, &mark);
520 }
521 break;
522 case ALPROTO_SIP:
523 JsonSIPAddMetadata(jb, p->flow, tx_id);
524 break;
525 case ALPROTO_RFB:
526 jb_get_mark(jb, &mark);
527 if (!JsonRFBAddMetadata(p->flow, tx_id, jb)) {
528 jb_restore_mark(jb, &mark);
529 }
530 break;
531 case ALPROTO_FTPDATA:
532 jb_get_mark(jb, &mark);
533 jb_open_object(jb, "ftp_data");
534 EveFTPDataAddMetadata(p->flow, jb);
535 jb_close(jb);
536 break;
537 case ALPROTO_DNP3:
538 AlertJsonDnp3(p->flow, tx_id, jb);
539 break;
540 case ALPROTO_HTTP2:
541 AlertJsonHttp2(p->flow, tx_id, jb);
542 break;
543 case ALPROTO_DNS:
544 AlertJsonDns(p->flow, tx_id, jb);
545 break;
546 case ALPROTO_IKE:
547 jb_get_mark(jb, &mark);
548 if (!EveIKEAddMetadata(p->flow, tx_id, jb)) {
549 jb_restore_mark(jb, &mark);
550 }
551 break;
552 case ALPROTO_MQTT:
553 jb_get_mark(jb, &mark);
554 if (!JsonMQTTAddMetadata(p->flow, tx_id, jb)) {
555 jb_restore_mark(jb, &mark);
556 }
557 break;
558 case ALPROTO_QUIC:
559 jb_get_mark(jb, &mark);
560 if (!JsonQuicAddMetadata(p->flow, tx_id, jb)) {
561 jb_restore_mark(jb, &mark);
562 }
563 break;
564 case ALPROTO_SNMP:
565 AlertJsonSNMP(p->flow, tx_id, jb);
566 break;
567 case ALPROTO_RDP:
568 AlertJsonRDP(p->flow, tx_id, jb);
569 break;
570 case ALPROTO_MODBUS:
571 jb_get_mark(jb, &mark);
572 if (!JsonModbusAddMetadata(p->flow, tx_id, jb)) {
573 jb_restore_mark(jb, &mark);
574 }
575 break;
576 default:
577 break;
578 }

581static void AlertAddFiles(const Packet *p, JsonBuilder *jb, const uint64_t tx_id)
582{
583 FileContainer *ffc = AppLayerParserGetFiles(p->flow,
584 p->flowflags & FLOW_PKT_TOSERVER ? STREAM_TOSERVER:STREAM_TOCLIENT);
585 if (ffc != NULL) {
586 File *file = ffc->head;
587 bool isopen = false;
588 while (file) {
589 if (tx_id == file->txid) {
590 if (!isopen) {
591 isopen = true;
592 jb_open_array(jb, "files");
593 }
594 jb_start_object(jb);
595 EveFileInfo(jb, file, file->flags & FILE_STORED);
596 jb_close(jb);
597 }
598 file = file->next;
599 }
600 if (isopen) {
601 jb_close(jb);
602 }
603 }

606static void AlertAddFrame(const Packet *p, JsonBuilder *jb, const int64_t frame_id)
607{
608 if (p->flow == NULL || (p->proto == IPPROTO_TCP && p->flow->protoctx == NULL))
609 return;
610
611 FramesContainer *frames_container = AppLayerFramesGetContainer(p->flow);
612 if (frames_container == NULL)
613 return;
614
615 Frames *frames = NULL;
616 TcpStream *stream = NULL;
617 if (p->proto == IPPROTO_TCP) {
618 TcpSession *ssn = p->flow->protoctx;
619 if (PKT_IS_TOSERVER(p)) {
620 stream = &ssn->client;
621 frames = &frames_container->toserver;
622 } else {
623 stream = &ssn->server;
624 frames = &frames_container->toclient;
625 }
626 Frame *frame = FrameGetById(frames, frame_id);
627 if (frame != NULL) {
628 FrameJsonLogOneFrame(IPPROTO_TCP, frame, p->flow, stream, p, jb);
629 }
630 } else if (p->proto == IPPROTO_UDP) {
631 if (PKT_IS_TOSERVER(p)) {
632 frames = &frames_container->toserver;
633 } else {
634 frames = &frames_container->toclient;
635 }
636 Frame *frame = FrameGetById(frames, frame_id);
637 if (frame != NULL) {
638 FrameJsonLogOneFrame(IPPROTO_UDP, frame, p->flow, NULL, p, jb);
639 }
640 }
641}

643static int AlertJson(ThreadVars *tv, JsonAlertLogThread *aft, const Packet *p)
644{
645 MemBuffer *payload = aft->payload_buffer;
646 AlertJsonOutputCtx *json_output_ctx = aft->json_output_ctx;
647
648 if (p->alerts.cnt == 0 && !(p->flags & PKT_HAS_TAG))
649 return TM_ECODE_OK;
650
651 for (int i = 0; i < p->alerts.cnt; i++) {
652 const PacketAlert *pa = &p->alerts.alerts[i];
653 if (unlikely(pa->s == NULL)) {
654 continue;
655 }
656
657 /* First initialize the address info (5-tuple). */
658 JsonAddrInfo addr = json_addr_info_zero;
659 JsonAddrInfoInit(p, LOG_DIR_PACKET, &addr);
660
661 /* Check for XFF, overwriting address info if needed. */
662 HttpXFFCfg *xff_cfg = json_output_ctx->xff_cfg != NULL ?
663 json_output_ctx->xff_cfg : json_output_ctx->parent_xff_cfg;;
664 int have_xff_ip = 0;
665 char xff_buffer[XFF_MAXLEN];
666 xff_buffer[0] = 0;
667 if ((xff_cfg != NULL) && !(xff_cfg->flags & XFF_DISABLED) && p->flow != NULL) {
668 if (FlowGetAppProtocol(p->flow) == ALPROTO_HTTP1) {
669 if (pa->flags & PACKET_ALERT_FLAG_TX) {
670 have_xff_ip = HttpXFFGetIPFromTx(p->flow, pa->tx_id, xff_cfg,

674 XFF_MAXLEN);
675 }
676 }

679 if (p->flowflags & FLOW_PKT_TOCLIENT) {
680 strlcpy(addr.dst_ip, xff_buffer, JSON_ADDR_LEN);
681 } else {
682 strlcpy(addr.src_ip, xff_buffer, JSON_ADDR_LEN);
683 }
684 /* Clear have_xff_ip so the xff field does not get
685 * logged below. */
686 have_xff_ip = false;
687 }
688 if (have_xff_ip && !(xff_cfg->flags & XFF_EXTRADATA)) {
689 // reset xff_buffer so as not to log it
690 xff_buffer[0] = 0;
691 }
692 }
693
694 JsonBuilder *jb =
695 CreateEveHeader(p, LOG_DIR_PACKET, "alert", &addr, json_output_ctx->eve_ctx);
696 if (unlikely(jb == NULL))
697 return TM_ECODE_OK;
698
699
700 /* alert */
701 AlertJsonHeader(json_output_ctx, p, pa, jb, json_output_ctx->flags, &addr, xff_buffer);
702
703 if (IS_TUNNEL_PKT(p)) {
704 AlertJsonTunnel(p, jb);
705 }
706
707 if (p->flow != NULL) {
708 if (json_output_ctx->flags & LOG_JSON_APP_LAYER) {
709 AlertAddAppLayer(p, jb, pa->tx_id, json_output_ctx->flags);
710 }
711 /* including fileinfo data is configured by the metadata setting */
712 if (json_output_ctx->flags & LOG_JSON_RULE_METADATA) {
713 AlertAddFiles(p, jb, pa->tx_id);
714 }
715
716 EveAddAppProto(p->flow, jb);
717 if (json_output_ctx->flags & LOG_JSON_FLOW) {
718 jb_open_object(jb, "flow");
719 EveAddFlow(p->flow, jb);
720 jb_close(jb);
721 }
722 }
723
724 /* payload */
725 if (json_output_ctx->flags & (LOG_JSON_PAYLOAD | LOG_JSON_PAYLOAD_BASE64)) {
726 int stream = (p->proto == IPPROTO_TCP) ?
727 (pa->flags & (PACKET_ALERT_FLAG_STATE_MATCH | PACKET_ALERT_FLAG_STREAM_MATCH) ?
728 1 : 0) : 0;
729
730 /* Is this a stream? If so, pack part of it into the payload field */
731 if (stream) {
732 uint8_t flag;
733
734 MemBufferReset(payload);
735
736 if (p->flowflags & FLOW_PKT_TOSERVER) {
737 flag = STREAM_DUMP_TOCLIENT;
738 } else {
739 flag = STREAM_DUMP_TOSERVER;
740 }
741
742 StreamSegmentForEach((const Packet *)p, flag,
743 AlertJsonDumpStreamSegmentCallback,
744 (void *)payload);
745 if (payload->offset) {
746 if (json_output_ctx->flags & LOG_JSON_PAYLOAD_BASE64) {
747 jb_set_base64(jb, "payload", payload->buffer, payload->offset);
748 }
749
750 if (json_output_ctx->flags & LOG_JSON_PAYLOAD) {
751 uint8_t printable_buf[payload->offset + 1];
752 uint32_t offset = 0;
753 PrintStringsToBuffer(printable_buf, &offset,
754 sizeof(printable_buf),
755 payload->buffer, payload->offset);
756 jb_set_string(jb, "payload_printable", (char *)printable_buf);
757 }
758 } else if (p->payload_len) {
759 /* Fallback on packet payload */
760 AlertAddPayload(json_output_ctx, jb, p);
761 }
762 } else {
763 /* This is a single packet and not a stream */
764 AlertAddPayload(json_output_ctx, jb, p);
765 }
766
767 jb_set_uint(jb, "stream", stream);
768 }
769
770 if (pa->flags & PACKET_ALERT_FLAG_FRAME) {
771 AlertAddFrame(p, jb, pa->frame_id);
772 }
773
774 /* base64-encoded full packet */
775 if (json_output_ctx->flags & LOG_JSON_PACKET) {
776 EvePacket(p, jb, 0);
777 }
778
779 char *pcap_filename = PcapLogGetFilename();
780 if (pcap_filename != NULL) {
781 jb_set_string(jb, "capture_file", pcap_filename);
782 }
783
784 OutputJsonBuilderBuffer(jb, aft->ctx);
785 jb_free(jb);
786 }
787
788 if ((p->flags & PKT_HAS_TAG) && (json_output_ctx->flags &
789 LOG_JSON_TAGGED_PACKETS)) {
790 JsonBuilder *packetjs =
791 CreateEveHeader(p, LOG_DIR_PACKET, "packet", NULL, json_output_ctx->eve_ctx);
792 if (unlikely(packetjs != NULL)) {
793 EvePacket(p, packetjs, 0);
794 OutputJsonBuilderBuffer(packetjs, aft->ctx);
795 jb_free(packetjs);
796 }
797 }
798
799 return TM_ECODE_OK;
800}

940static void JsonAlertLogSetupMetadata(AlertJsonOutputCtx *json_output_ctx,
941 ConfNode *conf)
942{
943 static bool warn_no_meta = false;
944 uint32_t payload_buffer_size = JSON_STREAM_BUFFER_SIZE;
945 uint16_t flags = METADATA_DEFAULTS;
946
947 if (conf != NULL) {
948 /* Check for metadata to enable/disable. */
949 ConfNode *metadata = ConfNodeLookupChild(conf, "metadata");
950 if (metadata != NULL) {
951 if (metadata->val != NULL && ConfValIsFalse(metadata->val)) {
952 flags &= ~METADATA_DEFAULTS;
953 } else if (ConfNodeHasChildren(metadata)) {
954 ConfNode *rule_metadata = ConfNodeLookupChild(metadata, "rule");
955 if (rule_metadata) {
956 SetFlag(rule_metadata, "raw", LOG_JSON_RULE, &flags);
957 SetFlag(rule_metadata, "metadata", LOG_JSON_RULE_METADATA,
958 &flags);
959 }
960 SetFlag(metadata, "flow", LOG_JSON_FLOW, &flags);
961 SetFlag(metadata, "app-layer", LOG_JSON_APP_LAYER, &flags);
962 }
963 }
964
965 /* Non-metadata toggles. */
966 SetFlag(conf, "payload", LOG_JSON_PAYLOAD_BASE64, &flags);
967 SetFlag(conf, "packet", LOG_JSON_PACKET, &flags);
968 SetFlag(conf, "tagged-packets", LOG_JSON_TAGGED_PACKETS, &flags);
969 SetFlag(conf, "payload-printable", LOG_JSON_PAYLOAD, &flags);
970 SetFlag(conf, "http-body-printable", LOG_JSON_HTTP_BODY, &flags);
971 SetFlag(conf, "http-body", LOG_JSON_HTTP_BODY_BASE64, &flags);
972
973 /* Check for obsolete configuration flags to enable specific
974 * protocols. These are now just aliases for enabling
975 * app-layer logging. */
976 SetFlag(conf, "http", LOG_JSON_APP_LAYER, &flags);
977 SetFlag(conf, "tls", LOG_JSON_APP_LAYER, &flags);
978 SetFlag(conf, "ssh", LOG_JSON_APP_LAYER, &flags);
979 SetFlag(conf, "smtp", LOG_JSON_APP_LAYER, &flags);
980 SetFlag(conf, "dnp3", LOG_JSON_APP_LAYER, &flags);
981
982 /* And check for obsolete configuration flags for enabling
983 * app-layer and flow as these have been moved under the
984 * metadata key. */
985 SetFlag(conf, "app-layer", LOG_JSON_APP_LAYER, &flags);
986 SetFlag(conf, "flow", LOG_JSON_FLOW, &flags);
987
988 const char *payload_buffer_value = ConfNodeLookupChildValue(conf, "payload-buffer-size");
989
990 if (payload_buffer_value != NULL) {
991 uint32_t value;
992 if (ParseSizeStringU32(payload_buffer_value, &value) < 0) {
993 SCLogError(SC_ERR_ALERT_PAYLOAD_BUFFER, "Error parsing "
994 "payload-buffer-size - %s. Killing engine",
995 payload_buffer_value);
996 exit(EXIT_FAILURE);
997 } else {
998 payload_buffer_size = value;
999 }
1000 }
1001
1002 if (!warn_no_meta && flags & JSON_BODY_LOGGING) {
1003 if (((flags & LOG_JSON_APP_LAYER) == 0)) {
1004 SCLogWarning(SC_WARN_ALERT_CONFIG, "HTTP body logging has been configured, however, "
1005 "metadata logging has not been enabled. HTTP body logging will be disabled.");
1006 flags &= ~JSON_BODY_LOGGING;
1007 warn_no_meta = true;
1008 }
1009 }
1010
1011 json_output_ctx->payload_buffer_size = payload_buffer_size;
1012 }
1013
1014 if (flags & LOG_JSON_RULE_METADATA) {
1015 DetectEngineSetParseMetadata();
1016 }
1017
1018 json_output_ctx->flags |= flags;
1019}
