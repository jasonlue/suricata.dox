83JsonBuilder *JsonBuildFileInfoRecord(const Packet *p, const File *ff, const bool stored,
84 uint8_t dir, HttpXFFCfg *xff_cfg, OutputJsonCtx *eve_ctx)
85{
86 enum OutputJsonLogDirection fdir = LOG_DIR_FLOW;
87
88 switch(dir) {
89 case STREAM_TOCLIENT:
90 fdir = LOG_DIR_FLOW_TOCLIENT;
91 break;
92 case STREAM_TOSERVER:
93 fdir = LOG_DIR_FLOW_TOSERVER;
94 break;
95 default:
96 DEBUG_VALIDATE_BUG_ON(1);
97 break;
98 }
99
100 JsonAddrInfo addr = json_addr_info_zero;
101 JsonAddrInfoInit(p, fdir, &addr);
102
103 /* Overwrite address info with XFF if needed. */
104 int have_xff_ip = 0;
105 char xff_buffer[XFF_MAXLEN];
106 if ((xff_cfg != NULL) && !(xff_cfg->flags & XFF_DISABLED)) {
107 if (FlowGetAppProtocol(p->flow) == ALPROTO_HTTP1) {
108 have_xff_ip = HttpXFFGetIPFromTx(p->flow, ff->txid, xff_cfg, xff_buffer, XFF_MAXLEN);
109 }
110 if (have_xff_ip && xff_cfg->flags & XFF_OVERWRITE) {
111 if (p->flowflags & FLOW_PKT_TOCLIENT) {
112 strlcpy(addr.dst_ip, xff_buffer, JSON_ADDR_LEN);
113 } else {
114 strlcpy(addr.src_ip, xff_buffer, JSON_ADDR_LEN);
115 }
116 have_xff_ip = 0;
117 }
118 }
119
120 JsonBuilder *js = CreateEveHeader(p, fdir, "fileinfo", &addr, eve_ctx);
121 if (unlikely(js == NULL))
122 return NULL;
123
124 JsonBuilderMark mark = { 0, 0, 0 };
125 switch (p->flow->alproto) {
126 case ALPROTO_HTTP1:
127 jb_open_object(js, "http");
128 EveHttpAddMetadata(p->flow, ff->txid, js);
129 jb_close(js);
130 break;
131 case ALPROTO_SMTP:
132 jb_get_mark(js, &mark);
133 jb_open_object(js, "smtp");
134 if (EveSMTPAddMetadata(p->flow, ff->txid, js)) {
135 jb_close(js);
136 } else {
137 jb_restore_mark(js, &mark);
138 }
139 jb_get_mark(js, &mark);
140 jb_open_object(js, "email");
141 if (EveEmailAddMetadata(p->flow, ff->txid, js)) {
142 jb_close(js);
143 } else {
144 jb_restore_mark(js, &mark);
145 }
146 break;
147 case ALPROTO_NFS:
148 /* rpc */
149 jb_get_mark(js, &mark);
150 jb_open_object(js, "rpc");
151 if (EveNFSAddMetadataRPC(p->flow, ff->txid, js)) {
152 jb_close(js);
153 } else {
154 jb_restore_mark(js, &mark);
155 }
156 /* nfs */
157 jb_get_mark(js, &mark);
158 jb_open_object(js, "nfs");
159 if (EveNFSAddMetadata(p->flow, ff->txid, js)) {
160 jb_close(js);
161 } else {
162 jb_restore_mark(js, &mark);
163 }
164 break;
165 case ALPROTO_SMB:
166 jb_get_mark(js, &mark);
167 jb_open_object(js, "smb");
168 if (EveSMBAddMetadata(p->flow, ff->txid, js)) {
169 jb_close(js);
170 } else {
171 jb_restore_mark(js, &mark);
172 }
173 break;
174 case ALPROTO_HTTP2:
175 jb_get_mark(js, &mark);
176 jb_open_object(js, "http2");
177 if (EveHTTP2AddMetadata(p->flow, ff->txid, js)) {
178 jb_close(js);
179 } else {
180 jb_restore_mark(js, &mark);
181 }
182 break;
183 }
184
185 jb_set_string(js, "app_proto", AppProtoToString(p->flow->alproto));
186
187 jb_open_object(js, "fileinfo");
188 EveFileInfo(js, ff, stored);
189 jb_close(js);
190
191 /* xff header */
192 if (have_xff_ip && xff_cfg->flags & XFF_EXTRADATA) {
193 jb_set_string(js, "xff", xff_buffer);
194 }
195
196 return js;
197}

203static void FileWriteJsonRecord(JsonFileLogThread *aft, const Packet *p, const File *ff,
204 uint8_t dir, OutputJsonCtx *eve_ctx)
205{
206 HttpXFFCfg *xff_cfg = aft->filelog_ctx->xff_cfg != NULL ?
207 aft->filelog_ctx->xff_cfg : aft->filelog_ctx->parent_xff_cfg;;
208 JsonBuilder *js = JsonBuildFileInfoRecord(
209 p, ff, ff->flags & FILE_STORED ? true : false, dir, xff_cfg, eve_ctx);
210 if (unlikely(js == NULL)) {
211 return;
212 }
213
214 OutputJsonBuilderBuffer(js, aft->ctx);
215 jb_free(js);
216}

218static int JsonFileLogger(ThreadVars *tv, void *thread_data, const Packet *p,
219 const File *ff, uint8_t dir)
220{
221 SCEnter();
222 JsonFileLogThread *aft = (JsonFileLogThread *)thread_data;

233static TmEcode JsonFileLogThreadInit(ThreadVars *t, const void *initdata, void **data)
234{
235 JsonFileLogThread *aft = SCCalloc(1, sizeof(JsonFileLogThread));
236 if (unlikely(aft == NULL))
237 return TM_ECODE_FAILED;
238
239 if(initdata == NULL)
240 {
241 SCLogDebug("Error getting context for EveLogFile. \"initdata\" argument NULL");
242 goto error_exit;
243 }
244
245 /* Use the Ouptut Context (file pointer and mutex) */
246 aft->filelog_ctx = ((OutputCtx *)initdata)->data;
247 aft->ctx = CreateEveThreadCtx(t, aft->filelog_ctx->eve_ctx);
248 if (!aft->ctx) {
249 goto error_exit;
250 }
251
252 *data = (void *)aft;
253 return TM_ECODE_OK;
254
255error_exit:
256 SCFree(aft);
257 return TM_ECODE_FAILED;
258}

260static TmEcode JsonFileLogThreadDeinit(ThreadVars *t, void *data)
261{
262 JsonFileLogThread *aft = (JsonFileLogThread *)data;
263 if (aft == NULL) {
264 return TM_ECODE_OK;
265 }
266
267 FreeEveThreadCtx(aft->ctx);
268
269 /* clear memory */
270 memset(aft, 0, sizeof(JsonFileLogThread));
271
272 SCFree(aft);
273 return TM_ECODE_OK;
274}

286/** \brief Create a new http log LogFileCtx.
287 * \param conf Pointer to ConfNode containing this loggers configuration.
288 * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
289 * */
290static OutputInitResult OutputFileLogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
291{
292 OutputInitResult result = { NULL, false };
293 OutputJsonCtx *ojc = parent_ctx->data;
294
295 OutputFileCtx *output_file_ctx = SCCalloc(1, sizeof(OutputFileCtx));
296 if (unlikely(output_file_ctx == NULL))
297 return result;
298
299 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
300 if (unlikely(output_ctx == NULL)) {
301 SCFree(output_file_ctx);
302 return result;
303 }
304
305 if (conf) {
306 const char *force_filestore = ConfNodeLookupChildValue(conf, "force-filestore");
307 if (force_filestore != NULL && ConfValIsTrue(force_filestore)) {
308 FileForceFilestoreEnable();
309 SCLogConfig("forcing filestore of all files");
310 }
311
312 const char *force_magic = ConfNodeLookupChildValue(conf, "force-magic");
313 if (force_magic != NULL && ConfValIsTrue(force_magic)) {
314 FileForceMagicEnable();
315 SCLogConfig("forcing magic lookup for logged files");
316 }
317
318 FileForceHashParseCfg(conf);
319 }
320
321 if (conf != NULL && ConfNodeLookupChild(conf, "xff") != NULL) {
322 output_file_ctx->xff_cfg = SCCalloc(1, sizeof(HttpXFFCfg));
323 if (output_file_ctx->xff_cfg != NULL) {
324 HttpXFFGetCfg(conf, output_file_ctx->xff_cfg);
325 }
326 } else if (ojc->xff_cfg) {
327 output_file_ctx->parent_xff_cfg = ojc->xff_cfg;
328 }
329
330 output_file_ctx->eve_ctx = ojc;
331 output_ctx->data = output_file_ctx;
332 output_ctx->DeInit = OutputFileLogDeinitSub;
333
334 FileForceTrackingEnable();
335 result.ctx = output_ctx;
336 result.ok = true;
337 return result;
338}
