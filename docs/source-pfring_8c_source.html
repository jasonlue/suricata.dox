188static inline void PfringDumpCounters(PfringThreadVars *ptv)
189{
190 pfring_stat pfring_s;
191 if (likely((pfring_stats(ptv->pd, &pfring_s) >= 0))) {
192 /* pfring counter is per socket and is not cleared after read.
193 * So to get the number of packet on the interface we can add
194 * the newly seen packets and drops for this thread and add it
195 * to the interface counter */
196 uint64_t th_pkts = StatsGetLocalCounterValue(ptv->tv, ptv->capture_kernel_packets);
197 uint64_t th_drops = StatsGetLocalCounterValue(ptv->tv, ptv->capture_kernel_drops);
198 SC_ATOMIC_ADD(ptv->livedev->pkts, pfring_s.recv - th_pkts);
199 SC_ATOMIC_ADD(ptv->livedev->drop, pfring_s.drop - th_drops);
200 StatsSetUI64(ptv->tv, ptv->capture_kernel_packets, pfring_s.recv);
201 StatsSetUI64(ptv->tv, ptv->capture_kernel_drops, pfring_s.drop);
202
203#ifdef HAVE_PF_RING_FLOW_OFFLOAD
204 if (ptv->flags & PFRING_FLAGS_BYPASS) {
205 uint64_t th_bypassed = StatsGetLocalCounterValue(ptv->tv, ptv->capture_bypassed);
206 SC_ATOMIC_ADD(ptv->livedev->bypassed, pfring_s.shunt - th_bypassed);
207 StatsSetUI64(ptv->tv, ptv->capture_bypassed, pfring_s.shunt);
208 }
209#endif
210 }
211}

213/**
214 * \brief Pfring Packet Process function.
215 *
216 * This function fills in our packet structure from libpfring.
217 * From here the packets are picked up by the DecodePfring thread.
218 *
219 * \param user pointer to PfringThreadVars
220 * \param h pointer to pfring packet header
221 * \param p pointer to the current packet
222 */
223static inline void PfringProcessPacket(void *user, struct pfring_pkthdr *h, Packet *p)
224{
225 PfringThreadVars *ptv = (PfringThreadVars *)user;
226
227 ptv->bytes += h->caplen;
228 ptv->pkts++;
229 p->livedev = ptv->livedev;
230
231 /* PF_RING may fail to set timestamp */
232 if (h->ts.tv_sec == 0) {
233 gettimeofday((struct timeval *)&h->ts, NULL);
234 }
235
236 p->ts.tv_sec = h->ts.tv_sec;
237 p->ts.tv_usec = h->ts.tv_usec;
238
239 /* PF_RING all packets are marked as a link type of ethernet
240 * so that is what we do here. */
241 p->datalink = LINKTYPE_ETHERNET;
242
243 /* In the past, we needed this vlan handling in cases
244 * where the vlan header was stripped from the raw packet.
245 * With modern (at least >= 6) versions of PF_RING, the
246 * 'copy_data_to_ring' function (kernel/pf_ring.c) makes
247 * sure that if the hardware stripped the vlan header,
248 * it is put back by PF_RING.
249 *
250 * PF_RING should put it back in all cases, but as a extra
251 * precaution keep the check here. If the vlan header is
252 * part of the raw packet, the vlan_offset will be set.
253 * So if it is not set, use the parsed info from PF_RING's
254 * extended header.
255 */
256 if (ptv->vlan_in_ext_header &&
257 h->extended_hdr.parsed_pkt.offset.vlan_offset == 0 &&
258 h->extended_hdr.parsed_pkt.vlan_id)
259 {
260 p->vlan_id[0] = h->extended_hdr.parsed_pkt.vlan_id & 0x0fff;
261 p->vlan_idx = 1;
262
263 if (!ptv->vlan_hdr_warned) {
264 SCLogWarning(SC_ERR_PF_RING_VLAN, "no VLAN header in the raw "
265 "packet. See #2355.");
266 ptv->vlan_hdr_warned = true;
267 }
268 }
269
270 switch (ptv->checksum_mode) {
271 case CHECKSUM_VALIDATION_RXONLY:
272 if (h->extended_hdr.rx_direction == 0) {
273 p->flags |= PKT_IGNORE_CHECKSUM;
274 }
275 break;
276 case CHECKSUM_VALIDATION_DISABLE:
277 p->flags |= PKT_IGNORE_CHECKSUM;
278 break;
279 case CHECKSUM_VALIDATION_AUTO:
280 if (ChecksumAutoModeCheck(ptv->pkts,
281 SC_ATOMIC_GET(ptv->livedev->pkts),
282 SC_ATOMIC_GET(ptv->livedev->invalid_checksums))) {
283 ptv->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
284 p->flags |= PKT_IGNORE_CHECKSUM;
285 }
286 break;
287 default:
288 break;
289 }
290
291 SET_PKT_LEN(p, h->caplen);
292}

294#ifdef HAVE_PF_RING_FLOW_OFFLOAD
295/**
296 * \brief Pfring bypass callback function
297 *
298 * \param p a Packet to use information from to trigger bypass
299 * \return 1 if bypass is successful, 0 if not
300 */
301static int PfringBypassCallback(Packet *p)
302{
303 hw_filtering_rule r;
304
305 /* Only bypass TCP and UDP */
306 if (!(PKT_IS_TCP(p) || PKT_IS_UDP(p))) {
307 return 0;
308 }
309
310 /* Bypassing tunneled packets is currently not supported */
311 if (IS_TUNNEL_PKT(p)) {
312 return 0;
313 }
314
315 r.rule_family_type = generic_flow_id_rule;
316 r.rule_family.flow_id_rule.action = flow_drop_rule;
317 r.rule_family.flow_id_rule.thread = 0;
318 r.rule_family.flow_id_rule.flow_id = p->pfring_v.flow_id;
319
320 SCLogDebug("Bypass set for flow ID = %u", p->pfring_v.flow_id);
321
322 if (pfring_add_hw_rule(p->pfring_v.ptv->pd, &r) < 0) {
323 return 0;
324 }
325
326 return 1;
327}
328#endif

330/**
331 * \brief Recieves packets from an interface via libpfring.
332 *
333 * This function recieves packets from an interface and passes
334 * the packet on to the pfring callback function.
335 *
336 * \param tv pointer to ThreadVars
337 * \param data pointer that gets cast into PfringThreadVars for ptv
338 * \param slot slot containing task information
339 * \retval TM_ECODE_OK on success
340 * \retval TM_ECODE_FAILED on failure
341 */
342TmEcode ReceivePfringLoop(ThreadVars *tv, void *data, void *slot)
343{
344 SCEnter();
345
346 PfringThreadVars *ptv = (PfringThreadVars *)data;
347 Packet *p = NULL;
348 struct pfring_pkthdr hdr;
349 TmSlot *s = (TmSlot *)slot;
350 time_t last_dump = 0;
351 u_int buffer_size;
352 u_char *pkt_buffer;
353
354 ptv->slot = s->slot_next;
355
356 /* we have to enable the ring here as we need to do it after all
357 * the threads have called pfring_set_cluster(). */
358 int rc = pfring_enable_ring(ptv->pd);
359 if (rc != 0) {
360 SCLogError(SC_ERR_PF_RING_OPEN, "pfring_enable_ring failed returned %d ", rc);
361 SCReturnInt(TM_ECODE_FAILED);
362 }
363
364 while(1) {
365 if (suricata_ctl_flags & SURICATA_STOP) {
366 SCReturnInt(TM_ECODE_OK);
367 }
368
369 /* make sure we have at least one packet in the packet pool, to prevent
370 * us from alloc'ing packets at line rate */
371 PacketPoolWait();
372
373 p = PacketGetFromQueueOrAlloc();
374 if (p == NULL) {
375 SCReturnInt(TM_ECODE_FAILED);
376 }
377 PKT_SET_SRC(p, PKT_SRC_WIRE);
378
379 /* Some flavours of PF_RING may fail to set timestamp - see PF-RING-enabled libpcap code*/
380 hdr.ts.tv_sec = hdr.ts.tv_usec = 0;
381
382 /* Check for Zero-copy mode */
383 if (ptv->flags & PFRING_FLAGS_ZERO_COPY) {
384 buffer_size = 0;
385 pkt_buffer = NULL;
386 } else {
387 buffer_size = GET_PKT_DIRECT_MAX_SIZE(p);
388 pkt_buffer = GET_PKT_DIRECT_DATA(p);
389 }
390
391 int r = pfring_recv(ptv->pd, &pkt_buffer,
392 buffer_size,
393 &hdr,
394 LIBPFRING_WAIT_FOR_INCOMING);
395 if (likely(r == 1)) {
396 /* profiling started before blocking pfring_recv call, so
397 * reset it here */
398 PACKET_PROFILING_RESTART(p);
399
400#ifdef HAVE_PF_RING_FLOW_OFFLOAD
401 if (ptv->flags & PFRING_FLAGS_BYPASS) {
402 /* pkt hash contains the flow id in this configuration */
403 p->pfring_v.flow_id = hdr.extended_hdr.pkt_hash;
404 p->pfring_v.ptv = ptv;
405 p->BypassPacketsFlow = PfringBypassCallback;
406 }
407#endif
408
409 /* Check for Zero-copy mode */
410 if (ptv->flags & PFRING_FLAGS_ZERO_COPY) {
411 PacketSetData(p, pkt_buffer, hdr.caplen);
412 }
413
414 PfringProcessPacket(ptv, &hdr, p);
415
416 if (TmThreadsSlotProcessPkt(ptv->tv, ptv->slot, p) != TM_ECODE_OK) {
417 SCReturnInt(TM_ECODE_FAILED);
418 }
419
420 /* Trigger one dump of stats every second */
421 if (p->ts.tv_sec != last_dump) {
422 PfringDumpCounters(ptv);
423 last_dump = p->ts.tv_sec;
424 }
425 } else if (unlikely(r == 0)) {
426 if (suricata_ctl_flags & SURICATA_STOP) {
427 SCReturnInt(TM_ECODE_OK);
428 }
429
430 /* pfring didn't use the packet yet */
431 TmThreadsCaptureHandleTimeout(tv, p);
432
433 } else {
434 SCLogError(SC_ERR_PF_RING_RECV,"pfring_recv error %" PRId32 "", r);
435 TmqhOutputPacketpool(ptv->tv, p);
436 SCReturnInt(TM_ECODE_FAILED);
437 }
438 StatsSyncCountersIfSignalled(tv);
439 }
440
441 return TM_ECODE_OK;
442}

444/**
445 * \brief Stop function for ReceivePfringLoop.
446 *
447 * This function forces ReceivePfringLoop to stop the
448 * execution, exiting the packet capture loop.
449 *
450 * \param tv pointer to ThreadVars
451 * \param data pointer that gets cast into PfringThreadVars for ptv
452 * \retval TM_ECODE_OK on success
453 * \retval TM_ECODE_FAILED on failure
454 */
455TmEcode PfringBreakLoop(ThreadVars *tv, void *data)
456{
457 PfringThreadVars *ptv = (PfringThreadVars *)data;
458
459 /* Safety check */
460 if (ptv->pd == NULL) {
461 return TM_ECODE_FAILED;
462 }
463
464 pfring_breakloop(ptv->pd);
465
466 return TM_ECODE_OK;
467}

469/**
470 * \brief Init function for RecievePfring.
471 *
472 * This is a setup function for recieving packets
473 * via libpfring.
474 *
475 * \param tv pointer to ThreadVars
476 * \param initdata pointer to the interface passed from the user
477 * \param data pointer gets populated with PfringThreadVars
478 * \todo add a config option for setting cluster id
479 * \todo Create a general pfring setup function.
480 * \retval TM_ECODE_OK on success
481 * \retval TM_ECODE_FAILED on error
482 */
483TmEcode ReceivePfringThreadInit(ThreadVars *tv, const void *initdata, void **data)
484{
485 int rc;
486 u_int32_t version = 0;
487 PfringIfaceConfig *pfconf = (PfringIfaceConfig *) initdata;
488 unsigned int opflag;
489 char const *active_runmode = RunmodeGetActive();
490
491 if (pfconf == NULL)
492 return TM_ECODE_FAILED;
493
494 PfringThreadVars *ptv = SCMalloc(sizeof(PfringThreadVars));
495 if (unlikely(ptv == NULL)) {
496 pfconf->DerefFunc(pfconf);
497 return TM_ECODE_FAILED;
498 }
499 memset(ptv, 0, sizeof(PfringThreadVars));
500
501 ptv->tv = tv;
502 ptv->threads = 1;
503
504 ptv->interface = SCStrdup(pfconf->iface);
505 if (unlikely(ptv->interface == NULL)) {
506 SCLogError(SC_ERR_MEM_ALLOC, "Unable to allocate device string");
507 SCFree(ptv);
508 SCReturnInt(TM_ECODE_FAILED);
509 }
510
511 ptv->livedev = LiveGetDevice(pfconf->iface);
512 if (ptv->livedev == NULL) {
513 SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
514 SCFree(ptv);
515 SCReturnInt(TM_ECODE_FAILED);
516 }
517
518 /* enable zero-copy mode for workers runmode */
519 if (active_runmode && strcmp("workers", active_runmode) == 0) {
520 ptv->flags |= PFRING_FLAGS_ZERO_COPY;
521 SCLogPerf("Enabling zero-copy for %s", ptv->interface);
522 }
523
524 ptv->checksum_mode = pfconf->checksum_mode;
525
526 opflag = PF_RING_PROMISC;
527
528 /* if we have a recent kernel, we need to use parsed_pkt to get VLAN info */
529 if (ptv->vlan_in_ext_header) {
530 opflag |= PF_RING_LONG_HEADER;
531 }
532
533 if (ptv->checksum_mode == CHECKSUM_VALIDATION_RXONLY) {
534 if (strncmp(ptv->interface, "dna", 3) == 0) {
535 SCLogWarning(SC_ERR_INVALID_VALUE,
536 "Can't use rxonly checksum-checks on DNA interface,"
537 " resetting to auto");
538 ptv->checksum_mode = CHECKSUM_VALIDATION_AUTO;
539 } else {
540 opflag |= PF_RING_LONG_HEADER;
541 }
542 }
543
544#ifdef HAVE_PF_RING_FLOW_OFFLOAD
545 if (pfconf->flags & PFRING_CONF_FLAGS_BYPASS) {
546 opflag |= PF_RING_FLOW_OFFLOAD | PF_RING_FLOW_OFFLOAD_NOUPDATES;
547 ptv->flags |= PFRING_FLAGS_BYPASS;
548 }
549#endif
550
551 ptv->pd = pfring_open(ptv->interface, (uint32_t)default_packet_size, opflag);
552 if (ptv->pd == NULL) {
553 SCLogError(SC_ERR_PF_RING_OPEN,"Failed to open %s: pfring_open error."
554 " Check if %s exists and pf_ring module is loaded.",
555 ptv->interface,
556 ptv->interface);
557 pfconf->DerefFunc(pfconf);
558 SCFree(ptv);
559 return TM_ECODE_FAILED;
560 }
561
562 pfring_set_application_name(ptv->pd, (char *)PROG_NAME);
563 pfring_version(ptv->pd, &version);
564
565 /* We only set cluster info if the number of pfring threads is greater than 1 */
566 ptv->threads = pfconf->threads;
567
568 ptv->cluster_id = pfconf->cluster_id;
569
570 if ((ptv->threads == 1) && (strncmp(ptv->interface, "dna", 3) == 0)) {
571 SCLogInfo("DNA interface detected, not adding thread to cluster");
572 } else if (strncmp(ptv->interface, "zc", 2) == 0) {
573 SCLogInfo("ZC interface detected, not adding thread to cluster");
574 } else {
575 ptv->ctype = (cluster_type)pfconf->ctype;
576 rc = pfring_set_cluster(ptv->pd, ptv->cluster_id, ptv->ctype);
577
578 if (rc != 0) {
579 SCLogError(SC_ERR_PF_RING_SET_CLUSTER_FAILED, "pfring_set_cluster "
580 "returned %d for cluster-id: %d", rc, ptv->cluster_id);
581 if (rc != PF_RING_ERROR_NOT_SUPPORTED || (pfconf->flags & PFRING_CONF_FLAGS_CLUSTER)) {
582 /* cluster is mandatory as explicitly specified in the configuration */
583 pfconf->DerefFunc(pfconf);
584 return TM_ECODE_FAILED;
585 }
586 }
587 }
588
589 if (ptv->threads > 1) {
590 SCLogPerf("(%s) Using PF_RING v.%d.%d.%d, interface %s, cluster-id %d",
591 tv->name, (version & 0xFFFF0000) >> 16, (version & 0x0000FF00) >> 8,
592 version & 0x000000FF, ptv->interface, ptv->cluster_id);
593 } else {
594 SCLogPerf("(%s) Using PF_RING v.%d.%d.%d, interface %s, cluster-id %d, single-pfring-thread",
595 tv->name, (version & 0xFFFF0000) >> 16, (version & 0x0000FF00) >> 8,
596 version & 0x000000FF, ptv->interface, ptv->cluster_id);
597 }
598
599 if (pfconf->bpf_filter) {
600 ptv->bpf_filter = SCStrdup(pfconf->bpf_filter);
601 if (unlikely(ptv->bpf_filter == NULL)) {
602 SCLogError(SC_ERR_MEM_ALLOC, "Set PF_RING bpf filter failed.");
603 } else {
604 SCMutexLock(&pfring_bpf_set_filter_lock);
605 rc = pfring_set_bpf_filter(ptv->pd, ptv->bpf_filter);
606 SCMutexUnlock(&pfring_bpf_set_filter_lock);
607
608 if (rc < 0) {
609 SCLogError(SC_ERR_INVALID_VALUE, "Failed to compile BPF \"%s\"",
610 ptv->bpf_filter);
611 return TM_ECODE_FAILED;
612 }
613 }
614 }
615
616 ptv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
617 ptv->tv);
618 ptv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
619 ptv->tv);
620#ifdef HAVE_PF_RING_FLOW_OFFLOAD
621 ptv->capture_bypassed = StatsRegisterCounter("capture.bypassed",
622 ptv->tv);
623#endif
624
625 /* If kernel is older than 3.0, VLAN is not stripped so we don't
626 * get the info from packt extended header but we will use a standard
627 * parsing */
628 ptv->vlan_in_ext_header = 1;
629 if (! SCKernelVersionIsAtLeast(3, 0)) {
630 ptv->vlan_in_ext_header = 0;
631 }
632
633 /* If VLAN tags are not in the extended header, set cluster type to 5-tuple
634 * or in case of a ZC interface, do nothing */
635 if ((! ptv->vlan_in_ext_header) && ptv->ctype == CLUSTER_FLOW &&
636 strncmp(ptv->interface, "zc", 2) != 0) {
637 SCLogPerf("VLAN not in extended header, setting cluster type to CLUSTER_FLOW_5_TUPLE");
638 rc = pfring_set_cluster(ptv->pd, ptv->cluster_id, CLUSTER_FLOW_5_TUPLE);
639
640 if (rc != 0) {
641 SCLogError(SC_ERR_PF_RING_SET_CLUSTER_FAILED, "pfring_set_cluster "
642 "returned %d for cluster-id: %d", rc, ptv->cluster_id);
643 pfconf->DerefFunc(pfconf);
644 return TM_ECODE_FAILED;
645 }
646 }
647
648 DatalinkSetGlobalType(LINKTYPE_ETHERNET);
649
650 *data = (void *)ptv;
651 pfconf->DerefFunc(pfconf);
652
653 return TM_ECODE_OK;
654}
