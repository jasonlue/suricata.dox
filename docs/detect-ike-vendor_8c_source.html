57static InspectionBuffer *IkeVendorGetData(DetectEngineThreadCtx *det_ctx,
58 const DetectEngineTransforms *transforms, Flow *f, struct IkeVendorGetDataArgs *cbdata,
59 int list_id, bool first)
60{
61 SCEnter();
62
63 InspectionBuffer *buffer =
64 InspectionBufferMultipleForListGet(det_ctx, list_id, cbdata->local_id);
65 if (buffer == NULL)
66 return NULL;
67 if (!first && buffer->inspect != NULL)
68 return buffer;
69
70 const uint8_t *data;
71 uint32_t data_len;
72 if (rs_ike_tx_get_vendor(cbdata->txv, cbdata->local_id, &data, &data_len) == 0) {
73 return NULL;
74 }
75
76 InspectionBufferSetupMulti(buffer, transforms, data, data_len);
77
78 SCReturnPtr(buffer, "InspectionBuffer");
79}

81/** \brief IkeVendor Mpm prefilter callback
82 *
83 * \param det_ctx detection engine thread ctx
84 * \param p packet to inspect
85 * \param f flow to inspect
86 * \param txv tx to inspect
87 * \param pectx inspection context
88 */
89static void PrefilterTxIkeVendor(DetectEngineThreadCtx *det_ctx, const void *pectx, Packet *p,
90 Flow *f, void *txv, const uint64_t idx, const uint8_t flags)
91{
92 SCEnter();
93
94 const PrefilterMpmIkeVendor *ctx = (const PrefilterMpmIkeVendor *)pectx;
95 const MpmCtx *mpm_ctx = ctx->mpm_ctx;
96 const int list_id = ctx->list_id;
97
98 uint32_t local_id = 0;
99 while (1) {
100 struct IkeVendorGetDataArgs cbdata = { local_id, txv };
101 InspectionBuffer *buffer =
102 IkeVendorGetData(det_ctx, ctx->transforms, f, &cbdata, list_id, true);
103 if (buffer == NULL)
104 break;
105
106 if (buffer->inspect_len >= mpm_ctx->minlen) {
107 (void)mpm_table[mpm_ctx->mpm_type].Search(
108 mpm_ctx, &det_ctx->mtcu, &det_ctx->pmq, buffer->inspect, buffer->inspect_len);
109 PREFILTER_PROFILING_ADD_BYTES(det_ctx, buffer->inspect_len);
110 }
111 local_id++;
112 }
113
114 SCReturn;
115}

137static uint8_t DetectEngineInspectIkeVendor(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
138 const DetectEngineAppInspectionEngine *engine, const Signature *s, Flow *f, uint8_t flags,
139 void *alstate, void *txv, uint64_t tx_id)
140{
141 uint32_t local_id = 0;
142
143 const DetectEngineTransforms *transforms = NULL;
144 if (!engine->mpm) {
145 transforms = engine->v2.transforms;
146 }
147
148 while (1) {
149 struct IkeVendorGetDataArgs cbdata = {
150 local_id,
151 txv,
152 };
153 InspectionBuffer *buffer =
154 IkeVendorGetData(det_ctx, transforms, f, &cbdata, engine->sm_list, false);
155 if (buffer == NULL || buffer->inspect == NULL)
156 break;
157
158 det_ctx->buffer_offset = 0;
159 det_ctx->discontinue_matching = 0;
160 det_ctx->inspection_recursion_counter = 0;
161
162 const int match = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd, NULL, f,
163 (uint8_t *)buffer->inspect, buffer->inspect_len, buffer->inspect_offset,
164 DETECT_CI_FLAGS_SINGLE, DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
165 if (match == 1) {
166 return DETECT_ENGINE_INSPECT_SIG_MATCH;
167 }
168 local_id++;
169 }
170 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
171}

173/**
174 * \brief Registration function for ike.vendor keyword.
175 */
176void DetectIkeVendorRegister(void)
177{
178 sigmatch_table[DETECT_AL_IKE_VENDOR].name = "ike.vendor";
179 sigmatch_table[DETECT_AL_IKE_VENDOR].desc = "match IKE Vendor";
180 sigmatch_table[DETECT_AL_IKE_VENDOR].url = "/rules/ike-keywords.html#ike-vendor";
181 sigmatch_table[DETECT_AL_IKE_VENDOR].Setup = DetectIkeVendorSetup;
182 sigmatch_table[DETECT_AL_IKE_VENDOR].flags |= SIGMATCH_NOOPT;
183 sigmatch_table[DETECT_AL_IKE_VENDOR].flags |= SIGMATCH_INFO_STICKY_BUFFER;
184
185 DetectAppLayerMpmRegister2("ike.vendor", SIG_FLAG_TOSERVER, 1, PrefilterMpmIkeVendorRegister,
186 NULL, ALPROTO_IKE, 1);
187
188 DetectAppLayerInspectEngineRegister2(
189 "ike.vendor", ALPROTO_IKE, SIG_FLAG_TOSERVER, 1, DetectEngineInspectIkeVendor, NULL);
190
191 g_ike_vendor_buffer_id = DetectBufferTypeGetByName("ike.vendor");
192}
