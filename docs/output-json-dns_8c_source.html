261JsonBuilder *JsonDNSLogQuery(void *txptr, uint64_t tx_id)
262{
263 JsonBuilder *queryjb = jb_new_array();
264 if (queryjb == NULL) {
265 return NULL;
266 }
267
268 for (uint16_t i = 0; i < UINT16_MAX; i++) {
269 JsonBuilder *js = jb_new_object();
270 if (!rs_dns_log_json_query((void *)txptr, i, LOG_ALL_RRTYPES, js)) {
271 jb_free(js);
272 break;
273 }
274 jb_close(js);
275 jb_append_object(queryjb, js);
276 jb_free(js);
277 }
278
279 jb_close(queryjb);
280 return queryjb;
281}
282
283JsonBuilder *JsonDNSLogAnswer(void *txptr, uint64_t tx_id)
284{
285 if (!rs_dns_do_log_answer(txptr, LOG_ALL_RRTYPES)) {
286 return NULL;
287 } else {
288 JsonBuilder *js = jb_new_object();
289 rs_dns_log_json_answer(txptr, LOG_ALL_RRTYPES, js);
290 jb_close(js);
291 return js;
292 }
293}

295static int JsonDnsLoggerToServer(ThreadVars *tv, void *thread_data,
296 const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
297{
298 SCEnter();
299
300 LogDnsLogThread *td = (LogDnsLogThread *)thread_data;
301 LogDnsFileCtx *dnslog_ctx = td->dnslog_ctx;
302
303 if (unlikely(dnslog_ctx->flags & LOG_QUERIES) == 0) {
304 return TM_ECODE_OK;
305 }
306
307 for (uint16_t i = 0; i < 0xffff; i++) {
308 JsonBuilder *jb = CreateEveHeader(p, LOG_DIR_FLOW, "dns", NULL, dnslog_ctx->eve_ctx);
309 if (unlikely(jb == NULL)) {
310 return TM_ECODE_OK;
311 }
312
313 jb_open_object(jb, "dns");
314 if (!rs_dns_log_json_query(txptr, i, td->dnslog_ctx->flags, jb)) {
315 jb_free(jb);
316 break;
317 }
318 jb_close(jb);
319
320 OutputJsonBuilderBuffer(jb, td->ctx);
321 jb_free(jb);
322 }
323
324 SCReturnInt(TM_ECODE_OK);
325}

327static int JsonDnsLoggerToClient(ThreadVars *tv, void *thread_data,
328 const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
329{
330 SCEnter();
331
332 LogDnsLogThread *td = (LogDnsLogThread *)thread_data;
333 LogDnsFileCtx *dnslog_ctx = td->dnslog_ctx;
334
335 if (unlikely(dnslog_ctx->flags & LOG_ANSWERS) == 0) {
336 return TM_ECODE_OK;
337 }
338
339 if (rs_dns_do_log_answer(txptr, td->dnslog_ctx->flags)) {
340 JsonBuilder *jb = CreateEveHeader(p, LOG_DIR_FLOW, "dns", NULL, dnslog_ctx->eve_ctx);
341 if (unlikely(jb == NULL)) {
342 return TM_ECODE_OK;
343 }
344
345 jb_open_object(jb, "dns");
346 rs_dns_log_json_answer(txptr, td->dnslog_ctx->flags, jb);
347 jb_close(jb);
348 OutputJsonBuilderBuffer(jb, td->ctx);
349 jb_free(jb);
350 }
351
352 SCReturnInt(TM_ECODE_OK);
353}

355static int JsonDnsLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *alstate,
356 void *txptr, uint64_t tx_id)
357{
358 if (rs_dns_tx_is_request(txptr)) {
359 return JsonDnsLoggerToServer(tv, thread_data, p, f, alstate, txptr, tx_id);
360 } else if (rs_dns_tx_is_response(txptr)) {
361 return JsonDnsLoggerToClient(tv, thread_data, p, f, alstate, txptr, tx_id);
362 }
363 return TM_ECODE_OK;
364}

366static TmEcode LogDnsLogThreadInit(ThreadVars *t, const void *initdata, void **data)
367{
368 LogDnsLogThread *aft = SCCalloc(1, sizeof(LogDnsLogThread));
369 if (unlikely(aft == NULL))
370 return TM_ECODE_FAILED;
371
372 if(initdata == NULL)
373 {
374 SCLogDebug("Error getting context for EveLogDNS. \"initdata\" argument NULL");
375 goto error_exit;
376 }
377
378 /* Use the Ouptut Context (file pointer and mutex) */
379 aft->dnslog_ctx = ((OutputCtx *)initdata)->data;
380 aft->ctx = CreateEveThreadCtx(t, aft->dnslog_ctx->eve_ctx);
381 if (!aft->ctx) {
382 goto error_exit;
383 }
384
385 *data = (void *)aft;
386 return TM_ECODE_OK;
387
388error_exit:
389 SCFree(aft);
390 return TM_ECODE_FAILED;
391}

393static TmEcode LogDnsLogThreadDeinit(ThreadVars *t, void *data)
394{

401 /* clear memory */
402 memset(aft, 0, sizeof(LogDnsLogThread));
403
404 SCFree(aft);
405 return TM_ECODE_OK;
406}

416static void JsonDnsLogParseConfig(LogDnsFileCtx *dnslog_ctx, ConfNode *conf,
417 const char *query_key, const char *answer_key,
418 const char *answer_types_key)
419{
420 const char *query = ConfNodeLookupChildValue(conf, query_key);
421 if (query != NULL) {
422 if (ConfValIsTrue(query)) {
423 dnslog_ctx->flags |= LOG_QUERIES;
424 } else {
425 dnslog_ctx->flags &= ~LOG_QUERIES;
426 }
427 } else {
428 dnslog_ctx->flags |= LOG_QUERIES;
429 }
430
431 const char *response = ConfNodeLookupChildValue(conf, answer_key);
432 if (response != NULL) {
433 if (ConfValIsTrue(response)) {
434 dnslog_ctx->flags |= LOG_ANSWERS;
435 } else {
436 dnslog_ctx->flags &= ~LOG_ANSWERS;
437 }
438 } else {
439 dnslog_ctx->flags |= LOG_ANSWERS;
440 }
441
442 ConfNode *custom;
443 if ((custom = ConfNodeLookupChild(conf, answer_types_key)) != NULL) {
444 dnslog_ctx->flags &= ~LOG_ALL_RRTYPES;
445 ConfNode *field;
446 TAILQ_FOREACH(field, &custom->head, next)
447 {
448 if (field != NULL)
449 {
450 DnsRRTypes f;
451 for (f = DNS_RRTYPE_A; f < DNS_RRTYPE_MAX; f++)
452 {
453 if (strcasecmp(dns_rrtype_fields[f].config_rrtype,
454 field->val) == 0)
455 {
456 dnslog_ctx->flags |= dns_rrtype_fields[f].flags;
457 break;
458 }
459 }
460 }
461 }
462 } else {
463 dnslog_ctx->flags |= LOG_ALL_RRTYPES;
464 }
465}

467static void JsonDnsCheckVersion(ConfNode *conf)
468{
469 if (conf == NULL) {
470 return;
471 }
472
473 static bool v1_deprecation_warned = false;
474 const ConfNode *has_version = ConfNodeLookupChild(conf, "version");
475 if (has_version != NULL) {
476 bool invalid = false;
477 intmax_t config_version;
478 if (ConfGetChildValueInt(conf, "version", &config_version)) {
479 switch(config_version) {
480 case 2:
481 break;
482 case 1:
483 if (!v1_deprecation_warned) {
484 SCLogError(SC_WARN_DEPRECATED,
485 "DNS EVE v1 logging has been removed, will use v2");
486 v1_deprecation_warned = true;
487 }
488 break;
489 default:
490 invalid = true;
491 break;
492 }
493 } else {
494 invalid = true;
495 }
496 if (invalid) {
497 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid EVE DNS version \"%s\", will use v2",
498 has_version->val);
499 }
500 }
501}

503static void JsonDnsLogInitFilters(LogDnsFileCtx *dnslog_ctx, ConfNode *conf)
504{
505 dnslog_ctx->flags = ~0ULL;
506
507 if (conf) {
508 JsonDnsLogParseConfig(dnslog_ctx, conf, "requests", "responses", "types");
509 if (dnslog_ctx->flags & LOG_ANSWERS) {
510 ConfNode *format;
511 if ((format = ConfNodeLookupChild(conf, "formats")) != NULL) {
512 dnslog_ctx->flags &= ~LOG_FORMAT_ALL;
513 ConfNode *field;
514 TAILQ_FOREACH (field, &format->head, next) {
515 if (strcasecmp(field->val, "detailed") == 0) {
516 dnslog_ctx->flags |= LOG_FORMAT_DETAILED;
517 } else if (strcasecmp(field->val, "grouped") == 0) {
518 dnslog_ctx->flags |= LOG_FORMAT_GROUPED;
519 }
520 }
521 } else {
522 dnslog_ctx->flags |= LOG_FORMAT_ALL;
523 }
524 }
525 }
526}

528static OutputInitResult JsonDnsLogInitCtxSub(ConfNode *conf, OutputCtx *parent_ctx)
529{
530 OutputInitResult result = { NULL, false };
531 const char *enabled = ConfNodeLookupChildValue(conf, "enabled");
532 if (enabled != NULL && !ConfValIsTrue(enabled)) {
533 result.ok = true;
534 return result;
535 }
536
537 /* As only a single version of logging is supported, this exists to warn about
538 * unsupported versions. */
539 JsonDnsCheckVersion(conf);
540
541 OutputJsonCtx *ojc = parent_ctx->data;
542
543 LogDnsFileCtx *dnslog_ctx = SCMalloc(sizeof(LogDnsFileCtx));
544 if (unlikely(dnslog_ctx == NULL)) {
545 return result;
546 }
547 memset(dnslog_ctx, 0x00, sizeof(LogDnsFileCtx));
548
549 dnslog_ctx->eve_ctx = ojc;
550
551 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
552 if (unlikely(output_ctx == NULL)) {
553 SCFree(dnslog_ctx);
554 return result;
555 }
556
557 output_ctx->data = dnslog_ctx;
558 output_ctx->DeInit = LogDnsLogDeInitCtxSub;
559
560 JsonDnsLogInitFilters(dnslog_ctx, conf);
561
562 SCLogDebug("DNS log sub-module initialized");
563
564 AppLayerParserRegisterLogger(IPPROTO_UDP, ALPROTO_DNS);
565 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNS);
566
567 result.ctx = output_ctx;
568 result.ok = true;
569 return result;
570}
