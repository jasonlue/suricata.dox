220/**
221 * Run or list unittests
222 *
223 * \param list_unittests If set to 1, list unittests. Run them if set to 0.
224 * \param regex_arg A regular expression to select unittests to run
225 *
226 * This function is terminal and will call exit after being called.
227 */
228
229void RunUnittests(int list_unittests, const char *regex_arg)
230{
231#ifdef UNITTESTS
232 /* Initializations for global vars, queues, etc (memsets, mutex init..) */
233 GlobalsInitPreConfig();
234
235#ifdef HAVE_LUAJIT
236 if (LuajitSetupStatesPool() != 0) {
237 exit(EXIT_FAILURE);
238 }
239#endif
240
241 default_packet_size = DEFAULT_PACKET_SIZE;
242 /* load the pattern matchers */
243 MpmTableSetup();
244 SpmTableSetup();
245
246 StorageInit();
247 AppLayerSetup();
248
249 /* hardcoded initialization code */
250 SigTableSetup(); /* load the rule keywords */
251 TmqhSetup();
252
253 TagInitCtx();
254 SCReferenceConfInit();
255 SCClassConfInit();
256
257 UtInitialize();
258
259 RegisterAllModules();
260
261 HostBitInitCtx();
262
263 StorageFinalize();
264 /* test and initialize the unittesting subsystem */
265 if (regex_arg == NULL){
266 regex_arg = ".*";
267 UtRunSelftest(regex_arg); /* inits and cleans up again */
268 }
269
270 AppLayerHtpEnableRequestBodyCallback();
271 AppLayerHtpNeedFileInspection();
272
273 RegisterUnittests();
274
275 if (list_unittests) {
276 UtListTests(regex_arg);
277 } else {
278 /* global packet pool */
279 extern intmax_t max_pending_packets;
280 max_pending_packets = 128;
281 PacketPoolInit();
282
283 uint32_t failed = UtRunTests(regex_arg);
284 PacketPoolDestroy();
285 UtCleanup();
286#ifdef BUILD_HYPERSCAN
287 MpmHSGlobalCleanup();
288#endif
289 if (failed) {
290 exit(EXIT_FAILURE);
291 }
292 }
293
294#ifdef HAVE_LUAJIT
295 LuajitFreeStatesPool();
296#endif
297
298 exit(EXIT_SUCCESS);
299#else
300 FatalError(SC_ERR_FATAL, "Unittests are not build-in");
301#endif /* UNITTESTS */
302}
