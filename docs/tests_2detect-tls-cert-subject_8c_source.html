28/**
29 * \test Test that a signature containing a tls.cert_subject is correctly parsed
30 * and that the keyword is registered.
31 */
32static int DetectTlsSubjectTest01(void)
33{
34 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
35 FAIL_IF_NULL(de_ctx);
36
37 de_ctx->flags |= DE_QUIET;
38 de_ctx->sig_list = SigInit(de_ctx, "alert tls any any -> any any "
39 "(msg:\"Testing tls.cert_subject\"; "
40 "tls.cert_subject; content:\"test\"; sid:1;)");
41 FAIL_IF_NULL(de_ctx->sig_list);
42
43 /* sm should not be in the MATCH list */
44 SigMatch *sm = de_ctx->sig_list->sm_lists[DETECT_SM_LIST_MATCH];
45 FAIL_IF_NOT_NULL(sm);
46
47 sm = de_ctx->sig_list->sm_lists[g_tls_cert_subject_buffer_id];
48 FAIL_IF_NULL(sm);
49
50 FAIL_IF(sm->type != DETECT_CONTENT);
51 FAIL_IF_NOT_NULL(sm->next);
52
53 SigGroupCleanup(de_ctx);
54 DetectEngineCtxFree(de_ctx);
55
56 PASS;
57}
