70static int CreateFileName(const Packet *p, SSLState *state, char *filename, size_t filename_size)
71{
72 char path[PATH_MAX];
73 int file_id = SC_ATOMIC_ADD(cert_id, 1);
74
75 /* Use format : packet time + incremental ID
76 * When running on same pcap it will overwrite
77 * On a live device, we will not be able to overwrite */
78 if (snprintf(path, sizeof(path), "%s/%ld.%ld-%d.pem",
79 tls_logfile_base_dir,
80 (long int)p->ts.tv_sec,
81 (long int)p->ts.tv_usec,
82 file_id) == sizeof(path))
83 return 0;
84
85 strlcpy(filename, path, filename_size);
86 return 1;
87}

89static void LogTlsLogPem(LogTlsStoreLogThread *aft, const Packet *p, SSLState *state, int ipproto)
90{
91#define PEMHEADER "-----BEGIN CERTIFICATE-----\n"
92#define PEMFOOTER "-----END CERTIFICATE-----\n"
93 //Logging pem certificate
94 char filename[PATH_MAX] = "";
95 FILE* fp = NULL;
96 FILE* fpmeta = NULL;
97 unsigned long pemlen;
98 unsigned char* pembase64ptr = NULL;
99 int ret;
100 uint8_t *ptmp;
101 SSLCertsChain *cert;
102
103 if (TAILQ_EMPTY(&state->server_connp.certs))
104 SCReturn;
105
106 CreateFileName(p, state, filename, sizeof(filename));
107 if (strlen(filename) == 0) {
108 SCLogWarning(SC_ERR_FOPEN, "Can't create PEM filename");
109 SCReturn;
110 }
111
112 fp = fopen(filename, "w");
113 if (fp == NULL) {
114 if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
115 SCLogWarning(SC_ERR_FOPEN,
116 "Can't create PEM file '%s' in '%s' directory",
117 filename, tls_logfile_base_dir);
118 logging_dir_not_writable++;
119 }
120 SCReturn;
121 }
122
123 TAILQ_FOREACH(cert, &state->server_connp.certs, next) {
124 pemlen = Base64EncodeBufferSize(cert->cert_len);
125 if (pemlen > aft->enc_buf_len) {
126 ptmp = (uint8_t*) SCRealloc(aft->enc_buf, sizeof(uint8_t) * pemlen);
127 if (ptmp == NULL) {
128 SCFree(aft->enc_buf);
129 aft->enc_buf = NULL;
130 aft->enc_buf_len = 0;
131 SCLogWarning(SC_ERR_MEM_ALLOC, "Can't allocate data for base64 encoding");
132 goto end_fp;
133 }
134 aft->enc_buf = ptmp;
135 aft->enc_buf_len = pemlen;
136 }
137
138 memset(aft->enc_buf, 0, aft->enc_buf_len);
139
140 ret = Base64Encode((unsigned char*) cert->cert_data, cert->cert_len, aft->enc_buf, &pemlen);
141 if (ret != SC_BASE64_OK) {
142 SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "Invalid return of Base64Encode function");
143 goto end_fwrite_fp;
144 }
145
146 if (fprintf(fp, PEMHEADER) < 0)
147 goto end_fwrite_fp;
148
149 pembase64ptr = aft->enc_buf;
150 while (pemlen > 0) {
151 size_t loffset = pemlen >= 64 ? 64 : pemlen;
152 if (fwrite(pembase64ptr, 1, loffset, fp) != loffset)
153 goto end_fwrite_fp;
154 if (fwrite("\n", 1, 1, fp) != 1)
155 goto end_fwrite_fp;
156 pembase64ptr += 64;
157 if (pemlen < 64)
158 break;
159 pemlen -= 64;
160 }
161
162 if (fprintf(fp, PEMFOOTER) < 0)
163 goto end_fwrite_fp;
164 }
165 fclose(fp);
166
167 //Logging certificate informations
168 memcpy(filename + (strlen(filename) - 3), "meta", 4);
169 fpmeta = fopen(filename, "w");
170 if (fpmeta != NULL) {
171 #define PRINT_BUF_LEN 46
172 char srcip[PRINT_BUF_LEN], dstip[PRINT_BUF_LEN];
173 char timebuf[64];
174 Port sp, dp;
175 CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
176 if (!TLSGetIPInformations(p, srcip, PRINT_BUF_LEN, &sp, dstip, PRINT_BUF_LEN, &dp, ipproto))
177 goto end_fwrite_fpmeta;
178 if (fprintf(fpmeta, "TIME: %s\n", timebuf) < 0)
179 goto end_fwrite_fpmeta;
180 if (p->pcap_cnt > 0) {
181 if (fprintf(fpmeta, "PCAP PKT NUM: %"PRIu64"\n", p->pcap_cnt) < 0)
182 goto end_fwrite_fpmeta;
183 }
184 if (fprintf(fpmeta, "SRC IP: %s\n", srcip) < 0)
185 goto end_fwrite_fpmeta;
186 if (fprintf(fpmeta, "DST IP: %s\n", dstip) < 0)
187 goto end_fwrite_fpmeta;
188 if (fprintf(fpmeta, "PROTO: %" PRIu32 "\n", p->proto) < 0)
189 goto end_fwrite_fpmeta;
190 if (PKT_IS_TCP(p) || PKT_IS_UDP(p)) {
191 if (fprintf(fpmeta, "SRC PORT: %" PRIu16 "\n", sp) < 0)
192 goto end_fwrite_fpmeta;
193 if (fprintf(fpmeta, "DST PORT: %" PRIu16 "\n", dp) < 0)
194 goto end_fwrite_fpmeta;
195 }
196
197 if (fprintf(fpmeta, "TLS SUBJECT: %s\n"
198 "TLS ISSUERDN: %s\n"
199 "TLS FINGERPRINT: %s\n",
200 state->server_connp.cert0_subject,
201 state->server_connp.cert0_issuerdn,
202 state->server_connp.cert0_fingerprint) < 0)
203 goto end_fwrite_fpmeta;
204
205 fclose(fpmeta);
206 } else {
207 if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
208 SCLogWarning(SC_ERR_FOPEN,
209 "Can't create meta file '%s' in '%s' directory",
210 filename, tls_logfile_base_dir);
211 logging_dir_not_writable++;
212 }
213 SCReturn;
214 }
215
216 /* Reset the store flag */
217 state->server_connp.cert_log_flag &= ~SSL_TLS_LOG_PEM;
218 SCReturn;
219
220end_fwrite_fp:
221 fclose(fp);
222 if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
223 SCLogWarning(SC_ERR_FWRITE, "Unable to write certificate");
224 logging_dir_not_writable++;
225 }
226end_fwrite_fpmeta:
227 if (fpmeta) {
228 fclose(fpmeta);
229 if (logging_dir_not_writable < LOGGING_WRITE_ISSUE_LIMIT) {
230 SCLogWarning(SC_ERR_FWRITE, "Unable to write certificate metafile");
231 logging_dir_not_writable++;
232 }
233 }
234 SCReturn;
235end_fp:
236 fclose(fp);
237 SCReturn;
238}

244static int LogTlsStoreCondition(ThreadVars *tv, const Packet *p, void *state,
245 void *tx, uint64_t tx_id)
246{
247 if (p->flow == NULL) {
248 return FALSE;
249 }
250
251 if (!(PKT_IS_TCP(p))) {
252 return FALSE;
253 }
254
255 SSLState *ssl_state = (SSLState *)state;
256 if (ssl_state == NULL) {
257 SCLogDebug("no tls state, so no request logging");
258 goto dontlog;
259 }
260
261 if ((ssl_state->server_connp.cert_log_flag & SSL_TLS_LOG_PEM) == 0)
262 goto dontlog;
263
264 if (ssl_state->server_connp.cert0_issuerdn == NULL ||
265 ssl_state->server_connp.cert0_subject == NULL)
266 goto dontlog;
267
268 return TRUE;
269dontlog:
270 return FALSE;
271}

273static int LogTlsStoreLogger(ThreadVars *tv, void *thread_data, const Packet *p,
274 Flow *f, void *state, void *tx, uint64_t tx_id)
275{
276 LogTlsStoreLogThread *aft = (LogTlsStoreLogThread *)thread_data;
277 int ipproto = (PKT_IS_IPV4(p)) ? AF_INET : AF_INET6;
278
279 SSLState *ssl_state = (SSLState *)state;
280 if (unlikely(ssl_state == NULL)) {
281 return 0;
282 }
283
284 if (ssl_state->server_connp.cert_log_flag & SSL_TLS_LOG_PEM) {
285 LogTlsLogPem(aft, p, ssl_state, ipproto);
286 }
287
288 return 0;
289}

291static TmEcode LogTlsStoreLogThreadInit(ThreadVars *t, const void *initdata, void **data)
292{
293 LogTlsStoreLogThread *aft = SCMalloc(sizeof(LogTlsStoreLogThread));
294 if (unlikely(aft == NULL))
295 return TM_ECODE_FAILED;
296 memset(aft, 0, sizeof(LogTlsStoreLogThread));
297
298 if (initdata == NULL) {
299 SCLogDebug("Error getting context for LogTLSStore. \"initdata\" argument NULL");
300 SCFree(aft);
301 return TM_ECODE_FAILED;
302 }
303
304 struct stat stat_buf;
305 /* coverity[toctou] */
306 if (stat(tls_logfile_base_dir, &stat_buf) != 0) {
307 int ret;
308 /* coverity[toctou] */
309 ret = SCMkDir(tls_logfile_base_dir, S_IRWXU|S_IXGRP|S_IRGRP);
310 if (ret != 0) {
311 int err = errno;
312 if (err != EEXIST) {
313 SCLogError(SC_ERR_LOGDIR_CONFIG,
314 "Cannot create certs drop directory %s: %s",
315 tls_logfile_base_dir, strerror(err));
316 exit(EXIT_FAILURE);
317 }
318 } else {
319 SCLogInfo("Created certs drop directory %s",
320 tls_logfile_base_dir);
321 }
322
323 }
324
325 *data = (void *)aft;
326 return TM_ECODE_OK;
327}

368/** \brief Create a new http log LogFilestoreCtx.
369 * \param conf Pointer to ConfNode containing this loggers configuration.
370 * \return NULL if failure, LogFilestoreCtx* to the file_ctx if succesful
371 * */
372static OutputInitResult LogTlsStoreLogInitCtx(ConfNode *conf)
373{
374 OutputInitResult result = { NULL, false };
375 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
376 if (unlikely(output_ctx == NULL))
377 return result;
378
379 output_ctx->data = NULL;
380 output_ctx->DeInit = LogTlsStoreLogDeInitCtx;
381
382 /* FIXME we need to implement backward compability here */
383 const char *s_default_log_dir = NULL;
384 s_default_log_dir = ConfigGetLogDirectory();
385
386 const char *s_base_dir = NULL;
387 s_base_dir = ConfNodeLookupChildValue(conf, "certs-log-dir");
388 if (s_base_dir == NULL || strlen(s_base_dir) == 0) {
389 strlcpy(tls_logfile_base_dir,
390 s_default_log_dir, sizeof(tls_logfile_base_dir));
391 } else {
392 if (PathIsAbsolute(s_base_dir)) {
393 strlcpy(tls_logfile_base_dir,
394 s_base_dir, sizeof(tls_logfile_base_dir));
395 } else {
396 snprintf(tls_logfile_base_dir, sizeof(tls_logfile_base_dir),
397 "%s/%s", s_default_log_dir, s_base_dir);
398 }
399 }
400
401 SCLogInfo("storing certs in %s", tls_logfile_base_dir);
402
403 /* enable the logger for the app layer */
404 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_TLS);
405
406 result.ctx = output_ctx;
407 result.ok = true;
408 SCReturnCT(result, "OutputInitResult");
409}
