40static int DetectMQTTConnectUsernameSetup(DetectEngineCtx *de_ctx, Signature *s, const char *arg)
41{
42 if (DetectBufferSetActiveList(s, g_buffer_id) < 0)
43 return -1;
44
45 if (DetectSignatureSetAppProto(s, ALPROTO_MQTT) < 0)
46 return -1;
47
48 return 0;
49}

51static InspectionBuffer *GetData(DetectEngineThreadCtx *det_ctx,
52 const DetectEngineTransforms *transforms,
53 Flow *_f, const uint8_t _flow_flags,
54 void *txv, const int list_id)
55{
56 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
57 if (buffer->inspect == NULL) {
58 const uint8_t *b = NULL;
59 uint32_t b_len = 0;
60
61 if (rs_mqtt_tx_get_connect_username(txv, &b, &b_len) != 1)
62 return NULL;
63 if (b == NULL || b_len == 0)
64 return NULL;
65
66 InspectionBufferSetup(det_ctx, list_id, buffer, b, b_len);
67 InspectionBufferApplyTransforms(buffer, transforms);
68 }
69 return buffer;
70}

72void DetectMQTTConnectUsernameRegister(void)
73{
74 /* mqtt.connect.username sticky buffer */
75 sigmatch_table[DETECT_AL_MQTT_CONNECT_USERNAME].name = KEYWORD_NAME;
76 sigmatch_table[DETECT_AL_MQTT_CONNECT_USERNAME].desc = "sticky buffer to match on the MQTT CONNECT username";
77 sigmatch_table[DETECT_AL_MQTT_CONNECT_USERNAME].url = "/rules/" KEYWORD_DOC;
78 sigmatch_table[DETECT_AL_MQTT_CONNECT_USERNAME].Setup = DetectMQTTConnectUsernameSetup;
79 sigmatch_table[DETECT_AL_MQTT_CONNECT_USERNAME].flags |= SIGMATCH_NOOPT;
80
81 DetectAppLayerInspectEngineRegister2(BUFFER_NAME, ALPROTO_MQTT,
82 SIG_FLAG_TOSERVER, 0,
83 DetectEngineInspectBufferGeneric, GetData);
84
85 DetectAppLayerMpmRegister2(BUFFER_NAME, SIG_FLAG_TOSERVER, 2,
86 PrefilterGenericMpmRegister, GetData, ALPROTO_MQTT,
87 1);
88
89 DetectBufferTypeSetDescriptionByName(BUFFER_NAME, BUFFER_DESC);
90
91 g_buffer_id = DetectBufferTypeGetByName(BUFFER_NAME);
92
93 SCLogDebug("registering " BUFFER_NAME " rule option");
94}
