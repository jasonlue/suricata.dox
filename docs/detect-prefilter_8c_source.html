37void DetectPrefilterRegister(void)
38{
39 sigmatch_table[DETECT_PREFILTER].name = "prefilter";
40 sigmatch_table[DETECT_PREFILTER].desc = "force a condition to be used as prefilter";
41 sigmatch_table[DETECT_PREFILTER].url = "/rules/prefilter-keywords.html#prefilter";
42 sigmatch_table[DETECT_PREFILTER].Setup = DetectPrefilterSetup;
43 sigmatch_table[DETECT_PREFILTER].flags |= SIGMATCH_NOOPT;
44}

46/**
47 * \internal
48 * \brief Apply the prefilter keyword to the last match
49 * \param det_ctx detection engine ctx
50 * \param s signature
51 * \param nullstr should be null
52 * \retval 0 ok
53 * \retval -1 failure
54 */
55static int DetectPrefilterSetup (DetectEngineCtx *de_ctx, Signature *s, const char *nullstr)
56{
57 SCEnter();
58
59 if (nullstr != NULL) {
60 SCLogError(SC_ERR_INVALID_VALUE, "prefilter has value");
61 SCReturnInt(-1);
62 }
63
64 if (s->flags & SIG_FLAG_PREFILTER) {
65 SCLogError(SC_ERR_INVALID_SIGNATURE, "prefilter already set");
66 SCReturnInt(-1);
67 }
68
69 SigMatch *sm = DetectGetLastSM(s);
70 if (sm == NULL) {
71 SCLogError(SC_ERR_INVALID_SIGNATURE, "prefilter needs preceding match");
72 SCReturnInt(-1);
73 }
74
75 /* if the sig match is content, prefilter should act like
76 * 'fast_pattern' w/o options. */
77 if (sm->type == DETECT_CONTENT) {
78 DetectContentData *cd = (DetectContentData *)sm->ctx;
79 if ((cd->flags & DETECT_CONTENT_NEGATED) &&
80 ((cd->flags & DETECT_CONTENT_DISTANCE) ||
81 (cd->flags & DETECT_CONTENT_WITHIN) ||
82 (cd->flags & DETECT_CONTENT_OFFSET) ||
83 (cd->flags & DETECT_CONTENT_DEPTH)))
84 {
85 SCLogError(SC_ERR_INVALID_SIGNATURE, "prefilter; cannot be "
86 "used with negated content, along with relative modifiers");
87 SCReturnInt(-1);
88 }
89 cd->flags |= DETECT_CONTENT_FAST_PATTERN;
90 } else {
91 if (sigmatch_table[sm->type].SupportsPrefilter == NULL) {
92 SCLogError(SC_ERR_INVALID_SIGNATURE, "prefilter is not supported for %s",
93 sigmatch_table[sm->type].name);
94 SCReturnInt(-1);
95 }
96 s->flags |= SIG_FLAG_PREFILTER;
97
98 /* make sure setup function runs for this type. */
99 de_ctx->sm_types_prefilter[sm->type] = true;
100 }
101
102 s->init_data->prefilter_sm = sm;
103
104 SCReturnInt(0);
105}
