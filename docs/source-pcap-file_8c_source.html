89void CleanupPcapFileThreadVars(PcapFileThreadVars *tv)
90{

92 if (tv->is_directory == 0) {
93 if (tv->behavior.file != NULL) {
94 CleanupPcapFileFromThreadVars(tv, tv->behavior.file);
95 }
96 tv->behavior.file = NULL;
97 } else {
98 if (tv->behavior.directory != NULL) {
99 CleanupPcapDirectoryFromThreadVars(tv, tv->behavior.directory);
100 }
101 tv->behavior.directory = NULL;
102 }
103 if (tv->shared.bpf_string != NULL) {
104 SCFree(tv->shared.bpf_string);
105 tv->shared.bpf_string = NULL;
106 }
107 SCFree(tv);
108 }

155TmEcode ReceivePcapFileLoop(ThreadVars *tv, void *data, void *slot)
156{
157 SCEnter();
158
159 if(unlikely(data == NULL)) {
160 SCLogError(SC_ERR_INVALID_ARGUMENT, "pcap file reader thread failed to initialize");
161
162 PcapFileExit(TM_ECODE_FAILED, NULL);
163
164 SCReturnInt(TM_ECODE_DONE);
165 }
166
167 TmEcode status = TM_ECODE_OK;
168 PcapFileThreadVars *ptv = (PcapFileThreadVars *) data;
169 TmSlot *s = (TmSlot *)slot;
170
171 ptv->shared.slot = s->slot_next;
172 ptv->shared.cb_result = TM_ECODE_OK;
173
174 if(ptv->is_directory == 0) {
175 SCLogInfo("Starting file run for %s", ptv->behavior.file->filename);
176 status = PcapFileDispatch(ptv->behavior.file);
177 CleanupPcapFileFromThreadVars(ptv, ptv->behavior.file);
178 } else {
179 SCLogInfo("Starting directory run for %s", ptv->behavior.directory->filename);
180 PcapDirectoryDispatch(ptv->behavior.directory);
181 CleanupPcapDirectoryFromThreadVars(ptv, ptv->behavior.directory);
182 }
183
184 SCLogDebug("Pcap file loop complete with status %u", status);
185
186 status = PcapFileExit(status, &ptv->shared.last_processed);
187 SCReturnInt(status);
188}

190TmEcode ReceivePcapFileThreadInit(ThreadVars *tv, const void *initdata, void **data)
191{
192 SCEnter();
193
194 TmEcode status = TM_ECODE_OK;
195 const char *tmpstring = NULL;
196 const char *tmp_bpf_string = NULL;
197
198 if (initdata == NULL) {
199 SCLogError(SC_ERR_INVALID_ARGUMENT, "error: initdata == NULL");
200
201 SCReturnInt(TM_ECODE_OK);
202 }
203
204 PcapFileThreadVars *ptv = SCMalloc(sizeof(PcapFileThreadVars));
205 if (unlikely(ptv == NULL)) {
206 SCReturnInt(TM_ECODE_OK);
207 }
208 memset(ptv, 0, sizeof(PcapFileThreadVars));
209 memset(&ptv->shared.last_processed, 0, sizeof(struct timespec));
210
211 intmax_t tenant = 0;
212 if (ConfGetInt("pcap-file.tenant-id", &tenant) == 1) {
213 if (tenant > 0 && tenant < UINT_MAX) {
214 ptv->shared.tenant_id = (uint32_t)tenant;
215 SCLogInfo("tenant %u", ptv->shared.tenant_id);
216 } else {
217 SCLogError(SC_ERR_INVALID_ARGUMENT, "tenant out of range");
218 }
219 }
220
221 if (ConfGet("bpf-filter", &(tmp_bpf_string)) != 1) {
222 SCLogDebug("could not get bpf or none specified");
223 } else {
224 ptv->shared.bpf_string = SCStrdup(tmp_bpf_string);
225 if (unlikely(ptv->shared.bpf_string == NULL)) {
226 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate bpf_string");
227
228 CleanupPcapFileThreadVars(ptv);
229
230 SCReturnInt(TM_ECODE_OK);
231 }
232 }
233
234 int should_delete = 0;
235 ptv->shared.should_delete = false;
236 if (ConfGetBool("pcap-file.delete-when-done", &should_delete) == 1) {
237 ptv->shared.should_delete = should_delete == 1;
238 }
239
240 DIR *directory = NULL;
241 SCLogDebug("checking file or directory %s", (char*)initdata);
242 if(PcapDetermineDirectoryOrFile((char *)initdata, &directory) == TM_ECODE_FAILED) {
243 CleanupPcapFileThreadVars(ptv);
244 SCReturnInt(TM_ECODE_OK);
245 }
246
247 if(directory == NULL) {
248 SCLogDebug("argument %s was a file", (char *)initdata);
249 PcapFileFileVars *pv = SCMalloc(sizeof(PcapFileFileVars));
250 if (unlikely(pv == NULL)) {
251 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate file vars");
252 CleanupPcapFileThreadVars(ptv);
253 SCReturnInt(TM_ECODE_OK);
254 }
255 memset(pv, 0, sizeof(PcapFileFileVars));
256
257 pv->filename = SCStrdup((char *)initdata);
258 if (unlikely(pv->filename == NULL)) {
259 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate filename");
260 CleanupPcapFileFileVars(pv);
261 CleanupPcapFileThreadVars(ptv);
262 SCReturnInt(TM_ECODE_OK);
263 }
264
265 pv->shared = &ptv->shared;
266 status = InitPcapFile(pv);
267 if(status == TM_ECODE_OK) {
268 ptv->is_directory = 0;
269 ptv->behavior.file = pv;
270 } else {
271 SCLogWarning(SC_ERR_PCAP_DISPATCH,
272 "Failed to init pcap file %s, skipping", pv->filename);
273 CleanupPcapFileFileVars(pv);
274 CleanupPcapFileThreadVars(ptv);
275 SCReturnInt(TM_ECODE_OK);
276 }
277 } else {
278 SCLogInfo("Argument %s was a directory", (char *)initdata);
279 PcapFileDirectoryVars *pv = SCMalloc(sizeof(PcapFileDirectoryVars));
280 if (unlikely(pv == NULL)) {
281 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate directory vars");
282 closedir(directory);
283 CleanupPcapFileThreadVars(ptv);
284 SCReturnInt(TM_ECODE_OK);
285 }
286 memset(pv, 0, sizeof(PcapFileDirectoryVars));
287
288 pv->filename = SCStrdup((char*)initdata);
289 if (unlikely(pv->filename == NULL)) {
290 SCLogError(SC_ERR_MEM_ALLOC, "Failed to allocate filename");
291 CleanupPcapFileDirectoryVars(pv);
292 CleanupPcapFileThreadVars(ptv);
293 SCReturnInt(TM_ECODE_OK);
294 }
295 pv->cur_dir_depth = 0;
296
297 int should_recurse;
298 pv->should_recurse = false;
299 if (ConfGetBool("pcap-file.recursive", &should_recurse) == 1) {
300 pv->should_recurse = (should_recurse == 1);
301 }
302
303 int should_loop = 0;
304 pv->should_loop = false;
305 if (ConfGetBool("pcap-file.continuous", &should_loop) == 1) {
306 pv->should_loop = (should_loop == 1);
307 }
308
309 if (pv->should_recurse == true && pv->should_loop == true) {
310 SCLogError(SC_ERR_INVALID_ARGUMENT, "Error, --pcap-file-continuous and --pcap-file-recursive "
311 "cannot be used together.");
312 CleanupPcapFileDirectoryVars(pv);
313 CleanupPcapFileThreadVars(ptv);
314 SCReturnInt(TM_ECODE_FAILED);
315 }
316
317 pv->delay = 30;
318 intmax_t delay = 0;
319 if (ConfGetInt("pcap-file.delay", &delay) == 1) {
320 if (delay > 0 && delay < UINT_MAX) {
321 pv->delay = (time_t)delay;
322 SCLogDebug("delay %lu", pv->delay);
323 } else {
324 SCLogError(SC_ERR_INVALID_ARGUMENT, "delay out of range");
325 }
326 }
327
328 pv->poll_interval = 5;
329 intmax_t poll_interval = 0;
330 if (ConfGetInt("pcap-file.poll-interval", &poll_interval) == 1) {
331 if (poll_interval > 0 && poll_interval < UINT_MAX) {
332 pv->poll_interval = (time_t)poll_interval;
333 SCLogDebug("poll-interval %lu", pv->delay);
334 } else {
335 SCLogError(SC_ERR_INVALID_ARGUMENT, "poll-interval out of range");
336 }
337 }
338
339 pv->shared = &ptv->shared;
340 pv->directory = directory;
341 TAILQ_INIT(&pv->directory_content);
342
343 ptv->is_directory = 1;
344 ptv->behavior.directory = pv;
345 }
346
347 if (ConfGet("pcap-file.checksum-checks", &tmpstring) != 1) {
348 pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_AUTO;
349 } else {
350 if (strcmp(tmpstring, "auto") == 0) {
351 pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_AUTO;
352 } else if (ConfValIsTrue(tmpstring)){
353 pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_ENABLE;
354 } else if (ConfValIsFalse(tmpstring)) {
355 pcap_g.conf_checksum_mode = CHECKSUM_VALIDATION_DISABLE;
356 }
357 }
358 pcap_g.checksum_mode = pcap_g.conf_checksum_mode;
359
360 ptv->shared.tv = tv;
361 *data = (void *)ptv;
362
363 SCReturnInt(TM_ECODE_OK);
364}

366void ReceivePcapFileThreadExitStats(ThreadVars *tv, void *data)
367{
368 SCEnter();
369 if(data != NULL) {
370 PcapFileThreadVars *ptv = (PcapFileThreadVars *)data;
371
372 if (pcap_g.conf_checksum_mode == CHECKSUM_VALIDATION_AUTO &&
373 pcap_g.cnt < CHECKSUM_SAMPLE_COUNT &&
374 SC_ATOMIC_GET(pcap_g.invalid_checksums)) {
375 uint64_t chrate = pcap_g.cnt / SC_ATOMIC_GET(pcap_g.invalid_checksums);
376 if (chrate < CHECKSUM_INVALID_RATIO)
377 SCLogWarning(SC_ERR_INVALID_CHECKSUM,
378 "1/%" PRIu64 "th of packets have an invalid checksum,"
379 " consider setting pcap-file.checksum-checks variable to no"
380 " or use '-k none' option on command line.",
381 chrate);
382 else
383 SCLogInfo("1/%" PRIu64 "th of packets have an invalid checksum",
384 chrate);
385 }
386 SCLogNotice(
387 "Pcap-file module read %" PRIu64 " files, %" PRIu64 " packets, %" PRIu64 " bytes",
388 ptv->shared.files,
389 ptv->shared.pkts,
390 ptv->shared.bytes
391 );
392 }
393}

405static TmEcode DecodePcapFile(ThreadVars *tv, Packet *p, void *data)
406{
407 SCEnter();
408 DecodeThreadVars *dtv = (DecodeThreadVars *)data;
409
410 BUG_ON(PKT_IS_PSEUDOPKT(p));
411
412 /* update counters */
413 DecodeUpdatePacketCounters(tv, dtv, p);
414
415 DecoderFunc decoder;
416 if(ValidateLinkType(p->datalink, &decoder) == TM_ECODE_OK) {
417
418 /* call the decoder */
419 decoder(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p));
420
421#ifdef DEBUG
422 BUG_ON(p->pkt_src != PKT_SRC_WIRE && p->pkt_src != PKT_SRC_FFR);
423#endif
424
425 PacketDecodeFinalize(tv, dtv, p);
426
427 SCReturnInt(TM_ECODE_OK);
428 } else {
429 SCReturnInt(TM_ECODE_FAILED);
430 }
431}
