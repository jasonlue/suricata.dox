48void RunModeErfDagRegister(void)
49{
50 RunModeRegisterNewRunMode(RUNMODE_DAG, "autofp",
51 "Multi threaded DAG mode. Packets from "
52 "each flow are assigned to a single detect "
53 "thread, unlike \"dag_auto\" where packets "
54 "from the same flow can be processed by any "
55 "detect thread",
56 RunModeIdsErfDagAutoFp);
57
58 RunModeRegisterNewRunMode(RUNMODE_DAG, "single",
59 "Singled threaded DAG mode",
60 RunModeIdsErfDagSingle);
61
62 RunModeRegisterNewRunMode(RUNMODE_DAG, "workers",
63 "Workers DAG mode, each thread does all "
64 " tasks from acquisition to logging",
65 RunModeIdsErfDagWorkers);
66
67 return;
68}

70int RunModeIdsErfDagSingle(void)

73

78 TimeModeSetLive();

80 ret = RunModeSetLiveCaptureSingle(ParseDagConfig,
