56int DecodeNull(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p,
57 const uint8_t *pkt, uint32_t len)
58{
59 DEBUG_VALIDATE_BUG_ON(pkt == NULL);
60
61 StatsIncr(tv, dtv->counter_null);
62
63 if (unlikely(len < HDR_SIZE)) {
64 ENGINE_SET_INVALID_EVENT(p, LTNULL_PKT_TOO_SMALL);
65 return TM_ECODE_FAILED;
66 }
67
68 if (unlikely(GET_PKT_LEN(p) > HDR_SIZE + USHRT_MAX)) {
69 return TM_ECODE_FAILED;
70 }
71#if __BYTE_ORDER__ == __BIG_ENDIAN
72 uint32_t type = pkt[0] | pkt[1] << 8 | pkt[2] << 16 | pkt[3] << 24;
73#else
74 uint32_t type = *((uint32_t *)pkt);
75#endif
76 switch(type) {
77 case AF_INET:
78 SCLogDebug("IPV4 Packet");
79 if (GET_PKT_LEN(p) - HDR_SIZE > USHRT_MAX) {
80 return TM_ECODE_FAILED;
81 }
82 DecodeIPV4(
83 tv, dtv, p, GET_PKT_DATA(p) + HDR_SIZE, (uint16_t)(GET_PKT_LEN(p) - HDR_SIZE));
84 break;
85 case AF_INET6_BSD:
86 case AF_INET6_FREEBSD:
87 case AF_INET6_DARWIN:
88 case AF_INET6_LINUX:
89 case AF_INET6_SOLARIS:
90 case AF_INET6_WINSOCK:
91 SCLogDebug("IPV6 Packet");
92 if (GET_PKT_LEN(p) - HDR_SIZE > USHRT_MAX) {
93 return TM_ECODE_FAILED;
94 }
95 DecodeIPV6(
96 tv, dtv, p, GET_PKT_DATA(p) + HDR_SIZE, (uint16_t)(GET_PKT_LEN(p) - HDR_SIZE));
97 break;
98 default:
99 SCLogDebug("Unknown Null packet type version %" PRIu32 "", type);
100 ENGINE_SET_EVENT(p, LTNULL_UNSUPPORTED_TYPE);
101 break;
102 }
103 return TM_ECODE_OK;
104}
