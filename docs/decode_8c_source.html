18/**
19 * \defgroup decode Packet decoding
20 *
21 * \brief Code in charge of protocol decoding
22 *
23 * The task of decoding packets is made in different files and
24 * as Suricata is supporting encapsulation there is a potential
25 * recursivity in the call.
26 *
27 * For each protocol a DecodePROTO function is provided. For
28 * example we have DecodeIPV4() for IPv4 and DecodePPP() for
29 * PPP.
30 *
31 * These functions have all a pkt and a len argument which
32 * are respectively a pointer to the protocol data and the length
33 * of this protocol data.
34 *
35 * \attention The pkt parameter must point to the effective data because
36 * it will be used later to set per protocol pointer like Packet::tcph
37 *
38 * @{
39 */

101static int DecodeTunnel(ThreadVars *, DecodeThreadVars *, Packet *, const uint8_t *, uint32_t,
102 enum DecodeTunnelProto) WARN_UNUSED;
103
104static int DecodeTunnel(ThreadVars *tv, DecodeThreadVars *dtv, Packet *p, const uint8_t *pkt,
105 uint32_t len, enum DecodeTunnelProto proto)
106{
107 switch (proto) {
108 case DECODE_TUNNEL_PPP:
109 return DecodePPP(tv, dtv, p, pkt, len);
110 case DECODE_TUNNEL_IPV4:
111 DEBUG_VALIDATE_BUG_ON(len > UINT16_MAX);
112 return DecodeIPV4(tv, dtv, p, pkt, (uint16_t)len);
113 case DECODE_TUNNEL_IPV6:
114 case DECODE_TUNNEL_IPV6_TEREDO:
115 DEBUG_VALIDATE_BUG_ON(len > UINT16_MAX);
116 return DecodeIPV6(tv, dtv, p, pkt, (uint16_t)len);
117 case DECODE_TUNNEL_VLAN:
118 return DecodeVLAN(tv, dtv, p, pkt, len);
119 case DECODE_TUNNEL_ETHERNET:
120 return DecodeEthernet(tv, dtv, p, pkt, len);
121 case DECODE_TUNNEL_ERSPANII:
122 return DecodeERSPAN(tv, dtv, p, pkt, len);
123 case DECODE_TUNNEL_ERSPANI:
124 return DecodeERSPANTypeI(tv, dtv, p, pkt, len);
125 case DECODE_TUNNEL_NSH:
126 return DecodeNSH(tv, dtv, p, pkt, len);
127 default:
128 SCLogDebug("FIXME: DecodeTunnel: protocol %" PRIu32 " not supported.", proto);
129 break;
130 }
131 return TM_ECODE_OK;
132}

171/**
172 * \brief Get a malloced packet.
173 *
174 * \retval p packet, NULL on error
175 */
176Packet *PacketGetFromAlloc(void)
177{
178 Packet *p = SCMalloc(SIZE_OF_PACKET);
179 if (unlikely(p == NULL)) {
180 return NULL;
181 }
182
183 memset(p, 0, SIZE_OF_PACKET);
184 PACKET_INITIALIZE(p);
185 p->ReleasePacket = PacketFree;
186 p->flags |= PKT_ALLOC;
187
188 SCLogDebug("allocated a new packet only using alloc...");
189
190 PACKET_PROFILING_START(p);
191 return p;
192}

207/**
208 * \brief Get a packet. We try to get a packet from the packetpool first, but
209 * if that is empty we alloc a packet that is free'd again after
210 * processing.
211 *
212 * \retval p packet, NULL on error
213 */
214Packet *PacketGetFromQueueOrAlloc(void)
215{
216 /* try the pool first */
217 Packet *p = PacketPoolGetPacket();
218
219 if (p == NULL) {
220 /* non fatal, we're just not processing a packet then */
221 p = PacketGetFromAlloc();
222 } else {
223 DEBUG_VALIDATE_BUG_ON(p->ReleasePacket != PacketPoolReturnPacket);
224 PACKET_PROFILING_START(p);
225 }
226
227 return p;
228}

242/**
243 * \brief Copy data to Packet payload at given offset
244 *
245 * This function copies data/payload to a Packet. It uses the
246 * space allocated at Packet creation (pointed by Packet::pkt)
247 * or allocate some memory (pointed by Packet::ext_pkt) if the
248 * data size is to big to fit in initial space (of size
249 * default_packet_size).
250 *
251 * \param Pointer to the Packet to modify
252 * \param Offset of the copy relatively to payload of Packet
253 * \param Pointer to the data to copy
254 * \param Length of the data to copy
255 */
256inline int PacketCopyDataOffset(Packet *p, uint32_t offset, const uint8_t *data, uint32_t datalen)
257{
258 if (unlikely(offset + datalen > MAX_PAYLOAD_SIZE)) {
259 /* too big */
260 SET_PKT_LEN(p, 0);
261 return -1;
262 }
263
264 /* Do we have already an packet with allocated data */
265 if (! p->ext_pkt) {
266 uint32_t newsize = offset + datalen;
267 // check overflow
268 if (newsize < offset)
269 return -1;
270 if (newsize <= default_packet_size) {
271 /* data will fit in memory allocated with packet */
272 memcpy(GET_PKT_DIRECT_DATA(p) + offset, data, datalen);
273 } else {
274 /* here we need a dynamic allocation */
275 p->ext_pkt = SCMalloc(MAX_PAYLOAD_SIZE);
276 if (unlikely(p->ext_pkt == NULL)) {
277 SET_PKT_LEN(p, 0);
278 return -1;
279 }
280 /* copy initial data */
281 memcpy(p->ext_pkt, GET_PKT_DIRECT_DATA(p), GET_PKT_DIRECT_MAX_SIZE(p));
282 /* copy data as asked */
283 memcpy(p->ext_pkt + offset, data, datalen);
284 }
285 } else {
286 memcpy(p->ext_pkt + offset, data, datalen);
287 }
288 return 0;
289}

292 * \brief Copy data to Packet payload and set packet length

294 * \param Pointer to the Packet to modify

298inline int PacketCopyData(Packet *p, const uint8_t *pktdata, uint32_t pktlen)

300 SET_PKT_LEN(p, (size_t)pktlen);

304/**
305 * \brief Setup a pseudo packet (tunnel)
306 *
307 * \param parent parent packet for this pseudo pkt
308 * \param pkt raw packet data
309 * \param len packet data length
310 * \param proto protocol of the tunneled packet
311 *
312 * \retval p the pseudo packet or NULL if out of memory
313 */
314Packet *PacketTunnelPktSetup(ThreadVars *tv, DecodeThreadVars *dtv, Packet *parent,
315 const uint8_t *pkt, uint32_t len, enum DecodeTunnelProto proto)
316{
317 int ret;
318
319 SCEnter();
320
321 /* get us a packet */
322 Packet *p = PacketGetFromQueueOrAlloc();
323 if (unlikely(p == NULL)) {
324 SCReturnPtr(NULL, "Packet");
325 }
326
327 /* copy packet and set length, proto */
328 PacketCopyData(p, pkt, len);
329 p->recursion_level = parent->recursion_level + 1;
330 p->ts.tv_sec = parent->ts.tv_sec;
331 p->ts.tv_usec = parent->ts.tv_usec;
332 p->datalink = DLT_RAW;
333 p->tenant_id = parent->tenant_id;
334 p->livedev = parent->livedev;
335
336 /* set the root ptr to the lowest layer */
337 if (parent->root != NULL)
338 p->root = parent->root;
339 else
340 p->root = parent;
341
342 /* tell new packet it's part of a tunnel */
343 SET_TUNNEL_PKT(p);
344
345 ret = DecodeTunnel(tv, dtv, p, GET_PKT_DATA(p),
346 GET_PKT_LEN(p), proto);
347
348 if (unlikely(ret != TM_ECODE_OK) ||
349 (proto == DECODE_TUNNEL_IPV6_TEREDO && (p->flags & PKT_IS_INVALID)))
350 {
351 /* Not a (valid) tunnel packet */
352 SCLogDebug("tunnel packet is invalid");
353
354 p->root = NULL;
355 UNSET_TUNNEL_PKT(p);
356 TmqhOutputPacketpool(tv, p);
357 SCReturnPtr(NULL, "Packet");
358 }
359
360
361 /* tell parent packet it's part of a tunnel */
362 SET_TUNNEL_PKT(parent);
363
364 /* increment tunnel packet refcnt in the root packet */
365 TUNNEL_INCR_PKT_TPR(p);
366
367 /* disable payload (not packet) inspection on the parent, as the payload
368 * is the packet we will now run through the system separately. We do
369 * check it against the ip/port/other header checks though */
370 DecodeSetNoPayloadInspectionFlag(parent);
371 SCReturnPtr(p, "Packet");
372}

374/**
375 * \brief Setup a pseudo packet (reassembled frags)
376 *
377 * Difference with PacketPseudoPktSetup is that this func doesn't increment
378 * the recursion level. It needs to be on the same level as the frags because
379 * we run the flow engine against this and we need to get the same flow.
380 *
381 * \param parent parent packet for this pseudo pkt
382 * \param pkt raw packet data
383 * \param len packet data length
384 * \param proto protocol of the tunneled packet
385 *
386 * \retval p the pseudo packet or NULL if out of memory
387 */
388Packet *PacketDefragPktSetup(Packet *parent, const uint8_t *pkt, uint32_t len, uint8_t proto)
389{
390 SCEnter();
391
392 /* get us a packet */
393 Packet *p = PacketGetFromQueueOrAlloc();
394 if (unlikely(p == NULL)) {
395 SCReturnPtr(NULL, "Packet");
396 }
397
398 /* set the root ptr to the lowest layer */
399 if (parent->root != NULL)
400 p->root = parent->root;
401 else
402 p->root = parent;
403
404 /* copy packet and set length, proto */
405 if (pkt && len) {
406 PacketCopyData(p, pkt, len);
407 }
408 p->recursion_level = parent->recursion_level; /* NOT incremented */
409 p->ts.tv_sec = parent->ts.tv_sec;
410 p->ts.tv_usec = parent->ts.tv_usec;
411 p->datalink = DLT_RAW;
412 p->tenant_id = parent->tenant_id;
413 /* tell new packet it's part of a tunnel */
414 SET_TUNNEL_PKT(p);
415 p->vlan_id[0] = parent->vlan_id[0];
416 p->vlan_id[1] = parent->vlan_id[1];
417 p->vlan_idx = parent->vlan_idx;
418 p->livedev = parent->livedev;
419
420 SCReturnPtr(p, "Packet");
421}

441/**
442 * \note if p->flow is set, the flow is locked
443 */
444void PacketBypassCallback(Packet *p)
445{
446 if (PKT_IS_PSEUDOPKT(p))
447 return;
448
449#ifdef CAPTURE_OFFLOAD
450 /* Don't try to bypass if flow is already out or
451 * if we have failed to do it once */
452 if (p->flow) {
453 int state = p->flow->flow_state;
454 if ((state == FLOW_STATE_LOCAL_BYPASSED) ||
455 (state == FLOW_STATE_CAPTURE_BYPASSED)) {
456 return;
457 }
458
459 FlowBypassInfo *fc;
460
461 fc = FlowGetStorageById(p->flow, GetFlowBypassInfoID());
462 if (fc == NULL) {
463 fc = SCCalloc(sizeof(FlowBypassInfo), 1);
464 if (fc) {
465 FlowSetStorageById(p->flow, GetFlowBypassInfoID(), fc);
466 } else {
467 return;
468 }
469 }
470 }
471 if (p->BypassPacketsFlow && p->BypassPacketsFlow(p)) {
472 if (p->flow) {
473 FlowUpdateState(p->flow, FLOW_STATE_CAPTURE_BYPASSED);
474 }
475 } else {
476 if (p->flow) {
477 FlowUpdateState(p->flow, FLOW_STATE_LOCAL_BYPASSED);
478 }
479 }
480#else /* CAPTURE_OFFLOAD */
481 if (p->flow) {
482 int state = p->flow->flow_state;
483 if (state == FLOW_STATE_LOCAL_BYPASSED)
484 return;
485 FlowUpdateState(p->flow, FLOW_STATE_LOCAL_BYPASSED);
486 }
487#endif
488}

490/** \brief switch direction of a packet */
491void PacketSwap(Packet *p)
492{
493 if (PKT_IS_TOSERVER(p)) {
494 p->flowflags &= ~FLOW_PKT_TOSERVER;
495 p->flowflags |= FLOW_PKT_TOCLIENT;
496
497 if (p->flowflags & FLOW_PKT_TOSERVER_FIRST) {
498 p->flowflags &= ~FLOW_PKT_TOSERVER_FIRST;
499 p->flowflags |= FLOW_PKT_TOCLIENT_FIRST;
500 }
501 } else {
502 p->flowflags &= ~FLOW_PKT_TOCLIENT;
503 p->flowflags |= FLOW_PKT_TOSERVER;
504
505 if (p->flowflags & FLOW_PKT_TOCLIENT_FIRST) {
506 p->flowflags &= ~FLOW_PKT_TOCLIENT_FIRST;
507 p->flowflags |= FLOW_PKT_TOSERVER_FIRST;
508 }
509 }
510}

606 continue;

611 SCMutexLock(&g_counter_table_mutex);
612 if (g_counter_table == NULL) {

614 StringHashCompareFunc,
615 StringHashFreeFunc);

617 FatalError(SC_ERR_INITIALIZATION, "decoder counter hash "
618 "table init failed");
619 }
620 }
621
622 char name[256];
623 char *dot = strchr(DEvents[i].event_name, '.');
624 BUG_ON(!dot);
625 snprintf(name, sizeof(name), "%s.%s",
626 stats_decoder_events_prefix, dot+1);
627
628 const char *found = HashTableLookup(g_counter_table, name, 0);
629 if (!found) {
630 char *add = SCStrdup(name);
631 if (add == NULL)
632 FatalError(SC_ERR_INITIALIZATION, "decoder counter hash "
633 "table name init failed");
634 int r = HashTableAdd(g_counter_table, add, 0);
635 if (r != 0)
636 FatalError(SC_ERR_INITIALIZATION, "decoder counter hash "
637 "table name add failed");
638 found = add;
639 }
640 dtv->counter_engine_events[i] = StatsRegisterCounter(
641 found, tv);
642
643 SCMutexUnlock(&g_counter_table_mutex);
644 } else {
645 dtv->counter_engine_events[i] = StatsRegisterCounter(
646 DEvents[i].event_name, tv);
647 }
648 }
649
650 return;
651}
