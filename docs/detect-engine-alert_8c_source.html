63/**
64 * \brief Handle a packet and check if needs a threshold logic
65 * Also apply rule action if necessary.
66 *
67 * \param de_ctx Detection Context
68 * \param sig Signature pointer
69 * \param p Packet structure
70 *
71 * \retval 1 alert is not suppressed
72 * \retval 0 alert is suppressed
73 */
74static int PacketAlertHandle(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
75 const Signature *s, Packet *p, PacketAlert *pa)
76{
77 SCEnter();
78 int ret = 1;
79 const DetectThresholdData *td = NULL;
80 const SigMatchData *smd;
81
82 if (!(PKT_IS_IPV4(p) || PKT_IS_IPV6(p))) {
83 SCReturnInt(1);
84 }
85
86 /* handle suppressions first */
87 if (s->sm_arrays[DETECT_SM_LIST_SUPPRESS] != NULL) {
88 KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_SUPPRESS);
89 smd = NULL;
90 do {
91 td = SigGetThresholdTypeIter(s, &smd, DETECT_SM_LIST_SUPPRESS);
92 if (td != NULL) {
93 SCLogDebug("td %p", td);
94
95 /* PacketAlertThreshold returns 2 if the alert is suppressed but
96 * we do need to apply rule actions to the packet. */
97 KEYWORD_PROFILING_START;
98 ret = PacketAlertThreshold(de_ctx, det_ctx, td, p, s, pa);
99 if (ret == 0 || ret == 2) {
100 KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 0);
101 /* It doesn't match threshold, remove it */
102 SCReturnInt(ret);
103 }
104 KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 1);
105 }
106 } while (smd != NULL);
107 }
108
109 /* if we're still here, consider thresholding */
110 if (s->sm_arrays[DETECT_SM_LIST_THRESHOLD] != NULL) {
111 KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_THRESHOLD);
112 smd = NULL;
113 do {
114 td = SigGetThresholdTypeIter(s, &smd, DETECT_SM_LIST_THRESHOLD);
115 if (td != NULL) {
116 SCLogDebug("td %p", td);
117
118 /* PacketAlertThreshold returns 2 if the alert is suppressed but
119 * we do need to apply rule actions to the packet. */
120 KEYWORD_PROFILING_START;
121 ret = PacketAlertThreshold(de_ctx, det_ctx, td, p, s, pa);
122 if (ret == 0 || ret == 2) {
123 KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD ,0);
124 /* It doesn't match threshold, remove it */
125 SCReturnInt(ret);
126 }
127 KEYWORD_PROFILING_END(det_ctx, DETECT_THRESHOLD, 1);
128 }
129 } while (smd != NULL);
130 }
131 SCReturnInt(1);
132}

134/**
135 * \brief Check if a certain sid alerted, this is used in the test functions
136 *
137 * \param p Packet on which we want to check if the signature alerted or not
138 * \param sid Signature id of the signature that has to be checked for a match
139 *
140 * \retval match A value > 0 on a match; 0 on no match
141 */
142int PacketAlertCheck(Packet *p, uint32_t sid)
143{
144 uint16_t i = 0;
145 int match = 0;
146
147 for (i = 0; i < p->alerts.cnt; i++) {
148 if (p->alerts.alerts[i].s == NULL)
149 continue;
150
151 if (p->alerts.alerts[i].s->id == sid)
152 match++;
153 }
154
155 return match;
156}

158static inline void RuleActionToFlow(const uint8_t action, Flow *f)
159{
160 if (action & (ACTION_DROP | ACTION_REJECT_ANY | ACTION_PASS)) {
161 if (f->flags & (FLOW_ACTION_DROP | FLOW_ACTION_PASS)) {
162 /* drop or pass already set. First to set wins. */
163 SCLogDebug("not setting %s flow already set to %s",
164 (action & ACTION_PASS) ? "pass" : "drop",
165 (f->flags & FLOW_ACTION_DROP) ? "drop" : "pass");
166 } else {
167 if (action & (ACTION_DROP | ACTION_REJECT_ANY)) {
168 f->flags |= FLOW_ACTION_DROP;
169 SCLogDebug("setting flow action drop");
170 }
171 if (action & ACTION_PASS) {
172 f->flags |= FLOW_ACTION_PASS;
173 SCLogDebug("setting flow action pass");
174 FlowSetNoPacketInspectionFlag(f);
175 }
176 }
177 }
178}

180/** \brief Apply action(s) and Set 'drop' sig info,
181 * if applicable */
182static void PacketApplySignatureActions(Packet *p, const Signature *s, const uint8_t alert_flags)
183{
184 SCLogDebug("packet %" PRIu64 " sid %u action %02x alert_flags %02x", p->pcap_cnt, s->id,
185 s->action, alert_flags);
186
187 /* REJECT also sets ACTION_DROP, just make it more visible with this check */
188 if (s->action & (ACTION_DROP | ACTION_REJECT_ANY)) {
189 /* PacketDrop will update the packet action, too */
190 PacketDrop(p, s->action, PKT_DROP_REASON_RULES);
191
192 if (p->alerts.drop.action == 0) {
193 p->alerts.drop.num = s->num;
194 p->alerts.drop.action = s->action;
195 p->alerts.drop.s = (Signature *)s;
196 }
197 if ((p->flow != NULL) && (alert_flags & PACKET_ALERT_FLAG_APPLY_ACTION_TO_FLOW)) {
198 RuleActionToFlow(s->action, p->flow);
199 }
200
201 DEBUG_VALIDATE_BUG_ON(!PacketTestAction(p, ACTION_DROP));
202 } else {
203 PacketUpdateAction(p, s->action);
204
205 if ((s->action & ACTION_PASS) && (p->flow != NULL) &&
206 (alert_flags & PACKET_ALERT_FLAG_APPLY_ACTION_TO_FLOW)) {
207 RuleActionToFlow(s->action, p->flow);
208 }
209 }
210}

231/** \internal
232 * \retval the new capacity
233 */
234static uint16_t AlertQueueExpand(DetectEngineThreadCtx *det_ctx)
235{
236#ifdef DEBUG
237 if (unlikely(g_eps_is_alert_queue_fail_mode))
238 return det_ctx->alert_queue_capacity;
239#endif
240 uint16_t new_cap = det_ctx->alert_queue_capacity * 2;
241 void *tmp_queue = SCRealloc(det_ctx->alert_queue, (size_t)(sizeof(PacketAlert) * new_cap));
242 if (unlikely(tmp_queue == NULL)) {
243 /* queue capacity didn't change */
244 return det_ctx->alert_queue_capacity;
245 }
246 det_ctx->alert_queue = tmp_queue;
247 det_ctx->alert_queue_capacity = new_cap;
248 SCLogDebug("Alert queue size doubled: %u elements, bytes: %" PRIuMAX "",
249 det_ctx->alert_queue_capacity,
250 (uintmax_t)(sizeof(PacketAlert) * det_ctx->alert_queue_capacity));
251 return new_cap;
252}

256static inline PacketAlert PacketAlertSet(
257 DetectEngineThreadCtx *det_ctx, const Signature *s, uint64_t tx_id, uint8_t alert_flags)
258{
259 PacketAlert pa;
260 pa.num = s->num;
261 pa.action = s->action;
262 pa.s = (Signature *)s;
263 pa.flags = alert_flags;
264 /* Set tx_id if the frame has it */
265 pa.tx_id = (tx_id == UINT64_MAX) ? 0 : tx_id;
266 pa.frame_id = (alert_flags & PACKET_ALERT_FLAG_FRAME) ? det_ctx->frame_id : 0;
267 return pa;

270/**
271 * \brief Append signature to local packet alert queue for later preprocessing
272 */
273void AlertQueueAppend(DetectEngineThreadCtx *det_ctx, const Signature *s, Packet *p, uint64_t tx_id,
274 uint8_t alert_flags)
275{
276 /* first time we see a drop action signature, set that in the packet */
277 /* we do that even before inserting into the queue, so we save it even if appending fails */
278 if (p->alerts.drop.action == 0 && s->action & ACTION_DROP) {
279 p->alerts.drop = PacketAlertSet(det_ctx, s, tx_id, alert_flags);
280 SCLogDebug("Set PacketAlert drop action. s->num %" PRIu32 "", s->num);
281 }
282
283 uint16_t pos = det_ctx->alert_queue_size;
284 if (pos == det_ctx->alert_queue_capacity) {
285 /* we must grow the alert queue */
286 if (pos == AlertQueueExpand(det_ctx)) {
287 /* this means we failed to expand the queue */
288 p->alerts.discarded++;
289 return;
290 }
291 }
292 det_ctx->alert_queue[pos] = PacketAlertSet(det_ctx, s, tx_id, alert_flags);
293
294 SCLogDebug("Appending sid %" PRIu32 ", s->num %" PRIu32 " to alert queue", s->id, s->num);
295 det_ctx->alert_queue_size++;
296 return;
297}

299/** \internal
300 * \brief sort helper for sorting alerts by priority
301 *
302 * Sorting is done first based on num and then using tx_id, if nums are equal.
303 * The Signature::num field is set based on internal priority. Higher priority
304 * rules have lower nums.
305 */
306static int AlertQueueSortHelper(const void *a, const void *b)
307{
308 const PacketAlert *pa0 = a;
309 const PacketAlert *pa1 = b;
310 if (pa1->num == pa0->num)
311 return pa0->tx_id < pa1->tx_id ? 1 : -1;
312 else
313 return pa0->num > pa1->num ? 1 : -1;
314}

316/** \internal
317 * \brief Check if Signature action should be applied to flow and apply
318 *
319 */
320static inline void FlowApplySignatureActions(
321 Packet *p, PacketAlert *pa, const Signature *s, uint8_t alert_flags)
322{
323 /* For DROP and PASS sigs we need to apply the action to the flow if
324 * - sig is IP or PD only
325 * - match is in applayer
326 * - match is in stream */
327 if (s->action & (ACTION_DROP | ACTION_PASS)) {
328 if ((pa->flags & (PACKET_ALERT_FLAG_STATE_MATCH | PACKET_ALERT_FLAG_STREAM_MATCH)) ||
329 (s->flags & (SIG_FLAG_IPONLY | SIG_FLAG_LIKE_IPONLY | SIG_FLAG_PDONLY |
330 SIG_FLAG_APPLAYER))) {
331 pa->flags |= PACKET_ALERT_FLAG_APPLY_ACTION_TO_FLOW;
332 SCLogDebug("packet %" PRIu64 " sid %u action %02x alert_flags %02x (set "
333 "PACKET_ALERT_FLAG_APPLY_ACTION_TO_FLOW)",
334 p->pcap_cnt, s->id, s->action, pa->flags);
335 }
336 }
337}

339/**
340 * \brief Check the threshold of the sigs that match, set actions, break on pass action
341 * This function iterate the packet alerts array, removing those that didn't match
342 * the threshold, and those that match after a signature with the action "pass".
343 * The array is sorted by action priority/order
344 * \param de_ctx detection engine context
345 * \param det_ctx detection engine thread context
346 * \param p pointer to the packet
347 */
348void PacketAlertFinalize(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx, Packet *p)
349{
350 SCEnter();
351
352 /* sort the alert queue before thresholding and appending to Packet */
353 qsort(det_ctx->alert_queue, det_ctx->alert_queue_size, sizeof(PacketAlert),
354 AlertQueueSortHelper);
355
356 uint16_t i = 0;
357 uint16_t max_pos = det_ctx->alert_queue_size;
358
359 while (i < max_pos) {
360 const Signature *s = de_ctx->sig_array[det_ctx->alert_queue[i].num];
361 int res = PacketAlertHandle(de_ctx, det_ctx, s, p, &det_ctx->alert_queue[i]);
362
363 if (res > 0) {
364 /* Now, if we have an alert, we have to check if we want
365 * to tag this session or src/dst host */
366 if (s->sm_arrays[DETECT_SM_LIST_TMATCH] != NULL) {
367 KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_TMATCH);
368 SigMatchData *smd = s->sm_arrays[DETECT_SM_LIST_TMATCH];
369 while (1) {
370 /* tags are set only for alerts */
371 KEYWORD_PROFILING_START;
372 sigmatch_table[smd->type].Match(det_ctx, p, (Signature *)s, smd->ctx);
373 KEYWORD_PROFILING_END(det_ctx, smd->type, 1);
374 if (smd->is_last)
375 break;
376 smd++;
377 }
378 }
379
380 /* set actions on the flow */
381 FlowApplySignatureActions(
382 p, &det_ctx->alert_queue[i], s, det_ctx->alert_queue[i].flags);
383
384 /* set actions on packet */
385 PacketApplySignatureActions(p, s, det_ctx->alert_queue[i].flags);
386 }
387
388 /* Thresholding removes this alert */
389 if (res == 0 || res == 2 || (s->flags & SIG_FLAG_NOALERT)) {
390 /* we will not copy this to the AlertQueue */
391 p->alerts.suppressed++;
392 } else if (p->alerts.cnt < packet_alert_max) {
393 p->alerts.alerts[p->alerts.cnt] = det_ctx->alert_queue[i];
394 SCLogDebug("Appending sid %" PRIu32 " alert to Packet::alerts at pos %u", s->id, i);
395
396 if (PacketTestAction(p, ACTION_PASS)) {
397 /* Ok, reset the alert cnt to end in the previous of pass
398 * so we ignore the rest with less prio */
399 break;
400 }
401 p->alerts.cnt++;
402 } else {
403 p->alerts.discarded++;
404 }
405 i++;
406 }
407
408 /* At this point, we should have all the new alerts. Now check the tag
409 * keyword context for sessions and hosts */
410 if (!(p->flags & PKT_PSEUDO_STREAM_END))
411 TagHandlePacket(de_ctx, det_ctx, p);
412
413 /* Set flag on flow to indicate that it has alerts */
414 if (p->flow != NULL && p->alerts.cnt > 0) {
415 if (!FlowHasAlerts(p->flow)) {
416 FlowSetHasAlertsFlag(p->flow);
417 p->flags |= PKT_FIRST_ALERTS;
418 }
419 }
420}
