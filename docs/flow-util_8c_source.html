45/** \brief allocate a flow
46 *
47 * We check against the memuse counter. If it passes that check we increment
48 * the counter first, then we try to alloc.
49 *
50 * \retval f the flow or NULL on out of memory
51 */
52Flow *FlowAlloc(void)
53{
54 Flow *f;
55 size_t size = sizeof(Flow) + FlowStorageSize();
56
57 if (!(FLOW_CHECK_MEMCAP(size))) {
58 return NULL;
59 }
60
61 (void) SC_ATOMIC_ADD(flow_memuse, size);
62
63 f = SCMalloc(size);
64 if (unlikely(f == NULL)) {
65 (void)SC_ATOMIC_SUB(flow_memuse, size);
66 return NULL;
67 }
68 memset(f, 0, size);
69
70 /* coverity[missing_lock] */
71 FLOW_INITIALIZE(f);
72 return f;
73}

142/* initialize the flow from the first packet
143 * we see from it. */
144void FlowInit(Flow *f, const Packet *p)
145{
146 SCEnter();
147 SCLogDebug("flow %p", f);
148
149 f->proto = p->proto;
150 f->recursion_level = p->recursion_level;
151 f->vlan_id[0] = p->vlan_id[0];
152 f->vlan_id[1] = p->vlan_id[1];
153 f->vlan_idx = p->vlan_idx;
154 f->livedev = p->livedev;
155
156 if (PKT_IS_IPV4(p)) {
157 FLOW_SET_IPV4_SRC_ADDR_FROM_PACKET(p, &f->src);
158 FLOW_SET_IPV4_DST_ADDR_FROM_PACKET(p, &f->dst);
159 f->min_ttl_toserver = f->max_ttl_toserver = IPV4_GET_IPTTL((p));
160 f->flags |= FLOW_IPV4;
161 } else if (PKT_IS_IPV6(p)) {
162 FLOW_SET_IPV6_SRC_ADDR_FROM_PACKET(p, &f->src);
163 FLOW_SET_IPV6_DST_ADDR_FROM_PACKET(p, &f->dst);
164 f->min_ttl_toserver = f->max_ttl_toserver = IPV6_GET_HLIM((p));
165 f->flags |= FLOW_IPV6;
166 }
167#ifdef DEBUG
168 /* XXX handle default */
169 else {
170 printf("FIXME: %s:%s:%" PRId32 "\n", __FILE__, __FUNCTION__, __LINE__);
171 }
172#endif
173
174 if (p->tcph != NULL) { /* XXX MACRO */
175 SET_TCP_SRC_PORT(p,&f->sp);
176 SET_TCP_DST_PORT(p,&f->dp);
177 } else if (p->udph != NULL) { /* XXX MACRO */
178 SET_UDP_SRC_PORT(p,&f->sp);
179 SET_UDP_DST_PORT(p,&f->dp);
180 } else if (p->icmpv4h != NULL) {
181 f->icmp_s.type = p->icmp_s.type;
182 f->icmp_s.code = p->icmp_s.code;
183 FlowSetICMPv4CounterPart(f);
184 } else if (p->icmpv6h != NULL) {
185 f->icmp_s.type = p->icmp_s.type;
186 f->icmp_s.code = p->icmp_s.code;
187 FlowSetICMPv6CounterPart(f);
188 } else if (p->sctph != NULL) { /* XXX MACRO */
189 SET_SCTP_SRC_PORT(p,&f->sp);
190 SET_SCTP_DST_PORT(p,&f->dp);
191 } else if (p->esph != NULL) {
192 f->esp.spi = ESP_GET_SPI(p);
193 } /* XXX handle default */
194#ifdef DEBUG
195 else {
196 printf("FIXME: %s:%s:%" PRId32 "\n", __FILE__, __FUNCTION__, __LINE__);
197 }
198#endif
199 COPY_TIMESTAMP(&p->ts, &f->startts);
200
201 f->protomap = FlowGetProtoMapping(f->proto);
202 f->timeout_policy = FlowGetTimeoutPolicy(f);
203 const uint32_t timeout_at = (uint32_t)f->startts.tv_sec + f->timeout_policy;
204 f->timeout_at = timeout_at;
205
206 if (MacSetFlowStorageEnabled()) {
207 MacSet *ms = FlowGetStorageById(f, MacSetGetFlowStorageID());
208 if (ms != NULL) {
209 MacSetReset(ms);
210 } else {
211 ms = MacSetInit(10);
212 FlowSetStorageById(f, MacSetGetFlowStorageID(), ms);
213 }
214 }
215
216 SCReturn;
217}

245void FlowEndCountersRegister(ThreadVars *t, FlowEndCounters *fec)
246{
247 for (int i = 0; i < FLOW_STATE_SIZE; i++) {
248 const char *name = NULL;
249 if (i == FLOW_STATE_NEW) {
250 name = "flow.end.state.new";
251 } else if (i == FLOW_STATE_ESTABLISHED) {
252 name = "flow.end.state.established";
253 } else if (i == FLOW_STATE_CLOSED) {
254 name = "flow.end.state.closed";
255 } else if (i == FLOW_STATE_LOCAL_BYPASSED) {
256 name = "flow.end.state.local_bypassed";
257#ifdef CAPTURE_OFFLOAD
258 } else if (i == FLOW_STATE_CAPTURE_BYPASSED) {
259 name = "flow.end.state.capture_bypassed";
260#endif
261 }
262 if (name) {
263 fec->flow_state[i] = StatsRegisterCounter(name, t);
264 }
265 }
266
267 for (enum TcpState i = TCP_NONE; i <= TCP_CLOSED; i++) {
268 const char *name = NULL;
269 switch (i) {
270 case TCP_NONE:
271 name = "flow.end.tcp_state.none";
272 break;
273 case TCP_SYN_SENT:
274 name = "flow.end.tcp_state.syn_sent";
275 break;
276 case TCP_SYN_RECV:
277 name = "flow.end.tcp_state.syn_recv";
278 break;
279 case TCP_ESTABLISHED:
280 name = "flow.end.tcp_state.established";
281 break;
282 case TCP_FIN_WAIT1:
283 name = "flow.end.tcp_state.fin_wait1";
284 break;
285 case TCP_FIN_WAIT2:
286 name = "flow.end.tcp_state.fin_wait2";
287 break;
288 case TCP_TIME_WAIT:
289 name = "flow.end.tcp_state.time_wait";
290 break;
291 case TCP_LAST_ACK:
292 name = "flow.end.tcp_state.last_ack";
293 break;
294 case TCP_CLOSE_WAIT:
295 name = "flow.end.tcp_state.close_wait";
296 break;
297 case TCP_CLOSING:
298 name = "flow.end.tcp_state.closing";
299 break;
300 case TCP_CLOSED:
301 name = "flow.end.tcp_state.closed";
302 break;
303 }
304
305 fec->flow_tcp_state[i] = StatsRegisterCounter(name, t);
306 }
307 fec->flow_tcp_liberal = StatsRegisterCounter("flow.end.tcp_liberal", t);
308}
