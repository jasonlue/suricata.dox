47/**
48 * \brief Registration function for sameip: keyword
49 * \todo add support for no_stream and stream_only
50 */
51void DetectSameipRegister(void)
52{
53 sigmatch_table[DETECT_SAMEIP].name = "sameip";
54 sigmatch_table[DETECT_SAMEIP].desc = "check if the IP address of the source is the same as the IP address of the destination";
55 sigmatch_table[DETECT_SAMEIP].url = "/rules/header-keywords.html#sameip";
56 sigmatch_table[DETECT_SAMEIP].Match = DetectSameipMatch;
57 sigmatch_table[DETECT_SAMEIP].Setup = DetectSameipSetup;
58#ifdef UNITTESTS
59 sigmatch_table[DETECT_SAMEIP].RegisterTests = DetectSameipRegisterTests;
60#endif
61 sigmatch_table[DETECT_SAMEIP].flags = SIGMATCH_NOOPT;
62}

64/**
65 * \internal
66 * \brief This function is used to match packets with same src/dst IPs
67 *
68 * \param t pointer to thread vars
69 * \param det_ctx pointer to the pattern matcher thread
70 * \param p pointer to the current packet
71 * \param m pointer to the sigmatch that we will cast into DetectSameipData
72 *
73 * \retval 0 no match
74 * \retval 1 match
75 */
76static int DetectSameipMatch(DetectEngineThreadCtx *det_ctx,
77 Packet *p, const Signature *s, const SigMatchCtx *ctx)
78{
79 return CMP_ADDR(&p->src, &p->dst) ? 1 : 0;
80}
81
82/**
83 * \internal
84 * \brief this function is used to add the sameip option into the signature
85 *
86 * \param de_ctx pointer to the Detection Engine Context
87 * \param s pointer to the Current Signature
88 * \param optstr pointer to the user provided options
89 *
90 * \retval 0 on Success
91 * \retval -1 on Failure
92 */
93static int DetectSameipSetup(DetectEngineCtx *de_ctx, Signature *s, const char *optstr)
94{
95 SigMatch *sm = NULL;
96
97 /* Get this into a SigMatch and put it in the Signature. */
98 sm = SigMatchAlloc();
99 if (sm == NULL)
100 goto error;
101
102 sm->type = DETECT_SAMEIP;
103 sm->ctx = NULL;
104
105 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
106 s->flags |= SIG_FLAG_REQUIRE_PACKET;
107
108 return 0;
109
110error:
111 if (sm != NULL)
112 SCFree(sm);
113 return -1;
114
115}

123 * \brief This test tests sameip success and failure.

125static int DetectSameipSigTest01(void)
126{
127 uint8_t *buf = (uint8_t *)
128 "GET / HTTP/1.0\r\n"
129 "\r\n";
130 uint16_t buflen = strlen((char *)buf);
131 Packet *p1 = NULL;
132 Packet *p2 = NULL;
133 ThreadVars th_v;
134 DetectEngineThreadCtx *det_ctx;
135
136 memset(&th_v, 0, sizeof(th_v));
137
138 /* First packet has same IPs */
139 p1 = UTHBuildPacketSrcDst(buf, buflen, IPPROTO_TCP, "1.2.3.4", "1.2.3.4");
140
141 /* Second packet does not have same IPs */
142 p2 = UTHBuildPacketSrcDst(buf, buflen, IPPROTO_TCP, "1.2.3.4", "4.3.2.1");
143
144 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
145 FAIL_IF_NULL(de_ctx);
146
147 de_ctx->flags |= DE_QUIET;
148
149 de_ctx->sig_list = SigInit(de_ctx,
150 "alert tcp any any -> any any "
151 "(msg:\"Testing sameip\"; sameip; sid:1;)");
152 FAIL_IF_NULL(de_ctx->sig_list);
153
154 SigGroupBuild(de_ctx);
155 DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
156
157 SigMatchSignatures(&th_v, de_ctx, det_ctx, p1);
158 FAIL_IF(PacketAlertCheck(p1, 1) == 0);
159
160 SigMatchSignatures(&th_v, de_ctx, det_ctx, p2);
161 FAIL_IF(PacketAlertCheck(p2, 1) != 0);
162
163 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
164 DetectEngineCtxFree(de_ctx);
165
166 PASS;
167}
