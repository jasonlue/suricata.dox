43void DetectRawbytesRegister(void)
44{
45 sigmatch_table[DETECT_RAWBYTES].name = "rawbytes";
46 sigmatch_table[DETECT_RAWBYTES].desc =
47 "dummy keyword to be compatible with snort signatures without effect";
48 sigmatch_table[DETECT_RAWBYTES].url = "/rules/payload-keywords.html#rawbytes";
49 sigmatch_table[DETECT_RAWBYTES].Setup = DetectRawbytesSetup;
50 sigmatch_table[DETECT_RAWBYTES].flags |= SIGMATCH_NOOPT;
51}

53static int DetectRawbytesSetup(DetectEngineCtx *de_ctx, Signature *s, const char *nullstr)
54{
55 SCEnter();
56
57 if (nullstr != NULL) {
58 SCLogError(SC_ERR_INVALID_VALUE, "rawbytes has no value");
59 SCReturnInt(-1);
60 }
61
62 if (s->init_data->list != DETECT_SM_LIST_NOTSET) {
63 SCLogError(SC_ERR_RAWBYTES_BUFFER,
64 "\"rawbytes\" cannot be combined "
65 "with the \"%s\" sticky buffer",
66 DetectEngineBufferTypeGetNameById(de_ctx, s->init_data->list));
67 SCReturnInt(-1);
68 }
69
70 SigMatch *pm = DetectGetLastSMByListId(s, DETECT_SM_LIST_PMATCH, DETECT_CONTENT, -1);
71 if (pm == NULL) {
72 SCLogError(
73 SC_ERR_RAWBYTES_MISSING_CONTENT, "\"rawbytes\" needs a preceding content option");
74 SCReturnInt(-1);
75 }
76
77 switch (pm->type) {
78 case DETECT_CONTENT: {
79 DetectContentData *cd = (DetectContentData *)pm->ctx;
80 if (cd->flags & DETECT_CONTENT_RAWBYTES) {
81 SCLogError(SC_ERR_INVALID_SIGNATURE,
82 "can't use multiple rawbytes modifiers for the same content. ");
83 SCReturnInt(-1);
84 }
85 cd->flags |= DETECT_CONTENT_RAWBYTES;
86 break;
87 }
88 default:
89 SCLogError(SC_ERR_RAWBYTES_MISSING_CONTENT,
90 "\"rawbytes\" needs a preceding content option");
91 SCReturnInt(-1);
92 }
93
94 SCReturnInt(0);
95}
