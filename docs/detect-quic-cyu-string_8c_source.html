61static InspectionBuffer *QuicStringGetData(DetectEngineThreadCtx *det_ctx,
62 const DetectEngineTransforms *transforms, Flow *f, struct QuicStringGetDataArgs *cbdata,
63 int list_id, bool first)
64{
65 SCEnter();
66
67 InspectionBuffer *buffer =
68 InspectionBufferMultipleForListGet(det_ctx, list_id, cbdata->local_id);
69 if (buffer == NULL)
70 return NULL;
71 if (!first && buffer->inspect != NULL)
72 return buffer;
73
74 const uint8_t *data;
75 uint32_t data_len;
76 if (rs_quic_tx_get_cyu_string(cbdata->txv, cbdata->local_id, &data, &data_len) == 0) {
77 return NULL;
78 }
79
80 InspectionBufferSetupMulti(buffer, transforms, data, data_len);
81
82 SCReturnPtr(buffer, "InspectionBuffer");
83}

85static uint8_t DetectEngineInspectQuicString(DetectEngineCtx *de_ctx,
86 DetectEngineThreadCtx *det_ctx, const DetectEngineAppInspectionEngine *engine,
87 const Signature *s, Flow *f, uint8_t flags, void *alstate, void *txv, uint64_t tx_id)
88{
89 uint32_t local_id = 0;
90
91 const DetectEngineTransforms *transforms = NULL;
92 if (!engine->mpm) {
93 transforms = engine->v2.transforms;
94 }
95
96 while (1) {
97 struct QuicStringGetDataArgs cbdata = {
98 local_id,
99 txv,
100 };
101 InspectionBuffer *buffer =
102 QuicStringGetData(det_ctx, transforms, f, &cbdata, engine->sm_list, false);
103 if (buffer == NULL || buffer->inspect == NULL)
104 break;
105
106 det_ctx->buffer_offset = 0;
107 det_ctx->discontinue_matching = 0;
108 det_ctx->inspection_recursion_counter = 0;
109
110 const int match = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd, NULL, f,
111 (uint8_t *)buffer->inspect, buffer->inspect_len, buffer->inspect_offset,
112 DETECT_CI_FLAGS_SINGLE, DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
113 if (match == 1) {
114 return DETECT_ENGINE_INSPECT_SIG_MATCH;
115 }
116 local_id++;
117 }
118 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
119}

121/** \brief QuicString Mpm prefilter callback
122 *
123 * \param det_ctx detection engine thread ctx
124 * \param p packet to inspect
125 * \param f flow to inspect
126 * \param txv tx to inspect
127 * \param pectx inspection context
128 */
129static void PrefilterTxQuicString(DetectEngineThreadCtx *det_ctx, const void *pectx, Packet *p,
130 Flow *f, void *txv, const uint64_t idx, const uint8_t flags)
131{
132 SCEnter();
133
134 const PrefilterMpmListId *ctx = (const PrefilterMpmListId *)pectx;
135 const MpmCtx *mpm_ctx = ctx->mpm_ctx;
136 const int list_id = ctx->list_id;
137
138 uint32_t local_id = 0;
139 while (1) {
140 // loop until we get a NULL
141
142 struct QuicStringGetDataArgs cbdata = { local_id, txv };
143 InspectionBuffer *buffer =
144 QuicStringGetData(det_ctx, ctx->transforms, f, &cbdata, list_id, true);
145 if (buffer == NULL)
146 break;
147
148 if (buffer->inspect_len >= mpm_ctx->minlen) {
149 (void)mpm_table[mpm_ctx->mpm_type].Search(
150 mpm_ctx, &det_ctx->mtcu, &det_ctx->pmq, buffer->inspect, buffer->inspect_len);
151 PREFILTER_PROFILING_ADD_BYTES(det_ctx, buffer->inspect_len);
152 }
153
154 local_id++;
155 }
156}

163static int PrefilterMpmListIdRegister(DetectEngineCtx *de_ctx, SigGroupHead *sgh, MpmCtx *mpm_ctx,
164 const DetectBufferMpmRegistery *mpm_reg, int list_id)
165{
166 PrefilterMpmListId *pectx = SCCalloc(1, sizeof(*pectx));
167 if (pectx == NULL)
168 return -1;
169 pectx->list_id = list_id;
170 pectx->mpm_ctx = mpm_ctx;
171 pectx->transforms = &mpm_reg->transforms;
172
173 return PrefilterAppendTxEngine(de_ctx, sgh, PrefilterTxQuicString, mpm_reg->app_v2.alproto,
174 mpm_reg->app_v2.tx_min_progress, pectx, PrefilterMpmListIdFree, mpm_reg->pname);
175}

177void DetectQuicCyuStringRegister(void)
178{
179 /* quic.cyu.string sticky buffer */
180 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].name = KEYWORD_NAME;
181 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].desc =
182 "sticky buffer to match on the QUIC CYU string";
183 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].url = "/rules/" KEYWORD_DOC;
184 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].Setup = DetectQuicCyuStringSetup;
185 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].flags |= SIGMATCH_NOOPT;
186#ifdef UNITTESTS
187 sigmatch_table[DETECT_AL_QUIC_CYU_STRING].RegisterTests = DetectQuicCyuStringRegisterTests;
188#endif
189
190 DetectAppLayerMpmRegister2(
191 BUFFER_NAME, SIG_FLAG_TOSERVER, 2, PrefilterMpmListIdRegister, NULL, ALPROTO_QUIC, 1);
192
193 DetectAppLayerInspectEngineRegister2(
194 BUFFER_NAME, ALPROTO_QUIC, SIG_FLAG_TOSERVER, 0, DetectEngineInspectQuicString, NULL);
195
196 DetectBufferTypeSetDescriptionByName(BUFFER_NAME, BUFFER_DESC);
197
198 g_buffer_id = DetectBufferTypeGetByName(BUFFER_NAME);
199
200 SCLogDebug("registering " BUFFER_NAME " rule option");
201}
