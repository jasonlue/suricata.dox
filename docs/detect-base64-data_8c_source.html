48static int DetectBase64DataSetup(DetectEngineCtx *de_ctx, Signature *s,
49 const char *str)
50{
51 SigMatch *pm = NULL;
52
53 /* Check for a preceding base64_decode. */
54 pm = DetectGetLastSMFromLists(s, DETECT_BASE64_DECODE, -1);
55 if (pm == NULL) {
56 SCLogError(SC_ERR_INVALID_SIGNATURE,
57 "\"base64_data\" keyword seen without preceding base64_decode.");
58 return -1;
59 }
60
61 s->init_data->list = DETECT_SM_LIST_BASE64_DATA;
62 return 0;
63}

65int DetectBase64DataDoMatch(DetectEngineCtx *de_ctx,
66 DetectEngineThreadCtx *det_ctx, const Signature *s, Flow *f)
67{
68 if (det_ctx->base64_decoded_len) {
69 return DetectEngineContentInspection(de_ctx, det_ctx, s,
70 s->sm_arrays[DETECT_SM_LIST_BASE64_DATA], NULL, f, det_ctx->base64_decoded,
71 det_ctx->base64_decoded_len, 0, DETECT_CI_FLAGS_SINGLE,
72 DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
73 }
74
75 return 0;
76}
