47Packet *TmqhInputSimple(ThreadVars *t)
48{
49 PacketQueue *q = t->inq->pq;
50
51 StatsSyncCountersIfSignalled(t);
52
53 SCMutexLock(&q->mutex_q);
54
55 if (q->len == 0) {
56 /* if we have no packets in queue, wait... */
57 SCCondWait(&q->cond_q, &q->mutex_q);
58 }
59
60 if (q->len > 0) {
61 Packet *p = PacketDequeue(q);
62 SCMutexUnlock(&q->mutex_q);
63 return p;
64 } else {
65 /* return NULL if we have no pkt. Should only happen on signals. */
66 SCMutexUnlock(&q->mutex_q);
67 return NULL;
68 }
69}

71void TmqhInputSimpleShutdownHandler(ThreadVars *tv)
72{
73 int i;
74
75 if (tv == NULL || tv->inq == NULL) {
76 return;
77 }
78
79 for (i = 0; i < (tv->inq->reader_cnt + tv->inq->writer_cnt); i++)
80 SCCondSignal(&tv->inq->pq->cond_q);
81}

83void TmqhOutputSimple(ThreadVars *t, Packet *p)
84{
85 SCLogDebug("Packet %p, p->root %p, alloced %s", p, p->root, p->flags & PKT_ALLOC ? "true":"false");
86
87 PacketQueue *q = t->outq->pq;
88
89 SCMutexLock(&q->mutex_q);
90 PacketEnqueue(q, p);
91 SCCondSignal(&q->cond_q);
92 SCMutexUnlock(&q->mutex_q);
93}
