31 *
32 * \param arg The name of the variable being sought

34 * \param index When found, the value of the slot within the byte vars
35 *
36 * \retval true A match for the variable was found.
37 * \retval false
38 */
39bool DetectByteRetrieveSMVar(const char *arg, const Signature *s, DetectByteIndexType *index)
40{
41 SigMatch *bed_sm = DetectByteExtractRetrieveSMVar(arg, s);
42 if (bed_sm != NULL) {
43 *index = ((DetectByteExtractData *)bed_sm->ctx)->local_id;
44 return true;
45 }
