70void SCProfilingKeywordsGlobalInit(void)
71{
72 ConfNode *conf;
73
74 conf = ConfGetNode("profiling.keywords");
75 if (conf != NULL) {
76 if (ConfNodeChildValueIsTrue(conf, "enabled")) {
77 profiling_keyword_enabled = 1;
78 const char *filename = ConfNodeLookupChildValue(conf, "filename");
79 if (filename != NULL) {
80 const char *log_dir;
81 log_dir = ConfigGetLogDirectory();
82
83 snprintf(profiling_file_name, sizeof(profiling_file_name), "%s/%s",
84 log_dir, filename);
85
86 const char *v = ConfNodeLookupChildValue(conf, "append");
87 if (v == NULL || ConfValIsTrue(v)) {
88 profiling_file_mode = "a";
89 } else {
90 profiling_file_mode = "w";
91 }
92
93 profiling_keywords_output_to_file = 1;
94 }
95 }

99static void DoDump(SCProfileKeywordDetectCtx *rules_ctx, FILE *fp, const char *name)
100{
101 int i;
102 fprintf(fp, " ----------------------------------------------"
103 "------------------------------------------------------"
104 "----------------------------\n");
105 fprintf(fp, " Stats for: %s\n", name);
106 fprintf(fp, " ----------------------------------------------"
107 "------------------------------------------------------"
108 "----------------------------\n");
109 fprintf(fp, " %-16s %-15s %-15s %-15s %-15s %-15s %-15s %-15s\n", "Keyword", "Ticks", "Checks", "Matches", "Max Ticks", "Avg", "Avg Match", "Avg No Match");
110 fprintf(fp, " ---------------- "
111 "--------------- "
112 "--------------- "
113 "--------------- "
114 "--------------- "
115 "--------------- "
116 "--------------- "
117 "--------------- "
118 "\n");
119 for (i = 0; i < DETECT_TBLSIZE; i++) {
120 SCProfileKeywordData *d = &rules_ctx->data[i];
121 if (d == NULL || d->checks == 0)
122 continue;
123
124 uint64_t ticks = d->ticks_match + d->ticks_no_match;
125 double avgticks = 0;
126 double avgticks_match = 0;
127 double avgticks_no_match = 0;
128 if (ticks && d->checks) {
129 avgticks = (double)(ticks / d->checks);
130
131 if (d->ticks_match && d->matches)
132 avgticks_match = (double)(d->ticks_match / d->matches);
133 if (d->ticks_no_match && (d->checks - d->matches) != 0)
134 avgticks_no_match = (double)(d->ticks_no_match / (d->checks - d->matches));
135 }
136
137 fprintf(fp,
138 " %-16s %-15"PRIu64" %-15"PRIu64" %-15"PRIu64" %-15"PRIu64" %-15.2f %-15.2f %-15.2f\n",
139 sigmatch_table[i].name,
140 ticks,
141 d->checks,
142 d->matches,
143 d->max,
144 avgticks,
145 avgticks_match,
146 avgticks_no_match);
147 }
148}

150static void
151SCProfilingKeywordDump(DetectEngineCtx *de_ctx)
152{
153 int i;
154 FILE *fp;
155 struct timeval tval;
156 struct tm *tms;
157 struct tm local_tm;
158
159 if (profiling_keyword_enabled == 0)
160 return;
161
162 const int nlists = de_ctx->buffer_type_id;
163 gettimeofday(&tval, NULL);
164 tms = SCLocalTime(tval.tv_sec, &local_tm);
165
166 if (profiling_keywords_output_to_file == 1) {
167 SCLogDebug("file %s mode %s", profiling_file_name, profiling_file_mode);
168
169 fp = fopen(profiling_file_name, profiling_file_mode);
170
171 if (fp == NULL) {
172 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", profiling_file_name,
173 strerror(errno));
174 return;
175 }
176 } else {
177 fp = stdout;
178 }
179
180 fprintf(fp, " ----------------------------------------------"
181 "------------------------------------------------------"
182 "----------------------------\n");
183 fprintf(fp, " Date: %" PRId32 "/%" PRId32 "/%04d -- "
184 "%02d:%02d:%02d\n", tms->tm_mon + 1, tms->tm_mday, tms->tm_year + 1900,
185 tms->tm_hour,tms->tm_min, tms->tm_sec);
186
187 /* global stats first */
188 DoDump(de_ctx->profile_keyword_ctx, fp, "total");
189 /* per buffer stats next, but only if there are stats to print */
190 for (i = 0; i < nlists; i++) {
191 int j;
192 uint64_t checks = 0;
193 for (j = 0; j < DETECT_TBLSIZE; j++) {
194 checks += de_ctx->profile_keyword_ctx_per_list[i]->data[j].checks;
195 }
196
197 if (checks) {
198 const char *name = NULL;
199 if (i < DETECT_SM_LIST_DYNAMIC_START) {
200 name = DetectSigmatchListEnumToString(i);
201 } else {
202 name = DetectEngineBufferTypeGetNameById(de_ctx, i);
203 }
204
205 DoDump(de_ctx->profile_keyword_ctx_per_list[i], fp, name);
206 }
207 }
208
209 fprintf(fp,"\n");
210 if (fp != stdout)
211 fclose(fp);
212
213 SCLogPerf("Done dumping keyword profiling data.");
214}

216/**
217 * \brief Update a rule counter.
218 *
219 * \param id The ID of this counter.
220 * \param ticks Number of CPU ticks for this rule.
221 * \param match Did the rule match?
222 */
223void
224SCProfilingKeywordUpdateCounter(DetectEngineThreadCtx *det_ctx, int id, uint64_t ticks, int match)
225{
226 if (det_ctx != NULL && det_ctx->keyword_perf_data != NULL && id < DETECT_TBLSIZE) {
227 SCProfileKeywordData *p = &det_ctx->keyword_perf_data[id];
228
229 p->checks++;
230 p->matches += match;
231 if (ticks > p->max)
232 p->max = ticks;
233 if (match == 1)
234 p->ticks_match += ticks;
235 else
236 p->ticks_no_match += ticks;
237
238 /* store per list (buffer type) as well */
239 if (det_ctx->keyword_perf_list >= 0) {// && det_ctx->keyword_perf_list < DETECT_SM_LIST_MAX) {
240 p = &det_ctx->keyword_perf_data_per_list[det_ctx->keyword_perf_list][id];
241 p->checks++;
242 p->matches += match;
243 if (ticks > p->max)
244 p->max = ticks;
245 if (match == 1)
246 p->ticks_match += ticks;
247 else
248 p->ticks_no_match += ticks;
249 }
250 }
251}

294void SCProfilingKeywordThreadSetup(SCProfileKeywordDetectCtx *ctx, DetectEngineThreadCtx *det_ctx)
295{
296 if (ctx == NULL)
297 return;
298
299 SCProfileKeywordData *a = SCMalloc(sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
300 if (a != NULL) {
301 memset(a, 0x00, sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
302 det_ctx->keyword_perf_data = a;
303 }
304
305 const int nlists = det_ctx->de_ctx->buffer_type_id;
306 det_ctx->keyword_perf_data_per_list = SCCalloc(nlists, sizeof(SCProfileKeywordData *));
307 BUG_ON(det_ctx->keyword_perf_data_per_list == NULL);
308
309 int i;
310 for (i = 0; i < nlists; i++) {
311 SCProfileKeywordData *b = SCMalloc(sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
312 if (b != NULL) {
313 memset(b, 0x00, sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
314 det_ctx->keyword_perf_data_per_list[i] = b;
315 }
316
317 }

320static void SCProfilingKeywordThreadMerge(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx)
321{
322 if (de_ctx == NULL || de_ctx->profile_keyword_ctx == NULL ||
323 de_ctx->profile_keyword_ctx->data == NULL || det_ctx == NULL ||
324 det_ctx->keyword_perf_data == NULL)
325 return;
326
327 int i;
328 for (i = 0; i < DETECT_TBLSIZE; i++) {
329 de_ctx->profile_keyword_ctx->data[i].checks += det_ctx->keyword_perf_data[i].checks;
330 de_ctx->profile_keyword_ctx->data[i].matches += det_ctx->keyword_perf_data[i].matches;
331 de_ctx->profile_keyword_ctx->data[i].ticks_match += det_ctx->keyword_perf_data[i].ticks_match;
332 de_ctx->profile_keyword_ctx->data[i].ticks_no_match += det_ctx->keyword_perf_data[i].ticks_no_match;
333 if (det_ctx->keyword_perf_data[i].max > de_ctx->profile_keyword_ctx->data[i].max)
334 de_ctx->profile_keyword_ctx->data[i].max = det_ctx->keyword_perf_data[i].max;
335 }
336
337 const int nlists = det_ctx->de_ctx->buffer_type_id;
338 int j;
339 for (j = 0; j < nlists; j++) {
340 for (i = 0; i < DETECT_TBLSIZE; i++) {
341 de_ctx->profile_keyword_ctx_per_list[j]->data[i].checks += det_ctx->keyword_perf_data_per_list[j][i].checks;
342 de_ctx->profile_keyword_ctx_per_list[j]->data[i].matches += det_ctx->keyword_perf_data_per_list[j][i].matches;
343 de_ctx->profile_keyword_ctx_per_list[j]->data[i].ticks_match += det_ctx->keyword_perf_data_per_list[j][i].ticks_match;
344 de_ctx->profile_keyword_ctx_per_list[j]->data[i].ticks_no_match += det_ctx->keyword_perf_data_per_list[j][i].ticks_no_match;
345 if (det_ctx->keyword_perf_data_per_list[j][i].max > de_ctx->profile_keyword_ctx_per_list[j]->data[i].max)
346 de_ctx->profile_keyword_ctx_per_list[j]->data[i].max = det_ctx->keyword_perf_data_per_list[j][i].max;
347 }
348 }

351void SCProfilingKeywordThreadCleanup(DetectEngineThreadCtx *det_ctx)
352{
353 if (det_ctx == NULL || det_ctx->de_ctx == NULL || det_ctx->keyword_perf_data == NULL)
354 return;
355
356 pthread_mutex_lock(&det_ctx->de_ctx->profile_keyword_ctx->data_m);
357 SCProfilingKeywordThreadMerge(det_ctx->de_ctx, det_ctx);
358 pthread_mutex_unlock(&det_ctx->de_ctx->profile_keyword_ctx->data_m);
359
360 SCFree(det_ctx->keyword_perf_data);
361 det_ctx->keyword_perf_data = NULL;
362
363 const int nlists = det_ctx->de_ctx->buffer_type_id;
364 int i;
365 for (i = 0; i < nlists; i++) {
366 SCFree(det_ctx->keyword_perf_data_per_list[i]);
367 det_ctx->keyword_perf_data_per_list[i] = NULL;
368 }
369 SCFree(det_ctx->keyword_perf_data_per_list);
370}

372/**
373 * \brief Register the keyword profiling counters.
374 *
375 * \param de_ctx The active DetectEngineCtx, used to get at the loaded rules.
376 */
377void
378SCProfilingKeywordInitCounters(DetectEngineCtx *de_ctx)
379{
380 if (profiling_keyword_enabled == 0)
381 return;
382
383 const int nlists = de_ctx->buffer_type_id;
384
385 de_ctx->profile_keyword_ctx = SCProfilingKeywordInitCtx();
386 BUG_ON(de_ctx->profile_keyword_ctx == NULL);
387
388 de_ctx->profile_keyword_ctx->data = SCMalloc(sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
389 BUG_ON(de_ctx->profile_keyword_ctx->data == NULL);
390 memset(de_ctx->profile_keyword_ctx->data, 0x00, sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
391
392 de_ctx->profile_keyword_ctx_per_list = SCCalloc(nlists, sizeof(SCProfileKeywordDetectCtx *));
393 BUG_ON(de_ctx->profile_keyword_ctx_per_list == NULL);
394
395 int i;
396 for (i = 0; i < nlists; i++) {
397 de_ctx->profile_keyword_ctx_per_list[i] = SCProfilingKeywordInitCtx();
398 BUG_ON(de_ctx->profile_keyword_ctx_per_list[i] == NULL);
399 de_ctx->profile_keyword_ctx_per_list[i]->data = SCMalloc(sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
400 BUG_ON(de_ctx->profile_keyword_ctx_per_list[i]->data == NULL);
401 memset(de_ctx->profile_keyword_ctx_per_list[i]->data, 0x00, sizeof(SCProfileKeywordData) * DETECT_TBLSIZE);
402 }
403
404 SCLogPerf("Registered %"PRIu32" keyword profiling counters.", DETECT_TBLSIZE);
405}
