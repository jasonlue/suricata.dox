109/**
110 * \brief Maps the SC logging level to the syslog logging level
111 *
112 * \param The SC logging level that has to be mapped to the syslog_log_level
113 *
114 * \retval syslog_log_level The mapped syslog_api_log_level, for the logging
115 * module api's internal log_level
116 */
117static inline int SCLogMapLogLevelToSyslogLevel(int log_level)
118{
119 int syslog_log_level = 0;
120
121 switch (log_level) {
122 case SC_LOG_EMERGENCY:
123 syslog_log_level = LOG_EMERG;
124 break;
125 case SC_LOG_ALERT:
126 syslog_log_level = LOG_ALERT;
127 break;
128 case SC_LOG_CRITICAL:
129 syslog_log_level = LOG_CRIT;
130 break;
131 case SC_LOG_ERROR:
132 syslog_log_level = LOG_ERR;
133 break;
134 case SC_LOG_WARNING:
135 syslog_log_level = LOG_WARNING;
136 break;
137 case SC_LOG_NOTICE:
138 syslog_log_level = LOG_NOTICE;
139 break;
140 case SC_LOG_INFO:
141 syslog_log_level = LOG_INFO;
142 break;
143 case SC_LOG_CONFIG:
144 case SC_LOG_DEBUG:
145 case SC_LOG_PERF:
146 syslog_log_level = LOG_DEBUG;
147 break;
148 default:
149 syslog_log_level = LOG_EMERG;
150 break;
151 }
152
153 return syslog_log_level;
154}

156/**
157 * \brief Output function that logs a character string out to a file descriptor
158 *
159 * \param fd Pointer to the file descriptor
160 * \param msg Pointer to the character string that should be logged
161 */
162static inline void SCLogPrintToStream(FILE *fd, char *msg)
163{
164 /* Would only happen if the log file failed to re-open during rotation. */
165 if (fd == NULL) {
166 return;
167 }
168
169#if defined (OS_WIN32)
170 SCMutexLock(&sc_log_stream_lock);
171#endif /* OS_WIN32 */
172
173 if (fprintf(fd, "%s\n", msg) < 0)
174 printf("Error writing to stream using fprintf\n");
175
176 fflush(fd);
177
178#if defined (OS_WIN32)
179 SCMutexUnlock(&sc_log_stream_lock);
180#endif /* OS_WIN32 */
181
182 return;
183}

207static int SCLogMessageJSON(struct timeval *tval, char *buffer, size_t buffer_size,
208 SCLogLevel log_level, const char *file,
209 unsigned line, const char *function, SCError error_code,
210 const char *message)
211{
212 json_t *js = json_object();
213 if (unlikely(js == NULL))
214 goto error;
215 json_t *ejs = json_object();
216 if (unlikely(ejs == NULL))
217 goto error;
218
219 char timebuf[64];
220 CreateIsoTimeString(tval, timebuf, sizeof(timebuf));
221 json_object_set_new(js, "timestamp", json_string(timebuf));
222
223 const char *s = SCMapEnumValueToName(log_level, sc_log_level_map);
224 if (s != NULL) {
225 json_object_set_new(js, "log_level", json_string(s));
226 } else {
227 json_object_set_new(js, "log_level", json_string("INVALID"));
228 }
229
230 json_object_set_new(js, "event_type", json_string("engine"));
231
232 if (error_code > 0) {
233 json_object_set_new(ejs, "error_code", json_integer(error_code));
234 json_object_set_new(ejs, "error", json_string(SCErrorToString(error_code)));
235 }
236
237 if (message)
238 json_object_set_new(ejs, "message", json_string(message));
239
240 if (log_level >= SC_LOG_DEBUG) {
241 if (function)
242 json_object_set_new(ejs, "function", json_string(function));
243
244 if (file)
245 json_object_set_new(ejs, "file", json_string(file));
246
247 if (line > 0)
248 json_object_set_new(ejs, "line", json_integer(line));
249 }
250
251 json_object_set_new(js, "engine", ejs);
252
253 char *js_s = json_dumps(js,
254 JSON_PRESERVE_ORDER|JSON_COMPACT|JSON_ENSURE_ASCII|
255 JSON_ESCAPE_SLASH);
256 snprintf(buffer, buffer_size, "%s", js_s);
257 free(js_s);
258
259 json_object_del(js, "engine");
260 json_object_clear(js);
261 json_decref(js);
262
263 return 0;
264error:
265 return -1;
266}

268/**
269 * \brief Adds the global log_format to the outgoing buffer
270 *
271 * \param log_level log_level of the message that has to be logged
272 * \param msg Buffer containing the outgoing message
273 * \param file File_name from where the message originated
274 * \param function Function_name from where the message originated
275 * \param line Line_no from where the messaged originated
276 *
277 * \retval SC_OK on success; else an error code
278 */
279static SCError SCLogMessageGetBuffer(
280 struct timeval *tval, int color, SCLogOPType type,
281 char *buffer, size_t buffer_size,
282 const char *log_format,
283
284 const SCLogLevel log_level, const char *file,
285 const unsigned int line, const char *function,
286 const SCError error_code, const char *message)
287{
288 if (type == SC_LOG_OP_TYPE_JSON)
289 return SCLogMessageJSON(tval, buffer, buffer_size, log_level, file, line, function, error_code, message);
290
291 char *temp = buffer;
292 const char *s = NULL;
293 struct tm *tms = NULL;
294
295 const char *redb = "";
296 const char *red = "";
297 const char *yellowb = "";
298 const char *yellow = "";
299 const char *green = "";
300 const char *blue = "";
301 const char *reset = "";
302 if (color) {
303 redb = "\x1b[1;31m";
304 red = "\x1b[31m";
305 yellowb = "\x1b[1;33m";
306 yellow = "\x1b[33m";
307 green = "\x1b[32m";
308 blue = "\x1b[34m";
309 reset = "\x1b[0m";
310 }
311 /* no of characters_written(cw) by snprintf */
312 int cw = 0;
313
314 BUG_ON(sc_log_module_initialized != 1);
315
316 /* make a copy of the format string as it will be modified below */
317 char local_format[strlen(log_format) + 1];
318 strlcpy(local_format, log_format, sizeof(local_format));
319 char *temp_fmt = local_format;
320 char *substr = temp_fmt;
321
322 while ( (temp_fmt = strchr(temp_fmt, SC_LOG_FMT_PREFIX)) ) {
323 if ((temp - buffer) > SC_LOG_MAX_LOG_MSG_LEN) {
324 return SC_OK;
325 }
326 switch(temp_fmt[1]) {
327 case SC_LOG_FMT_TIME:
328 temp_fmt[0] = '\0';
329
330 struct tm local_tm;
331 tms = SCLocalTime(tval->tv_sec, &local_tm);
332
333 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
334 "%s%s%d/%d/%04d -- %02d:%02d:%02d%s",
335 substr, green, tms->tm_mday, tms->tm_mon + 1,
336 tms->tm_year + 1900, tms->tm_hour, tms->tm_min,
337 tms->tm_sec, reset);
338 if (cw < 0)
339 return SC_ERR_SPRINTF;
340 temp += cw;
341 temp_fmt++;
342 substr = temp_fmt;
343 substr++;
344 break;
345
346 case SC_LOG_FMT_PID:
347 temp_fmt[0] = '\0';
348 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
349 "%s%s%u%s", substr, yellow, getpid(), reset);
350 if (cw < 0)
351 return SC_ERR_SPRINTF;
352 temp += cw;
353 temp_fmt++;
354 substr = temp_fmt;
355 substr++;
356 break;
357
358 case SC_LOG_FMT_TID:
359 temp_fmt[0] = '\0';
360 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
361 "%s%s%lu%s", substr, yellow, SCGetThreadIdLong(), reset);
362 if (cw < 0)
363 return SC_ERR_SPRINTF;
364 temp += cw;
365 temp_fmt++;
366 substr = temp_fmt;
367 substr++;
368 break;
369
370 case SC_LOG_FMT_TM:
371 temp_fmt[0] = '\0';
372 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer), "%s%s%s%s", substr,
373 yellow, t_thread_name, reset);
374 if (cw < 0)
375 return SC_ERR_SPRINTF;
376 temp += cw;
377 temp_fmt++;
378 substr = temp_fmt;
379 substr++;
380 break;
381
382 case SC_LOG_FMT_LOG_LEVEL:
383 temp_fmt[0] = '\0';
384 s = SCMapEnumValueToName(log_level, sc_log_level_map);
385 if (s != NULL) {
386 if (log_level <= SC_LOG_ERROR)
387 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
388 "%s%s%s%s", substr, redb, s, reset);
389 else if (log_level == SC_LOG_WARNING)
390 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
391 "%s%s%s%s", substr, red, s, reset);
392 else if (log_level == SC_LOG_NOTICE)
393 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
394 "%s%s%s%s", substr, yellowb, s, reset);
395 else
396 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
397 "%s%s%s%s", substr, yellow, s, reset);
398 } else {
399 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
400 "%s%s", substr, "INVALID");
401 }
402 if (cw < 0)
403 return SC_ERR_SPRINTF;
404 temp += cw;
405 temp_fmt++;
406 substr = temp_fmt;
407 substr++;
408 break;
409
410 case SC_LOG_FMT_FILE_NAME:
411 temp_fmt[0] = '\0';
412 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
413 "%s%s%s%s", substr, blue, file, reset);
414 if (cw < 0)
415 return SC_ERR_SPRINTF;
416 temp += cw;
417 temp_fmt++;
418 substr = temp_fmt;
419 substr++;
420 break;
421
422 case SC_LOG_FMT_LINE:
423 temp_fmt[0] = '\0';
424 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
425 "%s%s%u%s", substr, green, line, reset);
426 if (cw < 0)
427 return SC_ERR_SPRINTF;
428 temp += cw;
429 temp_fmt++;
430 substr = temp_fmt;
431 substr++;
432 break;
433
434 case SC_LOG_FMT_FUNCTION:
435 temp_fmt[0] = '\0';
436 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
437 "%s%s%s%s", substr, green, function, reset);
438 if (cw < 0)
439 return SC_ERR_SPRINTF;
440 temp += cw;
441 temp_fmt++;
442 substr = temp_fmt;
443 substr++;
444 break;
445
446 }
447 temp_fmt++;
448 }
449 if ((temp - buffer) > SC_LOG_MAX_LOG_MSG_LEN) {
450 return SC_OK;
451 }
452 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer), "%s", substr);
453 if (cw < 0) {
454 return SC_ERR_SPRINTF;
455 }
456 temp += cw;
457 if ((temp - buffer) > SC_LOG_MAX_LOG_MSG_LEN) {
458 return SC_OK;
459 }
460
461 if (error_code != SC_OK) {
462 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer),
463 "[%sERRCODE%s: %s%s%s(%s%d%s)] - ", yellow, reset, red, SCErrorToString(error_code), reset, yellow, error_code, reset);
464 if (cw < 0) {
465 return SC_ERR_SPRINTF;
466 }
467 temp += cw;
468 if ((temp - buffer) > SC_LOG_MAX_LOG_MSG_LEN) {
469 return SC_OK;
470 }
471 }
472
473 const char *hi = "";
474 if (error_code > SC_OK)
475 hi = red;
476 else if (log_level <= SC_LOG_NOTICE)
477 hi = yellow;
478 cw = snprintf(temp, SC_LOG_MAX_LOG_MSG_LEN - (temp - buffer), "%s%s%s", hi, message, reset);
479 if (cw < 0) {
480 return SC_ERR_SPRINTF;
481 }
482 temp += cw;
483 if ((temp - buffer) > SC_LOG_MAX_LOG_MSG_LEN) {
484 return SC_OK;
485 }
486
487 if (sc_log_config->op_filter_regex != NULL) {
488 if (pcre2_match(sc_log_config->op_filter_regex, (PCRE2_SPTR8)buffer, strlen(buffer), 0, 0,
489 sc_log_config->op_filter_regex_match, NULL) < 0) {
490 return SC_ERR_LOG_FG_FILTER_MATCH; // bit hacky, but just return !0
491 }
492 }
493
494 return SC_OK;
495}

497/** \internal
498 * \brief try to reopen file
499 * \note no error reporting here, as we're called by SCLogMessage
500 * \retval status 0 ok, -1 error */
501static int SCLogReopen(SCLogOPIfaceCtx *op_iface_ctx)
502{
503 if (op_iface_ctx->file == NULL) {
504 return 0;
505 }
506
507 if (op_iface_ctx->file_d != NULL) {
508 fclose(op_iface_ctx->file_d);
509 }
510 op_iface_ctx->file_d = fopen(op_iface_ctx->file, "a");
511 if (op_iface_ctx->file_d == NULL) {
512 return -1;
513 }
514 return 0;
515}

517/**
518 * \brief Adds the global log_format to the outgoing buffer
519 *
520 * \param log_level log_level of the message that has to be logged
521 * \param msg Buffer containing the outgoing message
522 * \param file File_name from where the message originated
523 * \param function Function_name from where the message originated
524 * \param line Line_no from where the messaged originated
525 *
526 * \retval SC_OK on success; else an error code
527 */
528SCError SCLogMessage(const SCLogLevel log_level, const char *file,
529 const unsigned int line, const char *function,
530 const SCError error_code, const char *message)
531{
532 char buffer[SC_LOG_MAX_LOG_MSG_LEN] = "";
533 SCLogOPIfaceCtx *op_iface_ctx = NULL;
534
535 if (sc_log_module_initialized != 1) {
536 printf("Logging module not initialized. Call SCLogInitLogModule() "
537 "first before using the debug API\n");
538 return SC_OK;
539 }
540
541 /* get ts here so we log the same ts to each output */
542 struct timeval tval;
543 gettimeofday(&tval, NULL);
544
545 op_iface_ctx = sc_log_config->op_ifaces;
546 while (op_iface_ctx != NULL) {
547 if (log_level != SC_LOG_NOTSET && log_level > op_iface_ctx->log_level) {
548 op_iface_ctx = op_iface_ctx->next;
549 continue;
550 }
551
552 switch (op_iface_ctx->iface) {
553 case SC_LOG_OP_IFACE_CONSOLE:
554 if (SCLogMessageGetBuffer(&tval, op_iface_ctx->use_color, op_iface_ctx->type,
555 buffer, sizeof(buffer),
556 op_iface_ctx->log_format ?
557 op_iface_ctx->log_format : sc_log_config->log_format,
558 log_level, file, line, function,
559 error_code, message) == 0)
560 {
561 SCLogPrintToStream((log_level == SC_LOG_ERROR)? stderr: stdout, buffer);
562 }
563 break;
564 case SC_LOG_OP_IFACE_FILE:
565 if (SCLogMessageGetBuffer(&tval, 0, op_iface_ctx->type, buffer, sizeof(buffer),
566 op_iface_ctx->log_format ?
567 op_iface_ctx->log_format : sc_log_config->log_format,
568 log_level, file, line, function,
569 error_code, message) == 0)
570 {
571 int r = 0;
572 SCMutexLock(&op_iface_ctx->fp_mutex);
573 if (op_iface_ctx->rotation_flag) {
574 r = SCLogReopen(op_iface_ctx);
575 op_iface_ctx->rotation_flag = 0;
576 }
577 SCLogPrintToStream(op_iface_ctx->file_d, buffer);
578 SCMutexUnlock(&op_iface_ctx->fp_mutex);
579
580 /* report error outside of lock to avoid recursion */
581 if (r == -1) {
582 SCLogError(SC_ERR_FOPEN, "re-opening file \"%s\" failed: %s",
583 op_iface_ctx->file, strerror(errno));
584 }
585 }
586 break;
587 case SC_LOG_OP_IFACE_SYSLOG:
588 if (SCLogMessageGetBuffer(&tval, 0, op_iface_ctx->type, buffer, sizeof(buffer),
589 op_iface_ctx->log_format ?
590 op_iface_ctx->log_format : sc_log_config->log_format,
591 log_level, file, line, function,
592 error_code, message) == 0)
593 {
594 SCLogPrintToSyslog(SCLogMapLogLevelToSyslogLevel(log_level), buffer);
595 }
596 break;
597 default:
598 break;
599 }
600 op_iface_ctx = op_iface_ctx->next;
601 }
602 return SC_OK;
603}

605void SCLog(int x, const char *file, const char *func, const int line,
606 const char *fmt, ...)
607{
608 if (sc_log_global_log_level >= x &&
609 (sc_log_fg_filters_present == 0 ||
610 SCLogMatchFGFilterWL(file, func, line) == 1 ||
611 SCLogMatchFGFilterBL(file, func, line) == 1) &&
612 (sc_log_fd_filters_present == 0 ||
613 SCLogMatchFDFilter(func) == 1))
614 {
615 char msg[SC_LOG_MAX_LOG_MSG_LEN];
616 va_list ap;
617 va_start(ap, fmt);
618 vsnprintf(msg, sizeof(msg), fmt, ap);
619 va_end(ap);
620 SCLogMessage(x, file, line, func, SC_OK, msg);
621 }
622}

636 va_start(ap, fmt);

639 SCLogMessage(x, file, line, func, err, msg);

665 *
666 * \retval buffer Pointer to the newly created output_buffer
667 */
668SCLogOPBuffer *SCLogAllocLogOPBuffer(void)
669{
670 SCLogOPBuffer *buffer = NULL;
671 SCLogOPIfaceCtx *op_iface_ctx = NULL;
672 int i = 0;
673
674 if ( (buffer = SCMalloc(sc_log_config->op_ifaces_cnt *
675 sizeof(SCLogOPBuffer))) == NULL) {
676 FatalError(SC_ERR_FATAL,
677 "Fatal error encountered in SCLogAllocLogOPBuffer. Exiting...");
678 }
679
680 op_iface_ctx = sc_log_config->op_ifaces;
681 for (i = 0;
682 i < sc_log_config->op_ifaces_cnt;
683 i++, op_iface_ctx = op_iface_ctx->next) {
684 buffer[i].log_format = op_iface_ctx->log_format;
685 buffer[i].temp = buffer[i].msg;
686 }
687
688 return buffer;
689}

752#endif

755 goto error;
756 }
757

760 }

773 }
774 if (iface_ctx->log_format != NULL) {
775 SCFree((char *)iface_ctx->log_format);
776 iface_ctx->log_format = NULL;
777 }
778 if (iface_ctx->file_d != NULL) {

780 iface_ctx->file_d = NULL;

782 SCFree(iface_ctx);

1090 }

1329void SCLogLoadConfig(int daemon, int verbose, uint32_t userid, uint32_t groupid)
1330{
1331 ConfNode *outputs;
1332 SCLogInitData *sc_lid;
1333 int have_logging = 0;
1334 int max_level = 0;
1335 SCLogLevel min_level = 0;
1336
1337 /* If verbose logging was requested, set the minimum as
1338 * SC_LOG_NOTICE plus the extra verbosity. */
1339 if (verbose) {
1340 min_level = SC_LOG_NOTICE + verbose;
1341 }
1342
1343 outputs = ConfGetNode("logging.outputs");
1344 if (outputs == NULL) {
1345 SCLogDebug("No logging.output configuration section found.");
1346 return;
1347 }
1348
1349 sc_lid = SCLogAllocLogInitData();
1350 if (sc_lid == NULL) {
1351 SCLogDebug("Could not allocate memory for log init data");
1352 return;
1353 }
1354
1355 /* Get default log level and format. */
1356 const char *default_log_level_s = NULL;
1357 if (ConfGet("logging.default-log-level", &default_log_level_s) == 1) {
1358 SCLogLevel default_log_level =
1359 SCMapEnumNameToValue(default_log_level_s, sc_log_level_map);

1372 sc_lid->global_log_format = SCLogGetDefaultLogFormat();

1377 TAILQ_FOREACH(seq_node, &outputs->head, next) {

1379 SCLogOPIfaceCtx *op_iface_ctx = NULL;
1380 const char *format;
1381 const char *level_s;

1384 if (output == NULL)
1385 continue;
1386
1387 /* By default an output is enabled. */
1388 const char *enabled = ConfNodeLookupChildValue(output, "enabled");
1389 if (enabled != NULL && ConfValIsFalse(enabled))
1390 continue;
1391
1392 SCLogOPType type = SC_LOG_OP_TYPE_REGULAR;
1393 const char *type_s = ConfNodeLookupChildValue(output, "type");
1394 if (type_s != NULL) {
1395 if (strcmp(type_s, "regular") == 0)
1396 type = SC_LOG_OP_TYPE_REGULAR;
1397 else if (strcmp(type_s, "json") == 0) {
1398 type = SC_LOG_OP_TYPE_JSON;
1399 }
1400 }
1401
1402 /* if available use the log format setting for this output,
1403 * otherwise fall back to the global setting. */
1404 format = ConfNodeLookupChildValue(output, "format");
1405 if (format == NULL)
1406 format = sc_lid->global_log_format;
1407
1408 level_s = ConfNodeLookupChildValue(output, "level");
1409 if (level_s != NULL) {
1410 level = SCMapEnumNameToValue(level_s, sc_log_level_map);
1411 if (level == -1) {
1412 SCLogError(SC_ERR_INVALID_ARGUMENT, "Invalid log level: %s",
1413 level_s);
1414 exit(EXIT_FAILURE);
1415 }
1416 max_level = MAX(max_level, level);
1417 }
1418
1419 /* Increase the level of extra verbosity was requested. */
1420 level = MAX(min_level, level);
1421
1422 if (strcmp(output->name, "console") == 0) {
1423 op_iface_ctx = SCLogInitConsoleOPIface(format, level, type);
1424 }
1425 else if (strcmp(output->name, "file") == 0) {
1426 const char *filename = ConfNodeLookupChildValue(output, "filename");
1427 if (filename == NULL) {
1428 FatalError(SC_ERR_FATAL,
1429 "Logging to file requires a filename");
1430 }

1432 if (!(PathIsAbsolute(filename))) {
1433 path = SCLogGetLogFilename(filename);
1434 } else {
1435 path = SCStrdup(filename);
1436 }
1437 if (path == NULL)
1438 FatalError(SC_ERR_FATAL, "failed to setup output to file");
1439 have_logging = 1;
1440 op_iface_ctx = SCLogInitFileOPIface(path, userid, groupid, format, level, type);
1441 SCFree(path);
1442 }
1443 else if (strcmp(output->name, "syslog") == 0) {
1444 int facility = SC_LOG_DEF_SYSLOG_FACILITY;
1445 const char *facility_s = ConfNodeLookupChildValue(output,
1446 "facility");
1447 if (facility_s != NULL) {
1448 facility = SCMapEnumNameToValue(facility_s, SCSyslogGetFacilityMap());
1449 if (facility == -1) {
1450 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid syslog "
1451 "facility: \"%s\", now using \"%s\" as syslog "
1452 "facility", facility_s, SC_LOG_DEF_SYSLOG_FACILITY_STR);
1453 facility = SC_LOG_DEF_SYSLOG_FACILITY;
1454 }
1455 }
1456 SCLogDebug("Initializing syslog logging with format \"%s\"", format);
1457 have_logging = 1;
1458 op_iface_ctx = SCLogInitSyslogOPIface(facility, format, level, type);
1459 }
1460 else {
1461 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid logging method: %s, "
1462 "ignoring", output->name);
1463 }
1464 if (op_iface_ctx != NULL) {
1465 SCLogAppendOPIfaceCtx(op_iface_ctx, sc_lid);
1466 }
1467 }
1468
1469 if (daemon && (have_logging == 0)) {
1470 SCLogError(SC_ERR_MISSING_CONFIG_PARAM,
1471 "NO logging compatible with daemon mode selected,"
1472 " suricata won't be able to log. Please update "
1473 " 'logging.outputs' in the YAML.");
1474 }
1475
1476 /* Set the global log level to that of the max level used. */
1477 sc_lid->global_log_level = MAX(sc_lid->global_log_level, max_level);
1478 SCLogInitLogModule(sc_lid);
1479
1480 SCLogDebug("sc_log_global_log_level: %d", sc_log_global_log_level);
1481 SCLogDebug("sc_lc->log_format: %s", sc_log_config->log_format);
1482 SCLogDebug("SCLogSetOPFilter: filter: %s", sc_log_config->op_filter);
1483
1484 if (sc_lid != NULL)
1485 SCFree(sc_lid);
1486}
