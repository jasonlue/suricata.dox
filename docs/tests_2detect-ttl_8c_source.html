123/**
124 * \test DetectTtlSetupTest01 is a test for setting up an valid ttl values with
125 * valid "-" operator and include spaces arround the given values. In the
126 * test the values are setup with initializing the detection engine context
127 * setting up the signature itself.
128 */
129
130static int DetectTtlSetupTest01(void)
131{
132 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
133 FAIL_IF_NULL(de_ctx);
134 de_ctx->flags |= DE_QUIET;
135
136 Signature *s = DetectEngineAppendSig(
137 de_ctx, "alert ip any any -> any any (msg:\"with in ttl limit\"; ttl:1 - 3; sid:1;)");
138 FAIL_IF_NULL(s);
139 SigGroupBuild(de_ctx);
140 FAIL_IF_NULL(s->sm_arrays[DETECT_SM_LIST_MATCH]);
141 FAIL_IF_NULL(s->sm_arrays[DETECT_SM_LIST_MATCH]->ctx);
142 DetectU8Data *ttld = (DetectU8Data *)s->sm_arrays[DETECT_SM_LIST_MATCH]->ctx;
143
144 FAIL_IF_NOT(ttld->arg1 == 1);
145 FAIL_IF_NOT(ttld->arg2 == 3);
146 FAIL_IF_NOT(ttld->mode == DETECT_UINT_RA);
147 DetectEngineCtxFree(de_ctx);
148 PASS;
149}

151/**
152 * \test DetectTtlTestSig01 is a test for checking the working of ttl keyword
153 * by setting up the signature and later testing its working by matching
154 * the received packet against the sig.
155 */
156
157static int DetectTtlTestSig1(void)
158{
159 Packet *p = PacketGetFromAlloc();
160 FAIL_IF_NULL(p);
161 Signature *s = NULL;
162 ThreadVars th_v;
163 DetectEngineThreadCtx *det_ctx;
164 IPV4Hdr ip4h;
165
166 memset(&th_v, 0, sizeof(th_v));
167 memset(&ip4h, 0, sizeof(ip4h));
168
169 p->src.family = AF_INET;
170 p->dst.family = AF_INET;
171 p->proto = IPPROTO_TCP;
172 ip4h.ip_ttl = 15;
173 p->ip4h = &ip4h;
174
175 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
176 FAIL_IF_NULL(de_ctx);
177 de_ctx->flags |= DE_QUIET;
178
179 s = DetectEngineAppendSig(de_ctx,"alert ip any any -> any any (msg:\"with in ttl limit\"; ttl: >16; sid:1;)");
180 FAIL_IF_NULL(s);
181
182 s = DetectEngineAppendSig(de_ctx,"alert ip any any -> any any (msg:\"Less than 17\"; ttl: <17; sid:2;)");
183 FAIL_IF_NULL(s);
184
185 s = DetectEngineAppendSig(de_ctx,"alert ip any any -> any any (msg:\"Greater than 5\"; ttl:15; sid:3;)");
186 FAIL_IF_NULL(s);
187
188 s = DetectEngineAppendSig(de_ctx,"alert ip any any -> any any (msg:\"Equals tcp\"; ttl: 1-30; sid:4;)");
189 FAIL_IF_NULL(s);
190
191 SigGroupBuild(de_ctx);
192 DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
193
194 SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
195 FAIL_IF(PacketAlertCheck(p, 1));
196 FAIL_IF_NOT(PacketAlertCheck(p, 2));
197 FAIL_IF_NOT(PacketAlertCheck(p, 3));
198 FAIL_IF_NOT(PacketAlertCheck(p, 4));
199
200 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
201 DetectEngineCtxFree(de_ctx);
202
203 SCFree(p);
204 PASS;
205}
