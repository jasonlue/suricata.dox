32#define TEST_HEADER \
33 ThreadVars tv; \
34 memset(&tv, 0, sizeof(tv)); \
35 Flow f; \
36 memset(&f, 0, sizeof(f));
37
38#define TEST_RUN(buf, buflen, sig, match, steps) \
39{ \
40 DetectEngineCtx *de_ctx = DetectEngineCtxInit(); \
41 FAIL_IF_NULL(de_ctx); \
42 DetectEngineThreadCtx *det_ctx = NULL; \
43 char rule[2048]; \
44 snprintf(rule, sizeof(rule), "alert tcp any any -> any any (%s sid:1; rev:1;)", (sig)); \
45 Signature *s = DetectEngineAppendSig(de_ctx, rule); \
46 FAIL_IF_NULL(s); \
47 SigGroupBuild(de_ctx); \
48 DetectEngineThreadCtxInit(&tv, (void *)de_ctx, (void *)&det_ctx); \
49 FAIL_IF_NULL(det_ctx); \
50 int r = DetectEngineContentInspection(de_ctx, det_ctx, \
51 s, s->sm_arrays[DETECT_SM_LIST_PMATCH], NULL, &f, \
52 (uint8_t *)(buf), (buflen), 0, DETECT_CI_FLAGS_SINGLE, \
53 DETECT_ENGINE_CONTENT_INSPECTION_MODE_PAYLOAD); \
54 FAIL_IF_NOT(r == (match)); \
55 FAIL_IF_NOT(det_ctx->inspection_recursion_counter == (steps)); \
56 DetectEngineThreadCtxDeinit(&tv, det_ctx); \
57 DetectEngineCtxFree(de_ctx); \
58}
59#define TEST_FOOTER \
60 PASS
