76static int PacketPoolIsEmpty(PktPool *pool)
77{
78 /* Check local stack first. */
79 if (pool->head || pool->return_stack.head)
80 return 0;
81
82 return 1;

85void PacketPoolWait(void)
86{
87 PktPool *my_pool = GetThreadPacketPool();
88
89 if (PacketPoolIsEmpty(my_pool)) {
90 SCMutexLock(&my_pool->return_stack.mutex);
91 SC_ATOMIC_ADD(my_pool->return_stack.sync_now, 1);
92 SCCondWait(&my_pool->return_stack.cond, &my_pool->return_stack.mutex);
93 SCMutexUnlock(&my_pool->return_stack.mutex);
94 }
95
96 while(PacketPoolIsEmpty(my_pool))
97 cc_barrier();
98}

100/** \brief Wait until we have the requested amount of packets in the pool
101 *
102 * In some cases waiting for packets is undesirable. Especially when
103 * a wait would happen under a lock of some kind, other parts of the
104 * engine could have to wait.
105 *
106 * This function only returns when at least N packets are in our pool.
107 *
108 * If counting in our pool's main stack didn't give us the number we
109 * are seeking, we check if the return stack is filled and add those
110 * to our main stack. Then we retry.
111 *
112 * \param n number of packets needed
113 */
114void PacketPoolWaitForN(int n)
115{
116 PktPool *my_pool = GetThreadPacketPool();
117
118 while (1) {
119 PacketPoolWait();
120
121 /* count packets in our stack */
122 int i = 0;
123 Packet *p, *pp;
124 pp = p = my_pool->head;
125 while (p != NULL) {
126 if (++i == n)
127 return;
128
129 pp = p;
130 p = p->next;
131 }
132
133 /* check return stack, return to our pool and retry counting */
134 if (my_pool->return_stack.head != NULL) {
135 SCMutexLock(&my_pool->return_stack.mutex);
136 /* Move all the packets from the locked return stack to the local stack. */
137 if (pp) {
138 pp->next = my_pool->return_stack.head;
139 } else {
140 my_pool->head = my_pool->return_stack.head;
141 }
142 my_pool->return_stack.head = NULL;
143 SC_ATOMIC_RESET(my_pool->return_stack.sync_now);
144 SCMutexUnlock(&my_pool->return_stack.mutex);
145
146 /* or signal that we need packets and wait */
147 } else {
148 SCMutexLock(&my_pool->return_stack.mutex);
149 SC_ATOMIC_ADD(my_pool->return_stack.sync_now, 1);
150 SCCondWait(&my_pool->return_stack.cond, &my_pool->return_stack.mutex);
151 SCMutexUnlock(&my_pool->return_stack.mutex);
152 }
153 }
154}

160static void PacketPoolStorePacket(Packet *p)
161{
162 /* Clear the PKT_ALLOC flag, since that indicates to push back
163 * onto the ring buffer. */
164 p->flags &= ~PKT_ALLOC;
165 p->pool = GetThreadPacketPool();
166 p->ReleasePacket = PacketPoolReturnPacket;
167 PacketPoolReturnPacket(p);
168}

170static void PacketPoolGetReturnedPackets(PktPool *pool)
171{
172 SCMutexLock(&pool->return_stack.mutex);
173 /* Move all the packets from the locked return stack to the local stack. */
174 pool->head = pool->return_stack.head;
175 pool->return_stack.head = NULL;
176 SCMutexUnlock(&pool->return_stack.mutex);
177}

179/** \brief Get a new packet from the packet pool
180 *
181 * Only allocates from the thread's local stack, or mallocs new packets.
182 * If the local stack is empty, first move all the return stack packets to
183 * the local stack.
184 * \retval Packet pointer, or NULL on failure.
185 */
186Packet *PacketPoolGetPacket(void)
187{
188 PktPool *pool = GetThreadPacketPool();
189#ifdef DEBUG_VALIDATION
190 BUG_ON(pool->initialized == 0);
191 BUG_ON(pool->destroyed == 1);
192#endif /* DEBUG_VALIDATION */
193 if (pool->head) {
194 /* Stack is not empty. */
195 Packet *p = pool->head;
196 pool->head = p->next;
197 p->pool = pool;
198 PACKET_REINIT(p);
199 return p;
200 }
201
202 /* Local Stack is empty, so check the return stack, which requires
203 * locking. */
204 PacketPoolGetReturnedPackets(pool);
205
206 /* Try to allocate again. Need to check for not empty again, since the
207 * return stack might have been empty too.
208 */
209 if (pool->head) {
210 /* Stack is not empty. */
211 Packet *p = pool->head;
212 pool->head = p->next;
213 p->pool = pool;
214 PACKET_REINIT(p);
215 return p;
216 }
217
218 /* Failed to allocate a packet, so return NULL. */
219 /* Optionally, could allocate a new packet here. */
220 return NULL;
221}

223/** \brief Return packet to Packet pool
224 *
225 */
226void PacketPoolReturnPacket(Packet *p)
227{
228 PktPool *my_pool = GetThreadPacketPool();
229 PktPool *pool = p->pool;
230 if (pool == NULL) {
231 PacketFree(p);
232 return;
233 }
234
235 PACKET_RELEASE_REFS(p);
236
237#ifdef DEBUG_VALIDATION
238 BUG_ON(pool->initialized == 0);
239 BUG_ON(pool->destroyed == 1);
240 BUG_ON(my_pool->initialized == 0);
241 BUG_ON(my_pool->destroyed == 1);
242#endif /* DEBUG_VALIDATION */
243
244 if (pool == my_pool) {
245 /* Push back onto this thread's own stack, so no locking. */
246 p->next = my_pool->head;
247 my_pool->head = p;
248 } else {
249 PktPool *pending_pool = my_pool->pending_pool;
250 if (pending_pool == NULL) {
251 /* No pending packet, so store the current packet. */
252 p->next = NULL;
253 my_pool->pending_pool = pool;
254 my_pool->pending_head = p;
255 my_pool->pending_tail = p;
256 my_pool->pending_count = 1;
257 } else if (pending_pool == pool) {
258 /* Another packet for the pending pool list. */
259 p->next = my_pool->pending_head;
260 my_pool->pending_head = p;
261 my_pool->pending_count++;
262 if (SC_ATOMIC_GET(pool->return_stack.sync_now) || my_pool->pending_count > max_pending_return_packets) {
263 /* Return the entire list of pending packets. */
264 SCMutexLock(&pool->return_stack.mutex);
265 my_pool->pending_tail->next = pool->return_stack.head;
266 pool->return_stack.head = my_pool->pending_head;
267 SC_ATOMIC_RESET(pool->return_stack.sync_now);
268 SCMutexUnlock(&pool->return_stack.mutex);
269 SCCondSignal(&pool->return_stack.cond);
270 /* Clear the list of pending packets to return. */
271 my_pool->pending_pool = NULL;
272 my_pool->pending_head = NULL;
273 my_pool->pending_tail = NULL;
274 my_pool->pending_count = 0;
275 }
276 } else {
277 /* Push onto return stack for this pool */
278 SCMutexLock(&pool->return_stack.mutex);
279 p->next = pool->return_stack.head;
280 pool->return_stack.head = p;
281 SC_ATOMIC_RESET(pool->return_stack.sync_now);
282 SCMutexUnlock(&pool->return_stack.mutex);
283 SCCondSignal(&pool->return_stack.cond);
284 }
285 }

303void PacketPoolInit(void)
304{
305 extern intmax_t max_pending_packets;
306
307 PktPool *my_pool = GetThreadPacketPool();
308
309#ifdef DEBUG_VALIDATION
310 BUG_ON(my_pool->initialized);
311 my_pool->initialized = 1;
312 my_pool->destroyed = 0;
313#endif /* DEBUG_VALIDATION */
314
315 SCMutexInit(&my_pool->return_stack.mutex, NULL);
316 SCCondInit(&my_pool->return_stack.cond, NULL);
317 SC_ATOMIC_INIT(my_pool->return_stack.sync_now);
318
319 /* pre allocate packets */
320 SCLogDebug("preallocating packets... packet size %" PRIuMAX "",
321 (uintmax_t)SIZE_OF_PACKET);
322 int i = 0;
323 for (i = 0; i < max_pending_packets; i++) {
324 Packet *p = PacketGetFromAlloc();
325 if (unlikely(p == NULL)) {
326 FatalError(SC_ERR_FATAL,
327 "Fatal error encountered while allocating a packet. Exiting...");
328 }
329 PacketPoolStorePacket(p);
330 }
331
332 //SCLogInfo("preallocated %"PRIiMAX" packets. Total memory %"PRIuMAX"",
333 // max_pending_packets, (uintmax_t)(max_pending_packets*SIZE_OF_PACKET));
334}

336void PacketPoolDestroy(void)
337{
338 Packet *p = NULL;
339 PktPool *my_pool = GetThreadPacketPool();
340
341#ifdef DEBUG_VALIDATION
342 BUG_ON(my_pool && my_pool->destroyed);
343#endif /* DEBUG_VALIDATION */
344
345 if (my_pool && my_pool->pending_pool != NULL) {
346 p = my_pool->pending_head;
347 while (p) {
348 Packet *next_p = p->next;
349 PacketFree(p);
350 p = next_p;
351 my_pool->pending_count--;
352 }
353#ifdef DEBUG_VALIDATION
354 BUG_ON(my_pool->pending_count);
355#endif /* DEBUG_VALIDATION */
356 my_pool->pending_pool = NULL;
357 my_pool->pending_head = NULL;
358 my_pool->pending_tail = NULL;
359 }
360
361 while ((p = PacketPoolGetPacket()) != NULL) {
362 PacketFree(p);
363 }
364
365#ifdef DEBUG_VALIDATION
366 my_pool->initialized = 0;
367 my_pool->destroyed = 1;
368#endif /* DEBUG_VALIDATION */
369}

376void TmqhOutputPacketpool(ThreadVars *t, Packet *p)
377{
378 bool proot = false;
379
380 SCEnter();
381 SCLogDebug("Packet %p, p->root %p, alloced %s", p, p->root, p->flags & PKT_ALLOC ? "true" : "false");
382
383#ifdef DEBUG_VALIDATION
384 /* Check that the drop reason has been set, if we have a drop.
385 * Use macro to avoid testing Packet::root. */
386 if (PACKET_TEST_ACTION(p, ACTION_DROP)) {
387 DEBUG_VALIDATE_BUG_ON((p)->drop_reason == PKT_DROP_REASON_NOT_SET);
388 }
389#endif
390
391 if (IS_TUNNEL_PKT(p)) {
392 SCLogDebug("Packet %p is a tunnel packet: %s",
393 p,p->root ? "upper layer" : "tunnel root");
394
395 /* get a lock to access root packet fields */
396 SCMutex *m = p->root ? &p->root->tunnel_mutex : &p->tunnel_mutex;
397 SCMutexLock(m);
398
399 if (IS_TUNNEL_ROOT_PKT(p)) {
400 SCLogDebug("IS_TUNNEL_ROOT_PKT == TRUE");
401 const uint16_t outstanding = TUNNEL_PKT_TPR(p) - TUNNEL_PKT_RTV(p);
402 SCLogDebug("root pkt: outstanding %u", outstanding);
403 if (outstanding == 0) {
404 SCLogDebug("no tunnel packets outstanding, no more tunnel "
405 "packet(s) depending on this root");
406 /* if this packet is the root and there are no
407 * more tunnel packets to consider
408 *
409 * return it to the pool */
410 } else {
411 SCLogDebug("tunnel root Packet %p: outstanding > 0, so "
412 "packets are still depending on this root, setting "
413 "SET_TUNNEL_PKT_VERDICTED", p);
414 /* if this is the root and there are more tunnel
415 * packets, return this to the pool. It's still referenced
416 * by the tunnel packets, and we will return it
417 * when we handle them */
418 SET_TUNNEL_PKT_VERDICTED(p);
419
420 PACKET_PROFILING_END(p);
421 SCMutexUnlock(m);
422 SCReturn;
423 }
424 } else {
425 SCLogDebug("NOT IS_TUNNEL_ROOT_PKT, so tunnel pkt");
426
427 TUNNEL_INCR_PKT_RTV_NOLOCK(p);
428 const uint16_t outstanding = TUNNEL_PKT_TPR(p) - TUNNEL_PKT_RTV(p);
429 SCLogDebug("tunnel pkt: outstanding %u", outstanding);
430 /* all tunnel packets are processed except us. Root already
431 * processed. So return tunnel pkt and root packet to the
432 * pool. */
433 if (outstanding == 0 &&
434 p->root && IS_TUNNEL_PKT_VERDICTED(p->root))
435 {
436 SCLogDebug("root verdicted == true && no outstanding");
437
438 /* handle freeing the root as well*/
439 SCLogDebug("setting proot = 1 for root pkt, p->root %p "
440 "(tunnel packet %p)", p->root, p);
441 proot = true;
442
443 /* fall through */
444
445 } else {
446 /* root not ready yet, or not the last tunnel packet,
447 * so get rid of the tunnel pkt only */
448
449 SCLogDebug("NOT IS_TUNNEL_PKT_VERDICTED (%s) || "
450 "outstanding > 0 (%u)",
451 (p->root && IS_TUNNEL_PKT_VERDICTED(p->root)) ? "true" : "false",
452 outstanding);
453
454 /* fall through */
455 }
456 }
457 SCMutexUnlock(m);
458
459 SCLogDebug("tunnel stuff done, move on (proot %d)", proot);
460 }
461
462 /* we're done with the tunnel root now as well */
463 if (proot == true) {
464 SCLogDebug("getting rid of root pkt... alloc'd %s", p->root->flags & PKT_ALLOC ? "true" : "false");
465
466 PACKET_RELEASE_REFS(p->root);
467 p->root->ReleasePacket(p->root);
468 p->root = NULL;
469 }
470
471 PACKET_PROFILING_END(p);
472
473 PACKET_RELEASE_REFS(p);
474 p->ReleasePacket(p);
475
476 SCReturn;
477}

479/**
480 * \brief Release all the packets in the queue back to the packetpool. Mainly
481 * used by threads that have failed, and wants to return the packets back
482 * to the packetpool.
483 *
484 * \param pq Pointer to the packetqueue from which the packets have to be
485 * returned back to the packetpool
486 *
487 * \warning this function assumes that the pq does not use locking
488 */
489void TmqhReleasePacketsToPacketPool(PacketQueue *pq)
490{
491 Packet *p = NULL;
492
493 if (pq == NULL)
494 return;
495
496 while ( (p = PacketDequeue(pq)) != NULL)
497 TmqhOutputPacketpool(NULL, p);
498
499 return;
500}

502/** number of packets to keep reserved when calculating the the pending
503 * return packets count. This assumes we need at max 10 packets in one
504 * PacketPoolWaitForN call. The actual number is 9 now, so this has a
505 * bit of margin. */
506#define RESERVED_PACKETS 10
507
508/**
509 * \brief Set the max_pending_return_packets value
510 *
511 * Set it to the max pending packets value, devided by the number
512 * of lister threads. Normally, in autofp these are the stream/detect/log
513 * worker threads.
514 *
515 * The max_pending_return_packets value needs to stay below the packet
516 * pool size of the 'producers' (normally pkt capture threads but also
517 * flow timeout injection ) to avoid a deadlock where all the 'workers'
518 * keep packets in their return pools, while the capture thread can't
519 * continue because its pool is empty.
520 */
521void PacketPoolPostRunmodes(void)
522{
523 extern intmax_t max_pending_packets;
524 intmax_t pending_packets = max_pending_packets;
525 if (pending_packets < RESERVED_PACKETS) {
526 FatalError(SC_ERR_INVALID_ARGUMENT, "'max-pending-packets' setting "
527 "must be at least %d", RESERVED_PACKETS);
528 }
529 uint32_t threads = TmThreadCountThreadsByTmmFlags(TM_FLAG_DETECT_TM);
530 if (threads == 0)
531 return;
532
533 uint32_t packets = (pending_packets / threads) - 1;
534 if (packets < max_pending_return_packets)
535 max_pending_return_packets = packets;
536
537 /* make sure to have a margin in the return logic */
538 if (max_pending_return_packets >= RESERVED_PACKETS)
539 max_pending_return_packets -= RESERVED_PACKETS;
540
541 SCLogDebug("detect threads %u, max packets %u, max_pending_return_packets %u",
542 threads, packets, max_pending_return_packets);
543}
