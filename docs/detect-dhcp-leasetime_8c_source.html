29/**
30 * \internal
31 * \brief Function to match leasetime of a TX
32 *
33 * \param t Pointer to thread vars.
34 * \param det_ctx Pointer to the pattern matcher thread.
35 * \param f Pointer to the current flow.
36 * \param flags Flags.
37 * \param state App layer state.
38 * \param s Pointer to the Signature.
39 * \param m Pointer to the sigmatch that we will cast into
40 * DetectU64Data.
41 *
42 * \retval 0 no match.
43 * \retval 1 match.
44 */
45static int DetectDHCPLeaseTimeMatch(DetectEngineThreadCtx *det_ctx, Flow *f, uint8_t flags,
46 void *state, void *txv, const Signature *s, const SigMatchCtx *ctx)
47{
48 SCEnter();
49
50 uint64_t leasetime;
51 if (rs_dhcp_tx_get_leasetime(txv, &leasetime)) {
52 const DetectU64Data *dd = (const DetectU64Data *)ctx;
53 if (DetectU64Match(leasetime, dd)) {
54 SCReturnInt(1);
55 }
56 }
57 SCReturnInt(0);
58}

71/**
72 * \brief Function to add the parsed dhcp leasetime field into the current signature.
73 *
74 * \param de_ctx Pointer to the Detection Engine Context.
75 * \param s Pointer to the Current Signature.
76 * \param rawstr Pointer to the user provided flags options.
77 * \param type Defines if this is notBefore or notAfter.
78 *
79 * \retval 0 on Success.
80 * \retval -1 on Failure.
81 */
82static int DetectDHCPLeaseTimeSetup(DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
83{
84 if (DetectSignatureSetAppProto(s, ALPROTO_DHCP) != 0)
85 return -1;
86
87 DetectU64Data *dd = DetectU64Parse(rawstr);
88 if (dd == NULL) {
89 SCLogError(SC_ERR_INVALID_ARGUMENT, "Parsing \'%s\' failed for %s", rawstr,
90 sigmatch_table[DETECT_AL_DHCP_LEASETIME].name);
91 return -1;
92 }
93
94 /* okay so far so good, lets get this into a SigMatch
95 * and put it in the Signature. */
96 SigMatch *sm = SigMatchAlloc();
97 if (sm == NULL)
98 goto error;
99
100 sm->type = DETECT_AL_DHCP_LEASETIME;
101 sm->ctx = (void *)dd;
102
103 SigMatchAppendSMToList(s, sm, g_buffer_id);
104 return 0;
105
106error:
107 DetectDHCPLeaseTimeFree(de_ctx, dd);
108 return -1;
109}
