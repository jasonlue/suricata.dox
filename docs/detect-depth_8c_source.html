49{

64static int DetectDepthSetup (DetectEngineCtx *de_ctx, Signature *s, const char *depthstr)
65{
66 const char *str = depthstr;
67 SigMatch *pm = NULL;
68 int ret = -1;
69
70 /* retrive the sm to apply the depth against */
71 pm = DetectGetLastSMFromLists(s, DETECT_CONTENT, -1);
72 if (pm == NULL) {
73 SCLogError(SC_ERR_DEPTH_MISSING_CONTENT, "depth needs "
74 "preceding content, uricontent option, http_client_body, "
75 "http_server_body, http_header option, http_raw_header option, "
76 "http_method option, http_cookie, http_raw_uri, "
77 "http_stat_msg, http_stat_code, http_user_agent, "
78 "http_host, http_raw_host or "
79 "file_data/dce_stub_data sticky buffer options.");
80 goto end;
81 }
82
83 /* verify other conditions. */
84 DetectContentData *cd = (DetectContentData *)pm->ctx;
85
86 if (cd->flags & DETECT_CONTENT_DEPTH) {
87 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use multiple depths for the same content.");
88 goto end;
89 }
90 if ((cd->flags & DETECT_CONTENT_WITHIN) || (cd->flags & DETECT_CONTENT_DISTANCE)) {
91 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use a relative "
92 "keyword like within/distance with a absolute "
93 "relative keyword like depth/offset for the same "
94 "content." );
95 goto end;
96 }
97 if (cd->flags & DETECT_CONTENT_NEGATED && cd->flags & DETECT_CONTENT_FAST_PATTERN) {
98 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
99 "negated keyword set along with 'fast_pattern'.");
100 goto end;
101 }
102 if (cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
103 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
104 "keyword set along with 'fast_pattern:only;'.");
105 goto end;
106 }
107 if (str[0] != '-' && isalpha((unsigned char)str[0])) {
108 DetectByteIndexType index;
109 if (!DetectByteRetrieveSMVar(str, s, &index)) {
110 SCLogError(SC_ERR_INVALID_SIGNATURE, "unknown byte_ keyword var "
111 "seen in depth - %s.", str);
112 goto end;
113 }
114 cd->depth = index;
115 cd->flags |= DETECT_CONTENT_DEPTH_VAR;
116 } else {
117 if (StringParseUint16(&cd->depth, 0, 0, str) < 0)
118 {
119 SCLogError(SC_ERR_INVALID_SIGNATURE,
120 "invalid value for depth: %s.", str);
121 goto end;
122 }
123
124 if (cd->depth < cd->content_len) {
125 SCLogError(SC_ERR_INVALID_SIGNATURE, "depth:%u smaller than "
126 "content of len %u.", cd->depth, cd->content_len);
127 return -1;
128 }
129 /* Now update the real limit, as depth is relative to the offset */
130 cd->depth += cd->offset;
131 }
132 cd->flags |= DETECT_CONTENT_DEPTH;
133
134 ret = 0;
135 end:
136 return ret;
137}

139static int DetectStartsWithSetup (DetectEngineCtx *de_ctx, Signature *s, const char *unused)
140{
141 SigMatch *pm = NULL;
142 int ret = -1;
143
144 /* retrieve the sm to apply the depth against */
145 pm = DetectGetLastSMFromLists(s, DETECT_CONTENT, -1);
146 if (pm == NULL) {
147 SCLogError(SC_ERR_DEPTH_MISSING_CONTENT, "startswith needs a "
148 "preceding content option.");
149 goto end;
150 }
151
152 /* verify other conditions. */
153 DetectContentData *cd = (DetectContentData *)pm->ctx;
154
155 if (cd->flags & DETECT_CONTENT_DEPTH) {
156 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use multiple "
157 "depth/startswith settings for the same content.");
158 goto end;
159 }
160 if ((cd->flags & DETECT_CONTENT_WITHIN) || (cd->flags & DETECT_CONTENT_DISTANCE)) {
161 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't use a relative "
162 "keyword like within/distance with a absolute "
163 "relative keyword like depth/offset for the same "
164 "content.");
165 goto end;
166 }
167 if (cd->flags & DETECT_CONTENT_NEGATED && cd->flags & DETECT_CONTENT_FAST_PATTERN) {
168 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
169 "negated keyword set along with a 'fast_pattern'.");
170 goto end;
171 }
172 if (cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
173 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
174 "keyword set along with 'fast_pattern:only;'.");
175 goto end;
176 }
177 if (cd->flags & DETECT_CONTENT_OFFSET) {
178 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't mix offset "
179 "with startswith.");
180 goto end;
181 }
182
183 cd->depth = cd->content_len;
184 cd->flags |= DETECT_CONTENT_DEPTH;
185 cd->flags |= DETECT_CONTENT_STARTS_WITH;
186
187 ret = 0;
188 end:
189 return ret;
190}
