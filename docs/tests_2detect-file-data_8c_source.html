109static int DetectFiledataIsdataatParseTest1(void)
110{
111 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
112 FAIL_IF_NULL(de_ctx);
113 de_ctx->flags |= DE_QUIET;
114
115 Signature *s = DetectEngineAppendSig(de_ctx,
116 "alert tcp any any -> any any ("
117 "file_data; content:\"one\"; "
118 "isdataat:!4,relative; sid:1;)");
119 FAIL_IF_NULL(s);
120
121 SigMatch *sm = s->init_data->smlists[g_file_data_buffer_id];
122 FAIL_IF_NULL(sm);
123 FAIL_IF_NOT(sm->type == DETECT_CONTENT);
124 sm = sm->next;
125 FAIL_IF_NULL(sm);
126 FAIL_IF_NOT(sm->type == DETECT_ISDATAAT);
127
128 DetectIsdataatData *data = (DetectIsdataatData *)sm->ctx;
129 FAIL_IF_NOT(data->flags & ISDATAAT_RELATIVE);
130 FAIL_IF_NOT(data->flags & ISDATAAT_NEGATED);
131 FAIL_IF(data->flags & ISDATAAT_RAWBYTES);
132
133 DetectEngineCtxFree(de_ctx);
134 PASS;
135}
