97int DetectFlowvarMatch (DetectEngineThreadCtx *det_ctx, Packet *p,
98 const Signature *s, const SigMatchCtx *ctx)
99{
100 int ret = 0;
101 DetectFlowvarData *fd = (DetectFlowvarData *)ctx;
102
103 FlowVar *fv = FlowVarGet(p->flow, fd->idx);
104 if (fv != NULL) {
105 uint8_t *ptr = SpmSearch(fv->data.fv_str.value,
106 fv->data.fv_str.value_len,
107 fd->content, fd->content_len);
108 if (ptr != NULL)
109 ret = 1;
110 }
111
112 return ret;
113}

115static int DetectFlowvarSetup (DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
116{
117 DetectFlowvarData *fd = NULL;
118 SigMatch *sm = NULL;
119 char varname[64], varcontent[64];
120 int ret = 0, res = 0;
121 size_t pcre2len;
122 uint8_t *content = NULL;
123 uint16_t contentlen = 0;
124 uint32_t contentflags = s->init_data->negated ? DETECT_CONTENT_NEGATED : 0;
125
126 ret = DetectParsePcreExec(&parse_regex, rawstr, 0, 0);
127 if (ret != 3) {
128 SCLogError(SC_ERR_PCRE_MATCH, "\"%s\" is not a valid setting for flowvar.", rawstr);
129 return -1;
130 }
131
132 pcre2len = sizeof(varname);
133 res = pcre2_substring_copy_bynumber(parse_regex.match, 1, (PCRE2_UCHAR8 *)varname, &pcre2len);
134 if (res < 0) {
135 SCLogError(SC_ERR_PCRE_COPY_SUBSTRING, "pcre2_substring_copy_bynumber failed");
136 return -1;
137 }
138
139 pcre2len = sizeof(varcontent);
140 res = pcre2_substring_copy_bynumber(
141 parse_regex.match, 2, (PCRE2_UCHAR8 *)varcontent, &pcre2len);
142 if (res < 0) {
143 SCLogError(SC_ERR_PCRE_COPY_SUBSTRING, "pcre2_substring_copy_bynumber failed");
144 return -1;
145 }
146
147 int varcontent_index = 0;
148 if (strlen(varcontent) >= 2) {
149 if (varcontent[0] == '"')
150 varcontent_index++;
151 if (varcontent[strlen(varcontent)-1] == '"')
152 varcontent[strlen(varcontent)-1] = '\0';
153 }
154 SCLogDebug("varcontent %s", &varcontent[varcontent_index]);
155
156 res = DetectContentDataParse("flowvar", &varcontent[varcontent_index], &content, &contentlen);
157 if (res == -1)
158 goto error;
159
160 fd = SCMalloc(sizeof(DetectFlowvarData));
161 if (unlikely(fd == NULL))
162 goto error;
163 memset(fd, 0x00, sizeof(*fd));
164
165 fd->content = SCMalloc(contentlen);
166 if (unlikely(fd->content == NULL))
167 goto error;
168
169 memcpy(fd->content, content, contentlen);
170 fd->content_len = contentlen;
171 fd->flags = contentflags;
172
173 fd->name = SCStrdup(varname);
174 if (unlikely(fd->name == NULL))
175 goto error;
176 fd->idx = VarNameStoreSetupAdd(varname, VAR_TYPE_FLOW_VAR);
177
178 /* Okay so far so good, lets get this into a SigMatch
179 * and put it in the Signature. */
180 sm = SigMatchAlloc();
181 if (unlikely(sm == NULL))
182 goto error;
183
184 sm->type = DETECT_FLOWVAR;
185 sm->ctx = (SigMatchCtx *)fd;
186
187 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
188
189 SCFree(content);
190 return 0;
191
192error:
193 if (fd != NULL)
194 DetectFlowvarDataFree(de_ctx, fd);
195 if (sm != NULL)
196 SCFree(sm);
197 if (content != NULL)
198 SCFree(content);
199 return -1;
200}

222/** \brief Store flowvar in det_ctx so we can exec it post-match */
223int DetectVarStoreMatch(DetectEngineThreadCtx *det_ctx,
224 uint32_t idx,
225 uint8_t *buffer, uint16_t len, int type)
226{
227 DetectVarList *fs = det_ctx->varlist;
228
229 /* first check if we have had a previous match for this idx */
230 for ( ; fs != NULL; fs = fs->next) {
231 if (fs->idx == idx) {
232 /* we're replacing the older store */
233 SCFree(fs->buffer);
234 fs->buffer = NULL;
235 break;
236 }
237 }
238
239 if (fs == NULL) {
240 fs = SCCalloc(1, sizeof(*fs));
241 if (unlikely(fs == NULL))
242 return -1;
243
244 fs->idx = idx;
245
246 fs->next = det_ctx->varlist;
247 det_ctx->varlist = fs;
248 }
249
250 fs->len = len;
251 fs->type = type;
252 fs->buffer = buffer;
253 return 0;
254}

259int DetectFlowvarPostMatchSetup(DetectEngineCtx *de_ctx, Signature *s, uint32_t idx)
260{
261 SigMatch *sm = NULL;
262 DetectFlowvarData *fv = NULL;
263
264 fv = SCMalloc(sizeof(DetectFlowvarData));
265 if (unlikely(fv == NULL))
266 goto error;
267 memset(fv, 0x00, sizeof(*fv));
268
269 /* we only need the idx */
270 fv->idx = idx;
271
272 sm = SigMatchAlloc();
273 if (unlikely(sm == NULL))
274 goto error;
275
276 sm->type = DETECT_FLOWVAR_POSTMATCH;
277 sm->ctx = (SigMatchCtx *)fv;
278
279 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_POSTMATCH);
280 return 0;
281error:
282 if (fv != NULL)
283 DetectFlowvarDataFree(de_ctx, fv);
284 return -1;
285}

287/** \internal
288 * \brief post-match func to store flowvars in the flow
289 * \param sm sigmatch containing the idx to store
290 * \retval 1 or -1 in case of error
291 */
292static int DetectFlowvarPostMatch(
293 DetectEngineThreadCtx *det_ctx,
294 Packet *p, const Signature *s, const SigMatchCtx *ctx)
295{
296 DetectVarList *fs, *prev;
297 const DetectFlowvarData *fd;
298
299 if (det_ctx->varlist == NULL)
300 return 1;
301
302 fd = (const DetectFlowvarData *)ctx;
303
304 prev = NULL;
305 fs = det_ctx->varlist;
306 while (fs != NULL) {
307 if (fd->idx == 0 || fd->idx == fs->idx) {
308 SCLogDebug("adding to the flow %u:", fs->idx);
309 //PrintRawDataFp(stdout, fs->buffer, fs->len);
310
311 if (fs->type == DETECT_VAR_TYPE_FLOW_POSTMATCH && p && p->flow) {
312 FlowVarAddIdValue(p->flow, fs->idx, fs->buffer, fs->len);
313 /* memory at fs->buffer is now the responsibility of
314 * the flowvar code. */
315 } else if (fs->type == DETECT_VAR_TYPE_PKT_POSTMATCH && fs->key && p) {
316 /* pkt key/value */
317 if (PktVarAddKeyValue(p, (uint8_t *)fs->key, fs->key_len,
318 (uint8_t *)fs->buffer, fs->len) == -1)
319 {
320 SCFree(fs->key);
321 SCFree(fs->buffer);
322 /* the rest of fs is freed below */
323 }
324 } else if (fs->type == DETECT_VAR_TYPE_PKT_POSTMATCH && p) {
325 if (PktVarAdd(p, fs->idx, fs->buffer, fs->len) == -1) {
326 SCFree(fs->buffer);
327 /* the rest of fs is freed below */
328 }
329 }
330
331 if (fs == det_ctx->varlist) {
332 det_ctx->varlist = fs->next;
333 SCFree(fs);
334 fs = det_ctx->varlist;
335 } else {
336 prev->next = fs->next;
337 SCFree(fs);
338 fs = prev->next;
339 }
340 } else {
341 prev = fs;
342 fs = fs->next;
343 }
344 }
345 return 1;
346}
