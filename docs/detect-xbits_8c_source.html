166static int DetectXbitMatch (DetectEngineThreadCtx *det_ctx, Packet *p, const Signature *s, const SigMatchCtx *ctx)
167{
168 const DetectXbitsData *fd = (const DetectXbitsData *)ctx;
169 if (fd == NULL)
170 return 0;
171
172 switch (fd->type) {
173 case VAR_TYPE_HOST_BIT:
174 return DetectXbitMatchHost(p, (const DetectXbitsData *)fd);
175 break;
176 case VAR_TYPE_IPPAIR_BIT:
177 return DetectXbitMatchIPPair(p, (const DetectXbitsData *)fd);
178 break;
179 default:
180 break;
181 }
182 return 0;
183}

185/** \internal
186 * \brief parse xbits rule options
187 * \retval 0 ok
188 * \retval -1 bad
189 * \param[out] cdout return DetectXbitsData structure or NULL if noalert
190 */
191static int DetectXbitParse(DetectEngineCtx *de_ctx,
192 const char *rawstr, DetectXbitsData **cdout)
193{
194 DetectXbitsData *cd = NULL;
195 uint8_t fb_cmd = 0;
196 uint8_t hb_dir = 0;
197 int ret = 0, res = 0;
198 size_t pcre2len;
199 char fb_cmd_str[16] = "", fb_name[256] = "";
200 char hb_dir_str[16] = "";
201 enum VarTypes var_type = VAR_TYPE_NOT_SET;
202 uint32_t expire = DETECT_XBITS_EXPIRE_DEFAULT;
203
204 ret = DetectParsePcreExec(&parse_regex, rawstr, 0, 0);
205 if (ret != 2 && ret != 3 && ret != 4 && ret != 5) {
206 SCLogError(SC_ERR_PCRE_MATCH, "\"%s\" is not a valid setting for xbits.", rawstr);
207 return -1;
208 }
209 SCLogDebug("ret %d, %s", ret, rawstr);
210 pcre2len = sizeof(fb_cmd_str);
211 res = pcre2_substring_copy_bynumber(
212 parse_regex.match, 1, (PCRE2_UCHAR8 *)fb_cmd_str, &pcre2len);
213 if (res < 0) {
214 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
215 return -1;
216 }
217
218 if (ret >= 3) {
219 pcre2len = sizeof(fb_name);
220 res = pcre2_substring_copy_bynumber(
221 parse_regex.match, 2, (PCRE2_UCHAR8 *)fb_name, &pcre2len);
222 if (res < 0) {
223 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
224 return -1;
225 }
226 if (ret >= 4) {
227 pcre2len = sizeof(hb_dir_str);
228 res = pcre2_substring_copy_bynumber(
229 parse_regex.match, 3, (PCRE2_UCHAR8 *)hb_dir_str, &pcre2len);
230 if (res < 0) {
231 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
232 return -1;
233 }
234 SCLogDebug("hb_dir_str %s", hb_dir_str);
235 if (strlen(hb_dir_str) > 0) {
236 if (strcmp(hb_dir_str, "ip_src") == 0) {
237 hb_dir = DETECT_XBITS_TRACK_IPSRC;
238 var_type = VAR_TYPE_HOST_BIT;
239 } else if (strcmp(hb_dir_str, "ip_dst") == 0) {
240 hb_dir = DETECT_XBITS_TRACK_IPDST;
241 var_type = VAR_TYPE_HOST_BIT;
242 } else if (strcmp(hb_dir_str, "ip_pair") == 0) {
243 hb_dir = DETECT_XBITS_TRACK_IPPAIR;
244 var_type = VAR_TYPE_IPPAIR_BIT;
245 } else {
246 // TODO
247 return -1;
248 }
249 }
250
251 if (ret >= 5) {
252 char expire_str[16] = "";
253 pcre2len = sizeof(expire_str);
254 res = pcre2_substring_copy_bynumber(
255 parse_regex.match, 4, (PCRE2_UCHAR8 *)expire_str, &pcre2len);
256 if (res < 0) {
257 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
258 return -1;
259 }
260 SCLogDebug("expire_str %s", expire_str);
261 if (StringParseUint32(&expire, 10, 0, (const char *)expire_str) < 0) {
262 SCLogError(SC_ERR_INVALID_VALUE, "Invalid value for "
263 "expire: \"%s\"", expire_str);
264 return -1;
265 }
266 if (expire == 0) {
267 SCLogError(SC_ERR_INVALID_VALUE, "expire must be bigger than 0");
268 return -1;
269 }
270 SCLogDebug("expire %d", expire);
271 }
272 }
273 }
274
275 if (strcmp(fb_cmd_str,"noalert") == 0) {
276 fb_cmd = DETECT_XBITS_CMD_NOALERT;
277 } else if (strcmp(fb_cmd_str,"isset") == 0) {
278 fb_cmd = DETECT_XBITS_CMD_ISSET;
279 } else if (strcmp(fb_cmd_str,"isnotset") == 0) {
280 fb_cmd = DETECT_XBITS_CMD_ISNOTSET;
281 } else if (strcmp(fb_cmd_str,"set") == 0) {
282 fb_cmd = DETECT_XBITS_CMD_SET;
283 } else if (strcmp(fb_cmd_str,"unset") == 0) {
284 fb_cmd = DETECT_XBITS_CMD_UNSET;
285 } else if (strcmp(fb_cmd_str,"toggle") == 0) {
286 fb_cmd = DETECT_XBITS_CMD_TOGGLE;
287 } else {
288 SCLogError(SC_ERR_UNKNOWN_VALUE, "xbits action \"%s\" is not supported.", fb_cmd_str);
289 return -1;
290 }
291
292 switch (fb_cmd) {
293 case DETECT_XBITS_CMD_NOALERT: {
294 if (strlen(fb_name) != 0)
295 return -1;
296 /* return ok, cd is NULL. Flag sig. */
297 *cdout = NULL;
298 return 0;
299 }
300 case DETECT_XBITS_CMD_ISNOTSET:
301 case DETECT_XBITS_CMD_ISSET:
302 case DETECT_XBITS_CMD_SET:
303 case DETECT_XBITS_CMD_UNSET:
304 case DETECT_XBITS_CMD_TOGGLE:
305 if (strlen(fb_name) == 0)
306 return -1;
307 break;
308 }
309
310 cd = SCMalloc(sizeof(DetectXbitsData));
311 if (unlikely(cd == NULL))
312 return -1;
313
314 cd->idx = VarNameStoreSetupAdd(fb_name, var_type);
315 cd->cmd = fb_cmd;
316 cd->tracker = hb_dir;
317 cd->type = var_type;
318 cd->expire = expire;
319
320 SCLogDebug("idx %" PRIu32 ", cmd %s, name %s",
321 cd->idx, fb_cmd_str, strlen(fb_name) ? fb_name : "(none)");
322
323 *cdout = cd;
324 return 0;
325}

327int DetectXbitSetup (DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
328{
329 SigMatch *sm = NULL;
330 DetectXbitsData *cd = NULL;
331
332 int result = DetectXbitParse(de_ctx, rawstr, &cd);
333 if (result < 0) {
334 return -1;
335 /* noalert doesn't use a cd/sm struct. It flags the sig. We're done. */
336 } else if (result == 0 && cd == NULL) {
337 s->flags |= SIG_FLAG_NOALERT;
338 return 0;
339 }
340
341 /* Okay so far so good, lets get this into a SigMatch
342 * and put it in the Signature. */
343 sm = SigMatchAlloc();
344 if (sm == NULL)
345 goto error;
346
347 sm->type = DETECT_XBITS;
348 sm->ctx = (void *)cd;
349
350 switch (cd->cmd) {
351 /* case DETECT_XBITS_CMD_NOALERT can't happen here */
352
353 case DETECT_XBITS_CMD_ISNOTSET:
354 case DETECT_XBITS_CMD_ISSET:
355 /* checks, so packet list */
356 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
357 break;
358
359 case DETECT_XBITS_CMD_SET:
360 case DETECT_XBITS_CMD_UNSET:
361 case DETECT_XBITS_CMD_TOGGLE:
362 /* modifiers, only run when entire sig has matched */
363 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_POSTMATCH);
364 break;
365 }
366
367 return 0;
368
369error:
370 if (cd != NULL)
371 SCFree(cd);
372 return -1;
373}

405static int XBitsTestParse01(void)
406{
407 DetectEngineCtx *de_ctx = NULL;
408 de_ctx = DetectEngineCtxInit();
409 FAIL_IF_NULL(de_ctx);
410 de_ctx->flags |= DE_QUIET;
411 DetectXbitsData *cd = NULL;
412
413#define BAD_INPUT(str) \
414 FAIL_IF_NOT(DetectXbitParse(de_ctx, (str), &cd) == -1);
415
416 BAD_INPUT("alert");
417 BAD_INPUT("n0alert");
418 BAD_INPUT("nOalert");
419 BAD_INPUT("set,abc,track nonsense, expire 3600");
420 BAD_INPUT("set,abc,track ip_source, expire 3600");
421 BAD_INPUT("set,abc,track ip_src, expire -1");
422 BAD_INPUT("set,abc,track ip_src, expire 0");
423
424#undef BAD_INPUT
425
426#define GOOD_INPUT(str, command, trk, typ, exp) \
427 FAIL_IF_NOT(DetectXbitParse(de_ctx, (str), &cd) == 0); \
428 FAIL_IF_NULL(cd); \
429 FAIL_IF_NOT(cd->cmd == (command)); \
430 FAIL_IF_NOT(cd->tracker == (trk)); \
431 FAIL_IF_NOT(cd->type == (typ)); \
432 FAIL_IF_NOT(cd->expire == (exp)); \
433 DetectXbitFree(NULL, cd); \
434 cd = NULL;
435
436 GOOD_INPUT("set,abc,track ip_pair",
437 DETECT_XBITS_CMD_SET,
438 DETECT_XBITS_TRACK_IPPAIR, VAR_TYPE_IPPAIR_BIT,
439 DETECT_XBITS_EXPIRE_DEFAULT);
440 GOOD_INPUT("set,abc,track ip_pair, expire 3600",
441 DETECT_XBITS_CMD_SET,
442 DETECT_XBITS_TRACK_IPPAIR, VAR_TYPE_IPPAIR_BIT,
443 3600);
444 GOOD_INPUT("set,abc,track ip_src, expire 1234",
445 DETECT_XBITS_CMD_SET,
446 DETECT_XBITS_TRACK_IPSRC, VAR_TYPE_HOST_BIT,
447 1234);
448
449#undef GOOD_INPUT
450
451 DetectEngineCtxFree(de_ctx);
452 PASS;
453}

459static int XBitsTestSig01(void)
460{
461 uint8_t *buf = (uint8_t *)
462 "GET /one/ HTTP/1.1\r\n"
463 "Host: one.example.org\r\n"
464 "\r\n";
465 uint16_t buflen = strlen((char *)buf);
466 Packet *p = PacketGetFromAlloc();
467 FAIL_IF_NULL(p);
468 Signature *s = NULL;
469 ThreadVars th_v;
470 DetectEngineThreadCtx *det_ctx = NULL;
471 DetectEngineCtx *de_ctx = NULL;
472
473 memset(&th_v, 0, sizeof(th_v));
474 p->src.family = AF_INET;
475 p->dst.family = AF_INET;
476 p->payload = buf;
477 p->payload_len = buflen;
478 p->proto = IPPROTO_TCP;
479
480 XBitsTestSetup();
481
482 de_ctx = DetectEngineCtxInit();
483 FAIL_IF_NULL(de_ctx);
484 de_ctx->flags |= DE_QUIET;
485
486 s = DetectEngineAppendSig(de_ctx,
487 "alert ip any any -> any any (xbits:set,abc,track ip_pair; content:\"GET \"; sid:1;)");
488 FAIL_IF_NULL(s);
489
490 SigGroupBuild(de_ctx);
491 DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
492 SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
493 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
494 DetectEngineCtxFree(de_ctx);
495 XBitsTestShutdown();
496 SCFree(p);
497 StatsThreadCleanup(&th_v);
498 StatsReleaseResources();
499 PASS;
500}
