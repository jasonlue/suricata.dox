54/**
55 * \brief Inspect the file inspecting keywords.
56 *
57 * \param tv thread vars
58 * \param det_ctx detection engine thread ctx
59 * \param f flow
60 * \param s signature to inspect
61 *
62 * \retval 0 no match
63 * \retval 1 match
64 * \retval 2 can't match
65 * \retval 3 can't match filestore signature
66 */
67static uint8_t DetectFileInspect(DetectEngineThreadCtx *det_ctx, Flow *f, const Signature *s,
68 const SigMatchData *smd, uint8_t flags, FileContainer *ffc)
69{
70 uint8_t r = 0;
71 int match = 0;
72 int store_r = 0;
73
74 SCLogDebug("file inspection... %p", ffc);
75
76 if (ffc != NULL) {
77 File *file = ffc->head;
78 for (; file != NULL; file = file->next) {
79 SCLogDebug("file");
80
81 if (file->state == FILE_STATE_NONE) {
82 SCLogDebug("file state FILE_STATE_NONE");
83 continue;
84 }
85
86 if (file->txid < det_ctx->tx_id) {
87 SCLogDebug("file->txid < det_ctx->tx_id == %"PRIu64" < %"PRIu64, file->txid, det_ctx->tx_id);
88 continue;
89 }
90
91 if (file->txid > det_ctx->tx_id) {
92 SCLogDebug("file->txid > det_ctx->tx_id == %"PRIu64" > %"PRIu64, file->txid, det_ctx->tx_id);
93 break;
94 }
95
96 if ((s->file_flags & FILE_SIG_NEED_FILENAME) && file->name == NULL) {
97 SCLogDebug("sig needs filename, but we don't have any");
98 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
99 continue;
100 }
101
102 uint64_t file_size = FileDataSize(file);
103 if ((s->file_flags & FILE_SIG_NEED_MAGIC) && file_size == 0) {
104 SCLogDebug("sig needs file content, but we don't have any");
105 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
106 continue;
107 }
108
109 if ((s->file_flags & FILE_SIG_NEED_FILECONTENT) && file_size == 0) {
110 SCLogDebug("sig needs file content, but we don't have any");
111 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
112 continue;
113 }
114
115 if ((s->file_flags & FILE_SIG_NEED_MD5) && (!(file->flags & FILE_MD5))) {
116 SCLogDebug("sig needs file md5, but we don't have any");
117 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
118 continue;
119 }
120
121 if ((s->file_flags & FILE_SIG_NEED_SHA1) && (!(file->flags & FILE_SHA1))) {
122 SCLogDebug("sig needs file sha1, but we don't have any");
123 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
124 continue;
125 }
126
127 if ((s->file_flags & FILE_SIG_NEED_SHA256) && (!(file->flags & FILE_SHA256))) {
128 SCLogDebug("sig needs file sha256, but we don't have any");
129 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
130 continue;
131 }
132
133 if ((s->file_flags & FILE_SIG_NEED_SIZE) && file->state < FILE_STATE_CLOSED) {
134 SCLogDebug("sig needs filesize, but state < FILE_STATE_CLOSED");
135 r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
136 continue;
137 }
138
139 /* run the file match functions. */
140 while (1) {
141 SCLogDebug("smd %p", smd);
142
143 if (sigmatch_table[smd->type].FileMatch != NULL) {
144 KEYWORD_PROFILING_START;
145 match = sigmatch_table[smd->type].
146 FileMatch(det_ctx, f, flags, file, s, smd->ctx);
147 KEYWORD_PROFILING_END(det_ctx, smd->type, (match > 0));
148 if (match == 0) {
149 r = DETECT_ENGINE_INSPECT_SIG_CANT_MATCH_FILES;
150 break;
151 } else if (smd->is_last) {
152 r = DETECT_ENGINE_INSPECT_SIG_MATCH;
153 break;
154 }
155 }
156 if (smd->is_last)
157 break;
158 smd++;
159 }
160
161 /* continue inspection for other files as we may want to store
162 * those as well. We'll return 1 (match) regardless of their
163 * results though */
164 if (r == DETECT_ENGINE_INSPECT_SIG_MATCH)
165 store_r = DETECT_ENGINE_INSPECT_SIG_MATCH;
166
167 /* continue, this file may (or may not) be unable to match
168 * maybe we have more that can :) */
169 }
170 } else {
171 /* if we have a filestore sm with a scope > file (so tx, ssn) we
172 * run it here */
173 if (smd != NULL && smd->is_last && smd->type == DETECT_FILESTORE &&
174 smd->ctx != NULL)
175 {
176 DetectFilestoreData *fd = (DetectFilestoreData *)smd->ctx;
177 if (fd->scope > FILESTORE_SCOPE_DEFAULT) {
178 KEYWORD_PROFILING_START;
179 match = sigmatch_table[smd->type].
180 FileMatch(det_ctx, f, flags, /* no file */NULL, s, smd->ctx);
181 KEYWORD_PROFILING_END(det_ctx, smd->type, (match > 0));
182
183 if (match == 1) {
184 r = DETECT_ENGINE_INSPECT_SIG_MATCH;
185 }
186 }
187 }
188 }
189
190 if (r == DETECT_ENGINE_INSPECT_SIG_NO_MATCH && store_r == DETECT_ENGINE_INSPECT_SIG_MATCH) {
191 SCLogDebug("stored MATCH, current file NOMATCH");
192 SCReturnInt(DETECT_ENGINE_INSPECT_SIG_MATCH_MORE_FILES);
193 }
194
195 if (store_r == DETECT_ENGINE_INSPECT_SIG_MATCH)
196 r = DETECT_ENGINE_INSPECT_SIG_MATCH;
197 SCReturnInt(r);
198}

200/**
201 * \brief Inspect the file inspecting keywords against the state
202 *
203 * \param det_ctx detection engine thread ctx
204 * \param f flow
205 * \param s signature to inspect
206 * \param alstate state
207 * \param flags direction flag
208 *
209 * \retval 0 no match
210 * \retval 1 match
211 * \retval 2 can't match
212 * \retval 3 can't match filestore signature
213 *
214 * \note flow is not locked at this time
215 */
216uint8_t DetectFileInspectGeneric(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
217 const struct DetectEngineAppInspectionEngine_ *engine, const Signature *s, Flow *f,
218 uint8_t flags, void *_alstate, void *tx, uint64_t tx_id)
219{
220 SCEnter();
221 DEBUG_VALIDATE_BUG_ON(f->alstate != _alstate);
222
223 const uint8_t direction = flags & (STREAM_TOSERVER|STREAM_TOCLIENT);
224 FileContainer *ffc = AppLayerParserGetFiles(f, direction);
225 if (ffc == NULL || ffc->head == NULL) {
226 SCReturnInt(DETECT_ENGINE_INSPECT_SIG_NO_MATCH);
227 }
228
229 uint8_t r = DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
230 uint8_t match = DetectFileInspect(det_ctx, f, s, engine->smd, flags, ffc);
231 if (match == DETECT_ENGINE_INSPECT_SIG_MATCH) {
232 r = DETECT_ENGINE_INSPECT_SIG_MATCH;
233 } else if (match == DETECT_ENGINE_INSPECT_SIG_CANT_MATCH) {
234 SCLogDebug("sid %u can't match on this transaction", s->id);
235 r = DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
236 } else if (match == DETECT_ENGINE_INSPECT_SIG_CANT_MATCH_FILES) {
237 SCLogDebug("sid %u can't match on this transaction (file sig)", s->id);
238 r = DETECT_ENGINE_INSPECT_SIG_CANT_MATCH_FILES;
239 } else if (match == DETECT_ENGINE_INSPECT_SIG_MATCH_MORE_FILES) {
240 SCLogDebug("match with more files ahead");
241 r = match;
242 }
243
244 SCReturnInt(r);
245}
