38AppProto AppLayerRegisterProtocolDetection(const struct AppLayerParser *p, int enable_default)
39{
40 AppProto alproto;
41 const char *ip_proto_str = NULL;
42
43 if (p == NULL)
44 FatalError(SC_ERR_FATAL, "Call to %s with NULL pointer.", __FUNCTION__);
45
46 alproto = StringToAppProto(p->name);
47 if (alproto == ALPROTO_UNKNOWN || alproto == ALPROTO_FAILED)
48 FatalError(SC_ERR_FATAL, "Unknown or invalid AppProto '%s'.", p->name);
49
50 ip_proto_str = IpProtoToString(p->ip_proto);
51 if (ip_proto_str == NULL)
52 FatalError(SC_ERR_FATAL, "Unknown or unsupported ip_proto field in parser '%s'", p->name);
53
54 SCLogDebug("%s %s protocol detection enabled.", ip_proto_str, p->name);
55
56 AppLayerProtoDetectRegisterProtocol(alproto, p->name);
57
58 if (p->ProbeTS == NULL && p->ProbeTC == NULL) {
59 BUG_ON(p->default_port != NULL);
60 return alproto;
61 }
62
63 if (RunmodeIsUnittests()) {
64
65 SCLogDebug("Unittest mode, registering default configuration.");
66 AppLayerProtoDetectPPRegister(p->ip_proto, p->default_port,
67 alproto, p->min_depth, p->max_depth, STREAM_TOSERVER,
68 p->ProbeTS, p->ProbeTC);
69
70 }
71 else {
72
73 if (!AppLayerProtoDetectPPParseConfPorts(ip_proto_str, p->ip_proto,
74 p->name, alproto, p->min_depth, p->max_depth,
75 p->ProbeTS, p->ProbeTC)) {
76 if (enable_default != 0) {
77 SCLogDebug("No %s app-layer configuration, enabling %s"
78 " detection %s detection on port %s.",
79 p->name, p->name, ip_proto_str, p->default_port);
80 AppLayerProtoDetectPPRegister(p->ip_proto,
81 p->default_port, alproto,
82 p->min_depth, p->max_depth, STREAM_TOSERVER,
83 p->ProbeTS, p->ProbeTC);
84 } else {
85 SCLogDebug("No %s app-layer configuration for detection port (%s).",
86 p->name, ip_proto_str);
87 }
88 }
89
90 }
91
92 return alproto;
93}

95int AppLayerRegisterParser(const struct AppLayerParser *p, AppProto alproto)
96{
97 const char *ip_proto_str = NULL;
98
99 if (p == NULL)
100 FatalError(SC_ERR_FATAL, "Call to %s with NULL pointer.", __FUNCTION__);
101
102 if (alproto == ALPROTO_UNKNOWN || alproto >= ALPROTO_FAILED)
103 FatalError(SC_ERR_FATAL, "Unknown or invalid AppProto '%s'.", p->name);
104
105 ip_proto_str = IpProtoToString(p->ip_proto);
106 if (ip_proto_str == NULL)
107 FatalError(SC_ERR_FATAL, "Unknown or unsupported ip_proto field in parser '%s'", p->name);
108
109 SCLogDebug("Registering %s protocol parser.", p->name);
110
111 /* Register functions for state allocation and freeing. A
112 * state is allocated for every new flow. */
113 AppLayerParserRegisterStateFuncs(p->ip_proto, alproto,
114 p->StateAlloc, p->StateFree);
115
116 /* Register request parser for parsing frame from server to server. */
117 AppLayerParserRegisterParser(p->ip_proto, alproto,
118 STREAM_TOSERVER, p->ParseTS);
119
120 /* Register response parser for parsing frames from server to client. */
121 AppLayerParserRegisterParser(p->ip_proto, alproto,
122 STREAM_TOCLIENT, p->ParseTC);
123
124 /* Register a function to be called by the application layer
125 * when a transaction is to be freed. */
126 AppLayerParserRegisterTxFreeFunc(p->ip_proto, alproto,
127 p->StateTransactionFree);
128
129 /* Register a function to return the current transaction count. */
130 AppLayerParserRegisterGetTxCnt(p->ip_proto, alproto,
131 p->StateGetTxCnt);
132
133 /* Transaction handling. */
134 AppLayerParserRegisterStateProgressCompletionStatus(alproto, p->complete_ts, p->complete_tc);
135
136 AppLayerParserRegisterGetStateProgressFunc(p->ip_proto, alproto,
137 p->StateGetProgress);
138 AppLayerParserRegisterGetTx(p->ip_proto, alproto,
139 p->StateGetTx);
140
141 if (p->StateGetEventInfo) {
142 AppLayerParserRegisterGetEventInfo(p->ip_proto, alproto,
143 p->StateGetEventInfo);
144 }
145 if (p->StateGetEventInfoById) {
146 AppLayerParserRegisterGetEventInfoById(p->ip_proto, alproto,
147 p->StateGetEventInfoById);
148 }
149 if (p->LocalStorageAlloc && p->LocalStorageFree) {
150 AppLayerParserRegisterLocalStorageFunc(p->ip_proto, alproto,
151 p->LocalStorageAlloc, p->LocalStorageFree);
152 }
153 if (p->StateGetFiles) {
154 AppLayerParserRegisterGetFilesFunc(p->ip_proto, alproto,
155 p->StateGetFiles);
156 }
157
158 if (p->GetTxIterator) {
159 AppLayerParserRegisterGetTxIterator(p->ip_proto, alproto,
160 p->GetTxIterator);
161 }
162
163 if (p->GetTxData) {
164 AppLayerParserRegisterTxDataFunc(p->ip_proto, alproto,
165 p->GetTxData);
166 }
167
168 if (p->ApplyTxConfig) {
169 AppLayerParserRegisterApplyTxConfigFunc(p->ip_proto, alproto,
170 p->ApplyTxConfig);
171 }
172
173 if (p->flags) {
174 AppLayerParserRegisterOptionFlags(p->ip_proto, alproto,
175 p->flags);
176
177 }
178
179 if (p->Truncate) {
180 AppLayerParserRegisterTruncateFunc(p->ip_proto, alproto, p->Truncate);
181 }
182
183 if (p->GetFrameIdByName && p->GetFrameNameById) {
184 AppLayerParserRegisterGetFrameFuncs(
185 p->ip_proto, alproto, p->GetFrameIdByName, p->GetFrameNameById);
186 }
187
188 return 0;
189}

198static const char * IpProtoToString(int ip_proto)
199{
200 switch (ip_proto) {
201 case IPPROTO_TCP:
202 return "tcp";
203 case IPPROTO_UDP:
204 return "udp";
205 default:
206 return NULL;
207 };
