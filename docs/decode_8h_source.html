128#define COPY_ADDRESS(a, b) do { \
129 (b)->family = (a)->family; \
130 (b)->addr_data32[0] = (a)->addr_data32[0]; \
131 (b)->addr_data32[1] = (a)->addr_data32[1]; \
132 (b)->addr_data32[2] = (a)->addr_data32[2]; \
133 (b)->addr_data32[3] = (a)->addr_data32[3]; \
134 } while (0)

136/* Set the IPv4 addresses into the Addrs of the Packet.
137 * Make sure p->ip4h is initialized and validated.
138 *
139 * We set the rest of the struct to 0 so we can
140 * prevent using memset. */
141#define SET_IPV4_SRC_ADDR(p, a) do { \
142 (a)->family = AF_INET; \
143 (a)->addr_data32[0] = (uint32_t)(p)->ip4h->s_ip_src.s_addr; \
144 (a)->addr_data32[1] = 0; \
145 (a)->addr_data32[2] = 0; \
146 (a)->addr_data32[3] = 0; \
147 } while (0)
148
149#define SET_IPV4_DST_ADDR(p, a) do { \
150 (a)->family = AF_INET; \
151 (a)->addr_data32[0] = (uint32_t)(p)->ip4h->s_ip_dst.s_addr; \
152 (a)->addr_data32[1] = 0; \
153 (a)->addr_data32[2] = 0; \
154 (a)->addr_data32[3] = 0; \
155 } while (0)
156
157/* clear the address structure by setting all fields to 0 */
158#define CLEAR_ADDR(a) do { \
159 (a)->family = 0; \
160 (a)->addr_data32[0] = 0; \
161 (a)->addr_data32[1] = 0; \
162 (a)->addr_data32[2] = 0; \
163 (a)->addr_data32[3] = 0; \
164 } while (0)
165
166/* Set the IPv6 addresses into the Addrs of the Packet.
167 * Make sure p->ip6h is initialized and validated. */
168#define SET_IPV6_SRC_ADDR(p, a) do { \
169 (a)->family = AF_INET6; \
170 (a)->addr_data32[0] = (p)->ip6h->s_ip6_src[0]; \
171 (a)->addr_data32[1] = (p)->ip6h->s_ip6_src[1]; \
172 (a)->addr_data32[2] = (p)->ip6h->s_ip6_src[2]; \
173 (a)->addr_data32[3] = (p)->ip6h->s_ip6_src[3]; \
174 } while (0)
175
176#define SET_IPV6_DST_ADDR(p, a) do { \
177 (a)->family = AF_INET6; \
178 (a)->addr_data32[0] = (p)->ip6h->s_ip6_dst[0]; \
179 (a)->addr_data32[1] = (p)->ip6h->s_ip6_dst[1]; \
180 (a)->addr_data32[2] = (p)->ip6h->s_ip6_dst[2]; \
181 (a)->addr_data32[3] = (p)->ip6h->s_ip6_dst[3]; \
182 } while (0)

1274 SCLogDebug("tunnel: outstanding %u", outstanding);

1276 /* if there are packets outstanding, we won't verdict this one */
1277 if (IS_TUNNEL_ROOT_PKT(p) && !IS_TUNNEL_PKT_VERDICTED(p) && !outstanding) {
1278 // verdict
1279 SCLogDebug("root %p: verdict", p);
1280 } else if (!IS_TUNNEL_ROOT_PKT(p) && outstanding == 1 && p->root && IS_TUNNEL_PKT_VERDICTED(p->root)) {
1281 // verdict
1282 SCLogDebug("tunnel %p: verdict", p);
1283 } else {
1284 verdict = false;
1285 }
1286 SCMutexUnlock(m);
