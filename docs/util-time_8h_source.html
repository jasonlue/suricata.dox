36static inline struct timeval TimevalWithSeconds(const struct timeval *ts, const time_t sec_add)
37{
38#ifdef timeradd
39 struct timeval add = { .tv_sec = sec_add, .tv_usec = 0 };
40 struct timeval result;
41 timeradd(ts, &add, &result);
42 return result;
43#else
44 const time_t sec = ts->tv_sec + sec_add;
45 struct timeval result = { .tv_sec = sec, .tv_usec = ts->tv_usec };
46 return result;
47#endif
48}

50/** \brief compare two 'struct timeval' and return if the first is earlier than the second */
51static inline bool TimevalEarlier(struct timeval *first, struct timeval *second)
52{
53 /* from man timercmp on Linux: "Some systems (but not Linux/glibc), have a broken timercmp()
54 * implementation, in which CMP of >=, <=, and == do not work; portable applications can instead
55 * use ... !timercmp(..., >) */
56 return !timercmp(first, second, >);
57}

59#ifndef timeradd
60#define timeradd(a, b, r) \
61 do { \
62 (r)->tv_sec = (a)->tv_sec + (b)->tv_sec; \
63 (r)->tv_usec = (a)->tv_usec + (b)->tv_usec; \
64 if ((r)->tv_usec >= 1000000) { \
65 (r)->tv_sec++; \
66 (r)->tv_usec -= 1000000; \
67 } \
68 } while (0)
69#endif
