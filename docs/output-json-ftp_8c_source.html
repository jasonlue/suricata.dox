49static void EveFTPLogCommand(Flow *f, FTPTransaction *tx, JsonBuilder *jb)
50{
51 /* Preallocate array objects to simplify failure case */
52 JsonBuilder *js_resplist = NULL;
53 if (!TAILQ_EMPTY(&tx->response_list)) {
54 js_resplist = jb_new_array();
55
56 if (unlikely(js_resplist == NULL)) {
57 return;
58 }
59 }
60 jb_set_string(jb, "command", tx->command_descriptor->command_name);
61 uint32_t min_length = tx->command_descriptor->command_length + 1; /* command + space */
62 if (tx->request_length > min_length) {
63 jb_set_string_from_bytes(jb,
64 "command_data",
65 (const uint8_t *)tx->request + min_length,
66 tx->request_length - min_length - 1);
67 if (tx->request_truncated) {
68 JB_SET_TRUE(jb, "command_truncated");
69 } else {
70 JB_SET_FALSE(jb, "command_truncated");
71 }
72 }
73
74 bool reply_truncated = false;
75
76 if (!TAILQ_EMPTY(&tx->response_list)) {
77 int resp_code_cnt = 0;
78 int resp_cnt = 0;
79 FTPString *response;
80 bool is_cc_array_open = false;
81 TAILQ_FOREACH(response, &tx->response_list, next) {
82 /* handle multiple lines within the response, \r\n delimited */
83 uint8_t *where = response->str;
84 uint16_t length = 0;
85 uint16_t pos;
86 if (response->len > 0 && response->len <= UINT16_MAX) {
87 length = (uint16_t)response->len - 1;
88 } else if (response->len > UINT16_MAX) {
89 length = UINT16_MAX;
90 }
91 if (!reply_truncated && response->truncated) {
92 reply_truncated = true;
93 }
94 while ((pos = JsonGetNextLineFromBuffer((const char *)where, length)) != UINT16_MAX) {
95 uint16_t offset = 0;
96 /* Try to find a completion code for this line */
97 if (pos >= 3) {
98 /* Gather the completion code if present */
99 if (isdigit(where[0]) && isdigit(where[1]) && isdigit(where[2])) {
100 if (!is_cc_array_open) {
101 jb_open_array(jb, "completion_code");
102 is_cc_array_open = true;
103 }
104 jb_append_string_from_bytes(jb, (const uint8_t *)where, 3);
105 resp_code_cnt++;
106 offset = 4;
107 }
108 }
109 /* move past 3 character completion code */
110 if (pos >= offset) {
111 jb_append_string_from_bytes(js_resplist, (const uint8_t *)where + offset, pos - offset);
112 resp_cnt++;
113 }
114
115 where += pos;
116 length -= pos;
117 }
118 }
119
120 if (is_cc_array_open) {
121 jb_close(jb);
122 }
123 if (resp_cnt) {
124 jb_close(js_resplist);
125 jb_set_object(jb, "reply", js_resplist);
126 }
127 jb_free(js_resplist);
128 }
129
130 if (tx->dyn_port) {
131 jb_set_uint(jb, "dynamic_port", tx->dyn_port);
132 }
133
134 if (tx->command_descriptor->command == FTP_COMMAND_PORT ||
135 tx->command_descriptor->command == FTP_COMMAND_EPRT) {
136 if (tx->active) {
137 JB_SET_STRING(jb, "mode", "active");
138 } else {
139 JB_SET_STRING(jb, "mode", "passive");
140 }
141 }
142
143 if (tx->done) {
144 JB_SET_STRING(jb, "reply_received", "yes");
145 } else {
146 JB_SET_STRING(jb, "reply_received", "no");
147 }
148
149 if (reply_truncated) {
150 JB_SET_TRUE(jb, "reply_truncated");
151 } else {
152 JB_SET_FALSE(jb, "reply_truncated");
153 }
154}

157static int JsonFTPLogger(ThreadVars *tv, void *thread_data,
158 const Packet *p, Flow *f, void *state, void *vtx, uint64_t tx_id)
159{
160 SCEnter();
161 OutputJsonThreadCtx *thread = thread_data;
162
163 const char *event_type;
164 if (f->alproto == ALPROTO_FTPDATA) {
165 event_type = "ftp_data";
166 } else {
167 event_type = "ftp";
168 }
169 FTPTransaction *tx = vtx;
170
171 JsonBuilder *jb =
172 CreateEveHeaderWithTxId(p, LOG_DIR_FLOW, event_type, NULL, tx_id, thread->ctx);
173 if (likely(jb)) {
174 jb_open_object(jb, event_type);
175 if (f->alproto == ALPROTO_FTPDATA) {
176 EveFTPDataAddMetadata(f, jb);
177 } else {
178 EveFTPLogCommand(f, tx, jb);
179 }
180
181 if (!jb_close(jb)) {
182 goto fail;
183 }
184
185 OutputJsonBuilderBuffer(jb, thread);
186
187 jb_free(jb);
188 }
189 return TM_ECODE_OK;
190
191fail:
192 jb_free(jb);
193 return TM_ECODE_FAILED;
194}

209 NULL);

212 JsonLogThreadDeinit, NULL);
213
214 SCLogDebug("FTP JSON logger registered.");
215}
