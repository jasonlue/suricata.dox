21/* djb2 string hashing */
22uint32_t StringHashDjb2(const uint8_t *data, uint32_t datalen)
23{
24 uint32_t hash = 5381;
25 for (uint32_t i = 0; i < datalen; i++) {
26 uint32_t c = data[i];
27 hash = ((hash << 5) + hash) + c; /* hash * 33 + c */
28 }
29 return hash;
30}
