73/**
74 * \brief Registers the keyword handlers for the "http_request_line" keyword.
75 */
76void DetectHttpRequestLineRegister(void)
77{
78 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].name = "http.request_line";
79 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].alias = "http_request_line";
80 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].desc = "sticky buffer to match on the HTTP request line";
81 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].url = "/rules/http-keywords.html#http-request-line";
82 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].Match = NULL;
83 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].Setup = DetectHttpRequestLineSetup;
84#ifdef UNITTESTS
85 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].RegisterTests = DetectHttpRequestLineRegisterTests;
86#endif
87 sigmatch_table[DETECT_AL_HTTP_REQUEST_LINE].flags |= SIGMATCH_NOOPT|SIGMATCH_INFO_STICKY_BUFFER;
88
89 DetectAppLayerInspectEngineRegister2("http_request_line", ALPROTO_HTTP1, SIG_FLAG_TOSERVER,
90 HTP_REQUEST_LINE, DetectEngineInspectBufferGeneric, GetData);
91
92 DetectAppLayerMpmRegister2("http_request_line", SIG_FLAG_TOSERVER, 2,
93 PrefilterGenericMpmRegister, GetData, ALPROTO_HTTP1, HTP_REQUEST_LINE);
94
95 DetectBufferTypeSetDescriptionByName("http_request_line",
96 "http request line");
97
98 g_http_request_line_buffer_id = DetectBufferTypeGetByName("http_request_line");
99}

101/**
102 * \brief The setup function for the http_request_line keyword for a signature.
103 *
104 * \param de_ctx Pointer to the detection engine context.
105 * \param s Pointer to the signature for the current Signature being
106 * parsed from the rules.
107 * \param m Pointer to the head of the SigMatch for the current rule
108 * being parsed.
109 * \param arg Pointer to the string holding the keyword value.
110 *
111 * \retval 0 On success
112 * \retval -1 On failure
113 */
114static int DetectHttpRequestLineSetup(DetectEngineCtx *de_ctx, Signature *s, const char *arg)
115{
116 if (DetectBufferSetActiveList(s, g_http_request_line_buffer_id) < 0)
117 return -1;
118
119 if (DetectSignatureSetAppProto(s, ALPROTO_HTTP1) < 0)
120 return -1;
121
122 return 0;
123}

125static InspectionBuffer *GetData(DetectEngineThreadCtx *det_ctx,
126 const DetectEngineTransforms *transforms,
127 Flow *_f, const uint8_t _flow_flags,
128 void *txv, const int list_id)
129{
130 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
131 if (buffer->inspect == NULL) {
132 htp_tx_t *tx = (htp_tx_t *)txv;
133 if (unlikely(tx->request_line == NULL)) {
134 return NULL;
135 }
136 const uint32_t data_len = bstr_len(tx->request_line);
137 const uint8_t *data = bstr_ptr(tx->request_line);
138
139 InspectionBufferSetup(det_ctx, list_id, buffer, data, data_len);
140 InspectionBufferApplyTransforms(buffer, transforms);
141 }
142 return buffer;
143}
