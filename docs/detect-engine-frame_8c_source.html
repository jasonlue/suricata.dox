43void DetectRunPrefilterFrame(DetectEngineThreadCtx *det_ctx, const SigGroupHead *sgh, Packet *p,
44 const Frames *frames, const Frame *frame, const AppProto alproto, const uint32_t idx)
45{
46 SCLogDebug("pcap_cnt %" PRIu64, p->pcap_cnt);
47 PrefilterEngine *engine = sgh->frame_engines;
48 do {
49 BUG_ON(engine->alproto == ALPROTO_UNKNOWN);
50 if (engine->alproto == alproto && engine->ctx.frame_type == frame->type) {
51 SCLogDebug("frame %p engine %p", frame, engine);
52 PREFILTER_PROFILING_START(det_ctx);
53 engine->cb.PrefilterFrame(det_ctx, engine->pectx, p, frames, frame, idx);
54 PREFILTER_PROFILING_END(det_ctx, engine->gid);
55 }
56 if (engine->is_last)
57 break;
58 engine++;
59 } while (1);
60}

71/** \brief Generic Mpm prefilter callback

73 * \param det_ctx detection engine thread ctx

75 * \param frame frame to inspect
76 * \param pectx inspection context
77 */
78static void PrefilterMpmFrame(DetectEngineThreadCtx *det_ctx, const void *pectx, Packet *p,
79 const Frames *frames, const Frame *frame, const uint32_t idx)
80{
81 SCEnter();
82
83 const PrefilterMpmFrameCtx *ctx = (const PrefilterMpmFrameCtx *)pectx;
84 const MpmCtx *mpm_ctx = ctx->mpm_ctx;
85 SCLogDebug("running on list %d -> frame field type %u", ctx->list_id, frame->type);
86 // BUG_ON(frame->type != ctx->type);
87
88 InspectionBuffer *buffer = DetectFrame2InspectBuffer(
89 det_ctx, ctx->transforms, p, frames, frame, ctx->list_id, idx, true);
90 if (buffer == NULL)
91 return;
92 DEBUG_VALIDATE_BUG_ON(frame->len >= 0 && buffer->orig_len > frame->len);
93
94 const uint32_t data_len = buffer->inspect_len;
95 const uint8_t *data = buffer->inspect;
96
97 SCLogDebug("mpm'ing buffer:");
98 // SCLogDebug("frame: %p", frame);
99 // PrintRawDataFp(stdout, data, MIN(32, data_len));
100
101 if (data != NULL && data_len >= mpm_ctx->minlen) {
102 (void)mpm_table[mpm_ctx->mpm_type].Search(
103 mpm_ctx, &det_ctx->mtcu, &det_ctx->pmq, data, data_len);
104 SCLogDebug("det_ctx->pmq.rule_id_array_cnt %u", det_ctx->pmq.rule_id_array_cnt);
105 PREFILTER_PROFILING_ADD_BYTES(det_ctx, data_len);
106 }

114int PrefilterGenericMpmFrameRegister(DetectEngineCtx *de_ctx, SigGroupHead *sgh, MpmCtx *mpm_ctx,
115 const DetectBufferMpmRegistery *mpm_reg, int list_id)
116{
117 SCEnter();

123 pectx->mpm_ctx = mpm_ctx;
124 pectx->transforms = &mpm_reg->transforms;

128 if (r != 0) {
129 SCFree(pectx);
130 }
131 return r;

161/** \internal
162 * \brief setup buffer based on frame in UDP payload
163 */
164static InspectionBuffer *DetectFrame2InspectBufferUdp(DetectEngineThreadCtx *det_ctx,
165 const DetectEngineTransforms *transforms, Packet *p, InspectionBuffer *buffer,
166 const Frames *frames, const Frame *frame, const int list_id, const uint32_t idx,
167 const bool first)
168{
169 DEBUG_VALIDATE_BUG_ON(frame->rel_offset >= p->payload_len);
170 if (frame->rel_offset >= p->payload_len)
171 return NULL;
172
173 int frame_len = frame->len != -1 ? frame->len : p->payload_len - frame->rel_offset;
174 uint8_t ci_flags = DETECT_CI_FLAGS_START;
175
176 if (frame->rel_offset + frame_len > p->payload_len) {
177 frame_len = p->payload_len - frame->rel_offset;
178 } else {
179 ci_flags |= DETECT_CI_FLAGS_END;
180 }
181 const uint8_t *data = p->payload + frame->rel_offset;
182 const uint32_t data_len = frame_len;
183
184 SCLogDebug("packet %" PRIu64 " -> frame %p/%" PRIi64 "/%s rel_offset %" PRIi64
185 " type %u len %" PRIi64,
186 p->pcap_cnt, frame, frame->id,
187 AppLayerParserGetFrameNameById(p->flow->proto, p->flow->alproto, frame->type),
188 frame->rel_offset, frame->type, frame->len);
189 // PrintRawDataFp(stdout, data, MIN(64,data_len));
190
191 InspectionBufferSetupMulti(buffer, transforms, data, data_len);
192 buffer->inspect_offset = 0;
193 buffer->flags = ci_flags;
194 return buffer;
195}

207static int FrameStreamDataFunc(
208 void *cb_data, const uint8_t *input, const uint32_t input_len, const uint64_t offset)
209{
210 struct FrameStreamData *fsd = cb_data;
211 SCLogDebug("fsd %p { det_ct:%p, transforms:%p, frame:%p, list_id:%d, idx:%u, "
212 "frame_data_offset_abs:%" PRIu64 ", frame_start_offset_abs:%" PRIu64
213 " }, input: %p, input_len:%u, offset:%" PRIu64,
214 fsd, fsd->det_ctx, fsd->transforms, fsd->frame, fsd->list_id, fsd->idx,
215 fsd->frame_data_offset_abs, fsd->frame_start_offset_abs, input, input_len, offset);
216
217 InspectionBuffer *buffer =
218 InspectionBufferMultipleForListGet(fsd->det_ctx, fsd->list_id, fsd->idx);
219 BUG_ON(buffer == NULL);
220 SCLogDebug("buffer %p", buffer);
221
222 const Frame *frame = fsd->frame;
223 SCLogDebug("frame rel_offset:%" PRIi64, frame->rel_offset);
224 const uint8_t *data = input;
225 uint8_t ci_flags = 0;
226 uint32_t data_len;
227 if (fsd->frame_start_offset_abs == offset) {
228 ci_flags |= DETECT_CI_FLAGS_START;
229 SCLogDebug("have frame data start");
230
231 if (frame->len >= 0) {
232 data_len = MIN(input_len, frame->len);
233 } else {
234 data_len = input_len;
235 }
236
237 if (data_len == frame->len) {
238 ci_flags |= DETECT_CI_FLAGS_END;
239 SCLogDebug("have frame data end");
240 }
241 } else {
242 BUG_ON(offset < fsd->frame_data_offset_abs);
243
244 uint64_t frame_delta = offset - fsd->frame_start_offset_abs;
245 uint64_t request_delta =
246 offset -
247 fsd->frame_data_offset_abs; // diff between what we requested and what we got
248 BUG_ON(request_delta > frame_delta);
249
250 if (frame->len >= 0) {
251 if (frame_delta >= (uint64_t)frame->len) {
252 SCLogDebug("data entirely past frame");
253 return 1;
254 }
255 uint32_t adjusted_frame_len = (uint32_t)((uint64_t)frame->len - frame_delta);
256 SCLogDebug("frame len after applying offset %" PRIu64 ": %u", frame_delta,
257 adjusted_frame_len);
258
259 data_len = MIN(adjusted_frame_len, input_len);
260 SCLogDebug("usable data len for frame: %u", data_len);
261
262 if ((uint64_t)data_len + frame_delta == (uint64_t)frame->len) {
263 ci_flags |= DETECT_CI_FLAGS_END;
264 SCLogDebug("have frame data end");
265 }
266 } else {
267 data_len = input_len;
268 }
269 }
270 // PrintRawDataFp(stdout, data, data_len);
271 InspectionBufferSetupMulti(buffer, fsd->transforms, data, data_len);
272 buffer->inspect_offset = frame->rel_offset < 0 ? -1 * frame->rel_offset : 0; // TODO review/test
273 buffer->flags = ci_flags;
274 return 1; // for now only the first chunk
275}

277InspectionBuffer *DetectFrame2InspectBuffer(DetectEngineThreadCtx *det_ctx,
278 const DetectEngineTransforms *transforms, Packet *p, const Frames *frames,
279 const Frame *frame, const int list_id, const uint32_t idx, const bool first)
280{
281 // TODO do we really need multiple buffer support here?
282 InspectionBuffer *buffer = InspectionBufferMultipleForListGet(det_ctx, list_id, idx);
283 if (buffer == NULL)
284 return NULL;
285 if (!first && buffer->inspect != NULL)
286 return buffer;
287
288 BUG_ON(p->flow == NULL);
289
290 if (p->proto == IPPROTO_UDP) {
291 return DetectFrame2InspectBufferUdp(
292 det_ctx, transforms, p, buffer, frames, frame, list_id, idx, first);
293 }
294
295 BUG_ON(p->flow->protoctx == NULL);
296 TcpSession *ssn = p->flow->protoctx;
297 TcpStream *stream;
298 if (PKT_IS_TOSERVER(p)) {
299 stream = &ssn->client;
300 } else {
301 stream = &ssn->server;
302 }
303
304 /*
305 stream: [s ]
306 frame: [r ]
307 progress: |>p
308 rel_offset: 10, len 100
309 progress: 20
310 avail: 90 (complete)
311
312 stream: [s ]
313 frame: [r ]
314 progress: |>p
315 stream: 0, len 59
316 rel_offset: 10, len 100
317 progress: 20
318 avail: 30 (incomplete)
319
320 stream: [s ]
321 frame: [r ]
322 progress: |>p
323 stream: 0, len 200
324 rel_offset: -30, len 100
325 progress: 20
326 avail: 50 (complete)
327 */
328
329 SCLogDebug("frame %" PRIi64 ", len %" PRIi64 ", rel_offset %" PRIi64, frame->id, frame->len,
330 frame->rel_offset);
331
332 uint64_t offset = STREAM_BASE_OFFSET(stream);
333 if (frame->rel_offset > 0) {
334 offset += (uint64_t)frame->rel_offset;
335 }
336 const int64_t frame_start_abs_offset = frame->rel_offset + (int64_t)STREAM_BASE_OFFSET(stream);
337 BUG_ON(frame_start_abs_offset < 0);
338
339 const bool eof = ssn->state == TCP_CLOSED || PKT_IS_PSEUDOPKT(p);
340
341 const uint64_t usable = StreamTcpGetUsable(stream, eof);
342 if (usable <= offset)
343 return NULL;
344
345 struct FrameStreamData fsd = { det_ctx, transforms, frame, list_id, idx, offset,
346 (uint64_t)frame_start_abs_offset };
347 StreamReassembleForFrame(ssn, stream, FrameStreamDataFunc, &fsd, offset, eof);
348 SCLogDebug("offset %" PRIu64, offset);
349
350 InspectionBuffer *ret = InspectionBufferMultipleForListGet(det_ctx, list_id, idx);
351 SCLogDebug("ret %p", ret);
352 return ret;
353}

357 *
358 * \param de_ctx Detection engine context
359 * \param det_ctx Detection engine thread context

367int DetectEngineInspectFrameBufferGeneric(DetectEngineThreadCtx *det_ctx,
368 const DetectEngineFrameInspectionEngine *engine, const Signature *s, Packet *p,
369 const Frames *frames, const Frame *frame, const uint32_t idx)
370{
371 const int list_id = engine->sm_list;
372 SCLogDebug("running inspect on %d", list_id);
373
374 SCLogDebug("list %d transforms %p", engine->sm_list, engine->v1.transforms);
375
376 /* if prefilter didn't already run, we need to consider transformations */
377 const DetectEngineTransforms *transforms = NULL;
378 if (!engine->mpm) {
379 transforms = engine->v1.transforms;
380 }
381
382 const InspectionBuffer *buffer =
383 DetectFrame2InspectBuffer(det_ctx, transforms, p, frames, frame, list_id, idx, false);
384 if (unlikely(buffer == NULL)) {
385 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
386 }
387
388 const uint32_t data_len = buffer->inspect_len;
389 const uint8_t *data = buffer->inspect;
390 const uint64_t offset = buffer->inspect_offset;
391
392 det_ctx->discontinue_matching = 0;
393 det_ctx->buffer_offset = 0;
394 det_ctx->inspection_recursion_counter = 0;
395#ifdef DEBUG
396 const uint8_t ci_flags = buffer->flags;
397 SCLogDebug("frame %p rel_offset %" PRIi64 " type %u len %" PRIi64
398 " ci_flags %02x (start:%s, end:%s)",
399 frame, frame->rel_offset, frame->type, frame->len, ci_flags,
400 (ci_flags & DETECT_CI_FLAGS_START) ? "true" : "false",
401 (ci_flags & DETECT_CI_FLAGS_END) ? "true" : "false");
402 SCLogDebug("buffer %p offset %" PRIu64 " len %u ci_flags %02x (start:%s, end:%s)", buffer,
403 buffer->inspect_offset, buffer->inspect_len, ci_flags,
404 (ci_flags & DETECT_CI_FLAGS_START) ? "true" : "false",
405 (ci_flags & DETECT_CI_FLAGS_END) ? "true" : "false");
406 // PrintRawDataFp(stdout, data, data_len);
407 // PrintRawDataFp(stdout, data, MIN(64, data_len));
408#endif
409 BUG_ON(frame->len > 0 && (int64_t)data_len > frame->len);
410
411 // TODO don't call if matching needs frame end and DETECT_CI_FLAGS_END not set
412 // TODO same for start
413 int r = DetectEngineContentInspection(det_ctx->de_ctx, det_ctx, s, engine->smd, p, p->flow,
414 (uint8_t *)data, data_len, offset, buffer->flags,
415 DETECT_ENGINE_CONTENT_INSPECTION_MODE_FRAME);
416 if (r == 1) {
417 return DETECT_ENGINE_INSPECT_SIG_MATCH;
418 } else {
419 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
420 }
421}
