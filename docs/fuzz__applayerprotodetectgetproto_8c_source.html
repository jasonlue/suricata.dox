28{
29 Flow *f;

35 if (size < HEADER_LEN) {
36 return 0;
37 }
38
39 if (alpd_tctx == NULL) {
40 //global init
41 InitGlobal();
42 run_mode = RUNMODE_UNITTEST;
43 if (ConfYamlLoadString(configNoChecksum, strlen(configNoChecksum)) != 0) {
44 abort();
45 }
46 MpmTableSetup();
47 SpmTableSetup();

53
54 f = TestHelperBuildFlow(AF_INET, "1.2.3.4", "5.6.7.8", (uint16_t)((data[2] << 8) | data[3]),
55 (uint16_t)((data[4] << 8) | data[5]));
56 if (f == NULL) {
57 return 0;
58 }
59 f->proto = data[1];
60 memset(&ssn, 0, sizeof(TcpSession));
61 f->protoctx = &ssn;
62 f->protomap = FlowGetProtoMapping(f->proto);
63
64 uint8_t flags = STREAM_TOCLIENT;
65 if (data[0] & STREAM_TOSERVER) {
66 flags = STREAM_TOSERVER;
67 }
68 alproto = AppLayerProtoDetectGetProto(
69 alpd_tctx, f, data + HEADER_LEN, size - HEADER_LEN, f->proto, flags, &reverse);
70 if (alproto != ALPROTO_UNKNOWN && alproto != ALPROTO_FAILED && f->proto == IPPROTO_TCP) {
71 /* If we find a valid protocol at the start of a stream :
72 * check that with smaller input
73 * we find the same protocol or ALPROTO_UNKNOWN.
74 * Otherwise, we have evasion with TCP splitting
75 */
76 for (size_t i = 0; i < size-HEADER_LEN && i < PROTO_DETECT_MAX_LEN; i++) {
77 // reset detection at each try cf probing_parser_toserver_alproto_masks
78 AppLayerProtoDetectReset(f);
79 alproto2 = AppLayerProtoDetectGetProto(
80 alpd_tctx, f, data + HEADER_LEN, i, f->proto, flags, &reverse);
81 if (alproto2 != ALPROTO_UNKNOWN && alproto2 != alproto) {
82 printf("Failed with input length %" PRIuMAX " versus %" PRIuMAX
83 ", found %s instead of %s\n",
84 (uintmax_t)i, (uintmax_t)size - HEADER_LEN, AppProtoToString(alproto2),
85 AppProtoToString(alproto));
86 printf("Assertion failure: %s-%s\n", AppProtoToString(alproto2),
87 AppProtoToString(alproto));
88 fflush(stdout);
89 abort();
90 }
91 }
92 }
93 FlowFree(f);
94
95 return 0;
