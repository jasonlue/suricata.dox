84static int LogTcpDataLoggerDir(ThreadVars *tv, void *thread_data, const Flow *f,
85 const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
86{
87 SCEnter();
88 LogTcpDataLogThread *aft = thread_data;
89 LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
90 const char *mode = "a";
91
92 if (flags & OUTPUT_STREAMING_FLAG_OPEN)
93 mode = "w";
94
95 if (data && data_len) {
96 char srcip[46] = "", dstip[46] = "";
97 if (FLOW_IS_IPV4(f)) {
98 PrintInet(AF_INET, (const void *)&f->src.addr_data32[0], srcip, sizeof(srcip));
99 PrintInet(AF_INET, (const void *)&f->dst.addr_data32[0], dstip, sizeof(dstip));
100 } else if (FLOW_IS_IPV6(f)) {
101 PrintInet(AF_INET6, (const void *)f->src.addr_data32, srcip, sizeof(srcip));
102 PrintInet(AF_INET6, (const void *)f->dst.addr_data32, dstip, sizeof(dstip));
103 }
104
105 char name[PATH_MAX];
106
107 char tx[64] = { 0 };
108 if (flags & OUTPUT_STREAMING_FLAG_TRANSACTION) {
109 snprintf(tx, sizeof(tx), "%"PRIu64, tx_id);
110 }
111
112 snprintf(name, sizeof(name), "%s/%s/%s_%u-%s_%u-%s-%s.data",
113 td->log_dir,
114 td->type == STREAMING_HTTP_BODIES ? "http" : "tcp",
115 srcip, f->sp, dstip, f->dp, tx,
116 flags & OUTPUT_STREAMING_FLAG_TOSERVER ? "ts" : "tc");
117
118 FILE *fp = fopen(name, mode);
119 BUG_ON(fp == NULL);
120
121 // PrintRawDataFp(stdout, (uint8_t *)data, data_len);
122 fwrite(data, data_len, 1, fp);
123
124 fclose(fp);
125 }
126 SCReturnInt(TM_ECODE_OK);
127}

129static int LogTcpDataLoggerFile(ThreadVars *tv, void *thread_data, const Flow *f,
130 const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
131{
132 SCEnter();
133 LogTcpDataLogThread *aft = thread_data;
134 LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
135
136 if (data && data_len) {
137 MemBufferReset(aft->buffer);
138
139 char srcip[46] = "", dstip[46] = "";
140 if (FLOW_IS_IPV4(f)) {
141 PrintInet(AF_INET, (const void *)&f->src.addr_data32[0], srcip, sizeof(srcip));
142 PrintInet(AF_INET, (const void *)&f->dst.addr_data32[0], dstip, sizeof(dstip));
143 } else if (FLOW_IS_IPV6(f)) {
144 PrintInet(AF_INET6, (const void *)f->src.addr_data32, srcip, sizeof(srcip));
145 PrintInet(AF_INET6, (const void *)f->dst.addr_data32, dstip, sizeof(dstip));
146 }
147
148 char name[PATH_MAX];
149 snprintf(name, sizeof(name), "%s_%u-%s_%u-%s:",
150 srcip, f->sp, dstip, f->dp,
151 flags & OUTPUT_STREAMING_FLAG_TOSERVER ? "ts" : "tc");
152
153 PrintRawUriBuf((char *)aft->buffer->buffer, &aft->buffer->offset,
154 aft->buffer->size, (uint8_t *)name,strlen(name));
155 MemBufferWriteString(aft->buffer, "\n");
156
157 PrintRawDataToBuffer(aft->buffer->buffer, &aft->buffer->offset,
158 aft->buffer->size, (uint8_t *)data,data_len);
159
160 td->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
161 MEMBUFFER_OFFSET(aft->buffer), td->file_ctx);
162 }
163 SCReturnInt(TM_ECODE_OK);
164}

166int LogTcpDataLogger(ThreadVars *tv, void *thread_data, const Flow *f,
167 const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
168{
169 SCEnter();
170 LogTcpDataLogThread *aft = thread_data;
171 LogTcpDataFileCtx *td = aft->tcpdatalog_ctx;
172
173 if (td->dir == 1)
174 LogTcpDataLoggerDir(tv, thread_data, f, data, data_len, tx_id, flags);
175 if (td->file == 1)
176 LogTcpDataLoggerFile(tv, thread_data, f, data, data_len, tx_id, flags);
177
178 SCReturnInt(TM_ECODE_OK);
179}

181TmEcode LogTcpDataLogThreadInit(ThreadVars *t, const void *initdata, void **data)
182{
183 LogTcpDataLogThread *aft = SCMalloc(sizeof(LogTcpDataLogThread));
184 if (unlikely(aft == NULL))
185 return TM_ECODE_FAILED;
186 memset(aft, 0, sizeof(LogTcpDataLogThread));
187
188 if(initdata == NULL)
189 {
190 SCLogDebug("Error getting context. \"initdata\" argument NULL");
191 SCFree(aft);
192 return TM_ECODE_FAILED;
193 }
194
195 aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
196 if (aft->buffer == NULL) {
197 SCFree(aft);
198 return TM_ECODE_FAILED;
199 }
200
201 /* Use the Ouptut Context (file pointer and mutex) */
202 aft->tcpdatalog_ctx= ((OutputCtx *)initdata)->data;
203
204 *data = (void *)aft;
205 return TM_ECODE_OK;
206}

208TmEcode LogTcpDataLogThreadDeinit(ThreadVars *t, void *data)
209{
210 LogTcpDataLogThread *aft = (LogTcpDataLogThread *)data;
211 if (aft == NULL) {
212 return TM_ECODE_OK;
213 }
214
215 MemBufferFree(aft->buffer);
216 /* clear memory */
217 memset(aft, 0, sizeof(LogTcpDataLogThread));
218
219 SCFree(aft);
220 return TM_ECODE_OK;
221}

223/** \brief Create a new http log LogFileCtx.
224 * \param conf Pointer to ConfNode containing this loggers configuration.
225 * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
226 * */
227OutputInitResult LogTcpDataLogInitCtx(ConfNode *conf)
228{
229 OutputInitResult result = { NULL, false };
230 char filename[PATH_MAX] = "";
231 char dirname[32] = "";
232 strlcpy(filename, DEFAULT_LOG_FILENAME, sizeof(filename));
233
234 LogFileCtx *file_ctx = LogFileNewCtx();
235 if(file_ctx == NULL) {
236 SCLogError(SC_ERR_TCPDATA_LOG_GENERIC, "couldn't create new file_ctx");
237 return result;
238 }
239
240 LogTcpDataFileCtx *tcpdatalog_ctx = SCMalloc(sizeof(LogTcpDataFileCtx));
241 if (unlikely(tcpdatalog_ctx == NULL)) {
242 LogFileFreeCtx(file_ctx);
243 return result;
244 }
245 memset(tcpdatalog_ctx, 0x00, sizeof(LogTcpDataFileCtx));
246
247 tcpdatalog_ctx->file_ctx = file_ctx;
248
249 if (conf) {
250 if (conf->name) {
251 if (strcmp(conf->name, "tcp-data") == 0) {
252 tcpdatalog_ctx->type = STREAMING_TCP_DATA;
253 snprintf(filename, sizeof(filename), "%s.log", conf->name);
254 strlcpy(dirname, "tcp", sizeof(dirname));
255 } else if (strcmp(conf->name, "http-body-data") == 0) {
256 tcpdatalog_ctx->type = STREAMING_HTTP_BODIES;
257 snprintf(filename, sizeof(filename), "%s.log", conf->name);
258 strlcpy(dirname, "http", sizeof(dirname));
259 }
260 }
261
262 const char *logtype = ConfNodeLookupChildValue(conf, "type");
263 if (logtype == NULL)
264 logtype = "file";
265
266 if (strcmp(logtype, "file") == 0) {
267 tcpdatalog_ctx->file = 1;
268 } else if (strcmp(logtype, "dir") == 0) {
269 tcpdatalog_ctx->dir = 1;
270 } else if (strcmp(logtype, "both") == 0) {
271 tcpdatalog_ctx->file = 1;
272 tcpdatalog_ctx->dir = 1;
273 }
274 } else {
275 tcpdatalog_ctx->file = 1;
276 tcpdatalog_ctx->dir = 0;
277 }
278
279 if (tcpdatalog_ctx->file == 1) {
280 SCLogInfo("opening logfile");
281 if (SCConfLogOpenGeneric(conf, file_ctx, filename, 1) < 0) {
282 LogFileFreeCtx(file_ctx);
283 SCFree(tcpdatalog_ctx);
284 return result;
285 }
286 }
287
288 if (tcpdatalog_ctx->dir == 1) {
289 tcpdatalog_ctx->log_dir = ConfigGetLogDirectory();
290 char dirfull[PATH_MAX];
291
292 /* create the filename to use */
293 snprintf(dirfull, PATH_MAX, "%s/%s", tcpdatalog_ctx->log_dir, dirname);
294
295 SCLogInfo("using directory %s", dirfull);
296
297 /* if mkdir fails file open will fail, so deal with errors there */
298 (void)SCMkDir(dirfull, 0700);
299 }
300
301 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
302 if (unlikely(output_ctx == NULL)) {
303 goto parsererror;
304 }
305
306 output_ctx->data = tcpdatalog_ctx;
307 output_ctx->DeInit = LogTcpDataLogDeInitCtx;
308
309 SCLogDebug("Streaming log output initialized");
310 result.ctx = output_ctx;
311 result.ok = true;
312 return result;
313
314parsererror:
315 LogFileFreeCtx(file_ctx);
316 SCFree(tcpdatalog_ctx);
317 SCLogError(SC_ERR_INVALID_ARGUMENT,"Syntax error in custom http log format string.");
318 return result;
319
320}

327 SCFree(output_ctx);
