159/**
160 * \brief If we are on IPS mode, and got a drop action triggered from
161 * the IP only module, or from a reassembled msg and/or from an
162 * applayer detection, then drop the rest of the packets of the
163 * same stream and avoid inspecting it any further
164 * \param p pointer to the Packet to check
165 * \retval 1 if we must drop this stream
166 * \retval 0 if the stream still legal
167 */
168static inline int StreamTcpCheckFlowDrops(Packet *p)
169{
170 /* If we are on IPS mode, and got a drop action triggered from
171 * the IP only module, or from a reassembled msg and/or from an
172 * applayer detection, then drop the rest of the packets of the
173 * same stream and avoid inspecting it any further */
174 if (EngineModeIsIPS() && (p->flow->flags & FLOW_ACTION_DROP))
175 return 1;
176
177 return 0;
178}
