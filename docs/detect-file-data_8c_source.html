171static void SetupDetectEngineConfig(DetectEngineCtx *de_ctx) {
172 if (de_ctx->filedata_config_initialized)
173 return;
174
175 /* initialize default */
176 for (int i = 0; i < (int)ALPROTO_MAX; i++) {
177 de_ctx->filedata_config[i].content_limit = FILEDATA_CONTENT_LIMIT;
178 de_ctx->filedata_config[i].content_inspect_min_size = FILEDATA_CONTENT_INSPECT_MIN_SIZE;
179 de_ctx->filedata_config[i].content_inspect_window = FILEDATA_CONTENT_INSPECT_WINDOW;
180 }
181
182 /* add protocol specific settings here */
183
184 /* SMTP */
185 de_ctx->filedata_config[ALPROTO_SMTP].content_limit = smtp_config.content_limit;
186 de_ctx->filedata_config[ALPROTO_SMTP].content_inspect_min_size = smtp_config.content_inspect_min_size;
187 de_ctx->filedata_config[ALPROTO_SMTP].content_inspect_window = smtp_config.content_inspect_window;
188
189 de_ctx->filedata_config_initialized = true;
190}

192/**
193 * \brief this function is used to parse filedata options
194 * \brief into the current signature
195 *
196 * \param de_ctx pointer to the Detection Engine Context
197 * \param s pointer to the Current Signature
198 * \param str pointer to the user provided "filestore" option
199 *
200 * \retval 0 on Success
201 * \retval -1 on Failure
202 */
203static int DetectFiledataSetup (DetectEngineCtx *de_ctx, Signature *s, const char *str)
204{
205 SCEnter();
206
207 if (!DetectProtoContainsProto(&s->proto, IPPROTO_TCP) ||
208 (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_HTTP1 &&
209 s->alproto != ALPROTO_SMTP && s->alproto != ALPROTO_SMB &&
210 s->alproto != ALPROTO_HTTP2 && s->alproto != ALPROTO_FTP &&
211 s->alproto != ALPROTO_FTPDATA && s->alproto != ALPROTO_HTTP &&
212 s->alproto != ALPROTO_NFS)) {
213 SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting keywords.");
214 return -1;
215 }
216
217 if (s->alproto == ALPROTO_SMTP && (s->init_data->init_flags & SIG_FLAG_INIT_FLOW) &&
218 !(s->flags & SIG_FLAG_TOSERVER) && (s->flags & SIG_FLAG_TOCLIENT)) {
219 SCLogError(SC_ERR_INVALID_SIGNATURE, "Can't use file_data with "
220 "flow:to_client or flow:from_server with smtp.");
221 return -1;
222 }
223
224 if (DetectBufferSetActiveList(s, DetectBufferTypeGetByName("file_data")) < 0)
225 return -1;
226
227 s->init_data->init_flags |= SIG_FLAG_INIT_FILEDATA;
228 SetupDetectEngineConfig(de_ctx);
229 return 0;
230}

262static inline HtpBody *GetResponseBody(htp_tx_t *tx)
263{
264 HtpTxUserData *htud = (HtpTxUserData *)htp_tx_get_user_data(tx);
265 if (htud == NULL) {
266 SCLogDebug("no htud");
267 return NULL;
268 }
269
270 return &htud->response_body;
271}

274 const DetectEngineTransforms *transforms, const int list_id, InspectionBuffer *base_buffer)
275{

287static InspectionBuffer *HttpServerBodyGetDataCallback(DetectEngineThreadCtx *det_ctx,
288 const DetectEngineTransforms *transforms, Flow *f, const uint8_t flow_flags, void *txv,
289 const int list_id, const int base_id)
290{
291 SCEnter();
292
293 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, base_id);
294 if (base_id != list_id && buffer->inspect != NULL)
295 return HttpServerBodyXformsGetDataCallback(det_ctx, transforms, list_id, buffer);
296 else if (buffer->inspect != NULL)
297 return buffer;
298
299 htp_tx_t *tx = txv;
300 HtpState *htp_state = f->alstate;
301 const uint8_t flags = flow_flags;
302
303 HtpBody *body = GetResponseBody(tx);
304 if (body == NULL) {
305 return NULL;
306 }
307
308 /* no new data */
309 if (body->body_inspected == body->content_len_so_far) {
310 SCLogDebug("no new data");
311 return NULL;
312 }
313
314 HtpBodyChunk *cur = body->first;
315 if (cur == NULL) {
316 SCLogDebug("No http chunks to inspect for this transaction");
317 return NULL;
318 }
319
320 SCLogDebug("response.body_limit %u response_body.content_len_so_far %" PRIu64
321 ", response.inspect_min_size %" PRIu32 ", EOF %s, progress > body? %s",
322 htp_state->cfg->response.body_limit, body->content_len_so_far,
323 htp_state->cfg->response.inspect_min_size, flags & STREAM_EOF ? "true" : "false",
324 (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, tx, flags) >
325 HTP_RESPONSE_BODY)
326 ? "true"
327 : "false");
328
329 if (!htp_state->cfg->http_body_inline) {
330 /* inspect the body if the transfer is complete or we have hit
331 * our body size limit */
332 if ((htp_state->cfg->response.body_limit == 0 ||
333 body->content_len_so_far < htp_state->cfg->response.body_limit) &&
334 body->content_len_so_far < htp_state->cfg->response.inspect_min_size &&
335 !(AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, tx, flags) >
336 HTP_RESPONSE_BODY) &&
337 !(flags & STREAM_EOF)) {
338 SCLogDebug("we still haven't seen the entire response body. "
339 "Let's defer body inspection till we see the "
340 "entire body.");
341 return NULL;
342 }
343 }
344
345 /* get the inspect buffer
346 *
347 * make sure that we have at least the configured inspect_win size.
348 * If we have more, take at least 1/4 of the inspect win size before
349 * the new data.
350 */
351 uint64_t offset = 0;
352 if (body->body_inspected > htp_state->cfg->response.inspect_min_size) {
353 BUG_ON(body->content_len_so_far < body->body_inspected);
354 uint64_t inspect_win = body->content_len_so_far - body->body_inspected;
355 SCLogDebug("inspect_win %"PRIu64, inspect_win);
356 if (inspect_win < htp_state->cfg->response.inspect_window) {
357 uint64_t inspect_short = htp_state->cfg->response.inspect_window - inspect_win;
358 if (body->body_inspected < inspect_short)
359 offset = 0;
360 else
361 offset = body->body_inspected - inspect_short;
362 } else {
363 offset = body->body_inspected - (htp_state->cfg->response.inspect_window / 4);
364 }
365 }
366
367 const uint8_t *data;
368 uint32_t data_len;
369
370 StreamingBufferGetDataAtOffset(body->sb,
371 &data, &data_len, offset);
372 InspectionBufferSetup(det_ctx, base_id, buffer, data, data_len);
373 buffer->inspect_offset = offset;
374 body->body_inspected = body->content_len_so_far;
375 SCLogDebug("body->body_inspected now: %" PRIu64, body->body_inspected);
376
377 /* built-in 'transformation' */
378 if (htp_state->cfg->swf_decompression_enabled) {
379 int swf_file_type = FileIsSwfFile(data, data_len);
380 if (swf_file_type == FILE_SWF_ZLIB_COMPRESSION ||
381 swf_file_type == FILE_SWF_LZMA_COMPRESSION)
382 {
383 (void)FileSwfDecompression(data, data_len,
384 det_ctx,
385 buffer,
386 htp_state->cfg->swf_compression_type,
387 htp_state->cfg->swf_decompress_depth,
388 htp_state->cfg->swf_compress_depth);
389 }
390 }
391
392 if (base_id != list_id) {
393 buffer = HttpServerBodyXformsGetDataCallback(det_ctx, transforms, list_id, buffer);
394 }
395 SCReturnPtr(buffer, "InspectionBuffer");
396}

398static uint8_t DetectEngineInspectBufferHttpBody(DetectEngineCtx *de_ctx,
399 DetectEngineThreadCtx *det_ctx, const DetectEngineAppInspectionEngine *engine,
400 const Signature *s, Flow *f, uint8_t flags, void *alstate, void *txv, uint64_t tx_id)
401{
402 bool eof =
403 (AppLayerParserGetStateProgress(f->proto, f->alproto, txv, flags) > engine->progress);
404 const InspectionBuffer *buffer = HttpServerBodyGetDataCallback(
405 det_ctx, engine->v2.transforms, f, flags, txv, engine->sm_list, engine->sm_list_base);
406 if (buffer == NULL || buffer->inspect == NULL) {
407 return eof ? DETECT_ENGINE_INSPECT_SIG_CANT_MATCH : DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
408 }
409
410 const uint32_t data_len = buffer->inspect_len;
411 const uint8_t *data = buffer->inspect;
412 const uint64_t offset = buffer->inspect_offset;
413
414 uint8_t ci_flags = eof ? DETECT_CI_FLAGS_END : 0;
415 ci_flags |= (offset == 0 ? DETECT_CI_FLAGS_START : 0);
416 ci_flags |= buffer->flags;
417
418 det_ctx->discontinue_matching = 0;
419 det_ctx->buffer_offset = 0;
420 det_ctx->inspection_recursion_counter = 0;
421
422 /* Inspect all the uricontents fetched on each
423 * transaction at the app layer */
424 int r = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd, NULL, f, (uint8_t *)data,
425 data_len, offset, ci_flags, DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
426 if (r == 1) {
427 return DETECT_ENGINE_INSPECT_SIG_MATCH;
428 }
429
430 if (flags & STREAM_TOSERVER) {
431 if (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, txv, flags) >
432 HTP_REQUEST_BODY)
433 return DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
434 } else {
435 if (AppLayerParserGetStateProgress(IPPROTO_TCP, ALPROTO_HTTP1, txv, flags) >
436 HTP_RESPONSE_BODY)
437 return DETECT_ENGINE_INSPECT_SIG_CANT_MATCH;
438 }
439 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
440}

442/** \brief Filedata Filedata Mpm prefilter callback
443 *
444 * \param det_ctx detection engine thread ctx
445 * \param pectx inspection context
446 * \param p packet to inspect
447 * \param f flow to inspect
448 * \param txv tx to inspect
449 * \param idx transaction id
450 * \param flags STREAM_* flags including direction
451 */
452static void PrefilterTxHTTPFiledata(DetectEngineThreadCtx *det_ctx, const void *pectx, Packet *p,
453 Flow *f, void *txv, const uint64_t idx, const uint8_t flags)
454{
455 SCEnter();
456
457 const PrefilterMpmFiledata *ctx = (const PrefilterMpmFiledata *)pectx;

467 (void)mpm_table[mpm_ctx->mpm_type].Search(
468 mpm_ctx, &det_ctx->mtcu, &det_ctx->pmq, buffer->inspect, buffer->inspect_len);
469 PREFILTER_PROFILING_ADD_BYTES(det_ctx, buffer->inspect_len);
470 }
471}

511static InspectionBuffer *FiledataGetDataCallback(DetectEngineThreadCtx *det_ctx,
512 const DetectEngineTransforms *transforms, Flow *f, uint8_t flow_flags, File *cur_file,
513 const int list_id, const int base_id, int local_file_id, bool first)
514{
515 SCEnter();
516 SCLogDebug(
517 "starting: list_id %d base_id %d first %s", list_id, base_id, first ? "true" : "false");
518
519 InspectionBuffer *buffer = InspectionBufferMultipleForListGet(det_ctx, base_id, local_file_id);
520 SCLogDebug("base: buffer %p", buffer);
521 if (buffer == NULL)
522 return NULL;
523 if (base_id != list_id && buffer->inspect != NULL) {
524 SCLogDebug("handle xform %s", (list_id != base_id) ? "true" : "false");
525 return FiledataWithXformsGetDataCallback(
526 det_ctx, transforms, list_id, local_file_id, buffer, first);
527 }
528 if (!first && buffer->inspect != NULL) {
529 SCLogDebug("base_id: %d, not first: use %p", base_id, buffer);
530 return buffer;
531 }
532
533 const uint64_t file_size = FileDataSize(cur_file);
534 const DetectEngineCtx *de_ctx = det_ctx->de_ctx;
535 const uint32_t content_limit = de_ctx->filedata_config[f->alproto].content_limit;
536 const uint32_t content_inspect_min_size = de_ctx->filedata_config[f->alproto].content_inspect_min_size;
537 // TODO this is unused, is that right?
538 //const uint32_t content_inspect_window = de_ctx->filedata_config[f->alproto].content_inspect_window;
539
540 SCLogDebug("[list %d] first: %d, content_limit %u, content_inspect_min_size %u", list_id,
541 first ? 1 : 0, content_limit, content_inspect_min_size);
542
543 SCLogDebug("[list %d] file %p size %" PRIu64 ", state %d", list_id, cur_file, file_size,
544 cur_file->state);
545
546 /* no new data */
547 if (cur_file->content_inspected == file_size) {
548 SCLogDebug("no new data");
549 return NULL;
550 }
551
552 if (file_size == 0) {
553 SCLogDebug("no data to inspect for this transaction");
554 return NULL;
555 }
556
557 if ((content_limit == 0 || file_size < content_limit) &&
558 file_size < content_inspect_min_size &&
559 !(flow_flags & STREAM_EOF) && !(cur_file->state > FILE_STATE_OPENED)) {
560 SCLogDebug("we still haven't seen the entire content. "
561 "Let's defer content inspection till we see the "
562 "entire content.");
563 return NULL;
564 }
565
566 const uint8_t *data;
567 uint32_t data_len;
568
569 StreamingBufferGetDataAtOffset(cur_file->sb,
570 &data, &data_len,
571 cur_file->content_inspected);
572 InspectionBufferSetupMulti(buffer, NULL, data, data_len);
573 SCLogDebug("[list %d] [before] buffer offset %" PRIu64 "; buffer len %" PRIu32
574 "; data_len %" PRIu32 "; file_size %" PRIu64,
575 list_id, buffer->inspect_offset, buffer->inspect_len, data_len, file_size);
576 buffer->inspect_offset = cur_file->content_inspected;
577
578 /* update inspected tracker */
579 cur_file->content_inspected = buffer->inspect_len + buffer->inspect_offset;
580 SCLogDebug("content inspected: %" PRIu64, cur_file->content_inspected);
581
582 /* get buffer for the list id if it is different from the base id */
583 if (list_id != base_id) {
584 SCLogDebug("regular %d has been set up: now handle xforms id %d", base_id, list_id);
585 InspectionBuffer *tbuffer = FiledataWithXformsGetDataCallback(
586 det_ctx, transforms, list_id, local_file_id, buffer, first);
587 SCReturnPtr(tbuffer, "InspectionBuffer");
588 } else {
589 SCLogDebug("regular buffer %p size %u", buffer, buffer->inspect_len);
590 SCReturnPtr(buffer, "InspectionBuffer");
591 }
592}

594static uint8_t DetectEngineInspectFiledata(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
595 const DetectEngineAppInspectionEngine *engine, const Signature *s, Flow *f, uint8_t flags,
596 void *alstate, void *txv, uint64_t tx_id)
597{
598 int r = 0;
599 int match = 0;
600
601 const DetectEngineTransforms *transforms = NULL;
602 if (!engine->mpm) {
603 transforms = engine->v2.transforms;
604 }
605
606 FileContainer *ffc = AppLayerParserGetFiles(f, flags);
607 if (ffc == NULL) {
608 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
609 }
610
611 int local_file_id = 0;
612 File *file = ffc->head;
613 for (; file != NULL; file = file->next) {
614 if (file->txid != tx_id)
615 continue;
616
617 InspectionBuffer *buffer = FiledataGetDataCallback(det_ctx, transforms, f, flags, file,
618 engine->sm_list, engine->sm_list_base, local_file_id, false);
619 if (buffer == NULL)
620 continue;
621
622 bool eof = (file->state == FILE_STATE_CLOSED);
623 uint8_t ciflags = eof ? DETECT_CI_FLAGS_END : 0;
624 if (buffer->inspect_offset == 0)
625 ciflags |= DETECT_CI_FLAGS_START;
626
627 det_ctx->buffer_offset = 0;
628 det_ctx->discontinue_matching = 0;
629 det_ctx->inspection_recursion_counter = 0;
630 match = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd,
631 NULL, f,
632 (uint8_t *)buffer->inspect,
633 buffer->inspect_len,
634 buffer->inspect_offset, ciflags,
635 DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
636 if (match == 1) {
637 r = 1;
638 break;
639 }
640 local_file_id++;
641 }
642
643 if (r == 1)
644 return DETECT_ENGINE_INSPECT_SIG_MATCH;
645 else
646 return DETECT_ENGINE_INSPECT_SIG_NO_MATCH;
647}

689 }
690 }
