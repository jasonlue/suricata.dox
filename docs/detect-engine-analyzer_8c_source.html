132static FpPatternStats fp_pattern_stats[DETECT_SM_LIST_MAX];
133
134static void FpPatternStatsAdd(int list, uint16_t patlen)
135{
136 if (list < 0 || list >= DETECT_SM_LIST_MAX)
137 return;
138
139 FpPatternStats *f = &fp_pattern_stats[list];
140
141 if (f->min == 0)
142 f->min = patlen;
143 else if (patlen < f->min)
144 f->min = patlen;
145
146 if (patlen > f->max)
147 f->max = patlen;
148
149 f->cnt++;
150 f->tot += patlen;
151}

153void EngineAnalysisFP(const DetectEngineCtx *de_ctx, const Signature *s, char *line)
154{
155 int fast_pattern_set = 0;
156 int fast_pattern_only_set = 0;
157 int fast_pattern_chop_set = 0;
158 const DetectContentData *fp_cd = NULL;
159 const SigMatch *mpm_sm = s->init_data->mpm_sm;
160 const int mpm_sm_list = s->init_data->mpm_sm_list;
161
162 if (mpm_sm != NULL) {
163 fp_cd = (DetectContentData *)mpm_sm->ctx;
164 if (fp_cd->flags & DETECT_CONTENT_FAST_PATTERN) {
165 fast_pattern_set = 1;
166 if (fp_cd->flags & DETECT_CONTENT_FAST_PATTERN_ONLY) {
167 fast_pattern_only_set = 1;
168 } else if (fp_cd->flags & DETECT_CONTENT_FAST_PATTERN_CHOP) {
169 fast_pattern_chop_set = 1;
170 }
171 }
172 }
173
174 fprintf(fp_engine_analysis_FD, "== Sid: %u ==\n", s->id);
175 fprintf(fp_engine_analysis_FD, "%s\n", line);
176
177 fprintf(fp_engine_analysis_FD, " Fast Pattern analysis:\n");
178 if (s->init_data->prefilter_sm != NULL) {
179 fprintf(fp_engine_analysis_FD, " Prefilter on: %s\n",
180 sigmatch_table[s->init_data->prefilter_sm->type].name);
181 fprintf(fp_engine_analysis_FD, "\n");
182 return;
183 }
184
185 if (fp_cd == NULL) {
186 fprintf(fp_engine_analysis_FD, " No content present\n");
187 fprintf(fp_engine_analysis_FD, "\n");
188 return;
189 }
190
191 fprintf(fp_engine_analysis_FD, " Fast pattern matcher: ");
192 int list_type = mpm_sm_list;
193 if (list_type == DETECT_SM_LIST_PMATCH)
194 fprintf(fp_engine_analysis_FD, "content\n");
195 else {
196 const char *desc = DetectEngineBufferTypeGetDescriptionById(de_ctx, list_type);
197 const char *name = DetectEngineBufferTypeGetNameById(de_ctx, list_type);
198 if (desc && name) {
199 fprintf(fp_engine_analysis_FD, "%s (%s)\n", desc, name);
200 }
201 }
202
203 int flags_set = 0;
204 fprintf(fp_engine_analysis_FD, " Flags:");
205 if (fp_cd->flags & DETECT_CONTENT_OFFSET) {
206 fprintf(fp_engine_analysis_FD, " Offset");
207 flags_set = 1;
208 } if (fp_cd->flags & DETECT_CONTENT_DEPTH) {
209 fprintf(fp_engine_analysis_FD, " Depth");
210 flags_set = 1;
211 }
212 if (fp_cd->flags & DETECT_CONTENT_WITHIN) {
213 fprintf(fp_engine_analysis_FD, " Within");
214 flags_set = 1;
215 }
216 if (fp_cd->flags & DETECT_CONTENT_DISTANCE) {
217 fprintf(fp_engine_analysis_FD, " Distance");
218 flags_set = 1;
219 }
220 if (fp_cd->flags & DETECT_CONTENT_NOCASE) {
221 fprintf(fp_engine_analysis_FD, " Nocase");
222 flags_set = 1;
223 }
224 if (fp_cd->flags & DETECT_CONTENT_NEGATED) {
225 fprintf(fp_engine_analysis_FD, " Negated");
226 flags_set = 1;
227 }
228 if (flags_set == 0)
229 fprintf(fp_engine_analysis_FD, " None");
230 fprintf(fp_engine_analysis_FD, "\n");
231
232 fprintf(fp_engine_analysis_FD, " Fast pattern set: %s\n", fast_pattern_set ? "yes" : "no");
233 fprintf(fp_engine_analysis_FD, " Fast pattern only set: %s\n",
234 fast_pattern_only_set ? "yes" : "no");
235 fprintf(fp_engine_analysis_FD, " Fast pattern chop set: %s\n",
236 fast_pattern_chop_set ? "yes" : "no");
237 if (fast_pattern_chop_set) {
238 fprintf(fp_engine_analysis_FD, " Fast pattern offset, length: %u, %u\n",
239 fp_cd->fp_chop_offset, fp_cd->fp_chop_len);
240 }
241
242 uint16_t patlen = fp_cd->content_len;
243 uint8_t *pat = SCMalloc(fp_cd->content_len + 1);
244 if (unlikely(pat == NULL)) {
245 FatalError(SC_ERR_FATAL, "Error allocating memory");
246 }
247 memcpy(pat, fp_cd->content, fp_cd->content_len);
248 pat[fp_cd->content_len] = '\0';
249 fprintf(fp_engine_analysis_FD, " Original content: ");
250 PrintRawUriFp(fp_engine_analysis_FD, pat, patlen);
251 fprintf(fp_engine_analysis_FD, "\n");
252
253 if (fast_pattern_chop_set) {
254 SCFree(pat);
255 patlen = fp_cd->fp_chop_len;
256 pat = SCMalloc(fp_cd->fp_chop_len + 1);
257 if (unlikely(pat == NULL)) {
258 exit(EXIT_FAILURE);
259 }
260 memcpy(pat, fp_cd->content + fp_cd->fp_chop_offset, fp_cd->fp_chop_len);
261 pat[fp_cd->fp_chop_len] = '\0';
262 fprintf(fp_engine_analysis_FD, " Final content: ");
263 PrintRawUriFp(fp_engine_analysis_FD, pat, patlen);
264 fprintf(fp_engine_analysis_FD, "\n");
265
266 FpPatternStatsAdd(list_type, patlen);
267 } else {
268 fprintf(fp_engine_analysis_FD, " Final content: ");
269 PrintRawUriFp(fp_engine_analysis_FD, pat, patlen);
270 fprintf(fp_engine_analysis_FD, "\n");
271
272 FpPatternStatsAdd(list_type, patlen);
273 }
274 SCFree(pat);
275
276 fprintf(fp_engine_analysis_FD, "\n");
277 return;
278}

286int SetupFPAnalyzer(void)
287{
288 int fp_engine_analysis_set = 0;
289
290 if ((ConfGetBool("engine-analysis.rules-fast-pattern",
291 &fp_engine_analysis_set)) == 0) {
292 return 0;
293 }
294
295 if (fp_engine_analysis_set == 0)
296 return 0;
297
298 const char *log_dir;
299 log_dir = ConfigGetLogDirectory();
300 snprintf(log_path, sizeof(log_path), "%s/%s", log_dir,
301 "rules_fast_pattern.txt");
302
303 fp_engine_analysis_FD = fopen(log_path, "w");
304 if (fp_engine_analysis_FD == NULL) {
305 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", log_path,
306 strerror(errno));
307 return 0;
308 }
309
310 SCLogInfo("Engine-Analysis for fast_pattern printed to file - %s",
311 log_path);
312
313 struct timeval tval;
314 struct tm *tms;
315 gettimeofday(&tval, NULL);
316 struct tm local_tm;
317 tms = SCLocalTime(tval.tv_sec, &local_tm);
318 fprintf(fp_engine_analysis_FD, "----------------------------------------------"
319 "---------------------\n");
320 fprintf(fp_engine_analysis_FD, "Date: %" PRId32 "/%" PRId32 "/%04d -- "
321 "%02d:%02d:%02d\n",
322 tms->tm_mday, tms->tm_mon + 1, tms->tm_year + 1900, tms->tm_hour,
323 tms->tm_min, tms->tm_sec);
324 fprintf(fp_engine_analysis_FD, "----------------------------------------------"
325 "---------------------\n");
326
327 memset(&fp_pattern_stats, 0, sizeof(fp_pattern_stats));
328 return 1;
329}

337{
338 ConfNode *conf = ConfGetNode("engine-analysis");

342 if (value && ConfValIsTrue(value)) {

346 rule_warnings_only = 1;
347 }
348 if (enabled) {
349 const char *log_dir;
350 log_dir = ConfigGetLogDirectory();
351 snprintf(log_path, sizeof(log_path), "%s/%s", log_dir, "rules_analysis.txt");
352 rule_engine_analysis_FD = fopen(log_path, "w");
353 if (rule_engine_analysis_FD == NULL) {
354 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", log_path, strerror(errno));
355 return 0;
356 }
357
358 SCLogInfo("Engine-Analysis for rules printed to file - %s",
359 log_path);
360
361 struct timeval tval;
362 struct tm *tms;
363 gettimeofday(&tval, NULL);
364 struct tm local_tm;
365 tms = SCLocalTime(tval.tv_sec, &local_tm);
366 fprintf(rule_engine_analysis_FD, "----------------------------------------------"
367 "---------------------\n");
368 fprintf(rule_engine_analysis_FD, "Date: %" PRId32 "/%" PRId32 "/%04d -- "
369 "%02d:%02d:%02d\n",
370 tms->tm_mday, tms->tm_mon + 1, tms->tm_year + 1900, tms->tm_hour,
371 tms->tm_min, tms->tm_sec);
372 fprintf(rule_engine_analysis_FD, "----------------------------------------------"
373 "---------------------\n");
374
375 /*compile regex's for rule analysis*/
376 if (PerCentEncodingSetup()== 0) {
377 fprintf(rule_engine_analysis_FD, "Error compiling regex; can't check for percent encoding in normalized http content.\n");
378 }
379 }
380 }
381 else {
382 SCLogInfo("Conf parameter \"engine-analysis.rules\" not found. "
383 "Defaulting to not printing the rules analysis report.");
384 }
385 if (!enabled) {
386 SCLogInfo("Engine-Analysis for rules disabled in conf file.");
387 return 0;
388 }
389 return 1;
390}

429/**
430 * \brief Compiles regex for rule analysis
431 * \retval 1 if successful
432 * \retval 0 if on error
433 */
434int PerCentEncodingSetup ()
435{
436#define DETECT_PERCENT_ENCODING_REGEX "%[0-9|a-f|A-F]{2}"
437 int en;
438 PCRE2_SIZE eo = 0;
439 int opts = 0; // PCRE2_NEWLINE_ANY??
440
441 percent_re = pcre2_compile((PCRE2_SPTR8)DETECT_PERCENT_ENCODING_REGEX, PCRE2_ZERO_TERMINATED,
442 opts, &en, &eo, NULL);
443 if (percent_re == NULL) {
444 PCRE2_UCHAR errbuffer[256];
445 pcre2_get_error_message(en, errbuffer, sizeof(errbuffer));
446 SCLogError(SC_ERR_PCRE_COMPILE, "Compile of \"%s\" failed at offset %d: %s",
447 DETECT_PERCENT_ENCODING_REGEX, (int)eo, errbuffer);
448 return 0;
449 }
450
451 percent_re_match = pcre2_match_data_create_from_pattern(percent_re, NULL);
452 return 1;
453}

455/**
456 * \brief Checks for % encoding in content.
457 * \param Pointer to content
458 * \retval number of matches if content has % encoding
459 * \retval 0 if it doesn't have % encoding
460 * \retval -1 on error
461 */
462static int PerCentEncodingMatch(uint8_t *content, uint16_t content_len)
463{
464 int ret = 0;
465
466 ret = pcre2_match(percent_re, (PCRE2_SPTR8)content, content_len, 0, 0, percent_re_match, NULL);
467 if (ret == -1) {
468 return 0;
469 }
470 else if (ret < -1) {
471 SCLogError(SC_ERR_PCRE_MATCH, "Error parsing content - %s; error code is %d", content, ret);
472 return -1;
473 }
474 return ret;
475}

477static void EngineAnalysisRulesPrintFP(const DetectEngineCtx *de_ctx, const Signature *s)
478{
479 const DetectContentData *fp_cd = NULL;
480 const SigMatch *mpm_sm = s->init_data->mpm_sm;
481 const int mpm_sm_list = s->init_data->mpm_sm_list;
482
483 if (mpm_sm != NULL) {
484 fp_cd = (DetectContentData *)mpm_sm->ctx;
485 }
486
487 if (fp_cd == NULL) {
488 return;
489 }
490
491 uint16_t patlen = fp_cd->content_len;
492 uint8_t *pat = SCMalloc(fp_cd->content_len + 1);
493 if (unlikely(pat == NULL)) {
494 FatalError(SC_ERR_FATAL, "Error allocating memory");
495 }
496 memcpy(pat, fp_cd->content, fp_cd->content_len);
497 pat[fp_cd->content_len] = '\0';
498
499 if (fp_cd->flags & DETECT_CONTENT_FAST_PATTERN_CHOP) {
500 SCFree(pat);
501 patlen = fp_cd->fp_chop_len;
502 pat = SCMalloc(fp_cd->fp_chop_len + 1);
503 if (unlikely(pat == NULL)) {
504 exit(EXIT_FAILURE);
505 }
506 memcpy(pat, fp_cd->content + fp_cd->fp_chop_offset, fp_cd->fp_chop_len);
507 pat[fp_cd->fp_chop_len] = '\0';
508 fprintf(rule_engine_analysis_FD, " Fast Pattern \"");
509 PrintRawUriFp(rule_engine_analysis_FD, pat, patlen);
510 } else {
511 fprintf(rule_engine_analysis_FD, " Fast Pattern \"");
512 PrintRawUriFp(rule_engine_analysis_FD, pat, patlen);
513 }
514 SCFree(pat);
515
516 fprintf(rule_engine_analysis_FD, "\" on \"");
517
518 const int list_type = mpm_sm_list;
519 if (list_type == DETECT_SM_LIST_PMATCH) {
520 int payload = 0;
521 int stream = 0;
522 if (SignatureHasPacketContent(s))
523 payload = 1;
524 if (SignatureHasStreamContent(s))
525 stream = 1;
526 fprintf(rule_engine_analysis_FD, "%s",
527 payload ? (stream ? "payload and reassembled stream" : "payload") : "reassembled stream");
528 }
529 else {
530 const char *desc = DetectEngineBufferTypeGetDescriptionById(de_ctx, list_type);
531 const char *name = DetectEngineBufferTypeGetNameById(de_ctx, list_type);
532 if (desc && name) {
533 fprintf(rule_engine_analysis_FD, "%s (%s)", desc, name);
534 } else if (desc || name) {
535 fprintf(rule_engine_analysis_FD, "%s", desc ? desc : name);
536 }
537
538 }
539
540 fprintf(rule_engine_analysis_FD, "\" ");
541 const DetectBufferType *bt = DetectEngineBufferTypeGetById(de_ctx, list_type);
542 if (bt && bt->transforms.cnt) {
543 fprintf(rule_engine_analysis_FD, "(with %d transform(s)) ", bt->transforms.cnt);
544 }
545 fprintf(rule_engine_analysis_FD, "buffer.\n");
546
547 return;
548}

616static void DumpContent(JsonBuilder *js, const DetectContentData *cd)
617{
618 char pattern_str[1024] = "";
619 DetectContentPatternPrettyPrint(cd, pattern_str, sizeof(pattern_str));
620
621 jb_set_string(js, "pattern", pattern_str);
622 jb_set_uint(js, "length", cd->content_len);
623 jb_set_bool(js, "nocase", cd->flags & DETECT_CONTENT_NOCASE);
624 jb_set_bool(js, "negated", cd->flags & DETECT_CONTENT_NEGATED);
625 jb_set_bool(js, "starts_with", cd->flags & DETECT_CONTENT_STARTS_WITH);
626 jb_set_bool(js, "ends_with", cd->flags & DETECT_CONTENT_ENDS_WITH);
627 jb_set_bool(js, "is_mpm", cd->flags & DETECT_CONTENT_MPM);
628 jb_set_bool(js, "no_double_inspect", cd->flags & DETECT_CONTENT_NO_DOUBLE_INSPECTION_REQUIRED);
629 if (cd->flags & DETECT_CONTENT_OFFSET) {
630 jb_set_uint(js, "offset", cd->offset);
631 }
632 if (cd->flags & DETECT_CONTENT_DEPTH) {

679 jb_close(js);

1187 rule_pcre++;

1190

1211 warn_encoding_norm_http_buf += 1;

1216 rule_flow += 1;
1217 if ((s->flags & SIG_FLAG_TOSERVER) && !(s->flags & SIG_FLAG_TOCLIENT)) {
