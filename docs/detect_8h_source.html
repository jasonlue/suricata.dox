331/* inspection buffer is a simple structure that is passed between prefilter,
332 * transformation functions and inspection functions.
333 * Initially setup with 'orig' ptr and len, transformations can then take
334 * then and fill the 'buf'. Multiple transformations can update the buffer,
335 * both growing and shrinking it.
336 * Prefilter and inspection will only deal with 'inspect'. */
337
338typedef struct InspectionBuffer {
339 const uint8_t *inspect; /**< active pointer, points either to ::buf or ::orig */
340 uint64_t inspect_offset;
341 uint32_t inspect_len; /**< size of active data. See to ::len or ::orig_len */
342 uint8_t flags; /**< DETECT_CI_FLAGS_* for use with DetectEngineContentInspection */
343#ifdef DEBUG_VALIDATION
344 bool multi;
345#endif
346 uint32_t len; /**< how much is in use */
347 uint8_t *buf;
348 uint32_t size; /**< size of the memory allocation */
349
350 uint32_t orig_len;
351 const uint8_t *orig;
352} InspectionBuffer;
