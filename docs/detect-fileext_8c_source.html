92/**
93 * \brief match the specified file extension
94 *
95 * \param t thread local vars
96 * \param det_ctx pattern matcher thread local data
97 * \param f *LOCKED* flow
98 * \param flags direction flags
99 * \param file file being inspected
100 * \param s signature being inspected
101 * \param m sigmatch that we will cast into DetectFileextData
102 *
103 * \retval 0 no match
104 * \retval 1 match
105 */
106static int DetectFileextMatch (DetectEngineThreadCtx *det_ctx,
107 Flow *f, uint8_t flags, File *file, const Signature *s, const SigMatchCtx *m)
108{
109 SCEnter();
110 int ret = 0;
111
112 DetectFileextData *fileext = (DetectFileextData *)m;
113
114 if (file->name == NULL)
115 SCReturnInt(0);
116
117 if (file->name_len <= fileext->len)
118 SCReturnInt(0);
119
120 int offset = file->name_len - fileext->len;
121
122 /* fileext->ext is already in lowercase, as SCMemcmpLowercase requires */
123 if (file->name[offset - 1] == '.' &&
124 SCMemcmpLowercase(fileext->ext, file->name + offset, fileext->len) == 0)
125 {
126 if (!(fileext->flags & DETECT_CONTENT_NEGATED)) {
127 ret = 1;
128 SCLogDebug("File ext found");
129 }
130 } else if (fileext->flags & DETECT_CONTENT_NEGATED) {
131 SCLogDebug("negated match");
132 ret = 1;
133 }
134
135 SCReturnInt(ret);
136}

138/**
139 * \brief This function is used to parse fileet
140 *
141 * \param de_ctx Pointer to the detection engine context
142 * \param str Pointer to the fileext value string
143 *
144 * \retval pointer to DetectFileextData on success
145 * \retval NULL on failure
146 */
147static DetectFileextData *DetectFileextParse (DetectEngineCtx *de_ctx, const char *str, bool negate)
148{
149 DetectFileextData *fileext = NULL;
150
151 /* We have a correct filename option */
152 fileext = SCMalloc(sizeof(DetectFileextData));
153 if (unlikely(fileext == NULL))
154 goto error;
155
156 memset(fileext, 0x00, sizeof(DetectFileextData));
157
158 if (DetectContentDataParse("fileext", str, &fileext->ext, &fileext->len) == -1) {
159 goto error;
160 }
161 uint16_t u;
162 for (u = 0; u < fileext->len; u++)
163 fileext->ext[u] = u8_tolower(fileext->ext[u]);
164
165 if (negate) {
166 fileext->flags |= DETECT_CONTENT_NEGATED;
167 }
168
169 SCLogDebug("flags %02X", fileext->flags);
170 if (fileext->flags & DETECT_CONTENT_NEGATED) {
171 SCLogDebug("negated fileext");
172 }
173
174#ifdef DEBUG
175 if (SCLogDebugEnabled()) {
176 char *ext = SCMalloc(fileext->len + 1);
177 if (ext != NULL) {
178 memcpy(ext, fileext->ext, fileext->len);
179 ext[fileext->len] = '\0';
180 SCLogDebug("will look for fileext %s", ext);
181 SCFree(ext);
182 }
183 }
184#endif
185
186 return fileext;
187
188error:
189 if (fileext != NULL)
190 DetectFileextFree(de_ctx, fileext);
191 return NULL;
192
193}

195/**
196 * \brief this function is used to add the parsed "id" option
197 * into the current signature
198 *
199 * \param de_ctx pointer to the Detection Engine Context
200 * \param s pointer to the Current Signature
201 * \param idstr pointer to the user provided "id" option
202 *
203 * \retval 0 on Success
204 * \retval -1 on Failure
205 */
206static int DetectFileextSetup (DetectEngineCtx *de_ctx, Signature *s, const char *str)
207{
208 DetectFileextData *fileext = DetectFileextParse(de_ctx, str, s->init_data->negated);
209 if (fileext == NULL)
210 return -1;
211
212 /* Okay so far so good, lets get this into a SigMatch
213 * and put it in the Signature. */
214 SigMatch *sm = SigMatchAlloc();
215 if (sm == NULL)
216 goto error;
217
218 sm->type = DETECT_FILEEXT;
219 sm->ctx = (void *)fileext;
220
221 SigMatchAppendSMToList(s, sm, g_file_match_list_id);
222
223 s->file_flags |= (FILE_SIG_NEED_FILE|FILE_SIG_NEED_FILENAME);
224 return 0;
225
226error:
227 DetectFileextFree(de_ctx, fileext);
228 if (sm != NULL)
229 SCFree(sm);
230 return -1;
231}
