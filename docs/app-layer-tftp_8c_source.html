76/**
77 * \brief Probe the input to see if it looks like tftp.
78 *
79 * \retval ALPROTO_TFTP if it looks like tftp, otherwise
80 * ALPROTO_UNKNOWN.
81 */
82static AppProto TFTPProbingParser(Flow *f, uint8_t direction,
83 const uint8_t *input, uint32_t input_len, uint8_t *rdir)
84{
85 /* Very simple test - if there is input, this is tftp.
86 * Also check if it's starting by a zero */
87 if (input_len >= TFTP_MIN_FRAME_LEN && *input == 0) {
88 SCLogDebug("Detected as ALPROTO_TFTP.");
89 return ALPROTO_TFTP;
90 }
91
92 SCLogDebug("Protocol not detected as ALPROTO_TFTP.");
93 return ALPROTO_UNKNOWN;
94}

96static AppLayerResult TFTPParseRequest(Flow *f, void *state, AppLayerParserState *pstate,
97 StreamSlice stream_slice, void *local_data)
98{
99 const uint8_t *input = StreamSliceGetData(&stream_slice);
100 uint32_t input_len = StreamSliceGetDataLen(&stream_slice);
101
102 SCLogDebug("Parsing tftp request: len=%" PRIu32, input_len);
103
104 /* Likely connection closed, we can just return here. */
105 if ((input == NULL || input_len == 0) &&
106 AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF_TS)) {
107 SCReturnStruct(APP_LAYER_OK);
108 }
109
110 /* Probably don't want to create a transaction in this case
111 * either. */
112 if (input == NULL || input_len == 0) {
113 SCReturnStruct(APP_LAYER_OK);
114 }
115
116 int res = rs_tftp_request(state, input, input_len);
117 if (res < 0) {
118 SCReturnStruct(APP_LAYER_ERROR);
119 }
120 SCReturnStruct(APP_LAYER_OK);
121}

160void RegisterTFTPParsers(void)
161{
162 const char *proto_name = "tftp";
163
164 /* Check if TFTP UDP detection is enabled. If it does not exist in
165 * the configuration file then it will be enabled by default. */
166 if (AppLayerProtoDetectConfProtoDetectionEnabled("udp", proto_name)) {
167
168 SCLogDebug("TFTP UDP protocol detection enabled.");
169
170 AppLayerProtoDetectRegisterProtocol(ALPROTO_TFTP, proto_name);
171
172 if (RunmodeIsUnittests()) {
173 SCLogDebug("Unittest mode, registering default configuration.");
174 AppLayerProtoDetectPPRegister(IPPROTO_UDP, TFTP_DEFAULT_PORT,
175 ALPROTO_TFTP, 0, TFTP_MIN_FRAME_LEN,
176 STREAM_TOSERVER, TFTPProbingParser,
177 TFTPProbingParser);
178 } else {
179 if (!AppLayerProtoDetectPPParseConfPorts("udp", IPPROTO_UDP,
180 proto_name, ALPROTO_TFTP,
181 0, TFTP_MIN_FRAME_LEN,
182 TFTPProbingParser, TFTPProbingParser)) {
183 SCLogDebug("No tftp app-layer configuration, enabling tftp"
184 " detection UDP detection on port %s.",
185 TFTP_DEFAULT_PORT);
186 AppLayerProtoDetectPPRegister(IPPROTO_UDP,
187 TFTP_DEFAULT_PORT, ALPROTO_TFTP,
188 0, TFTP_MIN_FRAME_LEN,
189 STREAM_TOSERVER,TFTPProbingParser,
190 TFTPProbingParser);
191 }
192 }
193 } else {
194 SCLogDebug("Protocol detector and parser disabled for TFTP.");
195 return;
196 }
197
198 if (AppLayerParserConfParserEnabled("udp", proto_name)) {
199
200 SCLogDebug("Registering TFTP protocol parser.");
201
202 /* Register functions for state allocation and freeing. A
203 * state is allocated for every new TFTP flow. */
204 AppLayerParserRegisterStateFuncs(IPPROTO_UDP, ALPROTO_TFTP,
205 TFTPStateAlloc, TFTPStateFree);
206
207 /* Register request parser for parsing frame from server to client. */
208 AppLayerParserRegisterParser(IPPROTO_UDP, ALPROTO_TFTP,
209 STREAM_TOSERVER, TFTPParseRequest);
210
211 /* Register response parser for parsing frames from server to client. */
212 AppLayerParserRegisterParser(IPPROTO_UDP, ALPROTO_TFTP,
213 STREAM_TOCLIENT, TFTPParseResponse);
214
215 /* Register a function to be called by the application layer
216 * when a transaction is to be freed. */
217 AppLayerParserRegisterTxFreeFunc(IPPROTO_UDP, ALPROTO_TFTP,
218 TFTPStateTxFree);
219
220 /* Register a function to return the current transaction count. */
221 AppLayerParserRegisterGetTxCnt(IPPROTO_UDP, ALPROTO_TFTP,
222 TFTPGetTxCnt);
223
224 /* Transaction handling. */
225 AppLayerParserRegisterStateProgressCompletionStatus(ALPROTO_TFTP, 1, 1);
226 AppLayerParserRegisterGetStateProgressFunc(IPPROTO_UDP,
227 ALPROTO_TFTP,
228 TFTPGetStateProgress);
229 AppLayerParserRegisterGetTx(IPPROTO_UDP, ALPROTO_TFTP,
230 TFTPGetTx);
231
232 AppLayerParserRegisterGetEventInfo(IPPROTO_UDP, ALPROTO_TFTP,
233 TFTPStateGetEventInfo);
234
235 AppLayerParserRegisterTxDataFunc(IPPROTO_UDP, ALPROTO_TFTP,
236 rs_tftp_get_tx_data);
237 }
238 else {
239 SCLogDebug("TFTP protocol parsing disabled.");
240 }
241}
