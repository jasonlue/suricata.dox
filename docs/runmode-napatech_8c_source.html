101static int NapatechRegisterDeviceStreams(void)
102{
103 /* Display the configuration mode */
104 int use_all_streams;
105
106 if (ConfGetBool("napatech.use-all-streams", &use_all_streams) == 0) {
107 SCLogInfo("Could not find napatech.use-all-streams in config file. Defaulting to \"no\".");
108 use_all_streams = 0;
109 }
110
111 if (ConfGetBool("napatech.auto-config", &auto_config) == 0) {
112 SCLogInfo("napatech.auto-config not found in config file. Defaulting to disabled.");
113 }
114
115 if (ConfGetBool("napatech.hardware-bypass", &use_hw_bypass) == 0) {
116 SCLogInfo("napatech.hardware-bypass not found in config file. Defaulting to disabled.");
117 }
118
119 /* use_all_streams uses existing streams created prior to starting Suricata. auto_config
120 * automatically creates streams. Therefore, these two options are mutually exclusive.
121 */
122 if (use_all_streams && auto_config) {
123 FatalError(SC_ERR_FATAL,
124 "napatech.auto-config cannot be used in configuration file at the same time as napatech.use-all-streams.");
125 }
126
127 /* to use hardware_bypass we need to configure the streams to be consistent.
128 * with the rest of the configuration. Therefore auto_config is not a valid
129 * option.
130 */
131 if (use_hw_bypass && auto_config == 0) {
132 FatalError(SC_ERR_FATAL,
133 "napatech auto-config must be enabled when using napatech.use_hw_bypass.");
134 }
135
136 /* Get the stream ID's either from the conf or by querying Napatech */
137 NapatechStreamConfig stream_config[MAX_STREAMS];
138
139 uint16_t stream_cnt = NapatechGetStreamConfig(stream_config);
140 num_configured_streams = stream_cnt;
141 SCLogDebug("Configuring %d Napatech Streams...", stream_cnt);
142
143 for (uint16_t inst = 0; inst < stream_cnt; ++inst) {
144 char *plive_dev_buf = SCCalloc(1, 9);
145 if (unlikely(plive_dev_buf == NULL)) {
146 FatalError(SC_ERR_FATAL,
147 "Failed to allocate memory for NAPATECH stream counter.");
148 }
149 snprintf(plive_dev_buf, 9, "nt%d", stream_config[inst].stream_id);
150
151 if (auto_config) {
152 if (stream_config[inst].is_active) {
153 SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED,
154 "Registering Napatech device: %s - active stream found.",
155 plive_dev_buf);
156 SCLogError(SC_ERR_NAPATECH_STREAMS_REGISTER_FAILED,
157 "run /opt/napatech3/bin/ntpl -e \"delete=all\" to delete existing stream");
158 FatalError(SC_ERR_FATAL,
159 "or disable auto-config in the conf file before running.");
160 }
161 } else {
162 SCLogInfo("Registering Napatech device: %s - active stream%sfound.",
163 plive_dev_buf, stream_config[inst].is_active ? " " : " NOT ");
164 }
165 LiveRegisterDevice(plive_dev_buf);
166
167 if (first_stream == 0xffff) {
168 first_stream = stream_config[inst].stream_id;
169 }
170 last_stream = stream_config[inst].stream_id;
171 }
172
173 /* Napatech stats come from a separate thread. This will surpress
174 * the counters when suricata exits.
175 */
176 LiveDeviceHasNoStats();
177 return 0;
178}

180static void *NapatechConfigParser(const char *device)
181{
182 /* Expect device to be of the form nt%d where %d is the stream id to use */
183 int dev_len = strlen(device);
184 if (dev_len < 3 || dev_len > 5) {
185 SCLogError(SC_ERR_NAPATECH_PARSE_CONFIG,
186 "Could not parse config for device: %s - invalid length", device);
187 return NULL;
188 }
189
190 struct NapatechStreamDevConf *conf = SCCalloc(1, sizeof (struct NapatechStreamDevConf));
191 if (unlikely(conf == NULL)) {
192 SCLogError(SC_ERR_MEM_ALLOC,
193 "Failed to allocate memory for NAPATECH device name.");
194 return NULL;
195 }
196
197 /* device+2 is a pointer to the beginning of the stream id after the constant nt portion */
198 if (StringParseUint16(&conf->stream_id, 10, 0, device + 2) < 0) {
199 SCLogError(SC_ERR_INVALID_VALUE, "Invalid value for stream_id: %s", device + 2);
200 SCFree(conf);
201 return NULL;
202 }
203
204 /* Set the host buffer allowance for this stream
205 * Right now we just look at the global default - there is no per-stream hba configuration
206 */
207 if (ConfGetInt("napatech.hba", &conf->hba) == 0) {
208 conf->hba = -1;
209 } else {
210 SCLogWarning(SC_WARN_COMPATIBILITY,
211 "Napatech Host Buffer Allocation (hba) will be deprecated in Suricata v7.0.");
212 }
213 return (void *) conf;
214}

223static int NapatechInit(int runmode)
224{
225 int status;
226
227 RunModeInitialize();
228 TimeModeSetLive();
229
230 /* Initialize the API and check version compatibility */
231 if ((status = NT_Init(NTAPI_VERSION)) != NT_SUCCESS) {
232 NAPATECH_ERROR(SC_ERR_NAPATECH_INIT_FAILED, status);
233 exit(EXIT_FAILURE);
234 }
235
236 status = NapatechRegisterDeviceStreams();
237 if (status < 0 || num_configured_streams <= 0) {
238 FatalError(SC_ERR_FATAL,
239 "Unable to find existing Napatech Streams");
240 }
241
242 struct NapatechStreamDevConf *conf =
243 SCCalloc(1, sizeof (struct NapatechStreamDevConf));
244 if (unlikely(conf == NULL)) {
245 FatalError(SC_ERR_FATAL,
246 "Failed to allocate memory for NAPATECH device.");
247 }
248
249 if ((ConfGetInt("napatech.hba", &conf->hba) != 0) && (conf->hba > 0)) {
250 SCLogInfo("Host Buffer Allowance: %d", (int) conf->hba);
251 }
252
253 if (use_hw_bypass) {
254#ifdef NAPATECH_ENABLE_BYPASS
255 if (NapatechVerifyBypassSupport()) {
256 SCLogInfo("Napatech Hardware Bypass is supported and enabled.");
257 } else {
258 FatalError(SC_ERR_FATAL,
259 "Napatech Hardware Bypass requested in conf but is not supported by the hardware.");
260 }
261#else
262 FatalError(SC_ERR_FATAL,
263 "Napatech Hardware Bypass requested in conf but is not enabled by the software.");
264#endif
265 } else {
266 SCLogInfo("Hardware Bypass is disabled in the conf file.");
267 }
268
269 /* Start a thread to process the statistics */
270 NapatechStartStats();
271
272 switch (runmode) {
273 case NT_RUNMODE_WORKERS:
274 status = RunModeSetLiveCaptureWorkers(NapatechConfigParser,
275 NapatechGetThreadsCount,
276 "NapatechStream", "NapatechDecode",
277 thread_name_workers, NULL);
278 break;
279 default:
280 status = -1;
281 }
282
283 if (status != 0) {
284 FatalError(SC_ERR_FATAL, "Runmode start failed");
285 }
286 return 0;
287}
