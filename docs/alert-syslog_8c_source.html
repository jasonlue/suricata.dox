63/**
64 * \brief Function to clear the memory of the output context and closes the
65 * syslog interface
66 *
67 * \param output_ctx pointer to the output context to be cleared
68 */
69static void AlertSyslogDeInitCtx(OutputCtx *output_ctx)
70{
71 if (output_ctx != NULL) {
72 LogFileCtx *logfile_ctx = (LogFileCtx *)output_ctx->data;
73 if (logfile_ctx != NULL) {
74 LogFileFreeCtx(logfile_ctx);
75 }
76 SCFree(output_ctx);
77 }
78 closelog();
79}

81/**
82 * \brief Create a new LogFileCtx for "syslog" output style.
83 *
84 * \param conf The configuration node for this output.
85 * \return A OutputCtx pointer on success, NULL on failure.
86 */
87static OutputInitResult AlertSyslogInitCtx(ConfNode *conf)
88{
89 OutputInitResult result = { NULL, false };
90 const char *facility_s = ConfNodeLookupChildValue(conf, "facility");
91 if (facility_s == NULL) {
92 facility_s = DEFAULT_ALERT_SYSLOG_FACILITY_STR;
93 }
94
95 LogFileCtx *logfile_ctx = LogFileNewCtx();
96 if (logfile_ctx == NULL) {
97 SCLogDebug("AlertSyslogInitCtx: Could not create new LogFileCtx");
98 return result;
99 }
100
101 int facility = SCMapEnumNameToValue(facility_s, SCSyslogGetFacilityMap());
102 if (facility == -1) {
103 SCLogWarning(SC_ERR_INVALID_ARGUMENT, "Invalid syslog facility: \"%s\","
104 " now using \"%s\" as syslog facility", facility_s,
105 DEFAULT_ALERT_SYSLOG_FACILITY_STR);
106 facility = DEFAULT_ALERT_SYSLOG_FACILITY;
107 }
108
109 const char *level_s = ConfNodeLookupChildValue(conf, "level");
110 if (level_s != NULL) {
111 int level = SCMapEnumNameToValue(level_s, SCSyslogGetLogLevelMap());
112 if (level != -1) {
113 alert_syslog_level = level;
114 }
115 }
116
117 const char *ident = ConfNodeLookupChildValue(conf, "identity");
118 /* if null we just pass that to openlog, which will then
119 * figure it out by itself. */
120
121 openlog(ident, LOG_PID|LOG_NDELAY, facility);
122
123 OutputCtx *output_ctx = SCMalloc(sizeof(OutputCtx));
124 if (unlikely(output_ctx == NULL)) {
125 SCLogDebug("could not create new OutputCtx");
126 LogFileFreeCtx(logfile_ctx);
127 return result;
128 }
129 memset(output_ctx, 0x00, sizeof(OutputCtx));
130
131 output_ctx->data = logfile_ctx;
132 output_ctx->DeInit = AlertSyslogDeInitCtx;
133
134 SCLogInfo("Syslog output initialized");
135
136 result.ctx = output_ctx;
137 result.ok = true;
138 return result;
139}

141/**
142 * \brief Function to initialize the AlertSystlogThread and sets the output
143 * context pointer
144 *
145 * \param tv Pointer to the threadvars
146 * \param initdata Pointer to the output context
147 * \param data pointer to pointer to point to the AlertSyslogThread
148 */
149static TmEcode AlertSyslogThreadInit(ThreadVars *t, const void *initdata, void **data)
150{
151 if(initdata == NULL) {
152 SCLogDebug("Error getting context for AlertSyslog. \"initdata\" "
153 "argument NULL");
154 return TM_ECODE_FAILED;
155 }
156
157 AlertSyslogThread *ast = SCMalloc(sizeof(AlertSyslogThread));
158 if (unlikely(ast == NULL))
159 return TM_ECODE_FAILED;
160
161 memset(ast, 0, sizeof(AlertSyslogThread));
162
163 /** Use the Ouptut Context (file pointer and mutex) */
164 ast->file_ctx = ((OutputCtx *)initdata)->data;
165
166 *data = (void *)ast;
167 return TM_ECODE_OK;
168}

190/**
191 * \brief Function which is called to print the IPv4 alerts to the syslog
192 *
193 * \param tv Pointer to the threadvars
194 * \param p Pointer to the packet
195 * \param data pointer to the AlertSyslogThread
196 *
197 * \return On succes return TM_ECODE_OK
198 */
199static TmEcode AlertSyslogIPv4(ThreadVars *tv, const Packet *p, void *data)
200{
201 AlertSyslogThread *ast = (AlertSyslogThread *)data;
202 int i;
203 const char *action = "";
204
205 if (p->alerts.cnt == 0)
206 return TM_ECODE_OK;
207
208 char proto[16] = "";
209 const char *protoptr;
210 if (SCProtoNameValid(IPV4_GET_IPPROTO(p))) {
211 protoptr = known_proto[IPV4_GET_IPPROTO(p)];
212 } else {
213 snprintf(proto, sizeof(proto), "PROTO:%03" PRIu32, IPV4_GET_IPPROTO(p));
214 protoptr = proto;
215 }
216
217 /* Not sure if this mutex is needed around calls to syslog. */
218 SCMutexLock(&ast->file_ctx->fp_mutex);
219
220 for (i = 0; i < p->alerts.cnt; i++) {
221 const PacketAlert *pa = &p->alerts.alerts[i];
222 if (unlikely(pa->s == NULL)) {
223 continue;
224 }
225
226 char srcip[16], dstip[16];
227
228 PrintInet(AF_INET, (const void *)GET_IPV4_SRC_ADDR_PTR(p), srcip, sizeof(srcip));
229 PrintInet(AF_INET, (const void *)GET_IPV4_DST_ADDR_PTR(p), dstip, sizeof(dstip));
230
231 if ((pa->action & ACTION_DROP) && EngineModeIsIPS()) {
232 action = "[Drop] ";
233 } else if (pa->action & ACTION_DROP) {
234 action = "[wDrop] ";
235 }
236
237 syslog(alert_syslog_level, "%s[%" PRIu32 ":%" PRIu32 ":%"
238 PRIu32 "] %s [Classification: %s] [Priority: %"PRIu32"]"
239 " {%s} %s:%" PRIu32 " -> %s:%" PRIu32 "", action, pa->s->gid,
240 pa->s->id, pa->s->rev, pa->s->msg, pa->s->class_msg, pa->s->prio,
241 protoptr, srcip, p->sp, dstip, p->dp);
242 }
243 SCMutexUnlock(&ast->file_ctx->fp_mutex);
244
245 return TM_ECODE_OK;
246}

307/**
308 * \brief Function which is called to print the decode alerts to the syslog
309 *
310 * \param tv Pointer to the threadvars
311 * \param p Pointer to the packet
312 * \param data pointer to the AlertSyslogThread
313 *
314 * \return On succes return TM_ECODE_OK
315 */
316static TmEcode AlertSyslogDecoderEvent(ThreadVars *tv, const Packet *p, void *data)
317{
318 AlertSyslogThread *ast = (AlertSyslogThread *)data;
319 int i;
320 const char *action = "";
321
322 if (p->alerts.cnt == 0)
323 return TM_ECODE_OK;
324
325 SCMutexLock(&ast->file_ctx->fp_mutex);
326
327 char temp_buf_hdr[512];
328 char temp_buf_pkt[65] = "";
329 char temp_buf_tail[64];
330 char alert[2048] = "";
331
332 for (i = 0; i < p->alerts.cnt; i++) {
333 const PacketAlert *pa = &p->alerts.alerts[i];
334 if (unlikely(pa->s == NULL)) {
335 continue;
336 }
337
338 if ((pa->action & ACTION_DROP) && EngineModeIsIPS()) {
339 action = "[Drop] ";
340 } else if (pa->action & ACTION_DROP) {
341 action = "[wDrop] ";
342 }
343
344 snprintf(temp_buf_hdr, sizeof(temp_buf_hdr), "%s[%" PRIu32 ":%" PRIu32
345 ":%" PRIu32 "] %s [Classification: %s] [Priority: %" PRIu32
346 "] [**] [Raw pkt: ", action, pa->s->gid, pa->s->id, pa->s->rev, pa->s->msg,
347 pa->s->class_msg, pa->s->prio);
348 strlcpy(alert, temp_buf_hdr, sizeof(alert));
349
350 PrintRawLineHexBuf(temp_buf_pkt, sizeof(temp_buf_pkt), GET_PKT_DATA(p), GET_PKT_LEN(p) < 32 ? GET_PKT_LEN(p) : 32);
351 strlcat(alert, temp_buf_pkt, sizeof(alert));
352
353 if (p->pcap_cnt != 0) {
354 snprintf(temp_buf_tail, sizeof(temp_buf_tail), "] [pcap file packet: %"PRIu64"]",
355 p->pcap_cnt);
356 } else {
357 temp_buf_tail[0] = ']';
358 temp_buf_tail[1] = '\0';
359 }
360 strlcat(alert, temp_buf_tail, sizeof(alert));
361
362 syslog(alert_syslog_level, "%s", alert);
363 }
364 SCMutexUnlock(&ast->file_ctx->fp_mutex);
365
366 return TM_ECODE_OK;
367}

369static int AlertSyslogCondition(ThreadVars *tv, void *thread_data, const Packet *p)
370{
371 return (p->alerts.cnt > 0 ? TRUE : FALSE);
372}
373
374static int AlertSyslogLogger(ThreadVars *tv, void *thread_data, const Packet *p)
375{
376 if (PKT_IS_IPV4(p)) {
377 return AlertSyslogIPv4(tv, p, thread_data);
378 } else if (PKT_IS_IPV6(p)) {
379 return AlertSyslogIPv6(tv, p, thread_data);
380 } else if (p->events.cnt > 0) {
381 return AlertSyslogDecoderEvent(tv, p, thread_data);
382 }
383
384 return TM_ECODE_OK;
385}
