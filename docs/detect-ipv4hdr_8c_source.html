47/**
48 * \brief Registration function for ipv4.hdr: keyword
49 */
50void DetectIpv4hdrRegister(void)
51{
52 sigmatch_table[DETECT_IPV4HDR].name = "ipv4.hdr";
53 sigmatch_table[DETECT_IPV4HDR].desc = "sticky buffer to match on the IPV4 header";
54 sigmatch_table[DETECT_IPV4HDR].url = "/rules/header-keywords.html#ipv4hdr";
55 sigmatch_table[DETECT_IPV4HDR].Setup = DetectIpv4hdrSetup;
56 sigmatch_table[DETECT_IPV4HDR].flags |= SIGMATCH_NOOPT | SIGMATCH_INFO_STICKY_BUFFER;
57#ifdef UNITTESTS
58 sigmatch_table[DETECT_IPV4HDR].RegisterTests = DetectIpv4hdrRegisterTests;
59#endif
60
61 g_ipv4hdr_buffer_id = DetectBufferTypeRegister("ipv4.hdr");
62 BUG_ON(g_ipv4hdr_buffer_id < 0);
63
64 DetectBufferTypeSupportsPacket("ipv4.hdr");
65
66 DetectPktMpmRegister("ipv4.hdr", 2, PrefilterGenericMpmPktRegister, GetData);
67
68 DetectPktInspectEngineRegister("ipv4.hdr", GetData,
69 DetectEngineInspectPktBufferGeneric);
70
71 return;
72}

85{
86 s->proto.flags |= DETECT_PROTO_IPV4; // TODO

96static InspectionBuffer *GetData(DetectEngineThreadCtx *det_ctx,
97 const DetectEngineTransforms *transforms, Packet *p, const int list_id)
98{
99 SCEnter();
100
101 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
102 if (buffer->inspect == NULL) {
103 if (p->ip4h == NULL) {
104 // DETECT_PROTO_IPV4 does not prefilter
105 return NULL;
106 }
107 uint32_t hlen = IPV4_GET_HLEN(p);
108 if (((uint8_t *)p->ip4h + (ptrdiff_t)hlen) >
109 ((uint8_t *)GET_PKT_DATA(p) + (ptrdiff_t)GET_PKT_LEN(p)))
110 {
111 SCLogDebug("data out of range: %p > %p",
112 ((uint8_t *)p->ip4h + (ptrdiff_t)hlen),
113 ((uint8_t *)GET_PKT_DATA(p) + (ptrdiff_t)GET_PKT_LEN(p)));
114 return NULL;
115 }
116
117 const uint32_t data_len = hlen;
118 const uint8_t *data = (const uint8_t *)p->ip4h;
119
120 InspectionBufferSetup(det_ctx, list_id, buffer, data, data_len);
121 InspectionBufferApplyTransforms(buffer, transforms);
122 }
123
124 return buffer;
125}
