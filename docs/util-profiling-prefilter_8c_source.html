69void SCProfilingPrefilterGlobalInit(void)
70{
71 ConfNode *conf;
72
73 conf = ConfGetNode("profiling.prefilter");
74 if (conf != NULL) {
75 if (ConfNodeChildValueIsTrue(conf, "enabled")) {
76 profiling_prefilter_enabled = 1;
77 const char *filename = ConfNodeLookupChildValue(conf, "filename");
78 if (filename != NULL) {
79 const char *log_dir;
80 log_dir = ConfigGetLogDirectory();
81
82 snprintf(profiling_file_name, sizeof(profiling_file_name), "%s/%s",
83 log_dir, filename);
84
85 const char *v = ConfNodeLookupChildValue(conf, "append");
86 if (v == NULL || ConfValIsTrue(v)) {
87 profiling_file_mode = "a";
88 } else {
89 profiling_file_mode = "w";
90 }
91
92 profiling_prefilter_output_to_file = 1;
93 }
94 }
95 }

98static void DoDump(SCProfilePrefilterDetectCtx *rules_ctx, FILE *fp, const char *name)
99{
100 int i;
101 fprintf(fp, " ----------------------------------------------"
102 "------------------------------------------------------"
103 "----------------------------\n");
104 fprintf(fp, " Stats for: %s\n", name);
105 fprintf(fp, " ----------------------------------------------"
106 "------------------------------------------------------"
107 "----------------------------\n");
108 fprintf(fp, " %-32s %-15s %-15s %-15s %-15s %-15s %-15s %-15s %-15s %-15s\n", "Prefilter",
109 "Ticks", "Called", "Max Ticks", "Avg", "Bytes", "Called", "Max Bytes", "Avg Bytes",
110 "Ticks/Byte");
111 fprintf(fp, " -------------------------------- "
112 "--------------- "
113 "--------------- "
114 "--------------- "
115 "--------------- "
116 "--------------- "
117 "--------------- "
118 "--------------- "
119 "--------------- "
120 "--------------- "
121 "\n");
122 for (i = 0; i < (int)rules_ctx->size; i++) {
123 SCProfilePrefilterData *d = &rules_ctx->data[i];
124 if (d == NULL || d->called== 0)
125 continue;
126
127 uint64_t ticks = d->total;
128 double avgticks = 0;
129 if (ticks && d->called) {
130 avgticks = (double)(ticks / d->called);
131 }
132 double avgbytes = 0;
133 if (d->total_bytes && d->called) {
134 avgbytes = (double)(d->total_bytes / d->bytes_called);
135 }
136 double ticks_per_byte = 0;
137 if (ticks && d->total_bytes) {
138 ticks_per_byte = (double)(ticks / d->total_bytes);
139 }
140
141 fprintf(fp,
142 " %-32s %-15" PRIu64 " %-15" PRIu64 " %-15" PRIu64 " %-15.2f %-15" PRIu64
143 " %-15" PRIu64 " %-15" PRIu64 " %-15.2f %-15.2f\n",
144 d->name, ticks, d->called, d->max, avgticks, d->total_bytes, d->bytes_called,
145 d->max_bytes, avgbytes, ticks_per_byte);
146 }
147}

149static void
150SCProfilingPrefilterDump(DetectEngineCtx *de_ctx)
151{
152 FILE *fp;
153 struct timeval tval;
154 struct tm *tms;
155 struct tm local_tm;
156
157 if (profiling_prefilter_enabled == 0 || de_ctx->profile_prefilter_ctx == NULL)
158 return;
159
160 gettimeofday(&tval, NULL);
161 tms = SCLocalTime(tval.tv_sec, &local_tm);
162
163 if (profiling_prefilter_output_to_file == 1) {
164 SCLogDebug("file %s mode %s", profiling_file_name, profiling_file_mode);
165
166 fp = fopen(profiling_file_name, profiling_file_mode);
167
168 if (fp == NULL) {
169 SCLogError(SC_ERR_FOPEN, "failed to open %s: %s", profiling_file_name,
170 strerror(errno));
171 return;
172 }
173 } else {
174 fp = stdout;
175 }
176
177 fprintf(fp, " ----------------------------------------------"
178 "------------------------------------------------------"
179 "----------------------------\n");
180 fprintf(fp, " Date: %" PRId32 "/%" PRId32 "/%04d -- "
181 "%02d:%02d:%02d\n", tms->tm_mon + 1, tms->tm_mday, tms->tm_year + 1900,
182 tms->tm_hour,tms->tm_min, tms->tm_sec);
183
184 /* global stats first */
185 DoDump(de_ctx->profile_prefilter_ctx, fp, "total");
186
187 fprintf(fp,"\n");
188 if (fp != stdout)
189 fclose(fp);
190
191 SCLogPerf("Done dumping prefilter profiling data.");
192}

194/**
195 * \brief Update a rule counter.
196 *
197 * \param id The ID of this counter.
198 * \param ticks Number of CPU ticks for this rule.
199 * \param match Did the rule match?
200 */
201void SCProfilingPrefilterUpdateCounter(DetectEngineThreadCtx *det_ctx, int id, uint64_t ticks,
202 uint64_t bytes, uint64_t bytes_called)
203{
204 if (det_ctx != NULL && det_ctx->prefilter_perf_data != NULL &&
205 id < (int)det_ctx->de_ctx->prefilter_id)
206 {
207 SCProfilePrefilterData *p = &det_ctx->prefilter_perf_data[id];
208
209 p->called++;
210 if (ticks > p->max)
211 p->max = ticks;
212 p->total += ticks;
213
214 p->bytes_called += bytes_called;
215 if (bytes > p->max_bytes)
216 p->max_bytes = bytes;
217 p->total_bytes += bytes;
218 }
219}

221static SCProfilePrefilterDetectCtx *SCProfilingPrefilterInitCtx(void)
222{
223 SCProfilePrefilterDetectCtx *ctx = SCMalloc(sizeof(SCProfilePrefilterDetectCtx));
224 if (ctx != NULL) {
225 memset(ctx, 0x00, sizeof(SCProfilePrefilterDetectCtx));
226
227 if (pthread_mutex_init(&ctx->data_m, NULL) != 0) {
228 FatalError(SC_ERR_FATAL,
229 "Failed to initialize hash table mutex.");
230 }
231 }
232
233 return ctx;
234}

236static void DetroyCtx(SCProfilePrefilterDetectCtx *ctx)
237{
238 if (ctx) {
239 if (ctx->data != NULL)
240 SCFree(ctx->data);
241 pthread_mutex_destroy(&ctx->data_m);
242 SCFree(ctx);
243 }

255void SCProfilingPrefilterThreadSetup(SCProfilePrefilterDetectCtx *ctx, DetectEngineThreadCtx *det_ctx)
256{
257 if (ctx == NULL)
258 return;
259
260 const uint32_t size = det_ctx->de_ctx->prefilter_id;
261
262 SCProfilePrefilterData *a = SCMalloc(sizeof(SCProfilePrefilterData) * size);

264 memset(a, 0x00, sizeof(SCProfilePrefilterData) * size);
265 det_ctx->prefilter_perf_data = a;
266 }

269static void SCProfilingPrefilterThreadMerge(DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx)
270{
271 if (de_ctx == NULL || de_ctx->profile_prefilter_ctx == NULL ||
272 de_ctx->profile_prefilter_ctx->data == NULL || det_ctx == NULL ||
273 det_ctx->prefilter_perf_data == NULL)
274 return;
275
276 for (uint32_t i = 0; i < de_ctx->prefilter_id; i++) {
277 de_ctx->profile_prefilter_ctx->data[i].called += det_ctx->prefilter_perf_data[i].called;
278 de_ctx->profile_prefilter_ctx->data[i].total += det_ctx->prefilter_perf_data[i].total;
279 if (det_ctx->prefilter_perf_data[i].max > de_ctx->profile_prefilter_ctx->data[i].max)
280 de_ctx->profile_prefilter_ctx->data[i].max = det_ctx->prefilter_perf_data[i].max;
281 de_ctx->profile_prefilter_ctx->data[i].total_bytes +=
282 det_ctx->prefilter_perf_data[i].total_bytes;
283 if (det_ctx->prefilter_perf_data[i].max_bytes >
284 de_ctx->profile_prefilter_ctx->data[i].max_bytes)
285 de_ctx->profile_prefilter_ctx->data[i].max_bytes =
286 det_ctx->prefilter_perf_data[i].max_bytes;
287 de_ctx->profile_prefilter_ctx->data[i].bytes_called +=
288 det_ctx->prefilter_perf_data[i].bytes_called;
289 }
290}

292void SCProfilingPrefilterThreadCleanup(DetectEngineThreadCtx *det_ctx)
293{
294 if (det_ctx == NULL || det_ctx->de_ctx == NULL || det_ctx->prefilter_perf_data == NULL)
295 return;
296
297 pthread_mutex_lock(&det_ctx->de_ctx->profile_prefilter_ctx->data_m);
298 SCProfilingPrefilterThreadMerge(det_ctx->de_ctx, det_ctx);
299 pthread_mutex_unlock(&det_ctx->de_ctx->profile_prefilter_ctx->data_m);
300
301 SCFree(det_ctx->prefilter_perf_data);
302 det_ctx->prefilter_perf_data = NULL;
303}

305/**
306 * \brief Register the prefilter profiling counters.
307 *
308 * \param de_ctx The active DetectEngineCtx, used to get at the loaded rules.
309 */
310void
311SCProfilingPrefilterInitCounters(DetectEngineCtx *de_ctx)
312{
313 if (profiling_prefilter_enabled == 0)
314 return;
315
316 const uint32_t size = de_ctx->prefilter_id;
317 if (size == 0)
318 return;
319
320 de_ctx->profile_prefilter_ctx = SCProfilingPrefilterInitCtx();
321 BUG_ON(de_ctx->profile_prefilter_ctx == NULL);
322 de_ctx->profile_prefilter_ctx->size = size;
323
324 de_ctx->profile_prefilter_ctx->data = SCMalloc(sizeof(SCProfilePrefilterData) * size);
325 BUG_ON(de_ctx->profile_prefilter_ctx->data == NULL);
326 memset(de_ctx->profile_prefilter_ctx->data, 0x00, sizeof(SCProfilePrefilterData) * size);
327
328 HashListTableBucket *hb = HashListTableGetListHead(de_ctx->prefilter_hash_table);
329 for ( ; hb != NULL; hb = HashListTableGetListNext(hb)) {
330 PrefilterStore *ctx = HashListTableGetListData(hb);
331 de_ctx->profile_prefilter_ctx->data[ctx->id].name = ctx->name;
332 SCLogDebug("prefilter %s set up", de_ctx->profile_prefilter_ctx->data[ctx->id].name);
333 }
334 SCLogDebug("size alloc'd %u", (uint32_t)size * (uint32_t)sizeof(SCProfilePrefilterData));
335
336 SCLogPerf("Registered %"PRIu32" prefilter profiling counters.", size);
337}
