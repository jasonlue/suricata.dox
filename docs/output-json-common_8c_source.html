43OutputJsonThreadCtx *CreateEveThreadCtx(ThreadVars *t, OutputJsonCtx *ctx)
44{
45 OutputJsonThreadCtx *thread = SCCalloc(1, sizeof(*thread));
46 if (unlikely(thread == NULL)) {
47 return NULL;
48 }
49
50 thread->buffer = MemBufferCreateNew(JSON_OUTPUT_BUFFER_SIZE);
51 if (unlikely(thread->buffer == NULL)) {
52 goto error;
53 }
54
55 thread->file_ctx = LogFileEnsureExists(ctx->file_ctx, t->id);
56 if (!thread->file_ctx) {
57 goto error;
58 }
59
60 thread->ctx = ctx;
61
62 return thread;
63
64error:
65 if (thread->buffer) {
66 MemBufferFree(thread->buffer);
67 }
68 SCFree(thread);
69 return NULL;
70}

104TmEcode JsonLogThreadInit(ThreadVars *t, const void *initdata, void **data)
105{
106 if (initdata == NULL) {
107 return TM_ECODE_FAILED;
108 }
109
110 OutputJsonThreadCtx *thread = SCCalloc(1, sizeof(*thread));
111 if (unlikely(thread == NULL)) {
112 return TM_ECODE_FAILED;
113 }
114
115 thread->buffer = MemBufferCreateNew(JSON_OUTPUT_BUFFER_SIZE);
116 if (unlikely(thread->buffer == NULL)) {
117 goto error_exit;
118 }
119
120 thread->ctx = ((OutputCtx *)initdata)->data;
121 thread->file_ctx = LogFileEnsureExists(thread->ctx->file_ctx, t->id);
122 if (!thread->file_ctx) {
123 goto error_exit;
124 }
125
126 *data = (void *)thread;
127 return TM_ECODE_OK;
128
129error_exit:
130 if (thread->buffer) {
131 MemBufferFree(thread->buffer);
132 }
133 SCFree(thread);
134 return TM_ECODE_FAILED;
135}
