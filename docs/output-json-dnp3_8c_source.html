55static void JsonDNP3LogLinkControl(JsonBuilder *js, uint8_t lc)
56{
57 jb_set_bool(js, "dir", DNP3_LINK_DIR(lc));
58 jb_set_bool(js, "pri", DNP3_LINK_PRI(lc));
59 jb_set_bool(js, "fcb", DNP3_LINK_FCB(lc));

61 jb_set_uint(js, "function_code", DNP3_LINK_FC(lc));

64static void JsonDNP3LogIin(JsonBuilder *js, uint16_t iin)
65{
66 if (iin) {
67 jb_open_array(js, "indicators");
68
69 int mapping = 0;
70 do {
71 if (iin & DNP3IndicatorsMap[mapping].value) {
72 jb_append_string(js, DNP3IndicatorsMap[mapping].name);
73 }
74 mapping++;
75 } while (DNP3IndicatorsMap[mapping].name != NULL);
76 jb_close(js);
77 }
78}

95{
96 DNP3Point *item;
97

99 jb_start_object(js);
100
101 jb_set_uint(js, "prefix", item->prefix);
102 jb_set_uint(js, "index", item->index);
103 if (DNP3PrefixIsSize(object->prefix_code)) {
104 jb_set_uint(js, "size", item->size);
105 }
106
107 OutputJsonDNP3SetItem(js, object, item);
108 jb_close(js);
109 }

112/**
113 * \brief Log the application layer objects.
114 *
115 * \param objects A list of DNP3 objects.
116 * \param jb A JsonBuilder instance with an open array.
117 */
118static void JsonDNP3LogObjects(JsonBuilder *js, DNP3ObjectList *objects)
119{
120 DNP3Object *object;
121
122 TAILQ_FOREACH(object, objects, next) {
123 jb_start_object(js);
124 jb_set_uint(js, "group", object->group);
125 jb_set_uint(js, "variation", object->variation);
126 jb_set_uint(js, "qualifier", object->qualifier);
127 jb_set_uint(js, "prefix_code", object->prefix_code);
128 jb_set_uint(js, "range_code", object->range_code);
129 jb_set_uint(js, "start", object->start);
130 jb_set_uint(js, "stop", object->stop);
131 jb_set_uint(js, "count", object->count);
132
133 if (object->points != NULL && !TAILQ_EMPTY(object->points)) {
134 jb_open_array(js, "points");
135 JsonDNP3LogObjectItems(js, object);
136 jb_close(js);
137 }
138
139 jb_close(js);
140 }
141}

143void JsonDNP3LogRequest(JsonBuilder *js, DNP3Transaction *dnp3tx)
144{
145 JB_SET_STRING(js, "type", "request");
146
147 jb_open_object(js, "control");
148 JsonDNP3LogLinkControl(js, dnp3tx->request_lh.control);
149 jb_close(js);
150
151 jb_set_uint(js, "src", DNP3_SWAP16(dnp3tx->request_lh.src));
152 jb_set_uint(js, "dst", DNP3_SWAP16(dnp3tx->request_lh.dst));
153
154 jb_open_object(js, "application");
155
156 jb_open_object(js, "control");
157 JsonDNP3LogApplicationControl(js, dnp3tx->request_ah.control);
158 jb_close(js);
159
160 jb_set_uint(js, "function_code", dnp3tx->request_ah.function_code);
161
162 if (!TAILQ_EMPTY(&dnp3tx->request_objects)) {
163 jb_open_array(js, "objects");
164 JsonDNP3LogObjects(js, &dnp3tx->request_objects);
165 jb_close(js);
166 }
167
168 jb_set_bool(js, "complete", dnp3tx->request_complete);
169
170 /* Close application. */
171 jb_close(js);
172}

174void JsonDNP3LogResponse(JsonBuilder *js, DNP3Transaction *dnp3tx)
175{
176 if (dnp3tx->response_ah.function_code == DNP3_APP_FC_UNSOLICITED_RESP) {
177 JB_SET_STRING(js, "type", "unsolicited_response");
178 }
179 else {
180 JB_SET_STRING(js, "type", "response");
181 }
182
183 jb_open_object(js, "control");
184 JsonDNP3LogLinkControl(js, dnp3tx->response_lh.control);
185 jb_close(js);
186
187 jb_set_uint(js, "src", DNP3_SWAP16(dnp3tx->response_lh.src));
188 jb_set_uint(js, "dst", DNP3_SWAP16(dnp3tx->response_lh.dst));
189
190 jb_open_object(js, "application");
191
192 jb_open_object(js, "control");
193 JsonDNP3LogApplicationControl(js, dnp3tx->response_ah.control);
194 jb_close(js);
195
196 jb_set_uint(js, "function_code", dnp3tx->response_ah.function_code);
197
198 if (!TAILQ_EMPTY(&dnp3tx->response_objects)) {
199 jb_open_array(js, "objects");
200 JsonDNP3LogObjects(js, &dnp3tx->response_objects);
201 jb_close(js);
202 }
203
204 jb_set_bool(js, "complete", dnp3tx->response_complete);
205
206 /* Close application. */
207 jb_close(js);
208
209 jb_open_object(js, "iin");
210 JsonDNP3LogIin(js, (uint16_t)(dnp3tx->response_iin.iin1 << 8 | dnp3tx->response_iin.iin2));
211 jb_close(js);
212}

214static int JsonDNP3LoggerToServer(ThreadVars *tv, void *thread_data,
215 const Packet *p, Flow *f, void *state, void *vtx, uint64_t tx_id)
216{
217 SCEnter();
218 LogDNP3LogThread *thread = (LogDNP3LogThread *)thread_data;
219 DNP3Transaction *tx = vtx;
220
221 if (tx->has_request && tx->request_done) {
222 JsonBuilder *js =
223 CreateEveHeader(p, LOG_DIR_FLOW, "dnp3", NULL, thread->dnp3log_ctx->eve_ctx);
224 if (unlikely(js == NULL)) {
225 return TM_ECODE_OK;
226 }
227
228 jb_open_object(js, "dnp3");
229 JsonDNP3LogRequest(js, tx);
230 jb_close(js);
231 OutputJsonBuilderBuffer(js, thread->ctx);
232 jb_free(js);
233 }
234
235 SCReturnInt(TM_ECODE_OK);
236}

272static OutputInitResult OutputDNP3LogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
273{
274 OutputInitResult result = { NULL, false };
275 OutputJsonCtx *json_ctx = parent_ctx->data;
276
277 LogDNP3FileCtx *dnp3log_ctx = SCCalloc(1, sizeof(*dnp3log_ctx));
278 if (unlikely(dnp3log_ctx == NULL)) {
279 return result;
280 }
281 dnp3log_ctx->eve_ctx = json_ctx;
282
283 OutputCtx *output_ctx = SCCalloc(1, sizeof(*output_ctx));
284 if (unlikely(output_ctx == NULL)) {
285 SCFree(dnp3log_ctx);
286 return result;
287 }
288 output_ctx->data = dnp3log_ctx;
289 output_ctx->DeInit = OutputDNP3LogDeInitCtxSub;
290
291 SCLogInfo("DNP3 log sub-module initialized.");
292
293 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNP3);
294
295 result.ctx = output_ctx;
296 result.ok = true;
297 return result;
298}

301static TmEcode JsonDNP3LogThreadInit(ThreadVars *t, const void *initdata, void **data)
302{
303 LogDNP3LogThread *thread = SCCalloc(1, sizeof(*thread));
304 if (unlikely(thread == NULL)) {
305 return TM_ECODE_FAILED;
306 }
307
308 if (initdata == NULL) {
309 SCLogDebug("Error getting context for DNP3. \"initdata\" is NULL.");
310 goto error_exit;
311 }
312
313 thread->dnp3log_ctx = ((OutputCtx *)initdata)->data;
314 thread->ctx = CreateEveThreadCtx(t, thread->dnp3log_ctx->eve_ctx);
315 if (thread->ctx == NULL) {
316 goto error_exit;
317 }
318
319 *data = (void *)thread;
320
321 return TM_ECODE_OK;
322
323error_exit:
324 SCFree(thread);
325 return TM_ECODE_FAILED;
326}
