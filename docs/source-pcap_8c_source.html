52/**
53 * \brief 64bit pcap stats counters.
54 *
55 * libpcap only supports 32bit counters. They will eventually wrap around.
56 *
57 * Keep track of libpcap counters as 64bit counters to keep on counting even
58 * if libpcap's 32bit counters wrap around.
59 * Requires pcap_stats() to be called before 32bit stats wrap around twice,
60 * which we do.
61 */
62typedef struct PcapStats64_ {
63 uint64_t ps_recv;
64 uint64_t ps_drop;
65 uint64_t ps_ifdrop;
66} PcapStats64;

158/**
159 * \brief Update 64 bit |last| value from |current32| value taking one
160 * wrap-around into account.
161 */
162static inline void UpdatePcapStatsValue64(uint64_t *last, uint32_t current32)
163{
164 /* uint64_t -> uint32_t is defined behaviour. It slices lower 32bits. */
165 uint32_t last32 = *last;
166
167 /* Branchless code as wrap-around is defined for unsigned */
168 *last += (uint32_t)(current32 - last32);
169
170 /* Same calculation as:
171 if (likely(current32 >= last32)) {
172 *last += current32 - last32;
173 } else {
174 *last += (1ull << 32) + current32 - last32;
175 }
176 */
177}
178
179/**
180 * \brief Update 64 bit |last| stat values with values from |current|
181 * 32 bit pcap_stat.
182 */
183static inline void UpdatePcapStats64(
184 PcapStats64 *last, const struct pcap_stat *current)
185{
186 UpdatePcapStatsValue64(&last->ps_recv, current->ps_recv);
187 UpdatePcapStatsValue64(&last->ps_drop, current->ps_drop);
188 UpdatePcapStatsValue64(&last->ps_ifdrop, current->ps_ifdrop);
189}

191static inline void PcapDumpCounters(PcapThreadVars *ptv)
192{
193 struct pcap_stat pcap_s;
194 if (likely((pcap_stats(ptv->pcap_handle, &pcap_s) >= 0))) {
195 UpdatePcapStats64(&ptv->last_stats64, &pcap_s);
196
197 StatsSetUI64(ptv->tv, ptv->capture_kernel_packets,
198 ptv->last_stats64.ps_recv);
199 StatsSetUI64(
200 ptv->tv, ptv->capture_kernel_drops, ptv->last_stats64.ps_drop);
201 (void)SC_ATOMIC_SET(ptv->livedev->drop, ptv->last_stats64.ps_drop);
202 StatsSetUI64(ptv->tv, ptv->capture_kernel_ifdrops,
203 ptv->last_stats64.ps_ifdrop);
204 }
205}

207static int PcapTryReopen(PcapThreadVars *ptv)
208{
209 ptv->pcap_state = PCAP_STATE_DOWN;
210
211 int pcap_activate_r = pcap_activate(ptv->pcap_handle);
212 if (pcap_activate_r != 0 && pcap_activate_r != PCAP_ERROR_ACTIVATED) {
213 return pcap_activate_r;
214 }
215
216 /* set bpf filter if we have one */
217 if (ptv->bpf_filter != NULL) {
218 if (pcap_compile(ptv->pcap_handle, &ptv->filter,
219 (char *)ptv->bpf_filter, 1, 0) < 0)
220 {
221 SCLogError(SC_ERR_BPF, "bpf compilation error %s",
222 pcap_geterr(ptv->pcap_handle));
223 return -1;
224 }
225
226 if (pcap_setfilter(ptv->pcap_handle, &ptv->filter) < 0) {
227 SCLogError(SC_ERR_BPF, "could not set bpf filter %s",
228 pcap_geterr(ptv->pcap_handle));
229 return -1;
230 }
231 }
232
233 SCLogInfo("Recovering interface listening");
234 ptv->pcap_state = PCAP_STATE_UP;
235 return 0;
236}

238static void PcapCallbackLoop(char *user, struct pcap_pkthdr *h, u_char *pkt)
239{
240 SCEnter();
241
242 PcapThreadVars *ptv = (PcapThreadVars *)user;
243 Packet *p = PacketGetFromQueueOrAlloc();
244 struct timeval current_time;
245
246 if (unlikely(p == NULL)) {
247 SCReturn;
248 }
249
250 PKT_SET_SRC(p, PKT_SRC_WIRE);
251 p->ts.tv_sec = h->ts.tv_sec;
252 p->ts.tv_usec = h->ts.tv_usec;
253 SCLogDebug("p->ts.tv_sec %"PRIuMAX"", (uintmax_t)p->ts.tv_sec);
254 p->datalink = ptv->datalink;
255
256 ptv->pkts++;
257 ptv->bytes += h->caplen;
258 (void) SC_ATOMIC_ADD(ptv->livedev->pkts, 1);
259 p->livedev = ptv->livedev;
260
261 if (unlikely(PacketCopyData(p, pkt, h->caplen))) {
262 TmqhOutputPacketpool(ptv->tv, p);
263 SCReturn;
264 }
265
266 switch (ptv->checksum_mode) {
267 case CHECKSUM_VALIDATION_AUTO:
268 if (ChecksumAutoModeCheck(ptv->pkts,
269 SC_ATOMIC_GET(ptv->livedev->pkts),
270 SC_ATOMIC_GET(ptv->livedev->invalid_checksums))) {
271 ptv->checksum_mode = CHECKSUM_VALIDATION_DISABLE;
272 p->flags |= PKT_IGNORE_CHECKSUM;
273 }
274 break;
275 case CHECKSUM_VALIDATION_DISABLE:
276 p->flags |= PKT_IGNORE_CHECKSUM;
277 break;
278 default:
279 break;
280 }
281
282 if (TmThreadsSlotProcessPkt(ptv->tv, ptv->slot, p) != TM_ECODE_OK) {
283 pcap_breakloop(ptv->pcap_handle);
284 ptv->cb_result = TM_ECODE_FAILED;
285 }
286
287 /* Trigger one dump of stats every second */
288 TimeGet(&current_time);
289 if (current_time.tv_sec != ptv->last_stats_dump) {
290 PcapDumpCounters(ptv);
291 ptv->last_stats_dump = current_time.tv_sec;
292 }
293
294 SCReturn;
295}

301/**
302 * \brief Main PCAP reading Loop function
303 */
304static TmEcode ReceivePcapLoop(ThreadVars *tv, void *data, void *slot)
305{
306 SCEnter();
307
308 int packet_q_len = 64;
309 PcapThreadVars *ptv = (PcapThreadVars *)data;
310 TmSlot *s = (TmSlot *)slot;
311
312 ptv->slot = s->slot_next;
313 ptv->cb_result = TM_ECODE_OK;
314
315 while (1) {
316 if (suricata_ctl_flags & SURICATA_STOP) {
317 SCReturnInt(TM_ECODE_OK);
318 }
319
320 /* make sure we have at least one packet in the packet pool, to prevent
321 * us from alloc'ing packets at line rate */
322 PacketPoolWait();
323
324 int r = pcap_dispatch(ptv->pcap_handle, packet_q_len,
325 (pcap_handler)PcapCallbackLoop, (u_char *)ptv);
326 if (unlikely(r == 0 || r == PCAP_ERROR_BREAK)) {
327 if (r == PCAP_ERROR_BREAK && ptv->cb_result == TM_ECODE_FAILED) {
328 SCReturnInt(TM_ECODE_FAILED);
329 }
330 TmThreadsCaptureHandleTimeout(tv, NULL);
331 } else if (unlikely(r < 0)) {
332 int dbreak = 0;
333 SCLogError(SC_ERR_PCAP_DISPATCH, "error code %" PRId32 " %s",
334 r, pcap_geterr(ptv->pcap_handle));
335 do {
336 usleep(PCAP_RECONNECT_TIMEOUT);
337 if (suricata_ctl_flags != 0) {
338 dbreak = 1;
339 break;
340 }
341 r = PcapTryReopen(ptv);
342 } while (r < 0);
343 if (dbreak) {
344 break;
345 }
346 } else if (ptv->cb_result == TM_ECODE_FAILED) {
347 SCLogError(SC_ERR_PCAP_DISPATCH, "Pcap callback PcapCallbackLoop failed");
348 SCReturnInt(TM_ECODE_FAILED);
349 }
350
351 StatsSyncCountersIfSignalled(tv);
352 }
353
354 PcapDumpCounters(ptv);
355 StatsSyncCountersIfSignalled(tv);
356 SCReturnInt(TM_ECODE_OK);
357}

373/**
374 * \brief Init function for ReceivePcap.
375 *
376 * This is a setup function for recieving packets
377 * via libpcap. There are two versions of this function
378 * depending on the major version of libpcap used.
379 * For versions prior to 1.x we use open_pcap_live,
380 * for versions 1.x and greater we use pcap_create + pcap_activate.
381 *
382 * \param tv pointer to ThreadVars
383 * \param initdata pointer to the interface passed from the user
384 * \param data pointer gets populated with PcapThreadVars
385 *
386 * \todo Create a general pcap setup function.
387 */
388static TmEcode ReceivePcapThreadInit(ThreadVars *tv, const void *initdata, void **data)
389{
390 SCEnter();
391 PcapIfaceConfig *pcapconfig = (PcapIfaceConfig *)initdata;
392
393 if (initdata == NULL) {
394 SCLogError(SC_ERR_INVALID_ARGUMENT, "initdata == NULL");
395 SCReturnInt(TM_ECODE_FAILED);
396 }
397
398 PcapThreadVars *ptv = SCCalloc(1, sizeof(PcapThreadVars));
399 if (unlikely(ptv == NULL)) {
400 pcapconfig->DerefFunc(pcapconfig);
401 SCReturnInt(TM_ECODE_FAILED);
402 }
403
404 ptv->tv = tv;
405
406 ptv->livedev = LiveGetDevice(pcapconfig->iface);
407 if (ptv->livedev == NULL) {
408 SCLogError(SC_ERR_INVALID_VALUE, "unable to find Live device");
409 SCFree(ptv);
410 SCReturnInt(TM_ECODE_FAILED);
411 }
412 SCLogInfo("using interface %s", (char *)pcapconfig->iface);
413
414 if (LiveGetOffload() == 0) {
415 (void)GetIfaceOffloading((char *)pcapconfig->iface, 1, 1);
416 } else {
417 DisableIfaceOffloading(ptv->livedev, 1, 1);
418 }
419
420 ptv->checksum_mode = pcapconfig->checksum_mode;
421 if (ptv->checksum_mode == CHECKSUM_VALIDATION_AUTO) {
422 SCLogInfo("running in 'auto' checksum mode. Detection of interface "
423 "state will require " xstr(CHECKSUM_SAMPLE_COUNT) " packets");
424 }
425
426 char errbuf[PCAP_ERRBUF_SIZE];
427 ptv->pcap_handle = pcap_create((char *)pcapconfig->iface, errbuf);
428 if (ptv->pcap_handle == NULL) {
429 if (strlen(errbuf)) {
430 SCLogError(SC_ERR_PCAP_CREATE, "could not create a new "
431 "pcap handler for %s, error %s",
432 (char *)pcapconfig->iface, errbuf);
433 } else {
434 SCLogError(SC_ERR_PCAP_CREATE, "could not create a new "
435 "pcap handler for %s",
436 (char *)pcapconfig->iface);
437 }
438 SCFree(ptv);
439 pcapconfig->DerefFunc(pcapconfig);
440 SCReturnInt(TM_ECODE_FAILED);
441 }
442
443 if (pcapconfig->snaplen == 0) {
444 /* We set snaplen if we can get the MTU */
445 ptv->pcap_snaplen = GetIfaceMaxPacketSize(pcapconfig->iface);
446 } else {
447 ptv->pcap_snaplen = pcapconfig->snaplen;
448 }
449 if (ptv->pcap_snaplen > 0) {
450 /* set Snaplen. Must be called before pcap_activate */
451 int pcap_set_snaplen_r = pcap_set_snaplen(ptv->pcap_handle, ptv->pcap_snaplen);
452 if (pcap_set_snaplen_r != 0) {
453 SCLogError(SC_ERR_PCAP_SET_SNAPLEN, "could not set snaplen, "
454 "error: %s", pcap_geterr(ptv->pcap_handle));
455 SCFree(ptv);
456 pcapconfig->DerefFunc(pcapconfig);
457 SCReturnInt(TM_ECODE_FAILED);
458 }
459 SCLogInfo("Set snaplen to %d for '%s'", ptv->pcap_snaplen,
460 pcapconfig->iface);
461 }
462
463 /* set Promisc, and Timeout. Must be called before pcap_activate */
464 int pcap_set_promisc_r = pcap_set_promisc(ptv->pcap_handle, pcapconfig->promisc);
465 if (pcap_set_promisc_r != 0) {
466 SCLogError(SC_ERR_PCAP_SET_PROMISC, "could not set promisc mode, "
467 "error %s", pcap_geterr(ptv->pcap_handle));
468 SCFree(ptv);
469 pcapconfig->DerefFunc(pcapconfig);
470 SCReturnInt(TM_ECODE_FAILED);
471 }
472
473 int pcap_set_timeout_r = pcap_set_timeout(ptv->pcap_handle, LIBPCAP_COPYWAIT);
474 if (pcap_set_timeout_r != 0) {
475 SCLogError(SC_ERR_PCAP_SET_TIMEOUT, "could not set timeout, "
476 "error %s", pcap_geterr(ptv->pcap_handle));
477 SCFree(ptv);
478 pcapconfig->DerefFunc(pcapconfig);
479 SCReturnInt(TM_ECODE_FAILED);
480 }
481#ifdef HAVE_PCAP_SET_BUFF
482 ptv->pcap_buffer_size = pcapconfig->buffer_size;
483 if (ptv->pcap_buffer_size > 0) {
484 SCLogInfo("going to use pcap buffer size of %" PRId32,
485 ptv->pcap_buffer_size);
486
487 int pcap_set_buffer_size_r = pcap_set_buffer_size(ptv->pcap_handle,
488 ptv->pcap_buffer_size);
489 if (pcap_set_buffer_size_r != 0) {
490 SCLogError(SC_ERR_PCAP_SET_BUFF_SIZE, "could not set "
491 "pcap buffer size, error %s", pcap_geterr(ptv->pcap_handle));
492 SCFree(ptv);
493 pcapconfig->DerefFunc(pcapconfig);
494 SCReturnInt(TM_ECODE_FAILED);
495 }
496 }
497#endif /* HAVE_PCAP_SET_BUFF */
498
499 /* activate the handle */
500 int pcap_activate_r = pcap_activate(ptv->pcap_handle);
501 if (pcap_activate_r != 0) {
502 SCLogError(SC_ERR_PCAP_ACTIVATE_HANDLE, "could not activate the "
503 "pcap handler, error %s", pcap_geterr(ptv->pcap_handle));
504 SCFree(ptv);
505 pcapconfig->DerefFunc(pcapconfig);
506 SCReturnInt(TM_ECODE_FAILED);
507 }
508 ptv->pcap_state = PCAP_STATE_UP;
509
510 /* set bpf filter if we have one */
511 if (pcapconfig->bpf_filter) {
512 SCMutexLock(&pcap_bpf_compile_lock);
513
514 ptv->bpf_filter = pcapconfig->bpf_filter;
515
516 if (pcap_compile(ptv->pcap_handle, &ptv->filter,
517 (char *)ptv->bpf_filter, 1, 0) < 0)
518 {
519 SCLogError(SC_ERR_BPF, "bpf compilation error %s",
520 pcap_geterr(ptv->pcap_handle));
521
522 SCMutexUnlock(&pcap_bpf_compile_lock);
523 SCFree(ptv);
524 pcapconfig->DerefFunc(pcapconfig);
525 return TM_ECODE_FAILED;
526 }
527
528 if (pcap_setfilter(ptv->pcap_handle, &ptv->filter) < 0) {
529 SCLogError(SC_ERR_BPF, "could not set bpf filter %s",
530 pcap_geterr(ptv->pcap_handle));
531
532 SCMutexUnlock(&pcap_bpf_compile_lock);
533 SCFree(ptv);
534 pcapconfig->DerefFunc(pcapconfig);
535 return TM_ECODE_FAILED;
536 }
537
538 SCMutexUnlock(&pcap_bpf_compile_lock);
539 }
540
541 /* no offloading supported at all */
542 (void)GetIfaceOffloading(pcapconfig->iface, 1, 1);
543
544 ptv->datalink = pcap_datalink(ptv->pcap_handle);
545 DatalinkSetGlobalType(ptv->datalink);
546
547 pcapconfig->DerefFunc(pcapconfig);
548
549 ptv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
550 ptv->tv);
551 ptv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
552 ptv->tv);
553 ptv->capture_kernel_ifdrops = StatsRegisterCounter("capture.kernel_ifdrops",
554 ptv->tv);
555
556 *data = (void *)ptv;
557 SCReturnInt(TM_ECODE_OK);
558}

560/**
561 * \brief This function prints stats to the screen at exit.
562 * \param tv pointer to ThreadVars
563 * \param data pointer that gets cast into PcapThreadVars for ptv
564 */
565static void ReceivePcapThreadExitStats(ThreadVars *tv, void *data)
566{
567 SCEnter();
568 PcapThreadVars *ptv = (PcapThreadVars *)data;
569 struct pcap_stat pcap_s;
570
571 if (pcap_stats(ptv->pcap_handle, &pcap_s) < 0) {
572 SCLogError(SC_ERR_STAT,"(%s) Failed to get pcap_stats: %s",
573 tv->name, pcap_geterr(ptv->pcap_handle));
574 SCLogInfo("(%s) Packets %" PRIu64 ", bytes %" PRIu64 "", tv->name,
575 ptv->pkts, ptv->bytes);
576 } else {
577 SCLogInfo("(%s) Packets %" PRIu64 ", bytes %" PRIu64 "", tv->name,
578 ptv->pkts, ptv->bytes);
579
580 /* these numbers are not entirely accurate as ps_recv contains packets
581 * that are still waiting to be processed at exit. ps_drop only contains
582 * packets dropped by the driver and not any packets dropped by the interface.
583 * Additionally see http://tracker.icir.org/bro/ticket/18
584 *
585 * Note: ps_recv includes dropped packets and should be considered total.
586 * Unless we start to look at ps_ifdrop which isn't supported everywhere.
587 */
588 UpdatePcapStats64(&ptv->last_stats64, &pcap_s);
589 float drop_percent =
590 likely(ptv->last_stats64.ps_recv > 0)
591 ? (((float)ptv->last_stats64.ps_drop) /
592 (float)ptv->last_stats64.ps_recv) *
593 100
594 : 0;
595 SCLogInfo("(%s) Pcap Total:%" PRIu64 " Recv:%" PRIu64 " Drop:%" PRIu64
596 " (%02.1f%%).",
597 tv->name, ptv->last_stats64.ps_recv,
598 ptv->last_stats64.ps_recv - ptv->last_stats64.ps_drop,
599 ptv->last_stats64.ps_drop, drop_percent);
600 }
601}

603/**
604 * \brief This function passes off to link type decoders.
605 *
606 * DecodePcap decodes packets from libpcap and passes
607 * them off to the proper link type decoder.
608 *
609 * \param t pointer to ThreadVars
610 * \param p pointer to the current packet
611 * \param data pointer that gets cast into PcapThreadVars for ptv
612 */
613static TmEcode DecodePcap(ThreadVars *tv, Packet *p, void *data)
614{
615 SCEnter();
616 DecodeThreadVars *dtv = (DecodeThreadVars *)data;
617
618 BUG_ON(PKT_IS_PSEUDOPKT(p));
619
620 /* update counters */
621 DecodeUpdatePacketCounters(tv, dtv, p);
622
623 DecodeLinkLayer(tv, dtv, p->datalink, p, GET_PKT_DATA(p), GET_PKT_LEN(p));
624
625 PacketDecodeFinalize(tv, dtv, p);
626
627 SCReturnInt(TM_ECODE_OK);
628}

630static TmEcode DecodePcapThreadInit(ThreadVars *tv, const void *initdata, void **data)
631{
632 SCEnter();
633
634 DecodeThreadVars *dtv = DecodeThreadVarsAlloc(tv);

652void PcapTranslateIPToDevice(char *pcap_dev, size_t len)
653{
654 char errbuf[PCAP_ERRBUF_SIZE];
655 pcap_if_t *alldevsp = NULL;
656
657 struct addrinfo ai_hints;
658 struct addrinfo *ai_list = NULL;
659
660 memset(&ai_hints, 0, sizeof(ai_hints));
661 ai_hints.ai_family = AF_UNSPEC;
662 ai_hints.ai_flags = AI_NUMERICHOST;
663
664 /* try to translate IP */
665 if (getaddrinfo(pcap_dev, NULL, &ai_hints, &ai_list) != 0) {
666 return;
667 }
668
669 if (pcap_findalldevs(&alldevsp, errbuf)) {
670 freeaddrinfo(ai_list);
671 return;
672 }
673
674 for (pcap_if_t *devsp = alldevsp; devsp ; devsp = devsp->next) {
675 for (pcap_addr_t *ip = devsp->addresses; ip ; ip = ip->next) {
676
677 if (ai_list->ai_family != ip->addr->sa_family) {
678 continue;
679 }
680
681 if (ip->addr->sa_family == AF_INET) {
682 if (memcmp(&((struct sockaddr_in*)ai_list->ai_addr)->sin_addr,
683 &((struct sockaddr_in*)ip->addr)->sin_addr,
684 sizeof(struct in_addr)))
685 {
686 continue;
687 }
688 } else if (ip->addr->sa_family == AF_INET6) {
689 if (memcmp(&((struct sockaddr_in6*)ai_list->ai_addr)->sin6_addr,
690 &((struct sockaddr_in6*)ip->addr)->sin6_addr,
691 sizeof(struct in6_addr)))
692 {
693 continue;
694 }
695 } else {
696 continue;
697 }
698
699 freeaddrinfo(ai_list);
700
701 memset(pcap_dev, 0, len);
702 strlcpy(pcap_dev, devsp->name, len);
703
704 pcap_freealldevs(alldevsp);
705 return;
706 }
707 }
708
709 freeaddrinfo(ai_list);
710
711 pcap_freealldevs(alldevsp);
712}
