46/**
47 * \brief Gets a base64-decoded value from an encoded character
48 *
49 * \param c The encoded character
50 *
51 * \return The decoded value (0 or above), or -1 if the parameter is invalid
52 */
53static inline int GetBase64Value(uint8_t c)
54{
55 int val = -1;
56
57 /* Pull from conversion table */
58 if (c <= BASE64_TABLE_MAX) {
59 val = b64table[(int) c];
60 }
61
62 return val;
63}

65/**
66 * \brief Decodes a 4-byte base64-encoded block into a 3-byte ascii-encoded block
67 *
68 * \param ascii the 3-byte ascii output block
69 * \param b64 the 4-byte base64 input block
70 *
71 * \return none
72 */
73static inline void DecodeBase64Block(uint8_t ascii[ASCII_BLOCK], uint8_t b64[B64_BLOCK])
74{
75 ascii[0] = (uint8_t) (b64[0] << 2) | (b64[1] >> 4);
76 ascii[1] = (uint8_t) (b64[1] << 4) | (b64[2] >> 2);
77 ascii[2] = (uint8_t) (b64[2] << 6) | (b64[3]);
78}

80/**
81 * \brief Decodes a base64-encoded string buffer into an ascii-encoded byte buffer
82 *
83 * \param dest The destination byte buffer
84 * \param dest_size The destination byte buffer size
85 * \param src The source string
86 * \param len The length of the source string
87 * \param consumed_bytes The bytes that were actually processed/consumed
88 * \param decoded_bytes The bytes that were decoded
89 * \param mode The mode in which decoding should happen
90 *
91 * \return Error code indicating success or failures with parsing
92 */
93Base64Ecode DecodeBase64(uint8_t *dest, uint32_t dest_size, const uint8_t *src, uint32_t len,
94 uint32_t *consumed_bytes, uint32_t *decoded_bytes, Base64Mode mode)
95{
96 int val;
97 uint32_t padding = 0, bbidx = 0, sp = 0, leading_sp = 0;
98 uint8_t *dptr = dest;
99 uint8_t b64[B64_BLOCK] = { 0,0,0,0 };
100 bool valid = true;
101 Base64Ecode ecode = BASE64_ECODE_OK;
102 *decoded_bytes = 0;
103
104 /* Traverse through each alpha-numeric letter in the source array */
105 for (uint32_t i = 0; i < len; i++) {
106 /* Get decimal representation */
107 val = GetBase64Value(src[i]);
108 if (val < 0) {
109 if ((mode == BASE64_MODE_RFC2045) && (src[i] == ' ')) {
110 if (bbidx == 0) {
111 /* Special case where last block of data has a leading space */
112 leading_sp++;
113 }
114 sp++;
115 continue;
116 }
117 /* Invalid character found, so decoding fails */
118 if (src[i] != '=') {
119 valid = false;
120 if (mode != BASE64_MODE_RELAX) {
121 *decoded_bytes = 0;
122 }
123 ecode = BASE64_ECODE_ERR;
124 break;
125 }
126 padding++;
127 }
128
129 /* For each alpha-numeric letter in the source array, find the numeric
130 * value */
131 b64[bbidx++] = (val > 0 ? (uint8_t)val : 0);
132
133 /* Decode every 4 base64 bytes into 3 ascii bytes */
134 if (bbidx == B64_BLOCK) {
135
136 /* For every 4 bytes, add 3 bytes but deduct the '=' padded blocks */
137 uint32_t numDecoded_blk = ASCII_BLOCK - (padding < B64_BLOCK ? padding : ASCII_BLOCK);
138 if (dest_size < *decoded_bytes + numDecoded_blk) {
139 SCLogDebug("Destination buffer full");
140 ecode = BASE64_ECODE_BUF;
141 break;
142 }
143
144 /* Decode base-64 block into ascii block and move pointer */
145 DecodeBase64Block(dptr, b64);
146 dptr += ASCII_BLOCK;
147 *decoded_bytes += numDecoded_blk;
148 /* Reset base-64 block and index */
149 bbidx = 0;
150 padding = 0;
151 *consumed_bytes += B64_BLOCK + sp;
152 sp = 0;
153 leading_sp = 0;
154 memset(&b64, 0, sizeof(b64));
155 }
156 }
157 /* Finish remaining b64 bytes by padding */
158 if (valid && bbidx > 0 && (mode != BASE64_MODE_RFC2045)) {
159 /* Decode remaining */
160 *decoded_bytes += ASCII_BLOCK - (B64_BLOCK - bbidx);
161 DecodeBase64Block(dptr, b64);
162 }
163
164 if (*decoded_bytes == 0) {
165 SCLogDebug("base64 decoding failed");
166 }
167
168 *consumed_bytes += leading_sp;
169 return ecode;
170}

173static int B64DecodeCompleteString(void)
174{
175 /*
176 * SGVsbG8gV29ybGR6 : Hello Worldz
177 * */
178 const char *src = "SGVsbG8gV29ybGR6";
179 const char *fin_str = "Hello Worldz";
180 uint32_t consumed_bytes = 0, num_decoded = 0;
181 uint8_t dst[strlen(fin_str)];
182 Base64Ecode code = DecodeBase64(dst, strlen(fin_str), (const uint8_t *)src, strlen(src),
183 &consumed_bytes, &num_decoded, BASE64_MODE_RFC2045);
184 FAIL_IF(code != BASE64_ECODE_OK);
185 FAIL_IF(memcmp(dst, fin_str, strlen(fin_str)) != 0);
186 FAIL_IF(num_decoded != 12);
187 FAIL_IF(consumed_bytes != strlen(src));
188 PASS;
189}
