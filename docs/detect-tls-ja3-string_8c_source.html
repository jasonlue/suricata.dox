67static InspectionBuffer *GetJa3Data(DetectEngineThreadCtx *det_ctx,
68 const DetectEngineTransforms *transforms, Flow *_f, const uint8_t _flow_flags, void *txv,
69 const int list_id)
70{
71 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
72 if (buffer->inspect == NULL) {
73 uint32_t b_len = 0;
74 const uint8_t *b = NULL;
75
76 if (rs_quic_tx_get_ja3(txv, &b, &b_len) != 1)
77 return NULL;
78 if (b == NULL || b_len == 0)
79 return NULL;
80
81 InspectionBufferSetup(det_ctx, list_id, buffer, b, b_len);
82 InspectionBufferApplyTransforms(buffer, transforms);
83 }
84 return buffer;
85}

87/**
88 * \brief Registration function for keyword: ja3.string
89 */
90void DetectTlsJa3StringRegister(void)
91{
92 sigmatch_table[DETECT_AL_TLS_JA3_STRING].name = "ja3.string";
93 sigmatch_table[DETECT_AL_TLS_JA3_STRING].alias = "ja3_string";
94 sigmatch_table[DETECT_AL_TLS_JA3_STRING].desc = "sticky buffer to match the JA3 string buffer";
95 sigmatch_table[DETECT_AL_TLS_JA3_STRING].url = "/rules/ja3-keywords.html#ja3-string";
96 sigmatch_table[DETECT_AL_TLS_JA3_STRING].Setup = DetectTlsJa3StringSetup;
97 sigmatch_table[DETECT_AL_TLS_JA3_STRING].flags |= SIGMATCH_NOOPT;
98 sigmatch_table[DETECT_AL_TLS_JA3_STRING].flags |= SIGMATCH_INFO_STICKY_BUFFER;
99
100 DetectAppLayerInspectEngineRegister2("ja3.string", ALPROTO_TLS, SIG_FLAG_TOSERVER, 0,
101 DetectEngineInspectBufferGeneric, GetData);
102
103 DetectAppLayerMpmRegister2("ja3.string", SIG_FLAG_TOSERVER, 2,
104 PrefilterGenericMpmRegister, GetData, ALPROTO_TLS, 0);
105
106 DetectAppLayerMpmRegister2("ja3.string", SIG_FLAG_TOSERVER, 2, PrefilterGenericMpmRegister,
107 GetJa3Data, ALPROTO_QUIC, 1);
108
109 DetectAppLayerInspectEngineRegister2("ja3.string", ALPROTO_QUIC, SIG_FLAG_TOSERVER, 1,
110 DetectEngineInspectBufferGeneric, GetJa3Data);
111
112 DetectBufferTypeSetDescriptionByName("ja3.string", "TLS JA3 string");
113
114 g_tls_ja3_str_buffer_id = DetectBufferTypeGetByName("ja3.string");
115}

117/**
118 * \brief this function setup the ja3.string modifier keyword used in the rule
119 *
120 * \param de_ctx Pointer to the Detection Engine Context
121 * \param s Pointer to the Signature to which the current keyword belongs
122 * \param str Should hold an empty string always
123 *
124 * \retval 0 On success
125 * \retval -1 On failure
126 */
127static int DetectTlsJa3StringSetup(DetectEngineCtx *de_ctx, Signature *s, const char *str)
128{
129 if (DetectBufferSetActiveList(s, g_tls_ja3_str_buffer_id) < 0)
130 return -1;
131
132 if (s->alproto != ALPROTO_UNKNOWN && s->alproto != ALPROTO_TLS && s->alproto != ALPROTO_QUIC) {
133 SCLogError(SC_ERR_CONFLICTING_RULE_KEYWORDS, "rule contains conflicting protocols.");
134 return -1;
135 }
136
137 /* try to enable JA3 */
138 SSLEnableJA3();
139
140 /* Check if JA3 is disabled */
141 if (!RunmodeIsUnittests() && Ja3IsDisabled("rule")) {
142 if (!SigMatchSilentErrorEnabled(de_ctx, DETECT_AL_TLS_JA3_STRING)) {
143 SCLogError(SC_WARN_JA3_DISABLED, "ja3(s) support is not enabled");
144 }
145 return -2;
146 }
147
148 return 0;
149}

151static InspectionBuffer *GetData(DetectEngineThreadCtx *det_ctx,
152 const DetectEngineTransforms *transforms, Flow *f,
153 const uint8_t flow_flags, void *txv, const int list_id)
154{
155 InspectionBuffer *buffer = InspectionBufferGet(det_ctx, list_id);
156 if (buffer->inspect == NULL) {
157 const SSLState *ssl_state = (SSLState *)f->alstate;
158
159 if (ssl_state->client_connp.ja3_str == NULL ||
160 ssl_state->client_connp.ja3_str->data == NULL) {
161 return NULL;
162 }
163
164 const uint32_t data_len = strlen(ssl_state->client_connp.ja3_str->data);
165 const uint8_t *data = (uint8_t *)ssl_state->client_connp.ja3_str->data;
166
167 InspectionBufferSetup(det_ctx, list_id, buffer, data, data_len);
168 InspectionBufferApplyTransforms(buffer, transforms);
169 }
170
171 return buffer;
172}
