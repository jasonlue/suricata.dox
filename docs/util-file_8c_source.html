167/**
168 * \brief Function to parse forced file hashing configuration.
169 */
170void FileForceHashParseCfg(ConfNode *conf)
171{
172 BUG_ON(conf == NULL);
173
174 ConfNode *forcehash_node = NULL;
175
176 /* legacy option */
177 const char *force_md5 = ConfNodeLookupChildValue(conf, "force-md5");
178 if (force_md5 != NULL) {
179 SCLogWarning(SC_ERR_DEPRECATED_CONF, "deprecated 'force-md5' option "
180 "found. Please use 'force-hash: [md5]' instead");
181
182 if (ConfValIsTrue(force_md5)) {
183 if (g_disable_hashing) {
184 SCLogInfo(
185 "not forcing md5 calculation for logged files: hashing globally disabled");
186 } else {
187 FileForceMd5Enable();
188 SCLogInfo("forcing md5 calculation for logged files");
189 }
190 }
191 }
192
193 if (conf != NULL)
194 forcehash_node = ConfNodeLookupChild(conf, "force-hash");
195
196 if (forcehash_node != NULL) {
197 ConfNode *field = NULL;
198
199 TAILQ_FOREACH(field, &forcehash_node->head, next) {
200 if (strcasecmp("md5", field->val) == 0) {
201 if (g_disable_hashing) {
202 SCLogInfo("not forcing md5 calculation for logged files: hashing globally "
203 "disabled");
204 } else {
205 FileForceMd5Enable();
206 SCLogConfig("forcing md5 calculation for logged or stored files");
207 }
208 }
209
210 if (strcasecmp("sha1", field->val) == 0) {
211 if (g_disable_hashing) {
212 SCLogInfo("not forcing sha1 calculation for logged files: hashing globally "
213 "disabled");
214 } else {
215 FileForceSha1Enable();
216 SCLogConfig("forcing sha1 calculation for logged or stored files");
217 }
218 }
219
220 if (strcasecmp("sha256", field->val) == 0) {
221 if (g_disable_hashing) {
222 SCLogInfo("not forcing sha256 calculation for logged files: hashing globally "
223 "disabled");
224 } else {
225 FileForceSha256Enable();
226 SCLogConfig("forcing sha256 calculation for logged or stored files");
227 }
228 }
229 }
230 }
231}

233uint16_t FileFlowToFlags(const Flow *flow, uint8_t direction)
234{
235 uint16_t flags = 0;
236
237 if (direction == STREAM_TOSERVER) {
238 if (flow->file_flags & FLOWFILE_NO_STORE_TS) {
239 flags |= FILE_NOSTORE;
240 }
241
242 if (flow->file_flags & FLOWFILE_NO_MAGIC_TS) {
243 flags |= FILE_NOMAGIC;
244 }
245
246 if (flow->file_flags & FLOWFILE_NO_MD5_TS) {
247 flags |= FILE_NOMD5;
248 }
249
250 if (flow->file_flags & FLOWFILE_NO_SHA1_TS) {
251 flags |= FILE_NOSHA1;
252 }
253
254 if (flow->file_flags & FLOWFILE_NO_SHA256_TS) {
255 flags |= FILE_NOSHA256;
256 }
257 } else {
258 if (flow->file_flags & FLOWFILE_NO_STORE_TC) {
259 flags |= FILE_NOSTORE;
260 }
261
262 if (flow->file_flags & FLOWFILE_NO_MAGIC_TC) {
263 flags |= FILE_NOMAGIC;
264 }
265
266 if (flow->file_flags & FLOWFILE_NO_MD5_TC) {
267 flags |= FILE_NOMD5;
268 }
269
270 if (flow->file_flags & FLOWFILE_NO_SHA1_TC) {
271 flags |= FILE_NOSHA1;
272 }
273
274 if (flow->file_flags & FLOWFILE_NO_SHA256_TC) {
275 flags |= FILE_NOSHA256;
276 }
277 }
278 return flags;
279}

318/** \brief test if file is ready to be pruned
319 *
320 * If a file is in the 'CLOSED' state, it means it has been processed
321 * completely by the pipeline in the correct direction. So we can
322 * prune it then.
323 *
324 * For other states, as well as for files we may not need to track
325 * until the close state, more specific checks are done.
326 *
327 * Also does house keeping within the file: move streaming buffer
328 * forward if possible.
329 *
330 * \retval 1 prune (free) this file
331 * \retval 0 file not ready to be freed
332 */
333static int FilePruneFile(File *file)
334{
335 SCEnter();
336
337 /* file is done when state is closed+, logging/storing is done (if any) */
338 SCLogDebug("file->state %d. Is >= FILE_STATE_CLOSED: %s",
339 file->state, (file->state >= FILE_STATE_CLOSED) ? "yes" : "no");
340 if (file->state >= FILE_STATE_CLOSED) {
341 SCReturnInt(1);
342 }
343
344#ifdef HAVE_MAGIC
345 if (!(file->flags & FILE_NOMAGIC)) {
346 /* need magic but haven't set it yet, bail out */
347 if (file->magic == NULL)
348 SCReturnInt(0);
349 else
350 SCLogDebug("file->magic %s", file->magic);
351 } else {
352 SCLogDebug("file->flags & FILE_NOMAGIC == true");
353 }
354#endif
355 uint64_t left_edge = FileDataSize(file);
356 if (file->flags & FILE_STORE) {
357 left_edge = MIN(left_edge,file->content_stored);
358 }
359 if (file->flags & FILE_USE_DETECT) {
360 left_edge = MIN(left_edge, file->content_inspected);
361
362 /* if file has inspect window and min size set, we
363 * do some house keeping here */
364 if (file->inspect_window != 0 && file->inspect_min_size != 0) {
365 uint32_t window = file->inspect_window;
366 if (file->sb->stream_offset == 0)
367 window = MAX(window, file->inspect_min_size);
368
369 uint64_t file_size = FileDataSize(file);
370 uint64_t data_size = file_size - file->sb->stream_offset;
371
372 SCLogDebug("window %"PRIu32", file_size %"PRIu64", data_size %"PRIu64,
373 window, file_size, data_size);
374
375 if (data_size > (window * 3)) {
376 left_edge = file_size - window;
377 SCLogDebug("file->content_inspected now %"PRIu64, left_edge);
378 file->content_inspected = left_edge;
379 }
380 }
381 }
382
383 if (left_edge) {
384 StreamingBufferSlideToOffset(file->sb, left_edge);
385 }
386
387 SCReturnInt(0);
388}

390void FilePrune(FileContainer *ffc)
391{
392 File *file = ffc->head;
393 File *prev = NULL;
394
395 while (file) {
396 if (FilePruneFile(file) == 0) {
397 prev = file;
398 file = file->next;
399 continue;
400 }
401
402 SCLogDebug("removing file %p", file);
403
404 File *file_next = file->next;
405
406 if (prev)
407 prev->next = file_next;
408 /* update head and tail */
409 if (file == ffc->head)
410 ffc->head = file_next;
411 if (file == ffc->tail)
412 ffc->tail = prev;
413
414 FileFree(file);
415 file = file_next;
416 }
417}

476/**
477 * \brief Alloc a new File
478 *
479 * \param name character array containing the name (not a string)
480 * \param name_len length in bytes of the name
481 *
482 * \retval new File object or NULL on error
483 */
484static File *FileAlloc(const uint8_t *name, uint16_t name_len)
485{
486 File *new = SCMalloc(sizeof(File));
487 if (unlikely(new == NULL)) {
488 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating mem");
489 return NULL;
490 }
491 memset(new, 0, sizeof(File));
492
493 new->name = SCMalloc(name_len);
494 if (new->name == NULL) {
495 SCFree(new);
496 return NULL;
497 }
498
499 new->name_len = name_len;
500 memcpy(new->name, name, name_len);
501
502 new->sid_cnt = 0;
503 new->sid_max = 8;
504 /* SCMalloc() is allowed to fail here because sid well be checked later on */
505 new->sid = SCMalloc(sizeof(uint32_t) * new->sid_max);
506 if (new->sid == NULL)
507 new->sid_max = 0;
508
509 return new;
510}

512static void FileFree(File *ff)
513{
514 if (ff == NULL)
515 return;
516
517 if (ff->name != NULL)
518 SCFree(ff->name);
519 if (ff->sid != NULL)
520 SCFree(ff->sid);
521#ifdef HAVE_MAGIC
522 /* magic returned by libmagic is strdup'd by MagicLookup. */
523 if (ff->magic != NULL)
524 SCFree(ff->magic);
525#endif
526 if (ff->sb != NULL) {
527 StreamingBufferFree(ff->sb);
528 }
529
530 if (ff->md5_ctx)
531 SCMd5Free(ff->md5_ctx);
532 if (ff->sha1_ctx)
533 SCSha1Free(ff->sha1_ctx);
534 if (ff->sha256_ctx)
535 SCSha256Free(ff->sha256_ctx);
536 SCFree(ff);
537}

581/**
582 * \brief check if we have stored enough
583 *
584 * \param ff file
585 *
586 * \retval 0 limit not reached yet
587 * \retval 1 limit reached
588 */
589static int FileStoreNoStoreCheck(File *ff)
590{
591 SCEnter();
592
593 if (ff == NULL) {
594 SCReturnInt(0);
595 }
596
597 if (ff->flags & FILE_NOSTORE) {
598 if (ff->state == FILE_STATE_OPENED &&
599 FileDataSize(ff) >= (uint64_t)FileMagicSize())
600 {
601 SCReturnInt(1);
602 }
603 }
604
605 SCReturnInt(0);
606}

608static int AppendData(File *file, const uint8_t *data, uint32_t data_len)
609{
610 if (StreamingBufferAppendNoTrack(file->sb, data, data_len) != 0) {
611 SCReturnInt(-1);
612 }
613
614 if (file->md5_ctx) {
615 SCMd5Update(file->md5_ctx, data, data_len);
616 }
617 if (file->sha1_ctx) {
618 SCSha1Update(file->sha1_ctx, data, data_len);
619 }
620 if (file->sha256_ctx) {
621 SCSha256Update(file->sha256_ctx, data, data_len);
622 }
623 SCReturnInt(0);
624}

637/** \internal
638 * \brief Store/handle a chunk of file data in the File structure
639 *
640 * \param ff the file
641 * \param data data chunk
642 * \param data_len data chunk len
643 *
644 * \retval 0 ok
645 * \retval -1 error
646 * \retval -2 no store for this file
647 */
648static int FileAppendDataDo(File *ff, const uint8_t *data, uint32_t data_len)
649{
650 SCEnter();
651#ifdef DEBUG_VALIDATION
652 BUG_ON(ff == NULL);
653#endif
654
655 ff->size += data_len;
656 if (data == NULL) {
657 FileFlagGap(ff);
658 SCReturnInt(0);
659 }
660
661 if (ff->state != FILE_STATE_OPENED) {
662 if (ff->flags & FILE_NOSTORE) {
663 SCReturnInt(-2);
664 }
665 SCReturnInt(-1);
666 }
667
668 if ((ff->flags & FILE_USE_DETECT) == 0 &&
669 FileStoreNoStoreCheck(ff) == 1) {
670 int hash_done = 0;
671 /* no storage but forced hashing */
672 if (ff->md5_ctx) {
673 SCMd5Update(ff->md5_ctx, data, data_len);
674 hash_done = 1;
675 }
676 if (ff->sha1_ctx) {
677 SCSha1Update(ff->sha1_ctx, data, data_len);
678 hash_done = 1;
679 }
680 if (ff->sha256_ctx) {
681 SCSha256Update(ff->sha256_ctx, data, data_len);
682 hash_done = 1;
683 }
684
685 if (hash_done)
686 SCReturnInt(0);
687
688 if (g_file_force_tracking || (!(ff->flags & FILE_NOTRACK)))
689 SCReturnInt(0);
690
691 ff->state = FILE_STATE_TRUNCATED;
692 SCLogDebug("flowfile state transitioned to FILE_STATE_TRUNCATED");
693 SCReturnInt(-2);
694 }
695
696 SCLogDebug("appending %"PRIu32" bytes", data_len);
697
698 int r = AppendData(ff, data, data_len);
699 if (r != 0) {
700 ff->state = FILE_STATE_ERROR;
701 SCReturnInt(r);
702 }
703
704 SCReturnInt(0);
705}

707/**
708 * \brief Store/handle a chunk of file data in the File structure
709 * The last file in the FileContainer will be used.
710 *
711 * \param ffc FileContainer used to append to
712 * \param data data chunk
713 * \param data_len data chunk len
714 *
715 * \retval 0 ok
716 * \retval -1 error
717 * \retval -2 no store for this file
718 */
719int FileAppendData(FileContainer *ffc, const uint8_t *data, uint32_t data_len)
720{
721 SCEnter();
722
723 if (ffc == NULL || ffc->tail == NULL || data_len == 0) {
724 SCReturnInt(-1);
725 }
726 int r = FileAppendDataDo(ffc->tail, data, data_len);
727 SCReturnInt(r);
728}

730/**
731 * \brief Store/handle a chunk of file data in the File structure
732 * The file with 'track_id' in the FileContainer will be used.
733 *
734 * \param ffc FileContainer used to append to
735 * \param track_id id to lookup the file
736 * \param data data chunk
737 * \param data_len data chunk len
738 *
739 * \retval 0 ok
740 * \retval -1 error
741 * \retval -2 no store for this file
742 */
743int FileAppendDataById(FileContainer *ffc, uint32_t track_id,
744 const uint8_t *data, uint32_t data_len)
745{
746 SCEnter();
747
748 if (ffc == NULL || ffc->tail == NULL || data == NULL || data_len == 0) {
749 SCReturnInt(-1);
750 }
751 File *ff = ffc->head;
752 for ( ; ff != NULL; ff = ff->next) {
753 if (track_id == ff->file_track_id) {
754 int r = FileAppendDataDo(ff, data, data_len);
755 SCReturnInt(r);
756 }
757 }
758 SCReturnInt(-1);
759}

801/**
802 * \brief Sets the offset range for a file.
803 *
804 * \param ffc the container
805 * \param start start offset
806 * \param end end offset
807 *
808 * \retval 0 ok
809 * \retval -1 error
810 */
811int FileSetRange(FileContainer *ffc, uint64_t start, uint64_t end)
812{
813 SCEnter();
814
815 if (ffc == NULL || ffc->tail == NULL) {
816 SCReturnInt(-1);

931int FileCloseFilePtr(File *ff, const uint8_t *data,
932 uint32_t data_len, uint16_t flags)
933{
934 SCEnter();
935
936 if (ff == NULL) {
937 SCReturnInt(-1);
938 }
939
940 if (ff->state != FILE_STATE_OPENED) {
941 SCReturnInt(-1);
942 }
943
944 ff->size += data_len;
945 if (data != NULL) {
946 if (ff->flags & FILE_NOSTORE) {
947 /* no storage but hashing */
948 if (ff->md5_ctx)
949 SCMd5Update(ff->md5_ctx, data, data_len);
950 if (ff->sha1_ctx)
951 SCSha1Update(ff->sha1_ctx, data, data_len);
952 if (ff->sha256_ctx)
953 SCSha256Update(ff->sha256_ctx, data, data_len);
954 } else {
955 if (AppendData(ff, data, data_len) != 0) {
956 ff->state = FILE_STATE_ERROR;
957 SCReturnInt(-1);
958 }
959 }
960 }
961
962 if ((flags & FILE_TRUNCATED) || (ff->flags & FILE_HAS_GAPS)) {
963 ff->state = FILE_STATE_TRUNCATED;
964 SCLogDebug("flowfile state transitioned to FILE_STATE_TRUNCATED");
965
966 if (flags & FILE_NOSTORE) {
967 SCLogDebug("not storing this file");
968 ff->flags |= FILE_NOSTORE;
969 } else {
970 if (g_file_force_sha256 && ff->sha256_ctx) {
971 FileEndSha256(ff);
972 }
973 }
974 } else {
975 ff->state = FILE_STATE_CLOSED;
976 SCLogDebug("flowfile state transitioned to FILE_STATE_CLOSED");
977
978 if (ff->md5_ctx) {
979 SCMd5Finalize(ff->md5_ctx, ff->md5, sizeof(ff->md5));
980 ff->md5_ctx = NULL;
981 ff->flags |= FILE_MD5;
982 }
983 if (ff->sha1_ctx) {
984 SCSha1Finalize(ff->sha1_ctx, ff->sha1, sizeof(ff->sha1));
985 ff->sha1_ctx = NULL;
986 ff->flags |= FILE_SHA1;
987 }
988 if (ff->sha256_ctx) {
989 FileEndSha256(ff);
990 }
991 }
992
993 SCReturnInt(0);
994}

1042/** \brief set a flow's file flags
1043 * \param set_file_flags flags in both directions that are requested to set
1044 *
1045 * This function will ignore the flags for the irrelevant direction and
1046 * also mask the flags with the global settings.

1048void FileUpdateFlowFileFlags(Flow *f, uint16_t set_file_flags, uint8_t direction)
1049{
1050 SCEnter();
1051 DEBUG_ASSERT_FLOW_LOCKED(f);
1052
1053 /* remove flags not in our direction and
1054 don't disable what is globally enabled */
1055 if (direction == STREAM_TOSERVER) {
1056 set_file_flags &= ~(FLOWFILE_NONE_TC|g_file_flow_mask);
1057 } else {
1058 set_file_flags &= ~(FLOWFILE_NONE_TS|g_file_flow_mask);
1059 }
1060 f->file_flags |= set_file_flags;
1061
1062 SCLogDebug("f->file_flags %04x set_file_flags %04x g_file_flow_mask %04x",
1063 f->file_flags, set_file_flags, g_file_flow_mask);

1067#ifdef HAVE_MAGIC
1068 if (set_file_flags & (FLOWFILE_NO_MAGIC_TS|FLOWFILE_NO_MAGIC_TC))
1069 per_file_flags |= FILE_NOMAGIC;
1070#endif
1071 if (set_file_flags & (FLOWFILE_NO_MD5_TS|FLOWFILE_NO_MD5_TC))
1072 per_file_flags |= FILE_NOMD5;
