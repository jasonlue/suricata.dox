197static void EveHttpLogJSONBasic(JsonBuilder *js, htp_tx_t *tx)
198{
199 /* hostname */
200 if (tx->request_hostname != NULL) {
201 jb_set_string_from_bytes(
202 js, "hostname", bstr_ptr(tx->request_hostname), bstr_len(tx->request_hostname));
203 }
204
205 /* port */
206 /* NOTE: this field will be set ONLY if the port is present in the
207 * hostname. It may be present in the header "Host" or in the URL.
208 * There is no connection (from the suricata point of view) between this
209 * port and the TCP destination port of the flow.
210 */
211 if (tx->request_port_number >= 0) {
212 jb_set_uint(js, "http_port", tx->request_port_number);
213 }
214
215 /* uri */
216 if (tx->request_uri != NULL) {
217 jb_set_string_from_bytes(js, "url", bstr_ptr(tx->request_uri), bstr_len(tx->request_uri));
218 }
219
220 if (tx->request_headers != NULL) {
221 /* user agent */
222 htp_header_t *h_user_agent = htp_table_get_c(tx->request_headers, "user-agent");
223 if (h_user_agent != NULL) {
224 jb_set_string_from_bytes(js, "http_user_agent", bstr_ptr(h_user_agent->value),
225 bstr_len(h_user_agent->value));
226 }
227
228 /* x-forwarded-for */
229 htp_header_t *h_x_forwarded_for = htp_table_get_c(tx->request_headers, "x-forwarded-for");
230 if (h_x_forwarded_for != NULL) {
231 jb_set_string_from_bytes(js, "xff", bstr_ptr(h_x_forwarded_for->value),
232 bstr_len(h_x_forwarded_for->value));
233 }
234 }
235
236 /* content-type */
237 if (tx->response_headers != NULL) {
238 htp_header_t *h_content_type = htp_table_get_c(tx->response_headers, "content-type");
239 if (h_content_type != NULL) {
240 const size_t size = bstr_len(h_content_type->value) * 2 + 1;
241 char string[size];
242 BytesToStringBuffer(bstr_ptr(h_content_type->value), bstr_len(h_content_type->value), string, size);
243 char *p = strchr(string, ';');
244 if (p != NULL)
245 *p = '\0';
246 jb_set_string(js, "http_content_type", string);
247 }
248 htp_header_t *h_content_range = htp_table_get_c(tx->response_headers, "content-range");
249 if (h_content_range != NULL) {
250 jb_open_object(js, "content_range");
251 jb_set_string_from_bytes(
252 js, "raw", bstr_ptr(h_content_range->value), bstr_len(h_content_range->value));
253 HTTPContentRange crparsed;
254 if (HTPParseContentRange(h_content_range->value, &crparsed) == 0) {
255 if (crparsed.start >= 0)
256 jb_set_uint(js, "start", crparsed.start);
257 if (crparsed.end >= 0)
258 jb_set_uint(js, "end", crparsed.end);
259 if (crparsed.size >= 0)
260 jb_set_uint(js, "size", crparsed.size);
261 }
262 jb_close(js);
263 }
264 }
265}

267static void EveHttpLogJSONCustom(LogHttpFileCtx *http_ctx, JsonBuilder *js, htp_tx_t *tx)
268{
269 char *c;
270 HttpField f;
271
272 for (f = HTTP_FIELD_ACCEPT; f < HTTP_FIELD_SIZE; f++)
273 {
274 if ((http_ctx->fields & (1ULL<<f)) != 0)
275 {
276 /* prevent logging a field twice if extended logging is
277 enabled */
278 if (((http_ctx->flags & LOG_HTTP_EXTENDED) == 0) ||
279 ((http_ctx->flags & LOG_HTTP_EXTENDED) !=
280 (http_fields[f].flags & LOG_HTTP_EXTENDED)))
281 {
282 htp_header_t *h_field = NULL;
283 if ((http_fields[f].flags & LOG_HTTP_REQUEST) != 0)
284 {
285 if (tx->request_headers != NULL) {
286 h_field = htp_table_get_c(tx->request_headers,
287 http_fields[f].htp_field);
288 }
289 } else {
290 if (tx->response_headers != NULL) {
291 h_field = htp_table_get_c(tx->response_headers,
292 http_fields[f].htp_field);
293 }
294 }
295 if (h_field != NULL) {
296 c = bstr_util_strdup_to_c(h_field->value);
297 if (c != NULL) {
298 jb_set_string(js, http_fields[f].config_field, c);
299 SCFree(c);
300 }
301 }
302 }
303 }
304 }
305}

307static void EveHttpLogJSONExtended(JsonBuilder *js, htp_tx_t *tx)
308{
309 /* referer */
310 htp_header_t *h_referer = NULL;
311 if (tx->request_headers != NULL) {
312 h_referer = htp_table_get_c(tx->request_headers, "referer");
313 }
314 if (h_referer != NULL) {
315 jb_set_string_from_bytes(
316 js, "http_refer", bstr_ptr(h_referer->value), bstr_len(h_referer->value));
317 }
318
319 /* method */
320 if (tx->request_method != NULL) {
321 jb_set_string_from_bytes(
322 js, "http_method", bstr_ptr(tx->request_method), bstr_len(tx->request_method));
323 }
324
325 /* protocol */
326 if (tx->request_protocol != NULL) {
327 jb_set_string_from_bytes(
328 js, "protocol", bstr_ptr(tx->request_protocol), bstr_len(tx->request_protocol));
329 }
330
331 /* response status */
332 if (tx->response_status != NULL) {
333 const size_t status_size = bstr_len(tx->response_status) * 2 + 1;
334 char status_string[status_size];
335 BytesToStringBuffer(bstr_ptr(tx->response_status), bstr_len(tx->response_status),
336 status_string, status_size);
337 unsigned int val = strtoul(status_string, NULL, 10);
338 jb_set_uint(js, "status", val);
339
340 htp_header_t *h_location = htp_table_get_c(tx->response_headers, "location");
341 if (h_location != NULL) {
342 jb_set_string_from_bytes(
343 js, "redirect", bstr_ptr(h_location->value), bstr_len(h_location->value));
344 }
345 }
346
347 /* length */
348 jb_set_uint(js, "length", tx->response_message_len);
349}

351static void EveHttpLogJSONHeaders(JsonBuilder *js, uint32_t direction, htp_tx_t *tx)
352{
353 htp_table_t * headers = direction & LOG_HTTP_REQ_HEADERS ?
354 tx->request_headers : tx->response_headers;
355 char name[MAX_SIZE_HEADER_NAME] = {0};
356 char value[MAX_SIZE_HEADER_VALUE] = {0};
357 size_t n = htp_table_size(headers);
358 jb_open_array(js, direction & LOG_HTTP_REQ_HEADERS ? "request_headers" : "response_headers");
359 for (size_t i = 0; i < n; i++) {
360 htp_header_t * h = htp_table_get_index(headers, i, NULL);
361 if (h == NULL) {
362 continue;
363 }
364 jb_start_object(js);
365 size_t size_name = bstr_len(h->name) < MAX_SIZE_HEADER_NAME - 1 ?
366 bstr_len(h->name) : MAX_SIZE_HEADER_NAME - 1;
367 memcpy(name, bstr_ptr(h->name), size_name);
368 name[size_name] = '\0';
369 jb_set_string(js, "name", name);
370 size_t size_value = bstr_len(h->value) < MAX_SIZE_HEADER_VALUE - 1 ?
371 bstr_len(h->value) : MAX_SIZE_HEADER_VALUE - 1;
372 memcpy(value, bstr_ptr(h->value), size_value);
373 value[size_value] = '\0';
374 jb_set_string(js, "value", value);
375 jb_close(js);
376 }
377 // Close array.
378 jb_close(js);
379}

381static void BodyPrintableBuffer(JsonBuilder *js, HtpBody *body, const char *key)
382{
383 if (body->sb != NULL && body->sb->buf != NULL) {
384 uint32_t offset = 0;
385 const uint8_t *body_data;
386 uint32_t body_data_len;
387 uint64_t body_offset;
388
389 if (StreamingBufferGetData(body->sb, &body_data,
390 &body_data_len, &body_offset) == 0) {
391 return;
392 }
393
394 uint8_t printable_buf[body_data_len + 1];
395 PrintStringsToBuffer(printable_buf, &offset,
396 sizeof(printable_buf),
397 body_data, body_data_len);
398 if (offset > 0) {
399 jb_set_string(js, key, (char *)printable_buf);
400 }
401 }
402}

450/* JSON format logging */
451static void EveHttpLogJSON(JsonHttpLogThread *aft, JsonBuilder *js, htp_tx_t *tx, uint64_t tx_id)
452{
453 LogHttpFileCtx *http_ctx = aft->httplog_ctx;
454 jb_open_object(js, "http");
455
456 EveHttpLogJSONBasic(js, tx);
457 /* log custom fields if configured */
458 if (http_ctx->fields != 0)
459 EveHttpLogJSONCustom(http_ctx, js, tx);
460 if (http_ctx->flags & LOG_HTTP_EXTENDED)
461 EveHttpLogJSONExtended(js, tx);
462 if (http_ctx->flags & LOG_HTTP_REQ_HEADERS)
463 EveHttpLogJSONHeaders(js, LOG_HTTP_REQ_HEADERS, tx);
464 if (http_ctx->flags & LOG_HTTP_RES_HEADERS)
465 EveHttpLogJSONHeaders(js, LOG_HTTP_RES_HEADERS, tx);
466
467 jb_close(js);
468}

470static int JsonHttpLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
471{
472 SCEnter();
473
474 htp_tx_t *tx = txptr;
475 JsonHttpLogThread *jhl = (JsonHttpLogThread *)thread_data;
476
477 JsonBuilder *js = CreateEveHeaderWithTxId(
478 p, LOG_DIR_FLOW, "http", NULL, tx_id, jhl->httplog_ctx->eve_ctx);
479 if (unlikely(js == NULL))
480 return TM_ECODE_OK;
481
482 SCLogDebug("got a HTTP request and now logging !!");
483
484 EveHttpLogJSON(jhl, js, tx, tx_id);
485 HttpXFFCfg *xff_cfg = jhl->httplog_ctx->xff_cfg != NULL ?
486 jhl->httplog_ctx->xff_cfg : jhl->httplog_ctx->parent_xff_cfg;
487
488 /* xff header */
489 if ((xff_cfg != NULL) && !(xff_cfg->flags & XFF_DISABLED) && p->flow != NULL) {
490 int have_xff_ip = 0;
491 char buffer[XFF_MAXLEN];
492
493 have_xff_ip = HttpXFFGetIPFromTx(p->flow, tx_id, xff_cfg, buffer, XFF_MAXLEN);
494
495 if (have_xff_ip) {
496 if (xff_cfg->flags & XFF_EXTRADATA) {
497 jb_set_string(js, "xff", buffer);
498 }
499 else if (xff_cfg->flags & XFF_OVERWRITE) {
500 if (p->flowflags & FLOW_PKT_TOCLIENT) {
501 jb_set_string(js, "dest_ip", buffer);
502 } else {
503 jb_set_string(js, "src_ip", buffer);
504 }
505 }
506 }
507 }
508
509 OutputJsonBuilderBuffer(js, jhl->ctx);
510 jb_free(js);
511
512 SCReturnInt(TM_ECODE_OK);
513}

515bool EveHttpAddMetadata(const Flow *f, uint64_t tx_id, JsonBuilder *js)
516{
517 HtpState *htp_state = (HtpState *)FlowGetAppState(f);
518 if (htp_state) {
519 htp_tx_t *tx = AppLayerParserGetTx(IPPROTO_TCP, ALPROTO_HTTP1, htp_state, tx_id);
520
521 if (tx) {
522 EveHttpLogJSONBasic(js, tx);
523 EveHttpLogJSONExtended(js, tx);
524 return true;
525 }
526 }
527
528 return false;
529}

541static OutputInitResult OutputHttpLogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
542{
543 OutputInitResult result = { NULL, false };
544 OutputJsonCtx *ojc = parent_ctx->data;
545
546 LogHttpFileCtx *http_ctx = SCCalloc(1, sizeof(LogHttpFileCtx));
547 if (unlikely(http_ctx == NULL))
548 return result;
549 memset(http_ctx, 0x00, sizeof(*http_ctx));
550
551 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
552 if (unlikely(output_ctx == NULL)) {
553 SCFree(http_ctx);
554 return result;
555 }
556
557 http_ctx->flags = LOG_HTTP_DEFAULT;
558 http_ctx->eve_ctx = ojc;
559
560 if (conf) {
561 const char *extended = ConfNodeLookupChildValue(conf, "extended");
562
563 if (extended != NULL) {
564 if (ConfValIsTrue(extended)) {
565 http_ctx->flags = LOG_HTTP_EXTENDED;
566 }
567 }
568
569 ConfNode *custom;
570 if ((custom = ConfNodeLookupChild(conf, "custom")) != NULL) {
571 ConfNode *field;
572 TAILQ_FOREACH(field, &custom->head, next)
573 {
574 if (field != NULL)
575 {
576 HttpField f;
577 for (f = HTTP_FIELD_ACCEPT; f < HTTP_FIELD_SIZE; f++)
578 {
579 if ((strcmp(http_fields[f].config_field,
580 field->val) == 0) ||
581 (strcasecmp(http_fields[f].htp_field,
582 field->val) == 0))
583 {
584 http_ctx->fields |= (1ULL<<f);
585 break;
586 }
587 }
588 }
589 }
590 }
591 const char *all_headers = ConfNodeLookupChildValue(
592 conf, "dump-all-headers");
593 if (all_headers != NULL) {
594 if (strncmp(all_headers, "both", 4) == 0) {
595 http_ctx->flags |= LOG_HTTP_REQ_HEADERS;
596 http_ctx->flags |= LOG_HTTP_RES_HEADERS;
597 } else if (strncmp(all_headers, "request", 7) == 0) {
598 http_ctx->flags |= LOG_HTTP_REQ_HEADERS;
599 } else if (strncmp(all_headers, "response", 8) == 0) {
600 http_ctx->flags |= LOG_HTTP_RES_HEADERS;
601 }
602 }
603 }
604
605 if (conf != NULL && ConfNodeLookupChild(conf, "xff") != NULL) {
606 http_ctx->xff_cfg = SCCalloc(1, sizeof(HttpXFFCfg));
607 if (http_ctx->xff_cfg != NULL) {
608 HttpXFFGetCfg(conf, http_ctx->xff_cfg);
609 }
610 } else if (ojc->xff_cfg) {
611 http_ctx->parent_xff_cfg = ojc->xff_cfg;
612 }
613
614 output_ctx->data = http_ctx;
615 output_ctx->DeInit = OutputHttpLogDeinitSub;
616
617 /* enable the logger for the app layer */
618 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_HTTP1);
619
620 result.ctx = output_ctx;
621 result.ok = true;
622 return result;
623}

625static TmEcode JsonHttpLogThreadInit(ThreadVars *t, const void *initdata, void **data)
626{
627 JsonHttpLogThread *aft = SCCalloc(1, sizeof(JsonHttpLogThread));
628 if (unlikely(aft == NULL))
