39static uint32_t FeatureHashFunc(HashListTable *ht, void *data,
40 uint16_t datalen)
41{
42 FeatureEntryType *f = (FeatureEntryType *)data;
43 uint32_t hash = 0;
44 int len = strlen(f->feature);
45
46 for (int i = 0; i < len; i++)
47 hash += u8_tolower((unsigned char)f->feature[i]);
48
49 return (hash % ht->array_size);
50}

52static char FeatureHashCompareFunc(void *data1, uint16_t datalen1,
53 void *data2, uint16_t datalen2)
54{
55 FeatureEntryType *f1 = (FeatureEntryType *)data1;
56 FeatureEntryType *f2 = (FeatureEntryType *)data2;
57 int len1 = 0;
58 int len2 = 0;
59
60 if (f1 == NULL || f2 == NULL)
61 return 0;
62
63 if (f1->feature == NULL || f2->feature == NULL)
64 return 0;
65
66 len1 = strlen(f1->feature);
67 len2 = strlen(f2->feature);
68
69 return (len1 == len2 && memcmp(f1->feature, f2->feature, len1) == 0);
70}

72static void FeatureHashFreeFunc(void *data)

91static void FeatureAddEntry(const char *feature_name)
92{
93 int rc;
94
95 FeatureEntryType *feature = SCCalloc(1, sizeof(*feature));
96 if (!feature) {
97 FatalError(SC_ERR_MEM_ALLOC, "Unable to allocate feature entry memory.");
98 }
99
100 feature->feature = SCStrdup(feature_name);
101 if (feature->feature) {
102 rc = HashListTableAdd(feature_hash_table, feature, sizeof(*feature));
103 if (rc == 0)
104 return;
105 }
106
107 FeatureHashFreeFunc(feature);
108}

110void ProvidesFeature(const char *feature_name)
111{
112 FeatureEntryType f = { feature_name };
113
114 SCMutexLock(&feature_table_mutex);
115
116 FeatureEntryType *feature = HashListTableLookup(feature_hash_table, &f, sizeof(f));
117
118 if (!feature) {
119 FeatureAddEntry(feature_name);
120 }
121
122 SCMutexUnlock(&feature_table_mutex);
123}
