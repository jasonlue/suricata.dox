79/* not using the enum as that is a unsigned int, so 4 bytes */
80typedef uint16_t AppProto;
81
82static inline bool AppProtoIsValid(AppProto a)
83{
84 return ((a > ALPROTO_UNKNOWN && a < ALPROTO_FAILED));
85}
86
87// wether a signature AppProto matches a flow (or signature) AppProto
88static inline bool AppProtoEquals(AppProto sigproto, AppProto alproto)
89{
90 switch (sigproto) {
91 case ALPROTO_HTTP:
92 return (alproto == ALPROTO_HTTP1) || (alproto == ALPROTO_HTTP2) ||
93 (alproto == ALPROTO_HTTP);
94 case ALPROTO_DCERPC:
95 return (alproto == ALPROTO_DCERPC || alproto == ALPROTO_SMB);
96 }
97 return (sigproto == alproto);
98}
