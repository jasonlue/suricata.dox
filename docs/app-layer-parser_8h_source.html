82/** parser has successfully processed in the input, and has consumed
83 * all of it. */
84#define APP_LAYER_OK (AppLayerResult) { 0, 0, 0 }
85
86/** parser has hit an unrecoverable error. Returning this to the API
87 * leads to no further calls to the parser. */
88#define APP_LAYER_ERROR (AppLayerResult) { -1, 0, 0 }
89
90/** parser needs more data. Through 'c' it will indicate how many
91 * of the input bytes it has consumed. Through 'n' it will indicate
92 * how many more bytes it needs before getting called again.
93 * \note consumed (c) should never be more than the input len
94 * needed (n) + consumed (c) should be more than the input len
95 */
96#define APP_LAYER_INCOMPLETE(c,n) (AppLayerResult) { 1, (c), (n) }
