18/**
19 * \file
20 *
21 * \author OISF, Jason Ish <jason.ish@oisf.net>
22 * \author Endace Technology Limited, Jason Ish <jason.ish@endace.com>
23 *
24 * The root logging output for all non-application logging.
25 *
26 * The loggers are made up of a hierarchy of loggers. At the top we
27 * have the root logger which is the main entry point to
28 * logging. Under the root there exists parent loggers that are the
29 * entry point for specific types of loggers such as packet logger,
30 * transaction loggers, etc. Each parent logger may have 0 or more
31 * loggers that actual handle the job of producing output to something
32 * like a file.
33 */

140/**
141 * \brief Register an output module.
142 *
143 * This function will register an output module so it can be
144 * configured with the configuration file.
145 *
146 * \retval Returns 0 on success, -1 on failure.
147 */
148void OutputRegisterModule(const char *name, const char *conf_name,
149 OutputInitFunc InitFunc)
150{
151 OutputModule *module = SCCalloc(1, sizeof(*module));
152 if (unlikely(module == NULL))
153 goto error;
154
155 module->name = name;
156 module->conf_name = conf_name;
157 module->InitFunc = InitFunc;
158 TAILQ_INSERT_TAIL(&output_modules, module, entries);
159
160 SCLogDebug("Output module \"%s\" registered.", name);
161
162 return;
163
164error:
165 FatalError(SC_ERR_FATAL,
166 "Fatal error encountered in OutputRegisterModule. Exiting...");
167}

893
894TmEcode OutputLoggerThreadInit(ThreadVars *tv, const void *initdata, void **data)
895{
896 LoggerThreadStore *thread_store = SCCalloc(1, sizeof(*thread_store));
897 if (thread_store == NULL) {
898 return TM_ECODE_FAILED;
899 }
900 TAILQ_INIT(thread_store);
901 *data = (void *)thread_store;
902
903 RootLogger *logger;
904 TAILQ_FOREACH(logger, &active_loggers, entries) {
905
906 void *child_thread_data = NULL;
907 if (logger->ThreadInit != NULL) {
908 if (logger->ThreadInit(tv, initdata, &child_thread_data) == TM_ECODE_OK) {
909 LoggerThreadStoreNode *thread_store_node =
910 SCCalloc(1, sizeof(*thread_store_node));
911 if (thread_store_node == NULL) {
912 /* Undo everything, calling de-init will take care
913 * of that. */
914 OutputLoggerThreadDeinit(tv, thread_store);
915 return TM_ECODE_FAILED;
916 }
917 thread_store_node->thread_data = child_thread_data;
918 TAILQ_INSERT_TAIL(thread_store, thread_store_node, entries);
919 }
920 }
921 }
922 return TM_ECODE_OK;
923}
924
925TmEcode OutputLoggerThreadDeinit(ThreadVars *tv, void *thread_data)
926{
927 if (thread_data == NULL)
928 return TM_ECODE_FAILED;
929
930 LoggerThreadStore *thread_store = (LoggerThreadStore *)thread_data;
931 RootLogger *logger = TAILQ_FIRST(&active_loggers);
932 LoggerThreadStoreNode *thread_store_node = TAILQ_FIRST(thread_store);
933 while (logger && thread_store_node) {
934 if (logger->ThreadDeinit != NULL) {
935 logger->ThreadDeinit(tv, thread_store_node->thread_data);
936 }
937 logger = TAILQ_NEXT(logger, entries);
938 thread_store_node = TAILQ_NEXT(thread_store_node, entries);
939 }
940
941 /* Free the thread store. */
942 while ((thread_store_node = TAILQ_FIRST(thread_store)) != NULL) {
943 TAILQ_REMOVE(thread_store, thread_store_node, entries);
944 SCFree(thread_store_node);
945 }
946 SCFree(thread_store);
947
948 return TM_ECODE_OK;
949}
