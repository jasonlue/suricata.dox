46/**
47 * \brief Registers the handler functions for the "priority" keyword
48 */
49void DetectPriorityRegister (void)
50{
51 sigmatch_table[DETECT_PRIORITY].name = "priority";
52 sigmatch_table[DETECT_PRIORITY].desc = "rules with a higher priority will be examined first";
53 sigmatch_table[DETECT_PRIORITY].url = "/rules/meta.html#priority";
54 sigmatch_table[DETECT_PRIORITY].Setup = DetectPrioritySetup;
55#ifdef UNITTESTS
56 sigmatch_table[DETECT_PRIORITY].RegisterTests = PriorityRegisterTests;
57#endif
58 DetectSetupParseRegexes(PARSE_REGEX, &parse_regex);
59}

61static int DetectPrioritySetup (DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
62{
63 char copy_str[128] = "";
64
65 int ret = 0;
66 size_t pcre2len;
67
68 ret = DetectParsePcreExec(&parse_regex, rawstr, 0, 0);
69 if (ret < 0) {
70 SCLogError(SC_ERR_PCRE_MATCH, "Invalid Priority in Signature "
71 "- %s", rawstr);
72 return -1;
73 }
74
75 pcre2len = sizeof(copy_str);
76 ret = pcre2_substring_copy_bynumber(parse_regex.match, 1, (PCRE2_UCHAR8 *)copy_str, &pcre2len);
77 if (ret < 0) {
78 SCLogError(SC_ERR_PCRE_GET_SUBSTRING, "pcre2_substring_copy_bynumber failed");
79 return -1;
80 }
81
82 long prio = 0;
83 char *endptr = NULL;
84 prio = strtol(copy_str, &endptr, 10);
85 if (endptr == NULL || *endptr != '\0') {
86 SCLogError(SC_ERR_INVALID_SIGNATURE, "Saw an invalid character as arg "
87 "to priority keyword");
88 return -1;
89 }
90
91 if (s->init_data->init_flags & SIG_FLAG_INIT_PRIO_EXPLICT) {
92 SCLogWarning(SC_ERR_CONFLICTING_RULE_KEYWORDS, "duplicate priority "
93 "keyword. Using highest priority in the rule");
94 s->prio = MIN(s->prio, prio);
95 } else {
96 s->prio = prio;
97 s->init_data->init_flags |= SIG_FLAG_INIT_PRIO_EXPLICT;
98 }
99 return 0;
100}

106static int DetectPriorityTest01(void)
107{
108 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
109 FAIL_IF_NULL(de_ctx);
110
111 de_ctx->sig_list = SigInit(de_ctx, "alert tcp any any -> any any "
112 "(msg:\"Priority test\"; priority:2; sid:1;)");
113 FAIL_IF_NULL(de_ctx->sig_list);
114
115 FAIL_IF_NOT(de_ctx->sig_list->prio == 2);
116
117 DetectEngineCtxFree(de_ctx);
118 PASS;
119}
