67int DetectDatarepBufferMatch(DetectEngineThreadCtx *det_ctx,
68 const DetectDatarepData *sd,
69 const uint8_t *data, const uint32_t data_len)
70{
71 if (data == NULL || data_len == 0)
72 return 0;
73
74 DataRepResultType r = DatasetLookupwRep(sd->set, data, data_len, &sd->rep);
75 if (!r.found)
76 return 0;
77
78 switch (sd->op) {
79 case DATAREP_OP_GT:
80 if (r.rep.value > sd->rep.value)
81 return 1;
82 break;
83 case DATAREP_OP_LT:
84 if (r.rep.value < sd->rep.value)
85 return 1;
86 break;
87 case DATAREP_OP_EQ:
88 if (r.rep.value == sd->rep.value)
89 return 1;
90 break;
91 }
92 return 0;
93}

95static int DetectDatarepParse(const char *str, char *cmd, int cmd_len, char *name, int name_len,
96 enum DatasetTypes *type, char *load, size_t load_size, uint16_t *rep_value,
97 uint64_t *memcap, uint32_t *hashsize)
98{
99 bool cmd_set = false;
100 bool name_set = false;
101 bool value_set = false;
102
103 char copy[strlen(str)+1];
104 strlcpy(copy, str, sizeof(copy));
105 char *xsaveptr = NULL;
106 char *key = strtok_r(copy, ",", &xsaveptr);
107 while (key != NULL) {
108 while (*key != '\0' && isblank(*key)) {
109 key++;
110 }
111 char *val = strchr(key, ' ');
112 if (val != NULL) {
113 *val++ = '\0';
114 while (*val != '\0' && isblank(*val)) {
115 val++;
116 SCLogDebug("cmd %s val %s", key, val);
117 }
118 } else {
119 SCLogDebug("cmd %s", key);
120 }
121
122 if (strlen(key) == 0) {
123 goto next;
124 }
125
126 if (!name_set) {
127 if (val) {
128 return -1;
129 }
130 strlcpy(name, key, name_len);
131 name_set = true;
132 } else if (!cmd_set) {
133 if (val) {
134 return -1;
135 }
136 strlcpy(cmd, key, cmd_len);
137 cmd_set = true;
138 } else if (!value_set) {
139 if (val) {
140 return -1;
141 }
142
143 if (StringParseUint16(rep_value, 10, 0, key) < 0)
144 return -1;
145
146 value_set = true;
147 } else {
148 if (val == NULL) {
149 return -1;
150 }
151
152 if (strcmp(key, "type") == 0) {
153 SCLogDebug("type %s", val);
154
155 if (strcmp(val, "md5") == 0) {
156 *type = DATASET_TYPE_MD5;
157 } else if (strcmp(val, "sha256") == 0) {
158 *type = DATASET_TYPE_SHA256;
159 } else if (strcmp(val, "string") == 0) {
160 *type = DATASET_TYPE_STRING;
161 } else {
162 SCLogDebug("bad type %s", val);
163 return -1;
164 }
165
166 } else if (strcmp(key, "load") == 0) {
167 SCLogDebug("load %s", val);
168 strlcpy(load, val, load_size);
169 }
170 if (strcmp(key, "memcap") == 0) {
171 if (ParseSizeStringU64(val, memcap) < 0) {
172 SCLogWarning(SC_ERR_INVALID_VALUE,
173 "invalid value for memcap: %s,"
174 " resetting to default",
175 val);
176 *memcap = 0;
177 }
178 }
179 if (strcmp(key, "hashsize") == 0) {
180 if (ParseSizeStringU32(val, hashsize) < 0) {
181 SCLogWarning(SC_ERR_INVALID_VALUE,
182 "invalid value for hashsize: %s,"
183 " resetting to default",
184 val);
185 *hashsize = 0;
186 }
187 }
188 }
189
190 SCLogDebug("key: %s, value: %s", key, val);
191
192 next:
193 key = strtok_r(NULL, ",", &xsaveptr);
194 }
195
196 if (strlen(load) > 0 && *type == DATASET_TYPE_NOTSET) {
197 SCLogError(SC_ERR_INVALID_SIGNATURE,
198 "if load is used type must be set as well");
199 return 0;
200 }
201
202 if (!name_set || !cmd_set || !value_set) {
203 SCLogError(SC_ERR_INVALID_SIGNATURE,
204 "missing values");
205 return 0;
206 }
207
208 /* Trim trailing whitespace. */
209 while (strlen(name) > 0 && isblank(name[strlen(name) - 1])) {
210 name[strlen(name) - 1] = '\0';
211 }
212
213 /* Validate name, spaces are not allowed. */
214 for (size_t i = 0; i < strlen(name); i++) {
215 if (isblank(name[i])) {
216 SCLogError(SC_ERR_INVALID_SIGNATURE,
217 "spaces not allowed in dataset names");
218 return 0;
219 }
220 }
221
222 return 1;
223}

225/** \brief wrapper around dirname that does leave input untouched */
226static void GetDirName(const char *in, char *out, size_t outs)
227{
228 if (strlen(in) == 0) {
229 return;
230 }
231
232 size_t size = strlen(in) + 1;
233 char tmp[size];
234 strlcpy(tmp, in, size);
235
236 char *dir = dirname(tmp);
237 BUG_ON(dir == NULL);
238 strlcpy(out, dir, outs);
239 return;
240}

242static int SetupLoadPath(const DetectEngineCtx *de_ctx,
243 char *load, size_t load_size)
244{
245 SCLogDebug("load %s", load);
246
247 if (PathIsAbsolute(load)) {
248 return 0;
249 }
250
251 bool done = false;
252#ifdef HAVE_LIBGEN_H
253 BUG_ON(de_ctx->rule_file == NULL);
254
255 char dir[PATH_MAX] = "";
256 GetDirName(de_ctx->rule_file, dir, sizeof(dir));
257
258 SCLogDebug("rule_file %s dir %s", de_ctx->rule_file, dir);
259 char path[PATH_MAX];
260 if (snprintf(path, sizeof(path), "%s/%s", dir, load) >= (int)sizeof(path)) // TODO windows path
261 return -1;
262
263 if (SCPathExists(path)) {
264 done = true;
265 strlcpy(load, path, load_size);
266 SCLogDebug("using path '%s' (HAVE_LIBGEN_H)", load);
267 } else {
268 SCLogDebug("path '%s' does not exist (HAVE_LIBGEN_H)", path);
269 }
270#endif
271 if (!done) {
272 char *loadp = DetectLoadCompleteSigPath(de_ctx, load);
273 if (loadp == NULL) {
274 return -1;
275 }
276 SCLogDebug("loadp %s", loadp);
277
278 if (SCPathExists(loadp)) {
279 strlcpy(load, loadp, load_size);
280 SCLogDebug("using path '%s' (non-HAVE_LIBGEN_H)", load);
281 } else {
282 SCLogDebug("path '%s' does not exist (non-HAVE_LIBGEN_H)", loadp);
283 }
284 SCFree(loadp);
285
286 // TODO try data-dir as well?
287 }
288 return 0;
289}

291static int DetectDatarepSetup (DetectEngineCtx *de_ctx, Signature *s, const char *rawstr)
292{
293 SigMatch *sm = NULL;
294 char cmd_str[16] = "", name[64] = "";
295 enum DatasetTypes type = DATASET_TYPE_NOTSET;
296 char load[PATH_MAX] = "";
297 uint16_t value = 0;
298 uint64_t memcap = 0;
299 uint32_t hashsize = 0;
300
301 if (DetectBufferGetActiveList(de_ctx, s) == -1) {
302 SCLogError(SC_ERR_INVALID_SIGNATURE,
303 "datarep is only supported for sticky buffers");
304 SCReturnInt(-1);
305 }
306
307 int list = s->init_data->list;
308 if (list == DETECT_SM_LIST_NOTSET) {
309 SCLogError(SC_ERR_INVALID_SIGNATURE,
310 "datarep is only supported for sticky buffers");
311 SCReturnInt(-1);
312 }
313
314 if (!DetectDatarepParse(rawstr, cmd_str, sizeof(cmd_str), name, sizeof(name), &type, load,
315 sizeof(load), &value, &memcap, &hashsize)) {
316 return -1;
317 }
318
319 if (strlen(load) != 0) {
320 if (SetupLoadPath(de_ctx, load, sizeof(load)) != 0)
321 return -1;
322 }
323
324 enum DetectDatarepOp op;
325 if (strcmp(cmd_str,">") == 0) {
326 op = DATAREP_OP_GT;
327 } else if (strcmp(cmd_str,"<") == 0) {
328 op = DATAREP_OP_LT;
329 } else if (strcmp(cmd_str,"==") == 0) {
330 op = DATAREP_OP_EQ;
331 } else {
332 SCLogError(SC_ERR_UNKNOWN_VALUE,
333 "datarep operation \"%s\" is not supported.", cmd_str);
334 return -1;
335 }
336
337 Dataset *set = DatasetGet(name, type, /* no save */ NULL, load, memcap, hashsize);
338 if (set == NULL) {
339 SCLogError(SC_ERR_UNKNOWN_VALUE,
340 "failed to set up datarep set '%s'.", name);
341 return -1;
342 }
343
344 DetectDatarepData *cd = SCCalloc(1, sizeof(DetectDatarepData));
345 if (unlikely(cd == NULL))
346 goto error;
347
348 cd->set = set;
349 cd->op = op;
350 cd->rep.value = value;
351
352 SCLogDebug("cmd %s, name %s",
353 cmd_str, strlen(name) ? name : "(none)");
354
355 /* Okay so far so good, lets get this into a SigMatch
356 * and put it in the Signature. */
357 sm = SigMatchAlloc();
358 if (sm == NULL)
359 goto error;
360
361 sm->type = DETECT_DATAREP;
362 sm->ctx = (SigMatchCtx *)cd;
363 SigMatchAppendSMToList(s, sm, list);
364 return 0;
365
366error:
367 if (cd != NULL)
368 SCFree(cd);
369 if (sm != NULL)
370 SCFree(sm);
371 return -1;
372}
