73/**
74 * \internal
75 * \brief This function is used to match packets with a given Ack number
76 *
77 * \param t pointer to thread vars
78 * \param det_ctx pointer to the pattern matcher thread
79 * \param p pointer to the current packet
80 * \param m pointer to the sigmatch that we will cast into DetectAckData
81 *
82 * \retval 0 no match
83 * \retval 1 match
84 */
85static int DetectAckMatch(DetectEngineThreadCtx *det_ctx,
86 Packet *p, const Signature *s, const SigMatchCtx *ctx)
87{
88 const DetectAckData *data = (const DetectAckData *)ctx;
89
90 /* This is only needed on TCP packets */
91 if (!(PKT_IS_TCP(p)) || PKT_IS_PSEUDOPKT(p)) {
92 return 0;
93 }
94
95 return (data->ack == TCP_GET_ACK(p)) ? 1 : 0;
96}

98/**
99 * \internal
100 * \brief this function is used to add the ack option into the signature
101 *
102 * \param de_ctx pointer to the Detection Engine Context
103 * \param s pointer to the Current Signature
104 * \param m pointer to the Current SigMatch
105 * \param optstr pointer to the user provided options
106 *
107 * \retval 0 on Success
108 * \retval -1 on Failure
109 */
110static int DetectAckSetup(DetectEngineCtx *de_ctx, Signature *s, const char *optstr)
111{
112 DetectAckData *data = NULL;
113 SigMatch *sm = NULL;
114
115 data = SCMalloc(sizeof(DetectAckData));
116 if (unlikely(data == NULL))
117 goto error;
118
119 sm = SigMatchAlloc();
120 if (sm == NULL)
121 goto error;
122
123 sm->type = DETECT_ACK;
124
125 if (StringParseUint32(&data->ack, 10, 0, optstr) < 0) {
126 goto error;
127 }
128 sm->ctx = (SigMatchCtx*)data;
129
130 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_MATCH);
131 s->flags |= SIG_FLAG_REQUIRE_PACKET;
132
133 return 0;
134
135error:
136 if (data)
137 SCFree(data);
138 if (sm)
139 SigMatchFree(de_ctx, sm);
140 return -1;
141
142}

158static void
159PrefilterPacketAckMatch(DetectEngineThreadCtx *det_ctx, Packet *p, const void *pectx)
160{
161 const PrefilterPacketHeaderCtx *ctx = pectx;
162
163 if (!PrefilterPacketHeaderExtraMatch(ctx, p))
164 return;
165
166 if ((p->proto) == IPPROTO_TCP && !(PKT_IS_PSEUDOPKT(p)) &&
167 (p->tcph != NULL) && (TCP_GET_ACK(p) == ctx->v1.u32[0]))
168 {
169 SCLogDebug("packet matches TCP ack %u", ctx->v1.u32[0]);
170 PrefilterAddSids(&det_ctx->pmq, ctx->sigs_array, ctx->sigs_cnt);
171 }
172}

174static void
175PrefilterPacketAckSet(PrefilterPacketHeaderValue *v, void *smctx)
176{
177 const DetectAckData *a = smctx;
178 v->u32[0] = a->ack;
179}

181static bool
182PrefilterPacketAckCompare(PrefilterPacketHeaderValue v, void *smctx)
183{
184 const DetectAckData *a = smctx;
185 if (v.u32[0] == a->ack)
186 return true;
187 return false;
188}

190static int PrefilterSetupTcpAck(DetectEngineCtx *de_ctx, SigGroupHead *sgh)
191{
192 return PrefilterSetupPacketHeader(de_ctx, sgh, DETECT_ACK,
193 PrefilterPacketAckSet,
194 PrefilterPacketAckCompare,
195 PrefilterPacketAckMatch);
196}

198static bool PrefilterTcpAckIsPrefilterable(const Signature *s)
199{
200 const SigMatch *sm;
201 for (sm = s->init_data->smlists[DETECT_SM_LIST_MATCH] ; sm != NULL; sm = sm->next) {
202 switch (sm->type) {
203 case DETECT_ACK:
204 return true;
205 }
206 }
207 return false;
208}

215static int DetectAckSigTest01(void)
216{
217 Packet *p1 = NULL;
218 Packet *p2 = NULL;
219 Packet *p3 = NULL;
220 ThreadVars th_v;
221 DetectEngineThreadCtx *det_ctx;
222 int result = 0;
223
224 memset(&th_v, 0, sizeof(th_v));
225
226 /* TCP w/ack=42 */
227 p1 = UTHBuildPacket(NULL, 0, IPPROTO_TCP);
228 p1->tcph->th_ack = htonl(42);
229
230 /* TCP w/ack=100 */
231 p2 = UTHBuildPacket(NULL, 0, IPPROTO_TCP);
232 p2->tcph->th_ack = htonl(100);
233
234 /* ICMP */
235 p3 = UTHBuildPacket(NULL, 0, IPPROTO_ICMP);
236
237 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
238 if (de_ctx == NULL) {
239 goto end;
240 }
241
242 de_ctx->flags |= DE_QUIET;
243
244 /* These three are crammed in here as there is no Parse */
245 if (SigInit(de_ctx,
246 "alert tcp any any -> any any "
247 "(msg:\"Testing ack\";ack:foo;sid:1;)") != NULL)
248 {
249 printf("invalid ack accepted: ");
250 goto cleanup_engine;
251 }
252 if (SigInit(de_ctx,
253 "alert tcp any any -> any any "
254 "(msg:\"Testing ack\";ack:9999999999;sid:1;)") != NULL)
255 {
256 printf("overflowing ack accepted: ");
257 goto cleanup_engine;
258 }
259 if (SigInit(de_ctx,
260 "alert tcp any any -> any any "
261 "(msg:\"Testing ack\";ack:-100;sid:1;)") != NULL)
262 {
263 printf("negative ack accepted: ");
264 goto cleanup_engine;
265 }
266
267 de_ctx->sig_list = SigInit(de_ctx,
268 "alert tcp any any -> any any "
269 "(msg:\"Testing ack\";ack:41;sid:1;)");
270 if (de_ctx->sig_list == NULL) {
271 goto cleanup_engine;
272 }
273
274 de_ctx->sig_list->next = SigInit(de_ctx,
275 "alert tcp any any -> any any "
276 "(msg:\"Testing ack\";ack:42;sid:2;)");
277 if (de_ctx->sig_list->next == NULL) {
278 goto cleanup_engine;
279 }
280
281 SigGroupBuild(de_ctx);
282 DetectEngineThreadCtxInit(&th_v, (void *)de_ctx, (void *)&det_ctx);
283
284 SigMatchSignatures(&th_v, de_ctx, det_ctx, p1);
285 if (PacketAlertCheck(p1, 1) != 0) {
286 printf("sid 1 alerted, but should not have: ");
287 goto cleanup;
288 }
289 if (PacketAlertCheck(p1, 2) == 0) {
290 printf("sid 2 did not alert, but should have: ");
291 goto cleanup;
292 }
293
294 SigMatchSignatures(&th_v, de_ctx, det_ctx, p2);
295 if (PacketAlertCheck(p2, 1) != 0) {
296 printf("sid 1 alerted, but should not have: ");
297 goto cleanup;
298 }
299 if (PacketAlertCheck(p2, 2) != 0) {
300 printf("sid 2 alerted, but should not have: ");
301 goto cleanup;
302 }
303
304 SigMatchSignatures(&th_v, de_ctx, det_ctx, p3);
305 if (PacketAlertCheck(p3, 1) != 0) {
306 printf("sid 1 alerted, but should not have: ");
307 goto cleanup;
308 }
309 if (PacketAlertCheck(p3, 2) != 0) {
310 printf("sid 2 alerted, but should not have: ");
311 goto cleanup;
312 }
313
314 result = 1;
315
316cleanup:
317 SigGroupCleanup(de_ctx);
318 SigCleanSignatures(de_ctx);
319
320 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
321
322cleanup_engine:
323 DetectEngineCtxFree(de_ctx);
324
325end:
326 return result;
327}
