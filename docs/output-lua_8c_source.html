92/** \internal
93 * \brief TX logger for lua scripts
94 *
95 * A single call to this function will run one script on a single
96 * transaction.
97 *
98 * NOTE: The flow (f) also referenced by p->flow is locked.
99 */
100static int LuaTxLogger(ThreadVars *tv, void *thread_data, const Packet *p, Flow *f, void *alstate, void *txptr, uint64_t tx_id)
101{
102 SCEnter();
103
104 LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
105
106 SCMutexLock(&td->lua_ctx->m);
107
108 LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
109 LuaStateSetPacket(td->lua_ctx->luastate, (Packet *)p);
110 LuaStateSetTX(td->lua_ctx->luastate, txptr);
111 LuaStateSetFlow(td->lua_ctx->luastate, f);
112
113 /* prepare data to pass to script */
114 lua_getglobal(td->lua_ctx->luastate, "log");
115 lua_newtable(td->lua_ctx->luastate);
116 LuaPushTableKeyValueInt(td->lua_ctx->luastate, "tx_id", (int)(tx_id));
117
118 int retval = lua_pcall(td->lua_ctx->luastate, 1, 0, 0);
119 if (retval != 0) {
120 SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
121 }
122
123 SCMutexUnlock(&td->lua_ctx->m);
124 SCReturnInt(0);
125}

127/** \internal
128 * \brief Streaming logger for lua scripts
129 *
130 * Hooks into the Streaming Logger API. Gets called for each chunk of new
131 * streaming data.
132 */
133static int LuaStreamingLogger(ThreadVars *tv, void *thread_data, const Flow *f,
134 const uint8_t *data, uint32_t data_len, uint64_t tx_id, uint8_t flags)
135{
136 SCEnter();
137
138 void *txptr = NULL;
139 LuaStreamingBuffer b = { data, data_len, flags };
140
141 SCLogDebug("flags %02x", flags);
142
143 if (flags & OUTPUT_STREAMING_FLAG_TRANSACTION) {
144 if (f && f->alstate)
145 txptr = AppLayerParserGetTx(f->proto, f->alproto, f->alstate, tx_id);
146 }
147
148 LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
149
150 SCMutexLock(&td->lua_ctx->m);
151
152 LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
153 if (flags & OUTPUT_STREAMING_FLAG_TRANSACTION)
154 LuaStateSetTX(td->lua_ctx->luastate, txptr);
155 LuaStateSetFlow(td->lua_ctx->luastate, (Flow *)f);
156 LuaStateSetStreamingBuffer(td->lua_ctx->luastate, &b);
157
158 /* prepare data to pass to script */
159 lua_getglobal(td->lua_ctx->luastate, "log");
160 lua_newtable(td->lua_ctx->luastate);
161
162 if (flags & OUTPUT_STREAMING_FLAG_TRANSACTION)
163 LuaPushTableKeyValueInt(td->lua_ctx->luastate, "tx_id", (int)(tx_id));
164
165 int retval = lua_pcall(td->lua_ctx->luastate, 1, 0, 0);
166 if (retval != 0) {
167 SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
168 }
169
170 SCMutexUnlock(&td->lua_ctx->m);
171
172 SCReturnInt(TM_ECODE_OK);
173}

175/** \internal
176 * \brief Packet Logger for lua scripts, for alerts
177 *
178 * A single call to this function will run one script for a single
179 * packet. If it is called, it means that the registered condition
180 * function has returned TRUE.
181 *
182 * The script is called once for each alert stored in the packet.
183 *
184 * NOTE: p->flow is UNlocked
185 */
186static int LuaPacketLoggerAlerts(ThreadVars *tv, void *thread_data, const Packet *p)
187{
188 LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
189
190 char timebuf[64];
191 CreateTimeString(&p->ts, timebuf, sizeof(timebuf));
192
193 if (!(PKT_IS_IPV4(p)) && !(PKT_IS_IPV6(p))) {
194 /* decoder event */
195 goto not_supported;
196 }
197
198 /* loop through alerts stored in the packet */
199 SCMutexLock(&td->lua_ctx->m);
200 uint16_t cnt;
201 for (cnt = 0; cnt < p->alerts.cnt; cnt++) {
202 const PacketAlert *pa = &p->alerts.alerts[cnt];
203 if (unlikely(pa->s == NULL)) {
204 continue;
205 }
206
207 lua_getglobal(td->lua_ctx->luastate, "log");
208
209 void *txptr = NULL;
210 if (p->flow && p->flow->alstate && (pa->flags & PACKET_ALERT_FLAG_TX))
211 txptr = AppLayerParserGetTx(p->proto, p->flow->alproto, p->flow->alstate, pa->tx_id);
212
213 LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
214 LuaStateSetPacket(td->lua_ctx->luastate, (Packet *)p);
215 LuaStateSetTX(td->lua_ctx->luastate, txptr);
216 LuaStateSetFlow(td->lua_ctx->luastate, p->flow);
217 LuaStateSetPacketAlert(td->lua_ctx->luastate, (PacketAlert *)pa);
218
219 /* prepare data to pass to script */
220 //lua_newtable(td->lua_ctx->luastate);
221
222 int retval = lua_pcall(td->lua_ctx->luastate, 0, 0, 0);
223 if (retval != 0) {
224 SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
225 }
226 }
227 SCMutexUnlock(&td->lua_ctx->m);
228not_supported:
229 SCReturnInt(0);
230}

287/** \internal
288 * \brief File API Logger function for Lua scripts
289 *
290 * Executes a script once for one file.
291 *
292 * NOTE p->flow is locked at this point
293 */
294static int LuaFileLogger(ThreadVars *tv, void *thread_data, const Packet *p, const File *ff, uint8_t dir)
295{
296 SCEnter();
297 LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
298
299 if ((!(PKT_IS_IPV4(p))) && (!(PKT_IS_IPV6(p))))
300 return 0;
301
302 BUG_ON(ff->flags & FILE_LOGGED);
303
304 SCLogDebug("ff %p", ff);
305
306 SCMutexLock(&td->lua_ctx->m);
307
308 LuaStateSetThreadVars(td->lua_ctx->luastate, tv);
309 LuaStateSetPacket(td->lua_ctx->luastate, (Packet *)p);
310 if (p->flow && p->flow->alstate) {
311 void *txptr = AppLayerParserGetTx(p->proto, p->flow->alproto, p->flow->alstate, ff->txid);
312 if (txptr) {
313 LuaStateSetTX(td->lua_ctx->luastate, txptr);
314 }
315 }
316 LuaStateSetFlow(td->lua_ctx->luastate, p->flow);
317 LuaStateSetFile(td->lua_ctx->luastate, (File *)ff);
318
319 /* get the lua function to call */
320 lua_getglobal(td->lua_ctx->luastate, "log");
321
322 int retval = lua_pcall(td->lua_ctx->luastate, 0, 0, 0);
323 if (retval != 0) {
324 SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
325 }
326 SCMutexUnlock(&td->lua_ctx->m);
327 return 0;
328}

362static int LuaStatsLogger(ThreadVars *tv, void *thread_data, const StatsTable *st)
363{
364 SCEnter();
365 LogLuaThreadCtx *td = (LogLuaThreadCtx *)thread_data;
366
367 SCMutexLock(&td->lua_ctx->m);
368
369 lua_State *luastate = td->lua_ctx->luastate;
370 /* get the lua function to call */
371 lua_getglobal(td->lua_ctx->luastate, "log");
372
373 /* create lua array, which is really just a table. The key is an int (1-x),
374 * the value another table with named fields: name, tm_name, value, pvalue.
375 * { 1, { name=<name>, tmname=<tm_name>, value=<value>, pvalue=<pvalue>}}
376 * { 2, { name=<name>, tmname=<tm_name>, value=<value>, pvalue=<pvalue>}}
377 * etc
378 */
379 lua_newtable(luastate);
380 uint32_t u = 0;
381 for (; u < st->nstats; u++) {
382 lua_pushinteger(luastate, u + 1);
383
384 lua_newtable(luastate);
385
386 lua_pushstring(luastate, "name");
387 lua_pushstring(luastate, st->stats[u].name);
388 lua_settable(luastate, -3);
389
390 lua_pushstring(luastate, "tmname");
391 lua_pushstring(luastate, st->stats[u].tm_name);
392 lua_settable(luastate, -3);
393
394 lua_pushstring(luastate, "value");
395 lua_pushinteger(luastate, st->stats[u].value);
396 lua_settable(luastate, -3);
397
398 lua_pushstring(luastate, "pvalue");
399 lua_pushinteger(luastate, st->stats[u].pvalue);
400 lua_settable(luastate, -3);
401
402 lua_settable(luastate, -3);
403 }
404
405 int retval = lua_pcall(td->lua_ctx->luastate, 1, 0, 0);
406 if (retval != 0) {
407 SCLogInfo("failed to run script: %s", lua_tostring(td->lua_ctx->luastate, -1));
408 }
409 SCMutexUnlock(&td->lua_ctx->m);
410 return 0;
411
412}

426/** \brief load and evaluate the script
427 *
428 * This function parses the script, checks if all the required functions
429 * are defined and runs the 'init' function. The init function will inform
430 * us what the scripts needs are.
431 *
432 * \param filename filename of lua script file
433 * \param options struct to pass script requirements/options back to caller
434 * \retval errcode 0 ok, -1 error
435 */
436static int LuaScriptInit(const char *filename, LogLuaScriptOptions *options) {
437 lua_State *luastate = LuaGetState();
438 if (luastate == NULL)
439 goto error;
440 luaL_openlibs(luastate);
441
442 int status = luaL_loadfile(luastate, filename);
443 if (status) {
444 SCLogError(SC_ERR_LUA_ERROR, "couldn't load file: %s", lua_tostring(luastate, -1));
445 goto error;
446 }
447
448 /* prime the script (or something) */
449 if (lua_pcall(luastate, 0, 0, 0) != 0) {
450 SCLogError(SC_ERR_LUA_ERROR, "couldn't prime file: %s", lua_tostring(luastate, -1));
451 goto error;
452 }
453
454 lua_getglobal(luastate, "init");
455 if (lua_type(luastate, -1) != LUA_TFUNCTION) {
456 SCLogError(SC_ERR_LUA_ERROR, "no init function in script");
457 goto error;
458 }
459
460 lua_newtable(luastate); /* stack at -1 */
461 if (lua_gettop(luastate) == 0 || lua_type(luastate, 2) != LUA_TTABLE) {
462 SCLogError(SC_ERR_LUA_ERROR, "no table setup");
463 goto error;
464 }
465
466 lua_pushliteral(luastate, "script_api_ver");
467 lua_pushnumber (luastate, 1);
468 lua_settable(luastate, -3);
469
470 if (lua_pcall(luastate, 1, 1, 0) != 0) {
471 SCLogError(SC_ERR_LUA_ERROR, "couldn't run script 'init' function: %s", lua_tostring(luastate, -1));
472 goto error;
473 }
474
475 /* process returns from script */
476 if (lua_gettop(luastate) == 0) {
477 SCLogError(SC_ERR_LUA_ERROR, "init function in script should return table, nothing returned");
478 goto error;
479 }
480 if (lua_type(luastate, 1) != LUA_TTABLE) {
481 SCLogError(SC_ERR_LUA_ERROR, "init function in script should return table, returned is not table");
482 goto error;
483 }
484
485 lua_pushnil(luastate);
486 const char *k, *v;
487 while (lua_next(luastate, -2)) {
488 k = lua_tostring(luastate, -2);
489 if (k == NULL)
490 continue;
491
492 v = lua_tostring(luastate, -1);
493 lua_pop(luastate, 1);
494 if (v == NULL)
495 continue;
496
497 SCLogDebug("k='%s', v='%s'", k, v);
498
499 if (strcmp(k,"protocol") == 0 && strcmp(v, "http") == 0)
500 options->alproto = ALPROTO_HTTP1;
501 else if (strcmp(k,"protocol") == 0 && strcmp(v, "dns") == 0)
502 options->alproto = ALPROTO_DNS;
503 else if (strcmp(k,"protocol") == 0 && strcmp(v, "tls") == 0)
504 options->alproto = ALPROTO_TLS;
505 else if (strcmp(k,"protocol") == 0 && strcmp(v, "ssh") == 0)
506 options->alproto = ALPROTO_SSH;
507 else if (strcmp(k,"protocol") == 0 && strcmp(v, "smtp") == 0)
508 options->alproto = ALPROTO_SMTP;
509 else if (strcmp(k, "type") == 0 && strcmp(v, "packet") == 0)
510 options->packet = 1;
511 else if (strcmp(k, "filter") == 0 && strcmp(v, "alerts") == 0)
512 options->alerts = 1;
513 else if (strcmp(k, "type") == 0 && strcmp(v, "file") == 0)
514 options->file = 1;
515 else if (strcmp(k, "type") == 0 && strcmp(v, "streaming") == 0)
516 options->streaming = 1;
517 else if (strcmp(k, "type") == 0 && strcmp(v, "flow") == 0)
518 options->flow = 1;
519 else if (strcmp(k, "filter") == 0 && strcmp(v, "tcp") == 0)
520 options->tcp_data = 1;
521 else if (strcmp(k, "type") == 0 && strcmp(v, "stats") == 0)
522 options->stats = 1;
523 else
524 SCLogInfo("unknown key and/or value: k='%s', v='%s'", k, v);
525 }
526
527 if (((options->alproto != ALPROTO_UNKNOWN)) + options->packet + options->file > 1) {
528 SCLogError(SC_ERR_LUA_ERROR, "invalid combination of 'needs' in the script");
529 goto error;
530 }
531
532 lua_getglobal(luastate, "setup");
533 if (lua_type(luastate, -1) != LUA_TFUNCTION) {
534 SCLogError(SC_ERR_LUA_ERROR, "no setup function in script");
535 goto error;
536 }
537
538 lua_getglobal(luastate, "log");
539 if (lua_type(luastate, -1) != LUA_TFUNCTION) {
540 SCLogError(SC_ERR_LUA_ERROR, "no log function in script");
541 goto error;
542 }
543
544 lua_getglobal(luastate, "deinit");
545 if (lua_type(luastate, -1) != LUA_TFUNCTION) {
546 SCLogError(SC_ERR_LUA_ERROR, "no deinit function in script");
547 goto error;
548 }
549
550 LuaReturnState(luastate);
551 return 0;
552error:
553 if (luastate)
554 LuaReturnState(luastate);
555 return -1;
556}

558/** \brief setup a luastate for use at runtime
559 *
560 * This loads the script, primes it and then runs the 'setup' function.
561 *
562 * \retval state Returns the set up luastate on success, NULL on error
563 */
564static lua_State *LuaScriptSetup(const char *filename)
565{
566 lua_State *luastate = LuaGetState();
567 if (luastate == NULL) {
568 SCLogError(SC_ERR_LUA_ERROR, "luaL_newstate failed");
569 goto error;
570 }
571
572 luaL_openlibs(luastate);
573
574 int status = luaL_loadfile(luastate, filename);
575 if (status) {
576 SCLogError(SC_ERR_LUA_ERROR, "couldn't load file: %s", lua_tostring(luastate, -1));
577 goto error;
578 }
579
580 /* prime the script */
581 if (lua_pcall(luastate, 0, 0, 0) != 0) {
582 SCLogError(SC_ERR_LUA_ERROR, "couldn't prime file: %s", lua_tostring(luastate, -1));
583 goto error;
584 }
585
586 lua_getglobal(luastate, "setup");
587
588 /* register functions common to all */
589 LuaRegisterFunctions(luastate);
590 /* unconditionally register http function. They will only work
591 * if the tx is registered in the state at runtime though. */
592 LuaRegisterHttpFunctions(luastate);
593 LuaRegisterDnsFunctions(luastate);
594 LuaRegisterJa3Functions(luastate);
595 LuaRegisterTlsFunctions(luastate);
596 LuaRegisterSshFunctions(luastate);
597 LuaRegisterHasshFunctions(luastate);
598 LuaRegisterSmtpFunctions(luastate);
599
600 if (lua_pcall(luastate, 0, 0, 0) != 0) {
601 SCLogError(SC_ERR_LUA_ERROR, "couldn't run script 'setup' function: %s", lua_tostring(luastate, -1));
602 goto error;
603 }
604
605 SCLogDebug("lua_State %p is set up", luastate);
606 return luastate;
607error:
608 if (luastate)
609 LuaReturnState(luastate);
610 return NULL;
611}

619/** \brief initialize output for a script instance
620 *
621 * Runs script 'setup' function.
622 */
623static OutputInitResult OutputLuaLogInitSub(ConfNode *conf, OutputCtx *parent_ctx)
624{
625 OutputInitResult result = { NULL, false };
626 if (conf == NULL)
627 return result;
628
629 LogLuaCtx *lua_ctx = SCMalloc(sizeof(LogLuaCtx));
630 if (unlikely(lua_ctx == NULL))
631 return result;
632 memset(lua_ctx, 0x00, sizeof(*lua_ctx));
633
634 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
635 if (unlikely(output_ctx == NULL)) {
636 SCFree(lua_ctx);
637 return result;
638 }
639
640 SCMutexInit(&lua_ctx->m, NULL);
641
642 const char *dir = "";
643 if (parent_ctx && parent_ctx->data) {
644 LogLuaMasterCtx *mc = parent_ctx->data;
645 dir = mc->path;
646 }
647
648 char path[PATH_MAX] = "";
649 int ret = snprintf(path, sizeof(path),"%s%s%s", dir, strlen(dir) ? "/" : "", conf->val);
650 if (ret < 0 || ret == sizeof(path)) {
651 SCLogError(SC_ERR_SPRINTF,"failed to construct lua script path");
652 goto error;
653 }
654 SCLogDebug("script full path %s", path);
655
656 SCMutexLock(&lua_ctx->m);
657 lua_ctx->luastate = LuaScriptSetup(path);
658 SCMutexUnlock(&lua_ctx->m);
659 if (lua_ctx->luastate == NULL)
660 goto error;
661

689/** \internal
690 * \brief initialize output instance for lua module
691 *
692 * Parses nested script list, primes them to find out what they
693 * inspect, then fills the OutputCtx::submodules list with the
694 * proper Logger function for the data type the script needs.
695 */
696static OutputInitResult OutputLuaLogInit(ConfNode *conf)
697{
698 OutputInitResult result = { NULL, false };
699 const char *dir = ConfNodeLookupChildValue(conf, "scripts-dir");
700 if (dir == NULL)
701 dir = "";
702
703 ConfNode *scripts = ConfNodeLookupChild(conf, "scripts");
704 if (scripts == NULL) {
705 /* No "outputs" section in the configuration. */
706 SCLogInfo("scripts not defined");
707 return result;
708 }
709
710 /* global output ctx setup */
711 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
712 if (unlikely(output_ctx == NULL)) {
713 return result;
714 }
715 output_ctx->DeInit = LogLuaMasterFree;
716 output_ctx->data = SCCalloc(1, sizeof(LogLuaMasterCtx));
717 if (unlikely(output_ctx->data == NULL)) {
718 SCFree(output_ctx);
719 return result;
720 }
721 LogLuaMasterCtx *master_config = output_ctx->data;
722 strlcpy(master_config->path, dir, sizeof(master_config->path));
723 TAILQ_INIT(&output_ctx->submodules);
724
725 /* check the enables scripts and set them up as submodules */
726 ConfNode *script;
727 TAILQ_FOREACH(script, &scripts->head, next) {
728 SCLogInfo("enabling script %s", script->val);
729 LogLuaScriptOptions opts;
730 memset(&opts, 0x00, sizeof(opts));
731
732 char path[PATH_MAX] = "";
733 snprintf(path, sizeof(path),"%s%s%s", dir, strlen(dir) ? "/" : "", script->val);
734 SCLogDebug("script full path %s", path);
735
736 int r = LuaScriptInit(path, &opts);
737 if (r != 0) {
738 SCLogError(SC_ERR_LUA_ERROR, "couldn't initialize script");
739 goto error;
740 }
741
742 /* create an OutputModule for this script, based
743 * on it's needs. */
744 OutputModule *om = SCCalloc(1, sizeof(*om));
745 if (om == NULL) {
746 SCLogError(SC_ERR_MEM_ALLOC, "calloc() failed");
747 goto error;
748 }
749
750 om->name = MODULE_NAME;
751 om->conf_name = script->val;
752 om->InitSubFunc = OutputLuaLogInitSub;
753 om->ThreadInit = LuaLogThreadInit;
754 om->ThreadDeinit = LuaLogThreadDeinit;
755
756 if (opts.alproto == ALPROTO_HTTP1 && opts.streaming) {
757 om->StreamingLogFunc = LuaStreamingLogger;
758 om->stream_type = STREAMING_HTTP_BODIES;
759 om->alproto = ALPROTO_HTTP1;
760 AppLayerHtpEnableRequestBodyCallback();
761 AppLayerHtpEnableResponseBodyCallback();
762 } else if (opts.alproto == ALPROTO_HTTP1) {
763 om->TxLogFunc = LuaTxLogger;
764 om->alproto = ALPROTO_HTTP1;
765 om->ts_log_progress = -1;
766 om->tc_log_progress = -1;
767 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_HTTP1);
768 } else if (opts.alproto == ALPROTO_TLS) {
769 om->TxLogFunc = LuaTxLogger;
770 om->alproto = ALPROTO_TLS;
771 om->tc_log_progress = TLS_HANDSHAKE_DONE;
772 om->ts_log_progress = TLS_HANDSHAKE_DONE;
773 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_TLS);
774 } else if (opts.alproto == ALPROTO_DNS) {
775 om->TxLogFunc = LuaTxLogger;
776 om->alproto = ALPROTO_DNS;
777 om->ts_log_progress = -1;
778 om->tc_log_progress = -1;
779 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_DNS);
780 AppLayerParserRegisterLogger(IPPROTO_UDP, ALPROTO_DNS);
781 } else if (opts.alproto == ALPROTO_SSH) {
782 om->TxLogFunc = LuaTxLogger;
783 om->alproto = ALPROTO_SSH;
784 om->TxLogCondition = SSHTxLogCondition;
785 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_SSH);
786 } else if (opts.alproto == ALPROTO_SMTP) {
787 om->TxLogFunc = LuaTxLogger;
788 om->alproto = ALPROTO_SMTP;
789 om->ts_log_progress = -1;
790 om->tc_log_progress = -1;
791 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_SMTP);
792 } else if (opts.packet && opts.alerts) {
793 om->PacketLogFunc = LuaPacketLoggerAlerts;
794 om->PacketConditionFunc = LuaPacketConditionAlerts;
795 } else if (opts.packet && opts.alerts == 0) {
796 om->PacketLogFunc = LuaPacketLogger;
797 om->PacketConditionFunc = LuaPacketCondition;
798 } else if (opts.file) {
799 om->FileLogFunc = LuaFileLogger;
800 AppLayerHtpNeedFileInspection();
801 } else if (opts.streaming && opts.tcp_data) {
802 om->StreamingLogFunc = LuaStreamingLogger;
803 om->stream_type = STREAMING_TCP_DATA;
804 } else if (opts.flow) {
805 om->FlowLogFunc = LuaFlowLogger;
806 } else if (opts.stats) {
807 om->StatsLogFunc = LuaStatsLogger;
808 } else {
809 SCLogError(SC_ERR_LUA_ERROR, "failed to setup thread module");
810 SCFree(om);
811 goto error;
812 }
813
814 TAILQ_INSERT_TAIL(&output_ctx->submodules, om, entries);
815 }
816
817 result.ctx = output_ctx;
818 result.ok = true;
819 return result;
820
821error:
822 if (output_ctx->DeInit)
823 output_ctx->DeInit(output_ctx);
824
825 int failure_fatal = 0;
826 if (ConfGetBool("engine.init-failure-fatal", &failure_fatal) != 1) {
827 SCLogDebug("ConfGetBool could not load the value.");
828 }
829 if (failure_fatal) {
830 FatalError(SC_ERR_FATAL,
831 "Error during setup of lua output. Details should be "
832 "described in previous error messages. Shutting down...");
833 }
834
835 return result;
836}

884/** \internal
885 * \brief Deinit the thread storage for lua
886 *
887 * Calls OutputLuaLogDoDeinit if no-one else already did.
888 */
889static TmEcode LuaLogThreadDeinit(ThreadVars *t, void *data)
890{
891 LogLuaThreadCtx *td = (LogLuaThreadCtx *)data;
892 if (td == NULL) {
893 return TM_ECODE_OK;
894 }

899 td->lua_ctx->deinit_once = 1;
900 }
901 SCMutexUnlock(&td->lua_ctx->m);
902
903 /* clear memory */
904 memset(td, 0, sizeof(*td));
905
906 SCFree(td);
907 return TM_ECODE_OK;
908}
