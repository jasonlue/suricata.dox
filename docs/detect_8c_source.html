99/** \internal
100 */
101static void DetectRun(ThreadVars *th_v,
102 DetectEngineCtx *de_ctx, DetectEngineThreadCtx *det_ctx,
103 Packet *p)
104{
105 SCEnter();
106 SCLogDebug("p->pcap_cnt %" PRIu64 " direction %s flow %p", p->pcap_cnt,
107 p->flow ? (FlowGetPacketDirection(p->flow, p) == TOSERVER ? "toserver" : "toclient")
108 : "noflow",
109 p->flow);
110
111 /* bail early if packet should not be inspected */
112 if (p->flags & PKT_NOPACKET_INSPECTION) {
113 /* nothing to do */
114 SCReturn;
115 }
116
117 /* Load the Packet's flow early, even though it might not be needed.
118 * Mark as a constant pointer, although the flow itself can change. */
119 Flow * const pflow = p->flow;
120
121 DetectRunScratchpad scratch = DetectRunSetup(de_ctx, det_ctx, p, pflow);
122
123 /* run the IPonly engine */
124 DetectRunInspectIPOnly(th_v, de_ctx, det_ctx, pflow, p);
125
126 /* get our rule group */
127 DetectRunGetRuleGroup(de_ctx, p, pflow, &scratch);
128 /* if we didn't get a sig group head, we
129 * have nothing to do.... */
130 if (scratch.sgh == NULL) {
131 SCLogDebug("no sgh for this packet, nothing to match against");
132 goto end;
133 }
134
135 /* run the prefilters for packets */
136 DetectRunPrefilterPkt(th_v, de_ctx, det_ctx, p, &scratch);
137
138 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_RULES);
139 /* inspect the rules against the packet */
140 DetectRulePacketRules(th_v, de_ctx, det_ctx, p, pflow, &scratch);
141 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_RULES);
142
143 /* run tx/state inspection. Don't call for ICMP error msgs. */
144 if (pflow && pflow->alstate && likely(pflow->proto == p->proto)) {
145 if (p->proto == IPPROTO_TCP) {
146 const TcpSession *ssn = p->flow->protoctx;
147 if (ssn && (ssn->flags & STREAMTCP_FLAG_APP_LAYER_DISABLED) == 0) {
148 // PACKET_PROFILING_DETECT_START(p, PROF_DETECT_TX);
149 DetectRunFrames(th_v, de_ctx, det_ctx, p, pflow, &scratch);
150 // PACKET_PROFILING_DETECT_END(p, PROF_DETECT_TX);
151 }
152 } else if (p->proto == IPPROTO_UDP) {
153 DetectRunFrames(th_v, de_ctx, det_ctx, p, pflow, &scratch);
154 }
155
156 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_TX);
157 DetectRunTx(th_v, de_ctx, det_ctx, p, pflow, &scratch);
158 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_TX);
159 }
160
161end:
162 DetectRunPostRules(th_v, de_ctx, det_ctx, p, pflow, &scratch);
163
164 DetectRunCleanup(det_ctx, p, pflow);
165 SCReturn;
166}

168static void DetectRunPostMatch(ThreadVars *tv,
169 DetectEngineThreadCtx *det_ctx, Packet *p,
170 const Signature *s)
171{
172 /* run the packet match functions */
173 const SigMatchData *smd = s->sm_arrays[DETECT_SM_LIST_POSTMATCH];
174 if (smd != NULL) {
175 KEYWORD_PROFILING_SET_LIST(det_ctx, DETECT_SM_LIST_POSTMATCH);
176
177 SCLogDebug("running match functions, sm %p", smd);
178
179 while (1) {
180 KEYWORD_PROFILING_START;
181 (void)sigmatch_table[smd->type].Match(det_ctx, p, s, smd->ctx);
182 KEYWORD_PROFILING_END(det_ctx, smd->type, 1);
183 if (smd->is_last)
184 break;
185 smd++;
186 }
187 }
188}

190/**
191 * \brief Get the SigGroupHead for a packet.
192 *
193 * \param de_ctx detection engine context
194 * \param p packet
195 *
196 * \retval sgh the SigGroupHead or NULL if non applies to the packet
197 */
198const SigGroupHead *SigMatchSignaturesGetSgh(const DetectEngineCtx *de_ctx,
199 const Packet *p)
200{
201 SCEnter();
202
203 int f;
204 SigGroupHead *sgh = NULL;
205
206 /* if the packet proto is 0 (not set), we're inspecting it against
207 * the decoder events sgh we have. */
208 if (p->proto == 0 && p->events.cnt > 0) {
209 SCReturnPtr(de_ctx->decoder_event_sgh, "SigGroupHead");
210 } else if (p->proto == 0) {
211 if (!(PKT_IS_IPV4(p) || PKT_IS_IPV6(p))) {
212 /* not IP, so nothing to do */
213 SCReturnPtr(NULL, "SigGroupHead");
214 }
215 }
216
217 /* select the flow_gh */
218 if (p->flowflags & FLOW_PKT_TOCLIENT)
219 f = 0;
220 else
221 f = 1;
222
223 int proto = IP_GET_IPPROTO(p);
224 if (proto == IPPROTO_TCP) {
225 DetectPort *list = de_ctx->flow_gh[f].tcp;
226 SCLogDebug("tcp toserver %p, tcp toclient %p: going to use %p",
227 de_ctx->flow_gh[1].tcp, de_ctx->flow_gh[0].tcp, de_ctx->flow_gh[f].tcp);
228 uint16_t port = f ? p->dp : p->sp;
229 SCLogDebug("tcp port %u -> %u:%u", port, p->sp, p->dp);
230 DetectPort *sghport = DetectPortLookupGroup(list, port);
231 if (sghport != NULL)
232 sgh = sghport->sh;
233 SCLogDebug("TCP list %p, port %u, direction %s, sghport %p, sgh %p",
234 list, port, f ? "toserver" : "toclient", sghport, sgh);
235 } else if (proto == IPPROTO_UDP) {
236 DetectPort *list = de_ctx->flow_gh[f].udp;
237 uint16_t port = f ? p->dp : p->sp;
238 DetectPort *sghport = DetectPortLookupGroup(list, port);
239 if (sghport != NULL)
240 sgh = sghport->sh;
241 SCLogDebug("UDP list %p, port %u, direction %s, sghport %p, sgh %p",
242 list, port, f ? "toserver" : "toclient", sghport, sgh);
243 } else {
244 sgh = de_ctx->flow_gh[f].sgh[proto];
245 }
246
247 SCReturnPtr(sgh, "SigGroupHead");
248}

250static inline void DetectPrefilterMergeSort(DetectEngineCtx *de_ctx,
251 DetectEngineThreadCtx *det_ctx)
252{
253 SigIntId mpm, nonmpm;
254 SigIntId *mpm_ptr = det_ctx->pmq.rule_id_array;
255 SigIntId *nonmpm_ptr = det_ctx->non_pf_id_array;
256 uint32_t m_cnt = det_ctx->pmq.rule_id_array_cnt;
257 uint32_t n_cnt = det_ctx->non_pf_id_cnt;
258 SigIntId *final_ptr;
259 uint32_t final_cnt;
260 SigIntId id;
261 SigIntId previous_id = (SigIntId)-1;
262 Signature **sig_array = de_ctx->sig_array;
263 Signature **match_array = det_ctx->match_array;
264 Signature *s;
265
266 SCLogDebug("PMQ rule id array count %d", det_ctx->pmq.rule_id_array_cnt);
267
268 /* Load first values. */
269 if (likely(m_cnt)) {
270 mpm = *mpm_ptr;
271 } else {
272 /* mpm list is empty */
273 final_ptr = nonmpm_ptr;
274 final_cnt = n_cnt;
275 goto final;
276 }
277 if (likely(n_cnt)) {
278 nonmpm = *nonmpm_ptr;
279 } else {
280 /* non-mpm list is empty. */
281 final_ptr = mpm_ptr;
282 final_cnt = m_cnt;
283 goto final;
284 }
285 while (1) {
286 if (mpm < nonmpm) {
287 /* Take from mpm list */
288 id = mpm;
289
290 s = sig_array[id];
291 /* As the mpm list can contain duplicates, check for that here. */
292 if (likely(id != previous_id)) {
293 *match_array++ = s;
294 previous_id = id;
295 }
296 if (unlikely(--m_cnt == 0)) {
297 /* mpm list is now empty */
298 final_ptr = nonmpm_ptr;
299 final_cnt = n_cnt;
300 goto final;
301 }
302 mpm_ptr++;
303 mpm = *mpm_ptr;
304 } else if (mpm > nonmpm) {
305 id = nonmpm;
306
307 s = sig_array[id];
308 /* As the mpm list can contain duplicates, check for that here. */
309 if (likely(id != previous_id)) {
310 *match_array++ = s;
311 previous_id = id;
312 }
313 if (unlikely(--n_cnt == 0)) {
314 final_ptr = mpm_ptr;
315 final_cnt = m_cnt;
316 goto final;
317 }
318 nonmpm_ptr++;
319 nonmpm = *nonmpm_ptr;
320
321 } else { /* implied mpm == nonmpm */
322 /* special case: if on both lists, it's a negated mpm pattern */
323
324 /* mpm list may have dups, so skip past them here */
325 while (--m_cnt != 0) {
326 mpm_ptr++;
327 mpm = *mpm_ptr;
328 if (mpm != nonmpm)
329 break;
330 }
331 /* if mpm is done, update nonmpm_ptrs and jump to final */
332 if (unlikely(m_cnt == 0)) {
333 n_cnt--;
334
335 /* mpm list is now empty */
336 final_ptr = ++nonmpm_ptr;
337 final_cnt = n_cnt;
338 goto final;
339 }
340 /* otherwise, if nonmpm is done jump to final for mpm
341 * mpm ptrs alrady updated */
342 if (unlikely(--n_cnt == 0)) {
343 final_ptr = mpm_ptr;
344 final_cnt = m_cnt;
345 goto final;
346 }
347
348 /* not at end of the lists, update nonmpm. Mpm already
349 * updated in while loop above. */
350 nonmpm_ptr++;
351 nonmpm = *nonmpm_ptr;
352 }
353 }
354
355 final: /* Only one list remaining. Just walk that list. */
356
357 while (final_cnt-- > 0) {
358 id = *final_ptr++;
359 s = sig_array[id];
360
361 /* As the mpm list can contain duplicates, check for that here. */
362 if (likely(id != previous_id)) {
363 *match_array++ = s;
364 previous_id = id;
365 }
366 }
367
368 det_ctx->match_array_cnt = match_array - det_ctx->match_array;
369 DEBUG_VALIDATE_BUG_ON((det_ctx->pmq.rule_id_array_cnt + det_ctx->non_pf_id_cnt) < det_ctx->match_array_cnt);
370 PMQ_RESET(&det_ctx->pmq);
371}

373/** \internal
374 * \brief build non-prefilter list based on the rule group list we've set.
375 */
376static inline void DetectPrefilterBuildNonPrefilterList(
377 DetectEngineThreadCtx *det_ctx, const SignatureMask mask, const AppProto alproto)
378{
379 for (uint32_t x = 0; x < det_ctx->non_pf_store_cnt; x++) {
380 /* only if the mask matches this rule can possibly match,
381 * so build the non_mpm array only for match candidates */
382 const SignatureMask rule_mask = det_ctx->non_pf_store_ptr[x].mask;
383 const AppProto rule_alproto = det_ctx->non_pf_store_ptr[x].alproto;
384 if ((rule_mask & mask) == rule_mask &&
385 (rule_alproto == 0 || AppProtoEquals(rule_alproto, alproto))) {
386 det_ctx->non_pf_id_array[det_ctx->non_pf_id_cnt++] = det_ctx->non_pf_store_ptr[x].id;
387 }
388 }
389}

411/** \internal
412 * \brief update flow's file tracking flags based on the detection engine
413 * A set of flags is prepared that is sent to the File API. The
414 File API may reject one or more based on the global force settings.
415 */
416static inline void
417DetectPostInspectFileFlagsUpdate(Flow *f, const SigGroupHead *sgh, uint8_t direction)
418{
419 uint16_t flow_file_flags = FLOWFILE_INIT;
420
421 if (sgh == NULL) {
422 SCLogDebug("requesting disabling all file features for flow");
423 flow_file_flags = FLOWFILE_NONE;
424 } else {
425 if (sgh->filestore_cnt == 0) {
426 SCLogDebug("requesting disabling filestore for flow");
427 flow_file_flags |= (FLOWFILE_NO_STORE_TS|FLOWFILE_NO_STORE_TC);
428 }
429#ifdef HAVE_MAGIC
430 if (!(sgh->flags & SIG_GROUP_HEAD_HAVEFILEMAGIC)) {
431 SCLogDebug("requesting disabling magic for flow");
432 flow_file_flags |= (FLOWFILE_NO_MAGIC_TS|FLOWFILE_NO_MAGIC_TC);
433 }
434#endif
435 if (!(sgh->flags & SIG_GROUP_HEAD_HAVEFILEMD5)) {
436 SCLogDebug("requesting disabling md5 for flow");
437 flow_file_flags |= (FLOWFILE_NO_MD5_TS|FLOWFILE_NO_MD5_TC);
438 }
439 if (!(sgh->flags & SIG_GROUP_HEAD_HAVEFILESHA1)) {
440 SCLogDebug("requesting disabling sha1 for flow");
441 flow_file_flags |= (FLOWFILE_NO_SHA1_TS|FLOWFILE_NO_SHA1_TC);
442 }
443 if (!(sgh->flags & SIG_GROUP_HEAD_HAVEFILESHA256)) {
444 SCLogDebug("requesting disabling sha256 for flow");
445 flow_file_flags |= (FLOWFILE_NO_SHA256_TS|FLOWFILE_NO_SHA256_TC);
446 }
447 if (!(sgh->flags & SIG_GROUP_HEAD_HAVEFILESIZE)) {
448 SCLogDebug("requesting disabling filesize for flow");
449 flow_file_flags |= (FLOWFILE_NO_SIZE_TS|FLOWFILE_NO_SIZE_TC);
450 }
451 }
452 if (flow_file_flags != 0) {
453 FileUpdateFlowFileFlags(f, flow_file_flags, direction);
454 }
455}

457static inline void
458DetectRunPostGetFirstRuleGroup(const Packet *p, Flow *pflow, const SigGroupHead *sgh)
459{
460 if ((p->flowflags & FLOW_PKT_TOSERVER) && !(pflow->flags & FLOW_SGH_TOSERVER)) {
461 /* first time we see this toserver sgh, store it */
462 pflow->sgh_toserver = sgh;
463 pflow->flags |= FLOW_SGH_TOSERVER;
464
465 if (p->proto == IPPROTO_TCP && (sgh == NULL || !(sgh->flags & SIG_GROUP_HEAD_HAVERAWSTREAM))) {
466 if (pflow->protoctx != NULL) {
467 TcpSession *ssn = pflow->protoctx;
468 SCLogDebug("STREAMTCP_STREAM_FLAG_DISABLE_RAW ssn.client");
469 ssn->client.flags |= STREAMTCP_STREAM_FLAG_DISABLE_RAW;
470 }
471 }
472
473 DetectPostInspectFileFlagsUpdate(pflow,
474 pflow->sgh_toserver, STREAM_TOSERVER);
475
476 } else if ((p->flowflags & FLOW_PKT_TOCLIENT) && !(pflow->flags & FLOW_SGH_TOCLIENT)) {
477 pflow->sgh_toclient = sgh;
478 pflow->flags |= FLOW_SGH_TOCLIENT;
479
480 if (p->proto == IPPROTO_TCP && (sgh == NULL || !(sgh->flags & SIG_GROUP_HEAD_HAVERAWSTREAM))) {
481 if (pflow->protoctx != NULL) {
482 TcpSession *ssn = pflow->protoctx;
483 SCLogDebug("STREAMTCP_STREAM_FLAG_DISABLE_RAW ssn.server");
484 ssn->server.flags |= STREAMTCP_STREAM_FLAG_DISABLE_RAW;
485 }
486 }
487
488 DetectPostInspectFileFlagsUpdate(pflow,
489 pflow->sgh_toclient, STREAM_TOCLIENT);
490 }
491}

493static inline void DetectRunGetRuleGroup(
494 const DetectEngineCtx *de_ctx,
495 Packet * const p, Flow * const pflow,
496 DetectRunScratchpad *scratch)
497{
498 const SigGroupHead *sgh = NULL;
499
500 if (pflow) {
501 bool use_flow_sgh = false;
502 /* Get the stored sgh from the flow (if any). Make sure we're not using
503 * the sgh for icmp error packets part of the same stream. */
504 if (IP_GET_IPPROTO(p) == pflow->proto) { /* filter out icmp */
505 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_GETSGH);
506 if ((p->flowflags & FLOW_PKT_TOSERVER) && (pflow->flags & FLOW_SGH_TOSERVER)) {
507 sgh = pflow->sgh_toserver;
508 SCLogDebug("sgh = pflow->sgh_toserver; => %p", sgh);
509 use_flow_sgh = true;
510 } else if ((p->flowflags & FLOW_PKT_TOCLIENT) && (pflow->flags & FLOW_SGH_TOCLIENT)) {
511 sgh = pflow->sgh_toclient;
512 SCLogDebug("sgh = pflow->sgh_toclient; => %p", sgh);
513 use_flow_sgh = true;
514 }
515 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_GETSGH);
516 }
517
518 if (!(use_flow_sgh)) {
519 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_GETSGH);
520 sgh = SigMatchSignaturesGetSgh(de_ctx, p);
521 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_GETSGH);
522
523 /* HACK: prevent the wrong sgh (or NULL) from being stored in the
524 * flow's sgh pointers */
525 if (PKT_IS_ICMPV4(p) && ICMPV4_DEST_UNREACH_IS_VALID(p)) {
526 ; /* no-op */
527 } else {
528 /* store the found sgh (or NULL) in the flow to save us
529 * from looking it up again for the next packet.
530 * Also run other tasks */
531 DetectRunPostGetFirstRuleGroup(p, pflow, sgh);
532 }
533 }
534 } else { /* p->flags & PKT_HAS_FLOW */
535 /* no flow */
536
537 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_GETSGH);
538 sgh = SigMatchSignaturesGetSgh(de_ctx, p);
539 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_GETSGH);
540 }
541
542 scratch->sgh = sgh;
543}

545static void DetectRunInspectIPOnly(ThreadVars *tv, const DetectEngineCtx *de_ctx,
546 DetectEngineThreadCtx *det_ctx,
547 Flow * const pflow, Packet * const p)
548{
549 if (pflow) {
550 /* set the iponly stuff */
551 if (pflow->flags & FLOW_TOCLIENT_IPONLY_SET)
552 p->flowflags |= FLOW_PKT_TOCLIENT_IPONLY_SET;
553 if (pflow->flags & FLOW_TOSERVER_IPONLY_SET)
554 p->flowflags |= FLOW_PKT_TOSERVER_IPONLY_SET;
555
556 if (((p->flowflags & FLOW_PKT_TOSERVER) && !(p->flowflags & FLOW_PKT_TOSERVER_IPONLY_SET)) ||
557 ((p->flowflags & FLOW_PKT_TOCLIENT) && !(p->flowflags & FLOW_PKT_TOCLIENT_IPONLY_SET)))
558 {
559 SCLogDebug("testing against \"ip-only\" signatures");
560
561 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_IPONLY);
562 IPOnlyMatchPacket(tv, de_ctx, det_ctx, &de_ctx->io_ctx, &det_ctx->io_ctx, p);
563 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_IPONLY);
564
565 /* save in the flow that we scanned this direction... */
566 FlowSetIPOnlyFlag(pflow, p->flowflags & FLOW_PKT_TOSERVER ? 1 : 0);
567 }
568 } else { /* p->flags & PKT_HAS_FLOW */
569 /* no flow */
570
571 /* Even without flow we should match the packet src/dst */
572 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_IPONLY);
573 IPOnlyMatchPacket(tv, de_ctx, det_ctx, &de_ctx->io_ctx,
574 &det_ctx->io_ctx, p);
575 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_IPONLY);
576 }
577}

579/* returns 0 if no match, 1 if match */
580static inline int DetectRunInspectRuleHeader(
581 const Packet *p,
582 const Flow *f,
583 const Signature *s,
584 const uint32_t sflags,
585 const uint8_t s_proto_flags)
586{
587 /* check if this signature has a requirement for flowvars of some type
588 * and if so, if we actually have any in the flow. If not, the sig
589 * can't match and we skip it. */
590 if ((p->flags & PKT_HAS_FLOW) && (sflags & SIG_FLAG_REQUIRE_FLOWVAR)) {
591 DEBUG_VALIDATE_BUG_ON(f == NULL);
592
593 int m = f->flowvar ? 1 : 0;
594
595 /* no flowvars? skip this sig */
596 if (m == 0) {
597 SCLogDebug("skipping sig as the flow has no flowvars and sig "
598 "has SIG_FLAG_REQUIRE_FLOWVAR flag set.");
599 return 0;
600 }
601 }
602
603 if ((s_proto_flags & DETECT_PROTO_IPV4) && !PKT_IS_IPV4(p)) {
604 SCLogDebug("ip version didn't match");
605 return 0;
606 }
607 if ((s_proto_flags & DETECT_PROTO_IPV6) && !PKT_IS_IPV6(p)) {
608 SCLogDebug("ip version didn't match");
609 return 0;
610 }
611
612 if (DetectProtoContainsProto(&s->proto, IP_GET_IPPROTO(p)) == 0) {
613 SCLogDebug("proto didn't match");
614 return 0;
615 }
616
617 /* check the source & dst port in the sig */
618 if (p->proto == IPPROTO_TCP || p->proto == IPPROTO_UDP || p->proto == IPPROTO_SCTP) {
619 if (!(sflags & SIG_FLAG_DP_ANY)) {
620 if (p->flags & PKT_IS_FRAGMENT)
621 return 0;
622 DetectPort *dport = DetectPortLookupGroup(s->dp,p->dp);
623 if (dport == NULL) {
624 SCLogDebug("dport didn't match.");
625 return 0;
626 }
627 }
628 if (!(sflags & SIG_FLAG_SP_ANY)) {
629 if (p->flags & PKT_IS_FRAGMENT)
630 return 0;
631 DetectPort *sport = DetectPortLookupGroup(s->sp,p->sp);
632 if (sport == NULL) {
633 SCLogDebug("sport didn't match.");
634 return 0;
635 }
636 }
637 } else if ((sflags & (SIG_FLAG_DP_ANY|SIG_FLAG_SP_ANY)) != (SIG_FLAG_DP_ANY|SIG_FLAG_SP_ANY)) {
638 SCLogDebug("port-less protocol and sig needs ports");
639 return 0;
640 }
641
642 /* check the destination address */
643 if (!(sflags & SIG_FLAG_DST_ANY)) {
644 if (PKT_IS_IPV4(p)) {
645 if (DetectAddressMatchIPv4(s->addr_dst_match4, s->addr_dst_match4_cnt, &p->dst) == 0)
646 return 0;
647 } else if (PKT_IS_IPV6(p)) {
648 if (DetectAddressMatchIPv6(s->addr_dst_match6, s->addr_dst_match6_cnt, &p->dst) == 0)
649 return 0;
650 }
651 }
652 /* check the source address */
653 if (!(sflags & SIG_FLAG_SRC_ANY)) {
654 if (PKT_IS_IPV4(p)) {
655 if (DetectAddressMatchIPv4(s->addr_src_match4, s->addr_src_match4_cnt, &p->src) == 0)
656 return 0;
657 } else if (PKT_IS_IPV6(p)) {
658 if (DetectAddressMatchIPv6(s->addr_src_match6, s->addr_src_match6_cnt, &p->src) == 0)
659 return 0;
660 }
661 }
662
663 return 1;
664}

715static inline void DetectRulePacketRules(
716 ThreadVars * const tv,
717 DetectEngineCtx * const de_ctx,
718 DetectEngineThreadCtx * const det_ctx,
719 Packet * const p,
720 Flow * const pflow,
721 const DetectRunScratchpad *scratch
722)
723{
724 const Signature *s = NULL;
725 const Signature *next_s = NULL;
726
727 /* inspect the sigs against the packet */
728 /* Prefetch the next signature. */
729 SigIntId match_cnt = det_ctx->match_array_cnt;
730#ifdef PROFILING
731 if (tv) {
732 StatsAddUI64(tv, det_ctx->counter_match_list,
733 (uint64_t)match_cnt);
734 }
735#endif
736 Signature **match_array = det_ctx->match_array;
737
738 SGH_PROFILING_RECORD(det_ctx, scratch->sgh);
739#ifdef PROFILING
740 if (match_cnt >= de_ctx->profile_match_logging_threshold)
741 RulesDumpMatchArray(det_ctx, scratch->sgh, p);
742#endif
743
744 uint32_t sflags, next_sflags = 0;
745 if (match_cnt) {
746 next_s = *match_array++;
747 next_sflags = next_s->flags;
748 }
749 while (match_cnt--) {
750 RULE_PROFILING_START(p);
751 uint8_t alert_flags = 0;
752#ifdef PROFILING
753 bool smatch = false; /* signature match */
754#endif
755 s = next_s;
756 sflags = next_sflags;
757 if (match_cnt) {
758 next_s = *match_array++;
759 next_sflags = next_s->flags;
760 }
761 const uint8_t s_proto_flags = s->proto.flags;
762
763 SCLogDebug("inspecting signature id %"PRIu32"", s->id);
764
765 if (s->app_inspect != NULL) {
766 goto next; // handle sig in DetectRunTx
767 }
768 if (s->frame_inspect != NULL) {
769 goto next; // handle sig in DetectRunFrame
770 }
771
772 /* don't run mask check for stateful rules.
773 * There we depend on prefilter */
774 if ((s->mask & scratch->pkt_mask) != s->mask) {
775 SCLogDebug("mask mismatch %x & %x != %x", s->mask, scratch->pkt_mask, s->mask);
776 goto next;
777 }
778
779 if (SigDsizePrefilter(p, s, sflags))
780 goto next;
781
782 /* if the sig has alproto and the session as well they should match */
783 if (likely(sflags & SIG_FLAG_APPLAYER)) {
784 if (s->alproto != ALPROTO_UNKNOWN && !AppProtoEquals(s->alproto, scratch->alproto)) {
785 SCLogDebug("alproto mismatch");
786 goto next;
787 }
788 }
789
790 if (DetectRunInspectRuleHeader(p, pflow, s, sflags, s_proto_flags) == 0) {
791 goto next;
792 }
793
794 if (DetectEnginePktInspectionRun(tv, det_ctx, s, pflow, p, &alert_flags) == false) {
795 goto next;
796 }
797
798#ifdef PROFILING
799 smatch = true;
800#endif
801 DetectRunPostMatch(tv, det_ctx, p, s);
802
803 AlertQueueAppend(det_ctx, s, p, 0, alert_flags);
804next:
805 DetectVarProcessList(det_ctx, pflow, p);
806 DetectReplaceFree(det_ctx);
807 RULE_PROFILING_END(det_ctx, s, smatch, p);
808
809 det_ctx->flags = 0;
810 continue;
811 }
812}

814static DetectRunScratchpad DetectRunSetup(
815 const DetectEngineCtx *de_ctx,
816 DetectEngineThreadCtx *det_ctx,
817 Packet * const p, Flow * const pflow)
818{
819 AppProto alproto = ALPROTO_UNKNOWN;
820 uint8_t flow_flags = 0; /* flow/state flags */
821 bool app_decoder_events = false;
822
823 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_SETUP);
824
825#ifdef UNITTESTS
826 p->alerts.cnt = 0;
827 p->alerts.discarded = 0;
828 p->alerts.suppressed = 0;
829#endif
830 det_ctx->filestore_cnt = 0;
831 det_ctx->base64_decoded_len = 0;
832 det_ctx->raw_stream_progress = 0;
833 det_ctx->match_array_cnt = 0;
834
835 det_ctx->alert_queue_size = 0;
836 p->alerts.drop.action = 0;
837
838#ifdef DEBUG
839 if (p->flags & PKT_STREAM_ADD) {
840 det_ctx->pkt_stream_add_cnt++;
841 }
842#endif
843
844 /* grab the protocol state we will detect on */
845 if (p->flags & PKT_HAS_FLOW) {
846 DEBUG_VALIDATE_BUG_ON(pflow == NULL);
847
848 if (p->flowflags & FLOW_PKT_TOSERVER) {
849 flow_flags = STREAM_TOSERVER;
850 SCLogDebug("flag STREAM_TOSERVER set");
851 } else if (p->flowflags & FLOW_PKT_TOCLIENT) {
852 flow_flags = STREAM_TOCLIENT;
853 SCLogDebug("flag STREAM_TOCLIENT set");
854 }
855 SCLogDebug("p->flowflags 0x%02x", p->flowflags);
856
857 if (p->flags & PKT_STREAM_EOF) {
858 flow_flags |= STREAM_EOF;
859 SCLogDebug("STREAM_EOF set");
860 }
861
862 /* store tenant_id in the flow so that we can use it
863 * for creating pseudo packets */
864 if (p->tenant_id > 0 && pflow->tenant_id == 0) {
865 pflow->tenant_id = p->tenant_id;
866 }
867
868 /* live ruleswap check for flow updates */
869 if (pflow->de_ctx_version == 0) {
870 /* first time this flow is inspected, set id */
871 pflow->de_ctx_version = de_ctx->version;
872 } else if (pflow->de_ctx_version != de_ctx->version) {
873 /* first time we inspect flow with this de_ctx, reset */
874 pflow->flags &= ~FLOW_SGH_TOSERVER;
875 pflow->flags &= ~FLOW_SGH_TOCLIENT;
876 pflow->sgh_toserver = NULL;
877 pflow->sgh_toclient = NULL;
878
879 pflow->de_ctx_version = de_ctx->version;
880 GenericVarFree(pflow->flowvar);
881 pflow->flowvar = NULL;
882
883 DetectEngineStateResetTxs(pflow);
884 }
885
886 /* Retrieve the app layer state and protocol and the tcp reassembled
887 * stream chunks. */
888 if ((p->proto == IPPROTO_TCP && (p->flags & PKT_STREAM_EST)) ||
889 (p->proto == IPPROTO_UDP) ||
890 (p->proto == IPPROTO_SCTP && (p->flowflags & FLOW_PKT_ESTABLISHED)))
891 {
892 /* update flow flags with knowledge on disruptions */
893 flow_flags = FlowGetDisruptionFlags(pflow, flow_flags);
894 alproto = FlowGetAppProtocol(pflow);
895 if (p->proto == IPPROTO_TCP && pflow->protoctx &&
896 StreamReassembleRawHasDataReady(pflow->protoctx, p)) {
897 p->flags |= PKT_DETECT_HAS_STREAMDATA;
898 flow_flags |= STREAM_FLUSH;
899 }
900 SCLogDebug("alproto %u", alproto);
901 } else {
902 SCLogDebug("packet doesn't have established flag set (proto %d)", p->proto);
903 }
904
905 app_decoder_events = AppLayerParserHasDecoderEvents(pflow->alparser);
906 }
907
908 DetectRunScratchpad pad = { alproto, flow_flags, app_decoder_events, NULL, 0 };
909 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_SETUP);
910 return pad;
911}

1050 * If stored_flags is set it means we're continueing
1051 * inspection from an earlier run.
1052 *
1053 * \retval bool true sig matched, false didn't match
1054 */
1055static bool DetectRunTxInspectRule(ThreadVars *tv,
1056 DetectEngineCtx *de_ctx,
1057 DetectEngineThreadCtx *det_ctx,
1058 Packet *p,
1059 Flow *f,
1060 const uint8_t in_flow_flags, // direction, EOF, etc
1061 void *alstate,
1062 DetectTransaction *tx,
1063 const Signature *s,
1064 uint32_t *stored_flags,
1065 RuleMatchCandidateTx *can,
1066 DetectRunScratchpad *scratch)
1067{
1068 uint8_t flow_flags = in_flow_flags;
1069 const int direction = (flow_flags & STREAM_TOSERVER) ? 0 : 1;
1070 uint32_t inspect_flags = stored_flags ? *stored_flags : 0;
1071 int total_matches = 0;
1072 uint16_t file_no_match = 0;
1073 bool retval = false;
1074 bool mpm_before_progress = false; // is mpm engine before progress?
1075 bool mpm_in_progress = false; // is mpm engine in a buffer we will revisit?
1076
1077 /* see if we want to pass on the FLUSH flag */
1078 if ((s->flags & SIG_FLAG_FLUSH) == 0)
1079 flow_flags &=~ STREAM_FLUSH;
1080
1081 TRACE_SID_TXS(s->id, tx, "starting %s", direction ? "toclient" : "toserver");
1082 TRACE_SID_TXS(s->id, tx, "FLUSH? %s", (flow_flags & STREAM_FLUSH)?"true":"false");
1083
1084 /* for a new inspection we inspect pkt header and packet matches */
1085 if (likely(stored_flags == NULL)) {
1086 TRACE_SID_TXS(s->id, tx, "first inspect, run packet matches");
1087 if (DetectRunInspectRuleHeader(p, f, s, s->flags, s->proto.flags) == 0) {
1088 TRACE_SID_TXS(s->id, tx, "DetectRunInspectRuleHeader() no match");
1089 return false;
1090 }
1091 if (DetectEnginePktInspectionRun(tv, det_ctx, s, f, p, NULL) == false) {
1092 TRACE_SID_TXS(s->id, tx, "DetectEnginePktInspectionRun no match");
1093 return false;
1094 }
1095 /* stream mpm and negated mpm sigs can end up here with wrong proto */
1096 if (!(AppProtoEquals(s->alproto, f->alproto) || s->alproto == ALPROTO_UNKNOWN)) {
1097 TRACE_SID_TXS(s->id, tx, "alproto mismatch");
1098 return false;
1099 }
1100 }
1101
1102 const DetectEngineAppInspectionEngine *engine = s->app_inspect;
1103 while (engine != NULL) { // TODO could be do {} while as s->app_inspect cannot be null
1104 TRACE_SID_TXS(s->id, tx, "engine %p inspect_flags %x", engine, inspect_flags);
1105 if (!(inspect_flags & BIT_U32(engine->id)) &&
1106 direction == engine->dir)
1107 {
1108 const bool skip_engine = (engine->alproto != 0 && engine->alproto != f->alproto);
1109 /* special case: file_data on 'alert tcp' will have engines
1110 * in the list that are not for us. */
1111 if (unlikely(skip_engine)) {
1112 engine = engine->next;
1113 continue;
1114 }

1116 /* engines are sorted per progress, except that the one with
1117 * mpm/prefilter enabled is first */
1118 if (tx->tx_progress < engine->progress) {
1119 SCLogDebug("tx progress %d < engine progress %d",
1120 tx->tx_progress, engine->progress);
1121 break;
1122 }
1123 if (engine->mpm) {
1124 if (tx->tx_progress > engine->progress) {
1125 TRACE_SID_TXS(s->id, tx,
1126 "engine->mpm: t->tx_progress %u > engine->progress %u, so set "
1127 "mpm_before_progress",
1128 tx->tx_progress, engine->progress);
1129 mpm_before_progress = true;
1130 } else if (tx->tx_progress == engine->progress) {
1131 TRACE_SID_TXS(s->id, tx,
1132 "engine->mpm: t->tx_progress %u == engine->progress %u, so set "
1133 "mpm_in_progress",
1134 tx->tx_progress, engine->progress);
1135 mpm_in_progress = true;
1136 }
1137 }
1138
1139 /* run callback: but bypass stream callback if we can */
1140 uint8_t match;
1141 if (unlikely(engine->stream && can->stream_stored)) {
1142 match = can->stream_result;
1143 TRACE_SID_TXS(s->id, tx, "stream skipped, stored result %d used instead", match);
1144 } else {
1145 KEYWORD_PROFILING_SET_LIST(det_ctx, engine->sm_list);
1146 DEBUG_VALIDATE_BUG_ON(engine->v2.Callback == NULL);
1147 match = engine->v2.Callback(
1148 de_ctx, det_ctx, engine, s, f, flow_flags, alstate, tx->tx_ptr, tx->tx_id);
1149 TRACE_SID_TXS(s->id, tx, "engine %p match %d", engine, match);
1150 if (engine->stream) {
1151 can->stream_stored = true;
1152 can->stream_result = match;
1153 TRACE_SID_TXS(s->id, tx, "stream ran, store result %d for next tx (if any)", match);
1154 }
1155 }
1156 if (match == DETECT_ENGINE_INSPECT_SIG_MATCH) {
1157 inspect_flags |= BIT_U32(engine->id);
1158 engine = engine->next;
1159 total_matches++;
1160 continue;
1161 } else if (match == DETECT_ENGINE_INSPECT_SIG_MATCH_MORE_FILES) {
1162 /* if the file engine matched, but indicated more
1163 * files are still in progress, we don't set inspect
1164 * flags as these would end inspection for this tx */
1165 engine = engine->next;
1166 total_matches++;
1167 continue;
1168 } else if (match == DETECT_ENGINE_INSPECT_SIG_CANT_MATCH) {
1169 inspect_flags |= DE_STATE_FLAG_SIG_CANT_MATCH;
1170 inspect_flags |= BIT_U32(engine->id);
1171 } else if (match == DETECT_ENGINE_INSPECT_SIG_CANT_MATCH_FILES) {
1172 inspect_flags |= DE_STATE_FLAG_SIG_CANT_MATCH;
1173 inspect_flags |= BIT_U32(engine->id);
1174 file_no_match = 1;
1175 }
1176 /* implied DETECT_ENGINE_INSPECT_SIG_NO_MATCH */
1177 if (engine->mpm && mpm_before_progress) {
1178 inspect_flags |= DE_STATE_FLAG_SIG_CANT_MATCH;
1179 inspect_flags |= BIT_U32(engine->id);
1180 }
1181 break;
1182 }
1183 engine = engine->next;
1184 }
1185 TRACE_SID_TXS(s->id, tx, "inspect_flags %x, total_matches %u, engine %p",
1186 inspect_flags, total_matches, engine);
1187
1188 if (engine == NULL && total_matches) {
1189 inspect_flags |= DE_STATE_FLAG_FULL_INSPECT;
1190 TRACE_SID_TXS(s->id, tx, "MATCH");
1191 retval = true;
1192 }
1193
1194 if (stored_flags) {
1195 *stored_flags = inspect_flags;
1196 TRACE_SID_TXS(s->id, tx, "continue inspect flags %08x", inspect_flags);
1197 } else {
1198 // store... or? If tx is done we might not want to come back to this tx
1199
1200 // also... if mpmid tracking is enabled, we won't do a sig again for this tx...
1201 TRACE_SID_TXS(s->id, tx, "start inspect flags %08x", inspect_flags);
1202 if (inspect_flags & DE_STATE_FLAG_SIG_CANT_MATCH) {
1203 if (file_no_match) {
1204 /* if we have a mismatch on a file sig, we need to keep state.
1205 * We may get another file on the same tx (for http and smtp
1206 * at least), so for a new file we need to re-eval the sig.
1207 * Thoughts / TODO:
1208 * - not for some protos that have 1 file per tx (e.g. nfs)
1209 * - maybe we only need this for file sigs that mix with
1210 * other matches? E.g. 'POST + filename', is different than
1211 * just 'filename'.
1212 */
1213 DetectRunStoreStateTx(scratch->sgh, f, tx->tx_ptr, tx->tx_id, s,
1214 inspect_flags, flow_flags, file_no_match);
1215 }
1216 } else if ((inspect_flags & DE_STATE_FLAG_FULL_INSPECT) && mpm_before_progress) {
1217 TRACE_SID_TXS(s->id, tx, "no need to store match sig, "
1218 "mpm won't trigger for it anymore");
1219
1220 if (inspect_flags & DE_STATE_FLAG_FILE_INSPECT) {
1221 TRACE_SID_TXS(s->id, tx, "except that for new files, "
1222 "we may have to revisit anyway");
1223 DetectRunStoreStateTx(scratch->sgh, f, tx->tx_ptr, tx->tx_id, s,
1224 inspect_flags, flow_flags, file_no_match);
1225 }
1226 } else if ((inspect_flags & DE_STATE_FLAG_FULL_INSPECT) == 0 && mpm_in_progress) {
1227 TRACE_SID_TXS(s->id, tx, "no need to store no-match sig, "
1228 "mpm will revisit it");
1229 } else {
1230 TRACE_SID_TXS(s->id, tx, "storing state: flags %08x", inspect_flags);
1231 DetectRunStoreStateTx(scratch->sgh, f, tx->tx_ptr, tx->tx_id, s,
1232 inspect_flags, flow_flags, file_no_match);
1233 }
1234 }
1235
1236 return retval;
1237}

1296static void DetectRunTx(ThreadVars *tv,
1297 DetectEngineCtx *de_ctx,
1298 DetectEngineThreadCtx *det_ctx,
1299 Packet *p,
1300 Flow *f,
1301 DetectRunScratchpad *scratch)
1302{
1303 const uint8_t flow_flags = scratch->flow_flags;
1304 const SigGroupHead * const sgh = scratch->sgh;
1305 void * const alstate = f->alstate;
1306 const uint8_t ipproto = f->proto;
1307 const AppProto alproto = f->alproto;
1308
1309 const uint64_t total_txs = AppLayerParserGetTxCnt(f, alstate);
1310 uint64_t tx_id_min = AppLayerParserGetTransactionInspectId(f->alparser, flow_flags);
1311 const int tx_end_state = AppLayerParserGetStateProgressCompletionStatus(alproto, flow_flags);
1312
1313 AppLayerGetTxIteratorFunc IterFunc = AppLayerGetTxIterator(ipproto, alproto);
1314 AppLayerGetTxIterState state;
1315 memset(&state, 0, sizeof(state));
1316
1317 while (1) {
1318 AppLayerGetTxIterTuple ires = IterFunc(ipproto, alproto, alstate, tx_id_min, total_txs, &state);
1319 if (ires.tx_ptr == NULL)
1320 break;
1321
1322 DetectTransaction tx = GetDetectTx(ipproto, alproto,
1323 alstate, ires.tx_id, ires.tx_ptr, tx_end_state, flow_flags);
1324 if (tx.tx_ptr == NULL) {
1325 SCLogDebug("%p/%"PRIu64" no transaction to inspect",
1326 tx.tx_ptr, tx_id_min);
1327
1328 tx_id_min++; // next (if any) run look for +1
1329 goto next;
1330 }
1331 tx_id_min = tx.tx_id + 1; // next look for cur + 1
1332
1333 bool do_sort = false; // do we need to sort the tx candidate list?
1334 uint32_t array_idx = 0;
1335 uint32_t total_rules = det_ctx->match_array_cnt;
1336 total_rules += (tx.de_state ? tx.de_state->cnt : 0);
1337
1338 /* run prefilter engines and merge results into a candidates array */
1339 if (sgh->tx_engines) {
1340 PACKET_PROFILING_DETECT_START(p, PROF_DETECT_PF_TX);
1341 DetectRunPrefilterTx(det_ctx, sgh, p, ipproto, flow_flags, alproto,
1342 alstate, &tx);
1343 PACKET_PROFILING_DETECT_END(p, PROF_DETECT_PF_TX);
1344 SCLogDebug("%p/%"PRIu64" rules added from prefilter: %u candidates",
1345 tx.tx_ptr, tx.tx_id, det_ctx->pmq.rule_id_array_cnt);
1346
1347 total_rules += det_ctx->pmq.rule_id_array_cnt;
1348 if (!(RuleMatchCandidateTxArrayHasSpace(det_ctx, total_rules))) {
1349 RuleMatchCandidateTxArrayExpand(det_ctx, total_rules);
1350 }
1351
1352 for (uint32_t i = 0; i < det_ctx->pmq.rule_id_array_cnt; i++) {
1353 const Signature *s = de_ctx->sig_array[det_ctx->pmq.rule_id_array[i]];
1354 const SigIntId id = s->num;
1355 det_ctx->tx_candidates[array_idx].s = s;
1356 det_ctx->tx_candidates[array_idx].id = id;
1357 det_ctx->tx_candidates[array_idx].flags = NULL;
1358 det_ctx->tx_candidates[array_idx].stream_reset = 0;
1359 array_idx++;
1360 }
1361 PMQ_RESET(&det_ctx->pmq);
1362 } else {
1363 if (!(RuleMatchCandidateTxArrayHasSpace(det_ctx, total_rules))) {
1364 RuleMatchCandidateTxArrayExpand(det_ctx, total_rules);
1365 }
1366 }
1367
1368 /* merge 'state' rules from the regular prefilter */
1369 uint32_t x = array_idx;
1370 for (uint32_t i = 0; i < det_ctx->match_array_cnt; i++) {
1371 const Signature *s = det_ctx->match_array[i];
1372 if (s->app_inspect != NULL) {
1373 const SigIntId id = s->num;
1374 det_ctx->tx_candidates[array_idx].s = s;
1375 det_ctx->tx_candidates[array_idx].id = id;
1376 det_ctx->tx_candidates[array_idx].flags = NULL;
1377 det_ctx->tx_candidates[array_idx].stream_reset = 0;
1378 array_idx++;
1379
1380 SCLogDebug("%p/%"PRIu64" rule %u (%u) added from 'match' list",
1381 tx.tx_ptr, tx.tx_id, s->id, id);
1382 }
1383 }
1384 do_sort = (array_idx > x); // sort if match added anything
1385 SCLogDebug("%p/%" PRIu64 " rules added from 'match' list: %u", tx.tx_ptr, tx.tx_id,
1386 array_idx - x);
1387
1388 /* merge stored state into results */
1389 if (tx.de_state != NULL) {
1390 const uint32_t old = array_idx;
1391
1392 /* if tx.de_state->flags has 'new file' set and sig below has
1393 * 'file inspected' flag, reset the file part of the state */
1394 const bool have_new_file = (tx.de_state->flags & DETECT_ENGINE_STATE_FLAG_FILE_NEW);
1395 if (have_new_file) {
1396 SCLogDebug("%p/%"PRIu64" destate: need to consider new file",
1397 tx.tx_ptr, tx.tx_id);
1398 tx.de_state->flags &= ~DETECT_ENGINE_STATE_FLAG_FILE_NEW;
1399 }
1400
1401 SigIntId state_cnt = 0;
1402 DeStateStore *tx_store = tx.de_state->head;
1403 for (; tx_store != NULL; tx_store = tx_store->next) {
1404 SCLogDebug("tx_store %p", tx_store);
1405
1406 SigIntId store_cnt = 0;
1407 for (store_cnt = 0;
1408 store_cnt < DE_STATE_CHUNK_SIZE && state_cnt < tx.de_state->cnt;
1409 store_cnt++, state_cnt++)
1410 {
1411 DeStateStoreItem *item = &tx_store->store[store_cnt];
1412 SCLogDebug("rule id %u, inspect_flags %u", item->sid, item->flags);
1413 if (have_new_file && (item->flags & DE_STATE_FLAG_FILE_INSPECT)) {
1414 /* remove part of the state. File inspect engine will now
1415 * be able to run again */
1416 item->flags &= ~(DE_STATE_FLAG_SIG_CANT_MATCH|DE_STATE_FLAG_FULL_INSPECT|DE_STATE_FLAG_FILE_INSPECT);
1417 SCLogDebug("rule id %u, post file reset inspect_flags %u", item->sid, item->flags);
1418 }
1419 det_ctx->tx_candidates[array_idx].s = de_ctx->sig_array[item->sid];
1420 det_ctx->tx_candidates[array_idx].id = item->sid;
1421 det_ctx->tx_candidates[array_idx].flags = &item->flags;
1422 det_ctx->tx_candidates[array_idx].stream_reset = 0;
1423 array_idx++;
1424 }
1425 }
1426 do_sort |= (old && old != array_idx); // sort if continue list adds sids
1427 SCLogDebug("%p/%" PRIu64 " rules added from 'continue' list: %u", tx.tx_ptr, tx.tx_id,
1428 array_idx - old);
1429 }
1430 if (do_sort) {
1431 qsort(det_ctx->tx_candidates, array_idx, sizeof(RuleMatchCandidateTx),
1432 DetectRunTxSortHelper);
1433 }
1434
1435#ifdef PROFILING
1436 if (array_idx >= de_ctx->profile_match_logging_threshold)
1437 RulesDumpTxMatchArray(det_ctx, scratch->sgh, p, tx.tx_id, array_idx, x);
1438#endif
1439 det_ctx->tx_id = tx.tx_id;
1440 det_ctx->tx_id_set = true;
1441 det_ctx->p = p;
1442
1443#ifdef DEBUG
1444 for (uint32_t i = 0; i < array_idx; i++) {
1445 RuleMatchCandidateTx *can = &det_ctx->tx_candidates[i];
1446 const Signature *s = det_ctx->tx_candidates[i].s;
1447 SCLogDebug("%u: sid %u flags %p", i, s->id, can->flags);
1448 }
1449#endif
1450 /* run rules: inspect the match candidates */
1451 for (uint32_t i = 0; i < array_idx; i++) {
1452 RuleMatchCandidateTx *can = &det_ctx->tx_candidates[i];
1453 const Signature *s = det_ctx->tx_candidates[i].s;
1454 uint32_t *inspect_flags = det_ctx->tx_candidates[i].flags;
1455
1456 /* deduplicate: rules_array is sorted, but not deduplicated:
1457 * both mpm and stored state could give us the same sid.
1458 * As they are back to back in that case we can check for it
1459 * here. We select the stored state one as that comes first
1460 * in the array. */
1461 while ((i + 1) < array_idx &&
1462 det_ctx->tx_candidates[i].s == det_ctx->tx_candidates[i + 1].s) {
1463 SCLogDebug("%p/%" PRIu64 " inspecting SKIP NEXT: sid %u (%u), flags %08x",
1464 tx.tx_ptr, tx.tx_id, s->id, s->num, inspect_flags ? *inspect_flags : 0);
1465 i++;
1466 }
1467
1468 SCLogDebug("%p/%"PRIu64" inspecting: sid %u (%u), flags %08x",
1469 tx.tx_ptr, tx.tx_id, s->id, s->num, inspect_flags ? *inspect_flags : 0);
1470
1471 if (inspect_flags) {
1472 if (*inspect_flags & (DE_STATE_FLAG_FULL_INSPECT|DE_STATE_FLAG_SIG_CANT_MATCH)) {
1473 SCLogDebug("%p/%"PRIu64" inspecting: sid %u (%u), flags %08x ALREADY COMPLETE",
1474 tx.tx_ptr, tx.tx_id, s->id, s->num, *inspect_flags);
1475 continue;
1476 }
1477 }
1478
1479 if (inspect_flags) {
1480 /* continue previous inspection */
1481 SCLogDebug("%p/%"PRIu64" Continueing sid %u", tx.tx_ptr, tx.tx_id, s->id);
1482 } else {
1483 /* start new inspection */
1484 SCLogDebug("%p/%"PRIu64" Start sid %u", tx.tx_ptr, tx.tx_id, s->id);
1485 }
1486
1487 /* call individual rule inspection */
1488 RULE_PROFILING_START(p);
1489 const int r = DetectRunTxInspectRule(tv, de_ctx, det_ctx, p, f, flow_flags,
1490 alstate, &tx, s, inspect_flags, can, scratch);
1491 if (r == 1) {
1492 /* match */
1493 DetectRunPostMatch(tv, det_ctx, p, s);
1494
1495 const uint8_t alert_flags = (PACKET_ALERT_FLAG_STATE_MATCH | PACKET_ALERT_FLAG_TX);
1496 SCLogDebug("%p/%"PRIu64" sig %u (%u) matched", tx.tx_ptr, tx.tx_id, s->id, s->num);
1497 AlertQueueAppend(det_ctx, s, p, tx.tx_id, alert_flags);
1498 }
1499 DetectVarProcessList(det_ctx, p->flow, p);
1500 RULE_PROFILING_END(det_ctx, s, r, p);
1501 }
1502
1503 det_ctx->tx_id = 0;
1504 det_ctx->tx_id_set = false;
1505 det_ctx->p = NULL;
1506
1507 /* see if we have any updated state to store in the tx */
1508
1509 uint64_t new_detect_flags = 0;
1510 /* this side of the tx is done */
1511 if (tx.tx_progress >= tx.tx_end_state) {
1512 new_detect_flags |= APP_LAYER_TX_INSPECTED_FLAG;
1513 SCLogDebug("%p/%"PRIu64" tx is done for direction %s. Flag %016"PRIx64,
1514 tx.tx_ptr, tx.tx_id,
1515 flow_flags & STREAM_TOSERVER ? "toserver" : "toclient",
1516 new_detect_flags);
1517 }
1518 if (tx.prefilter_flags != tx.prefilter_flags_orig) {
1519 new_detect_flags |= tx.prefilter_flags;
1520 DEBUG_VALIDATE_BUG_ON(new_detect_flags & APP_LAYER_TX_RESERVED_FLAGS);
1521 SCLogDebug("%p/%"PRIu64" updated prefilter flags %016"PRIx64" "
1522 "(was: %016"PRIx64") for direction %s. Flag %016"PRIx64,
1523 tx.tx_ptr, tx.tx_id, tx.prefilter_flags, tx.prefilter_flags_orig,
1524 flow_flags & STREAM_TOSERVER ? "toserver" : "toclient",
1525 new_detect_flags);
1526 }
1527 if (new_detect_flags != 0 &&
1528 (new_detect_flags | tx.detect_flags) != tx.detect_flags)
1529 {
1530 new_detect_flags |= tx.detect_flags;
1531 DEBUG_VALIDATE_BUG_ON(new_detect_flags & APP_LAYER_TX_RESERVED_FLAGS);
1532 SCLogDebug("%p/%"PRIu64" Storing new flags %016"PRIx64" (was %016"PRIx64")",
1533 tx.tx_ptr, tx.tx_id, new_detect_flags, tx.detect_flags);
1534
1535 StoreDetectFlags(&tx, flow_flags, ipproto, alproto, new_detect_flags);
1536 }
1537next:
1538 InspectionBufferClean(det_ctx);
1539
1540 if (!ires.has_next)
1541 break;
1542 }
1543}

1666 * update the inspect_id forward. So test for the condition here,
1667 * and call the update code if necessary. */
1668 const int pass = ((p->flow->flags & FLOW_NOPACKET_INSPECTION));
1669 if (pass) {
1670 uint8_t flags;
1671 if (p->flowflags & FLOW_PKT_TOSERVER) {

1674 flags = STREAM_TOCLIENT;

1682 }

1711 * \param p packet to inspect

1713 * \param pq packet queue
1714 * \retval TM_ECODE_FAILED error
1715 * \retval TM_ECODE_OK ok
1716 */
1717TmEcode Detect(ThreadVars *tv, Packet *p, void *data)
1718{
1719 DEBUG_VALIDATE_PACKET(p);
1720
1721 DetectEngineCtx *de_ctx = NULL;
1722 DetectEngineThreadCtx *det_ctx = (DetectEngineThreadCtx *)data;
1723 if (det_ctx == NULL) {
1724 printf("ERROR: Detect has no thread ctx\n");
1725 goto error;
1726 }
1727
1728 if (unlikely(SC_ATOMIC_GET(det_ctx->so_far_used_by_detect) == 0)) {
1729 (void)SC_ATOMIC_SET(det_ctx->so_far_used_by_detect, 1);
1730 SCLogDebug("Detect Engine using new det_ctx - %p",
1731 det_ctx);
1732 }
1733
1734 /* if in MT mode _and_ we have tenants registered, use
1735 * MT logic. */
1736 if (det_ctx->mt_det_ctxs_cnt > 0 && det_ctx->TenantGetId != NULL)
1737 {
1738 uint32_t tenant_id = p->tenant_id;
1739 if (tenant_id == 0)
1740 tenant_id = det_ctx->TenantGetId(det_ctx, p);
1741 if (tenant_id > 0 && tenant_id < det_ctx->mt_det_ctxs_cnt) {
1742 p->tenant_id = tenant_id;
1743 det_ctx = GetTenantById(det_ctx->mt_det_ctxs_hash, tenant_id);
1744 if (det_ctx == NULL)
1745 return TM_ECODE_OK;
1746 de_ctx = det_ctx->de_ctx;
1747 if (de_ctx == NULL)
1748 return TM_ECODE_OK;
1749
1750 if (unlikely(SC_ATOMIC_GET(det_ctx->so_far_used_by_detect) == 0)) {
1751 (void)SC_ATOMIC_SET(det_ctx->so_far_used_by_detect, 1);
1752 SCLogDebug("MT de_ctx %p det_ctx %p (tenant %u)", de_ctx, det_ctx, tenant_id);
1753 }
1754 } else {
1755 /* use default if no tenants are registered for this packet */
1756 de_ctx = det_ctx->de_ctx;
1757 }
1758 } else {
1759 de_ctx = det_ctx->de_ctx;
1760 }
1761
1762 if (p->flow) {
1763 DetectFlow(tv, de_ctx, det_ctx, p);
1764 } else {
1765 DetectNoFlow(tv, de_ctx, det_ctx, p);
1766 }
1767 return TM_ECODE_OK;
1768error:
1769 return TM_ECODE_FAILED;
1770}
