39/* Regex to parse the classtype argument from a Signature. The first substring
40 * holds the classtype name, the second substring holds the classtype the
41 * classtype description, and the third argument holds the priority */
42#define DETECT_CLASSCONFIG_REGEX "^\\s*config\\s*classification\\s*:\\s*([a-zA-Z][a-zA-Z0-9-_]*)\\s*,\\s*(.+)\\s*,\\s*(\\d+)\\s*$"
43
44/* Default path for the classification.config file */
45#if defined OS_WIN32 || defined __CYGWIN__
46#define SC_CLASS_CONF_DEF_CONF_FILEPATH CONFIG_DIR "\\\\classification.config"
47#else
48#define SC_CLASS_CONF_DEF_CONF_FILEPATH CONFIG_DIR "/classification.config"
49#endif

64void SCClassConfInit(void)
65{
66 int en;
67 PCRE2_SIZE eo;
68 int opts = 0;
69
70 regex = pcre2_compile(
71 (PCRE2_SPTR8)DETECT_CLASSCONFIG_REGEX, PCRE2_ZERO_TERMINATED, opts, &en, &eo, NULL);
72 if (regex == NULL) {
73 PCRE2_UCHAR errbuffer[256];
74 pcre2_get_error_message(en, errbuffer, sizeof(errbuffer));
75 SCLogWarning(SC_ERR_PCRE_COMPILE,
76 "pcre2 compile of \"%s\" failed at "
77 "offset %d: %s",
78 DETECT_CLASSCONFIG_REGEX, (int)eo, errbuffer);
79 return;
80 }
81 regex_match = pcre2_match_data_create_from_pattern(regex, NULL);
82 return;
83}
84
85void SCClassConfDeinit(void)
86{
87 if (regex != NULL) {
88 pcre2_code_free(regex);
89 regex = NULL;
90 }
91 if (regex_match != NULL) {
92 pcre2_match_data_free(regex_match);
93 regex_match = NULL;
94 }
95}

98/**
99 * \brief Inits the context to be used by the Classification Config parsing API.
100 *
101 * This function initializes the hash table to be used by the Detection
102 * Engine Context to hold the data from the classification.config file,
103 * obtains the file desc to parse the classification.config file, and
104 * inits the regex used to parse the lines from classification.config
105 * file.
106 *
107 * \param de_ctx Pointer to the Detection Engine Context.
108 * \param fd Pointer to already opened file
109 *
110 * \note even if the file open fails we will keep the de_ctx->class_conf_ht
111 * initialized.
112 *
113 * \retval fp NULL on error
114 */
115static FILE *SCClassConfInitContextAndLocalResources(DetectEngineCtx *de_ctx, FILE *fd)
116{
117 /* init the hash table to be used by the classification config Classtypes */
118 de_ctx->class_conf_ht = HashTableInit(128, SCClassConfClasstypeHashFunc,
119 SCClassConfClasstypeHashCompareFunc,
120 SCClassConfClasstypeHashFree);
121 if (de_ctx->class_conf_ht == NULL) {
122 SCLogError(SC_ERR_HASH_TABLE_INIT, "Error initializing the hash "
123 "table");
124 return NULL;
125 }
126
127 /* if it is not NULL, use the file descriptor. The hack so that we can
128 * avoid using a dummy classification file for testing purposes and
129 * instead use an input stream against a buffer containing the
130 * classification strings */
131 if (fd == NULL) {
132 const char *filename = SCClassConfGetConfFilename(de_ctx);
133 if ( (fd = fopen(filename, "r")) == NULL) {
134#ifdef UNITTESTS
135 if (RunmodeIsUnittests())
136 return NULL; // silently fail
137#endif
138 SCLogWarning(SC_ERR_FOPEN, "could not open: \"%s\": %s",
139 filename, strerror(errno));
140 return NULL;
141 }
142 }
143
144 return fd;
145}

148/**
149 * \brief Returns the path for the Classification Config file. We check if we
150 * can retrieve the path from the yaml conf file. If it is not present,
151 * return the default path for the classification file which is
152 * "./classification.config".
153 *
154 * \retval log_filename Pointer to a string containing the path for the
155 * Classification Config file.
156 */
157static const char *SCClassConfGetConfFilename(const DetectEngineCtx *de_ctx)
158{
159 const char *log_filename = NULL;
160
161 if (de_ctx != NULL && strlen(de_ctx->config_prefix) > 0) {
162 char config_value[256];
163 snprintf(config_value, sizeof(config_value),
164 "%s.classification-file", de_ctx->config_prefix);
165
166 /* try loading prefix setting, fall back to global if that
167 * fails. */
168 if (ConfGet(config_value, &log_filename) != 1) {
169 if (ConfGet("classification-file", &log_filename) != 1) {
170 log_filename = (char *)SC_CLASS_CONF_DEF_CONF_FILEPATH;
171 }
172 }
173 } else {
174 if (ConfGet("classification-file", &log_filename) != 1) {
175 log_filename = (char *)SC_CLASS_CONF_DEF_CONF_FILEPATH;
176 }
177 }
178
179 return log_filename;
180}

205/**
206 * \brief Converts a string to lowercase.
207 *
208 * \param str Pointer to the string to be converted.
209 */
210static char *SCClassConfStringToLowercase(const char *str)
211{
212 char *new_str = NULL;
213 char *temp_str = NULL;
214
215 if ( (new_str = SCStrdup(str)) == NULL) {
216 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
217 return NULL;
218 }
219
220 temp_str = new_str;
221 while (*temp_str != '\0') {
222 *temp_str = u8_tolower((unsigned char)*temp_str);
223 temp_str++;
224 }
225
226 return new_str;
227}

229/**
230 * \brief Parses a line from the classification file and adds it to Classtype
231 * hash table in DetectEngineCtx, i.e. DetectEngineCtx->class_conf_ht.
232 *
233 * \param rawstr Pointer to the string to be parsed.
234 * \param index Relative index of the string to be parsed.
235 * \param de_ctx Pointer to the Detection Engine Context.
236 *
237 * \retval 0 On success.
238 * \retval -1 On failure.
239 */
240int SCClassConfAddClasstype(DetectEngineCtx *de_ctx, char *rawstr, uint16_t index)
241{
242 char ct_name[CLASSTYPE_NAME_MAX_LEN];
243 char ct_desc[CLASSTYPE_DESC_MAX_LEN];
244 char ct_priority_str[16];
245 uint32_t ct_priority = 0;
246 uint16_t ct_id = index;
247
248 SCClassConfClasstype *ct_new = NULL;
249 SCClassConfClasstype *ct_lookup = NULL;
250
251 int ret = 0;
252
253 ret = pcre2_match(regex, (PCRE2_SPTR8)rawstr, strlen(rawstr), 0, 0, regex_match, NULL);
254 if (ret < 0) {
255 SCLogError(SC_ERR_INVALID_SIGNATURE,
256 "Invalid Classtype in "
257 "classification.config file %s: \"%s\"",
258 SCClassConfGetConfFilename(de_ctx), rawstr);
259 goto error;
260 }
261
262 size_t copylen = sizeof(ct_name);
263 /* retrieve the classtype name */
264 ret = pcre2_substring_copy_bynumber(regex_match, 1, (PCRE2_UCHAR8 *)ct_name, &copylen);
265 if (ret < 0) {
266 SCLogInfo("pcre2_substring_copy_bynumber() failed");
267 goto error;
268 }
269
270 /* retrieve the classtype description */
271 copylen = sizeof(ct_desc);
272 ret = pcre2_substring_copy_bynumber(regex_match, 2, (PCRE2_UCHAR8 *)ct_desc, &copylen);
273 if (ret < 0) {
274 SCLogInfo("pcre2_substring_copy_bynumber() failed");
275 goto error;
276 }
277
278 /* retrieve the classtype priority */
279 copylen = sizeof(ct_priority_str);
280 ret = pcre2_substring_copy_bynumber(regex_match, 3, (PCRE2_UCHAR8 *)ct_priority_str, &copylen);
281 if (ret < 0) {
282 SCLogInfo("pcre2_substring_copy_bynumber() failed");
283 goto error;
284 }
285 if (StringParseUint32(&ct_priority, 10, 0, (const char *)ct_priority_str) < 0) {
286 goto error;
287 }
288
289 /* Create a new instance of the parsed Classtype string */
290 ct_new = SCClassConfAllocClasstype(ct_id, ct_name, ct_desc, ct_priority);
291 if (ct_new == NULL)
292 goto error;
293
294 /* Check if the Classtype is present in the HashTable. In case it's present
295 * ignore it, as it is a duplicate. If not present, add it to the table */
296 ct_lookup = HashTableLookup(de_ctx->class_conf_ht, ct_new, 0);
297 if (ct_lookup == NULL) {
298 if (HashTableAdd(de_ctx->class_conf_ht, ct_new, 0) < 0)
299 SCLogDebug("HashTable Add failed");
300 } else {
301 SCLogDebug("Duplicate classtype found inside classification.config");
302 if (ct_new->classtype_desc) SCFree(ct_new->classtype_desc);
303 if (ct_new->classtype) SCFree(ct_new->classtype);
304 SCFree(ct_new);
305 }
306
307 return 0;
308
309 error:
310 return -1;
311}

313/**
314 * \brief Checks if a string is a comment or a blank line.
315 *
316 * Comments lines are lines of the following format -
317 * "# This is a comment string" or
318 * " # This is a comment string".
319 *
320 * \param line String that has to be checked
321 *
322 * \retval 1 On the argument string being a comment or blank line
323 * \retval 0 Otherwise
324 */
325static int SCClassConfIsLineBlankOrComment(char *line)
326{
327 while (*line != '\0') {
328 /* we have a comment */
329 if (*line == '#')
330 return 1;
331
332 /* this line is neither a comment line, nor a blank line */
333 if (!isspace((unsigned char)*line))
334 return 0;
335
336 line++;
337 }
338
339 /* we have a blank line */
340 return 1;
341}

343/**
344 * \brief Parses the Classification Config file and updates the
345 * DetectionEngineCtx->class_conf_ht with the Classtype information.
346 *
347 * \param de_ctx Pointer to the Detection Engine Context.
348 */
349static bool SCClassConfParseFile(DetectEngineCtx *de_ctx, FILE *fd)
350{
351 char line[1024];
352 uint16_t i = 1;
353
354 while (fgets(line, sizeof(line), fd) != NULL) {
355 if (SCClassConfIsLineBlankOrComment(line))
356 continue;
357
358 if (SCClassConfAddClasstype(de_ctx, line, i) == -1) {
359 return false;
360 }
361 i++;
362 }
363
364#ifdef UNITTESTS
365 SCLogInfo("Added \"%d\" classification types from the classification file",
366 de_ctx->class_conf_ht->count);
367#endif
368
369 return true;
370}

372/**
373 * \internal
374 * \brief Returns a new SCClassConfClasstype instance. The classtype string
375 * is converted into lowercase, before being assigned to the instance.
376 *
377 * \param classtype Pointer to the classification type.
378 * \param classtype_desc Pointer to the classification type description.
379 * \param priority Holds the priority for the classification type.
380 *
381 * \retval ct Pointer to the new instance of SCClassConfClasstype on success;
382 * NULL on failure.
383 */
384static SCClassConfClasstype *SCClassConfAllocClasstype(uint16_t classtype_id,
385 const char *classtype,
386 const char *classtype_desc,
387 int priority)
388{
389 SCClassConfClasstype *ct = NULL;
390
391 if (classtype == NULL)
392 return NULL;
393
394 if ( (ct = SCMalloc(sizeof(SCClassConfClasstype))) == NULL)
395 return NULL;
396 memset(ct, 0, sizeof(SCClassConfClasstype));
397
398 if ( (ct->classtype = SCClassConfStringToLowercase(classtype)) == NULL) {
399 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
400
401 SCClassConfDeAllocClasstype(ct);
402 return NULL;
403 }
404
405 if (classtype_desc != NULL &&
406 (ct->classtype_desc = SCStrdup(classtype_desc)) == NULL) {
407 SCLogError(SC_ERR_MEM_ALLOC, "Error allocating memory");
408
409 SCClassConfDeAllocClasstype(ct);
410 return NULL;
411 }
412
413 ct->classtype_id = classtype_id;
414 ct->priority = priority;
415
416 return ct;
417}

440/**
441 * \brief Hashing function to be used to hash the Classtype name. Would be
442 * supplied as an argument to the HashTableInit function for
443 * DetectEngineCtx->class_conf_ht.
444 *
445 * \param ht Pointer to the HashTable.
446 * \param data Pointer to the data to be hashed. In this case, the data
447 * would be a pointer to a SCClassConfClasstype instance.
448 * \param datalen Not used by this function.
449 */
450uint32_t SCClassConfClasstypeHashFunc(HashTable *ht, void *data, uint16_t datalen)
451{
452 SCClassConfClasstype *ct = (SCClassConfClasstype *)data;
453 uint32_t hash = 0;
454 int i = 0;
455
456 int len = strlen(ct->classtype);
457
458 for (i = 0; i < len; i++)
459 hash += u8_tolower((unsigned char)(ct->classtype)[i]);
460
461 hash = hash % ht->array_size;
462
463 return hash;
464}

466/**
467 * \brief Used to compare two Classtypes that have been stored in the HashTable.
468 * This function is supplied as an argument to the HashTableInit function
469 * for DetectionEngineCtx->class_conf_ct.
470 *
471 * \param data1 Pointer to the first SCClassConfClasstype to be compared.
472 * \param len1 Not used by this function.
473 * \param data2 Pointer to the second SCClassConfClasstype to be compared.
474 * \param len2 Not used by this function.
475 *
476 * \retval 1 On data1 and data2 being equal.
477 * \retval 0 On data1 and data2 not being equal.
478 */
479char SCClassConfClasstypeHashCompareFunc(void *data1, uint16_t datalen1,
480 void *data2, uint16_t datalen2)
481{
482 SCClassConfClasstype *ct1 = (SCClassConfClasstype *)data1;
483 SCClassConfClasstype *ct2 = (SCClassConfClasstype *)data2;
484 int len1 = 0;
485 int len2 = 0;
486
487 if (ct1 == NULL || ct2 == NULL)
488 return 0;
489
490 if (ct1->classtype == NULL || ct2->classtype == NULL)
491 return 0;
492
493 len1 = strlen(ct1->classtype);
494 len2 = strlen(ct2->classtype);
495
496 if (len1 == len2 && memcmp(ct1->classtype, ct2->classtype, len1) == 0) {
497 SCLogDebug("Match found inside Classification-Config hash function");
498 return 1;
499 }
500
501 return 0;
502}

517/**
518 * \brief Loads the Classtype info from the classification.config file.
519 *
520 * The classification.config file contains the different classtypes,
521 * that can be used to label Signatures. Each line of the file should
522 * have the following format -
523 * classtype_name, classtype_description, priority
524 * None of the above parameters should hold a quote inside the file.
525 *
526 * \param de_ctx Pointer to the Detection Engine Context that should be updated
527 * with Classtype information.
528 */
529bool SCClassConfLoadClassficationConfigFile(DetectEngineCtx *de_ctx, FILE *fd)
530{
531 fd = SCClassConfInitContextAndLocalResources(de_ctx, fd);
532 if (fd == NULL) {
533#ifdef UNITTESTS
534 if (RunmodeIsUnittests()) {
535 return false;
536 }
537#endif
538 SCLogError(SC_ERR_OPENING_FILE, "please check the \"classification-file\" "
539 "option in your suricata.yaml file");
540 return false;
541 }
542
543 bool ret = true;
544 if (!SCClassConfParseFile(de_ctx, fd)) {
545 SCLogWarning(SC_WARN_CLASSIFICATION_CONFIG,
546 "Error loading classification configuration from %s",
547 SCClassConfGetConfFilename(de_ctx));
548 ret = false;
549 }
550
551 SCClassConfDeInitLocalResources(de_ctx, fd);
552
553 return ret;
554}

556/**
557 * \brief Gets the classtype from the corresponding hash table stored
558 * in the Detection Engine Context's class conf ht, given the
559 * classtype name.
560 *
561 * \param ct_name Pointer to the classtype name that has to be looked up.
562 * \param de_ctx Pointer to the Detection Engine Context.
563 *
564 * \retval lookup_ct_info Pointer to the SCClassConfClasstype instance from
565 * the hash table on success; NULL on failure.
566 */
567SCClassConfClasstype *SCClassConfGetClasstype(const char *ct_name,
568 DetectEngineCtx *de_ctx)
569{
570 char name[strlen(ct_name) + 1];
571 size_t s;
572 for (s = 0; s < strlen(ct_name); s++)
573 name[s] = u8_tolower((unsigned char)ct_name[s]);
574 name[s] = '\0';
575
576 SCClassConfClasstype ct_lookup = {0, 0, name, NULL };
577 SCClassConfClasstype *lookup_ct_info = HashTableLookup(de_ctx->class_conf_ht,
578 &ct_lookup, 0);
579 return lookup_ct_info;
580}
