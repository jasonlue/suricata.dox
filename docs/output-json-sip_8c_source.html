51void JsonSIPAddMetadata(JsonBuilder *js, const Flow *f, uint64_t tx_id)
52{
53 SIPState *state = FlowGetAppState(f);
54 if (state) {
55 SIPTransaction *tx = AppLayerParserGetTx(f->proto, ALPROTO_SIP, state, tx_id);
56 if (tx) {

62static int JsonSIPLogger(ThreadVars *tv, void *thread_data,
63 const Packet *p, Flow *f, void *state, void *tx, uint64_t tx_id)
64{
65 SIPTransaction *siptx = tx;
66 OutputJsonThreadCtx *thread = thread_data;
67
68 JsonBuilder *js = CreateEveHeader((Packet *)p, LOG_DIR_PACKET, "sip", NULL, thread->ctx);
69 if (unlikely(js == NULL)) {
70 return TM_ECODE_OK;
71 }
72
73 if (!rs_sip_log_json(siptx, js)) {
74 goto error;
75 }
76
77 OutputJsonBuilderBuffer(js, thread);
78 jb_free(js);
79
80 return TM_ECODE_OK;
81
82error:
83 jb_free(js);
84 return TM_ECODE_FAILED;
85}
