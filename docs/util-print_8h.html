do { \
int cw = snprintf((buf) + *(buf_offset_ptr), \
(buf_size) - *(buf_offset_ptr), \
__VA_ARGS__); \
if (cw >= 0) { \
if ( (*(buf_offset_ptr) + cw) >= buf_size) { \
SCLogDebug("Truncating data write since it exceeded buffer " \
"limit of - %"PRIu32"\n", buf_size); \
*(buf_offset_ptr) = buf_size - 1; \
} else { \
*(buf_offset_ptr) += cw; \
} \
} \
} while (0)
