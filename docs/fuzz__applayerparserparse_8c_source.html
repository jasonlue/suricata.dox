25/* input buffer is structured this way :
26 * 6 bytes header,
27 * then sequence of buffers separated by magic bytes 01 D5 CA 7A */

29/* The 6 bytes header is
30 * alproto
31 * proto
32 * source port (uint16_t)
33 * destination port (uint16_t) */

39int LLVMFuzzerInitialize(int *argc, char ***argv)
40{
41 char *target_suffix = strrchr((*argv)[0], '_');
42 if (target_suffix != NULL) {
43 AppProto applayer = StringToAppProto(target_suffix + 1);
44 if (applayer != ALPROTO_UNKNOWN) {
45 forceLayer = applayer;
46 printf("Forcing %s=%" PRIu16 "\n", AppProtoToString(forceLayer), forceLayer);
47 return 0;
48 }
49 }
50 // else
51 const char *forceLayerStr = getenv("FUZZ_APPLAYER");
52 if (forceLayerStr) {
53 if (ByteExtractStringUint16(&forceLayer, 10, 0, forceLayerStr) < 0) {
54 forceLayer = 0;
55 printf("Invalid numeric value for FUZZ_APPLAYER environment variable");
56 } else {
57 printf("Forcing %s\n", AppProtoToString(forceLayer));
58 }
59 }
60 return 0;
61}

66int LLVMFuzzerTestOneInput(const uint8_t *data, size_t size)
67{
68 Flow * f;
69 TcpSession ssn;
70 const uint8_t * albuffer;
71 uint8_t * alnext;
72 size_t alsize;
73 // used to find under and overflows
74 // otherwise overflows do not fail as they read the next packet
75 uint8_t * isolatedBuffer;
76
77 if (size < HEADER_LEN) {
78 return 0;
79 }
80
81 if (alp_tctx == NULL) {
82 //Redirects logs to /dev/null
83 setenv("SC_LOG_OP_IFACE", "file", 0);
84 setenv("SC_LOG_FILE", "/dev/null", 0);
85
86 InitGlobal();
87 run_mode = RUNMODE_PCAP_FILE;
88 GlobalsInitPreConfig();
89
90 //redirect logs to /tmp
91 ConfigSetLogDirectory("/tmp/");
92 // disables checksums validation for fuzzing
93 if (ConfYamlLoadString(configNoChecksum, strlen(configNoChecksum)) != 0) {
94 abort();
95 }
96
97 PostConfLoadedSetup(&surifuzz);
98 alp_tctx = AppLayerParserThreadCtxAlloc();
99 }
100
101 if (data[0] >= ALPROTO_MAX) {
102 return 0;
103 }
104 //no UTHBuildFlow to have storage
105 f = FlowAlloc();
106 if (f == NULL) {
107 return 0;
108 }
109 f->flags |= FLOW_IPV4;
110 f->src.addr_data32[0] = 0x01020304;
111 f->dst.addr_data32[0] = 0x05060708;
112 f->sp = (uint16_t)((data[2] << 8) | data[3]);
113 f->dp = (uint16_t)((data[4] << 8) | data[5]);
114 f->proto = data[1];
115 memset(&ssn, 0, sizeof(TcpSession));
116 f->protoctx = &ssn;
117 f->protomap = FlowGetProtoMapping(f->proto);
118 if (forceLayer > 0) {
119 f->alproto = forceLayer;
120 } else {
121 f->alproto = data[0];
122 }
123
124 FLOWLOCK_WRLOCK(f);
125 /*
126 * We want to fuzz multiple calls to AppLayerParserParse
127 * because some parts of the code are only reached after
128 * multiple packets (in SMTP for example).
129 * So we treat our input as a list of buffers with magic separator.
130 */
131 albuffer = data + HEADER_LEN;
132 alsize = size - HEADER_LEN;
133 uint8_t flags = STREAM_START;
134 int flip = 0;
135 alnext = memmem(albuffer, alsize, separator, 4);
136 while (alnext) {
137 if (flip) {
138 flags |= STREAM_TOCLIENT;
139 flags &= ~(STREAM_TOSERVER);
140 flip = 0;
141 } else {
142 flags |= STREAM_TOSERVER;
143 flags &= ~(STREAM_TOCLIENT);
144 flip = 1;
145 }
146
147 if (alnext != albuffer) {
148 // only if we have some data
149 isolatedBuffer = malloc(alnext - albuffer);
150 if (isolatedBuffer == NULL) {
151 return 0;
152 }
153 memcpy(isolatedBuffer, albuffer, alnext - albuffer);
154 (void) AppLayerParserParse(NULL, alp_tctx, f, f->alproto, flags, isolatedBuffer, alnext - albuffer);
155 free(isolatedBuffer);
156 if (FlowChangeProto(f)) {
157 // exits if a protocol change is requested
158 alsize = 0;
159 break;
160 }
161 flags &= ~(STREAM_START);
162 if (f->alparser &&
163 (((flags & STREAM_TOSERVER) != 0 &&
164 AppLayerParserStateIssetFlag(f->alparser, APP_LAYER_PARSER_EOF_TS)) ||
165 ((flags & STREAM_TOCLIENT) != 0 &&
166 AppLayerParserStateIssetFlag(f->alparser, APP_LAYER_PARSER_EOF_TC)))) {
167 //no final chunk
168 alsize = 0;
169 break;
170 }
171
172 AppLayerParserTransactionsCleanup(f);
173 }
174 alsize -= alnext - albuffer + 4;
175 albuffer = alnext + 4;
176 if (alsize == 0) {
177 break;
178 }
179 alnext = memmem(albuffer, alsize, separator, 4);
180 }
181 if (alsize > 0 ) {
182 if (flip) {
183 flags |= STREAM_TOCLIENT;
184 flags &= ~(STREAM_TOSERVER);
185 flip = 0;
186 } else {
187 flags |= STREAM_TOSERVER;
188 flags &= ~(STREAM_TOCLIENT);
189 flip = 1;
190 }
191 flags |= STREAM_EOF;
192 isolatedBuffer = malloc(alsize);
193 if (isolatedBuffer == NULL) {
194 return 0;
195 }
196 memcpy(isolatedBuffer, albuffer, alsize);
197 (void) AppLayerParserParse(NULL, alp_tctx, f, f->alproto, flags, isolatedBuffer, alsize);
198 free(isolatedBuffer);
199 }
200
201 FLOWLOCK_UNLOCK(f);
202 FlowFree(f);
203
204 return 0;
205}
