137static void FTPParseMemcap(void)
138{
139 const char *conf_val;
140
141 /** set config values for memcap, prealloc and hash_size */
142 if ((ConfGet("app-layer.protocols.ftp.memcap", &conf_val)) == 1)
143 {
144 if (ParseSizeStringU64(conf_val, &ftp_config_memcap) < 0) {
145 SCLogError(SC_ERR_SIZE_PARSE, "Error parsing ftp.memcap "
146 "from conf file - %s. Killing engine",
147 conf_val);
148 exit(EXIT_FAILURE);
149 }
150 SCLogInfo("FTP memcap: %"PRIu64, ftp_config_memcap);
151 } else {
152 /* default to unlimited */
153 ftp_config_memcap = 0;
154 }
155
156 SC_ATOMIC_INIT(ftp_memuse);
157 SC_ATOMIC_INIT(ftp_memcap);
158
159 if ((ConfGet("app-layer.protocols.ftp.max-tx", &conf_val)) == 1) {
160 if (ParseSizeStringU32(conf_val, &ftp_config_maxtx) < 0) {
161 SCLogError(SC_ERR_SIZE_PARSE,
162 "Error parsing ftp.max-tx "
163 "from conf file - %s.",
164 conf_val);
165 }
166 SCLogInfo("FTP max tx: %" PRIu32, ftp_config_maxtx);
167 }
168
169 if ((ConfGet("app-layer.protocols.ftp.max-line-length", &conf_val)) == 1) {
170 if (ParseSizeStringU32(conf_val, &ftp_max_line_len) < 0) {
171 SCLogError(SC_ERR_SIZE_PARSE, "Error parsing ftp.max-line-length from conf file - %s.",
172 conf_val);
173 }
174 SCLogConfig("FTP max line length: %" PRIu32, ftp_max_line_len);
175 }
176}

202/**
203 * \brief Check if alloc'ing "size" would mean we're over memcap
204 *
205 * \retval 1 if in bounds
206 * \retval 0 if not in bounds
207 */
208static int FTPCheckMemcap(uint64_t size)
209{
210 if (ftp_config_memcap == 0 || size + SC_ATOMIC_GET(ftp_memuse) <= ftp_config_memcap)
211 return 1;
212 (void) SC_ATOMIC_ADD(ftp_memcap, 1);
213 return 0;
214}

216static void *FTPMalloc(size_t size)
217{
218 void *ptr = NULL;
219
220 if (FTPCheckMemcap((uint32_t)size) == 0)
221 return NULL;
222
223 ptr = SCMalloc(size);
224
225 if (unlikely(ptr == NULL))
226 return NULL;
227
228 FTPIncrMemuse((uint64_t)size);
229
230 return ptr;
231}

233static void *FTPCalloc(size_t n, size_t size)
234{
235 if (FTPCheckMemcap((uint32_t)(n * size)) == 0)
236 return NULL;
237
238 void *ptr = SCCalloc(n, size);
239
240 if (unlikely(ptr == NULL))
241 return NULL;
242
243 FTPIncrMemuse((uint64_t)(n * size));
244 return ptr;
245}

247static void *FTPRealloc(void *ptr, size_t orig_size, size_t size)
248{
249 void *rptr = NULL;
250
251 if (FTPCheckMemcap((uint32_t)(size - orig_size)) == 0)
252 return NULL;
253
254 rptr = SCRealloc(ptr, size);
255 if (rptr == NULL)
256 return NULL;
257
258 if (size > orig_size) {
259 FTPIncrMemuse(size - orig_size);
260 } else {
261 FTPDecrMemuse(orig_size - size);
262 }
263
264 return rptr;
265}

329static FTPTransaction *FTPTransactionCreate(FtpState *state)
330{
331 SCEnter();
332 FTPTransaction *firsttx = TAILQ_FIRST(&state->tx_list);
333 if (firsttx && state->tx_cnt - firsttx->tx_id > ftp_config_maxtx) {
334 // FTP does not set events yet...
335 return NULL;
336 }
337 FTPTransaction *tx = FTPCalloc(1, sizeof(*tx));
338 if (tx == NULL) {
339 return NULL;
340 }
341
342 TAILQ_INSERT_TAIL(&state->tx_list, tx, next);
343 tx->tx_id = state->tx_cnt++;
344
345 TAILQ_INIT(&tx->response_list);
346
347 SCLogDebug("new transaction %p (state tx cnt %"PRIu64")", tx, state->tx_cnt);
348 return tx;
349}

351static void FTPTransactionFree(FTPTransaction *tx)
352{
353 SCEnter();
354
355 if (tx->tx_data.de_state != NULL) {
356 DetectEngineStateFree(tx->tx_data.de_state);
357 }

360 FTPFree(tx->request, tx->request_length);
361 }
362
363 FTPString *str = NULL;
364 while ((str = TAILQ_FIRST(&tx->response_list))) {
365 TAILQ_REMOVE(&tx->response_list, str, next);
366 FTPStringFree(str);
367 }
368
369 FTPFree(tx, sizeof(*tx));
370}

377static int FTPGetLineForDirection(FtpState *state, FtpLineState *line_state, FtpInput *ftpi)
378{
379 void *ptmp;
380 if (line_state->current_line_lf_seen == 1) {
381 /* we have seen the lf for the previous line. Clear the parser
382 * details to parse new line */
383 line_state->current_line_lf_seen = 0;
384 state->current_line_truncated = false;
385 if (line_state->current_line_db == 1) {
386 line_state->current_line_db = 0;
387 FTPFree(line_state->db, line_state->db_len);
388 line_state->db = NULL;
389 line_state->db_len = 0;
390 state->current_line = NULL;
391 state->current_line_len = 0;
392 }
393 }
394
395 /* Should be guaranteed by the caller. */
396 DEBUG_VALIDATE_BUG_ON(ftpi->input_len <= 0);
397
398 uint8_t *lf_idx = memchr(ftpi->input, 0x0a, ftpi->input_len);
399
400 if (lf_idx == NULL) {
401 /* fragmented lines. Decoder event for special cases. Not all
402 * fragmented lines should be treated as a possible evasion
403 * attempt. With multi payload ftp chunks we can have valid
404 * cases of fragmentation. But within the same segment chunk
405 * if we see fragmentation then it's definitely something you
406 * should alert about */
407 if (line_state->current_line_db == 0) {
408 int32_t input_len = ftpi->input_len;
409 if ((uint32_t)input_len > ftp_max_line_len) {
410 input_len = ftp_max_line_len;
411 state->current_line_truncated = true;
412 }
413 line_state->db = FTPMalloc(input_len);
414 if (line_state->db == NULL) {
415 return -1;
416 }
417 line_state->current_line_db = 1;
418 memcpy(line_state->db, ftpi->input, input_len);
419 line_state->db_len = input_len;
420 } else if (!state->current_line_truncated) {
421 int32_t input_len = ftpi->input_len;
422 if (line_state->db_len + input_len > ftp_max_line_len) {
423 input_len = ftp_max_line_len - line_state->db_len;
424 DEBUG_VALIDATE_BUG_ON(input_len < 0);
425 state->current_line_truncated = true;
426 }
427 if (input_len > 0) {
428 ptmp = FTPRealloc(
429 line_state->db, line_state->db_len, (line_state->db_len + input_len));
430 if (ptmp == NULL) {
431 FTPFree(line_state->db, line_state->db_len);
432 line_state->db = NULL;
433 line_state->db_len = 0;
434 return -1;
435 }
436 line_state->db = ptmp;
437
438 memcpy(line_state->db + line_state->db_len, ftpi->input, input_len);
439 line_state->db_len += input_len;
440 }
441 }
442 ftpi->input += ftpi->input_len;
443 ftpi->input_len = 0;
444
445 return -1;
446
447 } else {
448 line_state->current_line_lf_seen = 1;
449
450 if (line_state->current_line_db == 1) {
451 if (!state->current_line_truncated) {
452 int32_t input_len = lf_idx + 1 - ftpi->input;
453 if (line_state->db_len + input_len > ftp_max_line_len) {
454 input_len = ftp_max_line_len - line_state->db_len;
455 DEBUG_VALIDATE_BUG_ON(input_len < 0);
456 state->current_line_truncated = true;
457 }
458 if (input_len > 0) {
459 ptmp = FTPRealloc(
460 line_state->db, line_state->db_len, (line_state->db_len + input_len));
461 if (ptmp == NULL) {
462 FTPFree(line_state->db, line_state->db_len);
463 line_state->db = NULL;
464 line_state->db_len = 0;
465 return -1;
466 }
467 line_state->db = ptmp;
468
469 memcpy(line_state->db + line_state->db_len, ftpi->input, input_len);
470 line_state->db_len += input_len;
471
472 if (line_state->db_len > 1 && line_state->db[line_state->db_len - 2] == 0x0D) {
473 line_state->db_len -= 2;
474 state->current_line_delimiter_len = 2;
475 } else {
476 line_state->db_len -= 1;
477 state->current_line_delimiter_len = 1;
478 }
479 }
480 }
481
482 state->current_line = line_state->db;
483 state->current_line_len = line_state->db_len;
484
485 } else {
486 state->current_line = ftpi->input;
487 if (lf_idx - ftpi->input > ftp_max_line_len) {
488 state->current_line_len = ftp_max_line_len;
489 state->current_line_truncated = true;
490 } else {
491 state->current_line_len = lf_idx - ftpi->input;
492 }
493
494 if (ftpi->input != lf_idx && *(lf_idx - 1) == 0x0D) {
495 state->current_line_len--;
496 state->current_line_delimiter_len = 2;
497 } else {
498 state->current_line_delimiter_len = 1;
499 }
500 }
501
502 ftpi->input_len -= (lf_idx - ftpi->input) + 1;
503 ftpi->input = (lf_idx + 1);
504
505 return 0;
506 }

510static int FTPGetLine(FtpState *state, int direction, FtpInput *ftpi)
511{
512 SCEnter();
513
514 /* we have run out of input */
515 if (ftpi->input_len <= 0)
516 return -1;
517
518 /* toserver */
519 if (direction == STREAM_TOSERVER)
520 return FTPGetLineForDirection(state, &state->line_state[0], ftpi);
521 else
522 return FTPGetLineForDirection(state, &state->line_state[1], ftpi);
523}

525/**
526 * \brief This function is called to determine and set which command is being
527 * transferred to the ftp server
528 * \param thread context
529 * \param input input line of the command
530 * \param len of the command
531 * \param cmd_descriptor when the command has been parsed
532 *
533 * \retval 1 when the command is parsed, 0 otherwise
534 */
535static int FTPParseRequestCommand(FTPThreadCtx *td,
536 const uint8_t *input, uint32_t input_len,
537 const FtpCommand **cmd_descriptor)
538{
539 SCEnter();
540
541 /* I don't like this pmq reset here. We'll devise a method later, that
542 * should make the use of the mpm very efficient */
543 PmqReset(td->pmq);
544 int mpm_cnt = mpm_table[FTP_MPM].Search(ftp_mpm_ctx, td->ftp_mpm_thread_ctx,
545 td->pmq, input, input_len);
546 if (mpm_cnt) {
547 *cmd_descriptor = &FtpCommands[td->pmq->rule_id_array[0]];
548 SCReturnInt(1);
549 }
550
551 *cmd_descriptor = NULL;
552 SCReturnInt(0);
553}

577static uint32_t CopyCommandLine(uint8_t **dest, const uint8_t *src, uint32_t length)
578{
579 if (likely(length)) {
580 uint8_t *where = FTPCalloc(length + 1, sizeof(char));
581 if (unlikely(where == NULL)) {
582 return 0;
583 }
584 memcpy(where, src, length);
585
586 /* Remove trailing newlines/carriage returns */
587 while (length && isspace((unsigned char) where[length - 1])) {
588 length--;
589 }
590
591 where[length] = '\0';
592 *dest = where;
593 }
594 /* either 0 or actual */
595 return length ? length + 1 : 0;
596}

599/**
600 * \brief This function is called to retrieve a ftp request
601 * \param ftp_state the ftp state structure for the parser
602 *
603 * \retval APP_LAYER_OK when input was process successfully
604 * \retval APP_LAYER_ERROR when a unrecoverable error was encountered
605 */
606static AppLayerResult FTPParseRequest(Flow *f, void *ftp_state, AppLayerParserState *pstate,
607 StreamSlice stream_slice, void *local_data)
608{
609 FTPThreadCtx *thread_data = local_data;
610
611 SCEnter();
612 /* PrintRawDataFp(stdout, input,input_len); */
613
614 FtpState *state = (FtpState *)ftp_state;
615 void *ptmp;
616
617 const uint8_t *input = StreamSliceGetData(&stream_slice);
618 uint32_t input_len = StreamSliceGetDataLen(&stream_slice);
619
620 if (input == NULL && AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF_TS)) {
621 SCReturnStruct(APP_LAYER_OK);
622 } else if (input == NULL || input_len == 0) {
623 SCReturnStruct(APP_LAYER_ERROR);
624 }
625
626 FtpInput ftpi = { .input = input, .input_len = input_len };
627
628 uint8_t direction = STREAM_TOSERVER;
629 while (FTPGetLine(state, STREAM_TOSERVER, &ftpi) >= 0) {
630 const FtpCommand *cmd_descriptor;
631
632 if (!FTPParseRequestCommand(thread_data,
633 state->current_line, state->current_line_len,
634 &cmd_descriptor)) {
635 state->command = FTP_COMMAND_UNKNOWN;
636 continue;
637 }
638
639 state->command = cmd_descriptor->command;
640
641 FTPTransaction *tx = FTPTransactionCreate(state);
642 if (unlikely(tx == NULL))
643 SCReturnStruct(APP_LAYER_ERROR);
644 state->curr_tx = tx;
645
646 tx->command_descriptor = cmd_descriptor;
647 tx->request_length = CopyCommandLine(&tx->request,
648 state->current_line, state->current_line_len);
649 tx->request_truncated = state->current_line_truncated;
650
651 /* change direction (default to server) so expectation will handle
652 * the correct message when expectation will match.
653 * For ftp active mode, data connection direction is opposite to
654 * control direction.
655 */
656 if ((state->active && state->command == FTP_COMMAND_STOR) ||
657 (!state->active && state->command == FTP_COMMAND_RETR)) {
658 direction = STREAM_TOCLIENT;
659 }
660
661 switch (state->command) {
662 case FTP_COMMAND_EPRT:
663 // fallthrough
664 case FTP_COMMAND_PORT:
665 if (state->current_line_len + 1 > state->port_line_size) {
666 /* Allocate an extra byte for a NULL terminator */
667 ptmp = FTPRealloc(state->port_line, state->port_line_size,
668 state->current_line_len);
669 if (ptmp == NULL) {
670 if (state->port_line) {
671 FTPFree(state->port_line, state->port_line_size);
672 state->port_line = NULL;
673 state->port_line_size = 0;
674 }
675 SCReturnStruct(APP_LAYER_OK);
676 }
677 state->port_line = ptmp;
678 state->port_line_size = state->current_line_len;
679 }
680 memcpy(state->port_line, state->current_line,
681 state->current_line_len);
682 state->port_line_len = state->current_line_len;
683 break;
684 case FTP_COMMAND_RETR:
685 // fallthrough
686 case FTP_COMMAND_STOR: {
687 /* Ensure that there is a negotiated dyn port and a file
688 * name -- need more than 5 chars: cmd [4], space, <filename>
689 */
690 if (state->dyn_port == 0 || state->current_line_len < 6) {
691 SCReturnStruct(APP_LAYER_ERROR);
692 }
693 struct FtpTransferCmd *data = FTPCalloc(1, sizeof(struct FtpTransferCmd));
694 if (data == NULL)
695 SCReturnStruct(APP_LAYER_ERROR);
696 data->DFree = FtpTransferCmdFree;
697 /*
698 * Min size has been checked in FTPParseRequestCommand
699 * SC_FILENAME_MAX includes the null
700 */
701 uint32_t file_name_len = MIN(SC_FILENAME_MAX - 1, state->current_line_len - 5);
702#if SC_FILENAME_MAX > UINT16_MAX
703#error SC_FILENAME_MAX is greater than UINT16_MAX
704#endif
705 data->file_name = FTPCalloc(file_name_len + 1, sizeof(char));
706 if (data->file_name == NULL) {
707 FtpTransferCmdFree(data);
708 SCReturnStruct(APP_LAYER_ERROR);
709 }
710 data->file_name[file_name_len] = 0;
711 data->file_len = (uint16_t)file_name_len;
712 memcpy(data->file_name, state->current_line + 5, file_name_len);
713 data->cmd = state->command;
714 data->flow_id = FlowGetId(f);
715 data->direction = direction;
716 int ret = AppLayerExpectationCreate(f, direction,
717 0, state->dyn_port, ALPROTO_FTPDATA, data);
718 if (ret == -1) {
719 FtpTransferCmdFree(data);
720 SCLogDebug("No expectation created.");
721 SCReturnStruct(APP_LAYER_ERROR);
722 } else {
723 SCLogDebug("Expectation created [direction: %s, dynamic port %"PRIu16"].",
724 state->active ? "to server" : "to client",
725 state->dyn_port);
726 }
727
728 /* reset the dyn port to avoid duplicate */
729 state->dyn_port = 0;
730 /* reset active/passive indicator */
731 state->active = false;
732 }
733 break;
734 default:
735 break;
736 }
737 }
738
739 SCReturnStruct(APP_LAYER_OK);
740}

786/**
787 * \brief This function is called to retrieve a ftp response
788 * \param ftp_state the ftp state structure for the parser
789 * \param input input line of the command
790 * \param input_len length of the request
791 * \param output the resulting output
792 *
793 * \retval 1 when the command is parsed, 0 otherwise
794 */
795static AppLayerResult FTPParseResponse(Flow *f, void *ftp_state, AppLayerParserState *pstate,
796 StreamSlice stream_slice, void *local_data)
797{
798 FtpState *state = (FtpState *)ftp_state;
799
800 const uint8_t *input = StreamSliceGetData(&stream_slice);
801 uint32_t input_len = StreamSliceGetDataLen(&stream_slice);
802
803 if (unlikely(input_len == 0)) {
804 SCReturnStruct(APP_LAYER_OK);
805 }
806 FtpInput ftpi = { .input = input, .input_len = input_len };
807
808 FTPTransaction *lasttx = TAILQ_FIRST(&state->tx_list);
809 while (FTPGetLine(state, STREAM_TOCLIENT, &ftpi) >= 0) {
810 FTPTransaction *tx = FTPGetOldestTx(state, lasttx);
811 if (tx == NULL) {
812 tx = FTPTransactionCreate(state);
813 }
814 if (unlikely(tx == NULL)) {
815 SCReturnStruct(APP_LAYER_ERROR);
816 }
817 lasttx = tx;
818 if (state->command == FTP_COMMAND_UNKNOWN || tx->command_descriptor == NULL) {
819 /* unknown */
820 tx->command_descriptor = &FtpCommands[FTP_COMMAND_MAX -1];
821 }
822
823 state->curr_tx = tx;
824 uint16_t dyn_port;
825 switch (state->command) {
826 case FTP_COMMAND_AUTH_TLS:
827 if (state->current_line_len >= 4 && SCMemcmp("234 ", state->current_line, 4) == 0) {
828 AppLayerRequestProtocolTLSUpgrade(f);
829 }
830 break;
831
832 case FTP_COMMAND_EPRT:
833 dyn_port = rs_ftp_active_eprt(state->port_line, state->port_line_len);
834 if (dyn_port == 0) {
835 goto tx_complete;
836 }
837 state->dyn_port = dyn_port;
838 state->active = true;
839 tx->dyn_port = dyn_port;
840 tx->active = true;
841 SCLogDebug("FTP active mode (v6): dynamic port %"PRIu16"", dyn_port);
842 break;
843
844 case FTP_COMMAND_PORT:
845 dyn_port = rs_ftp_active_port(state->port_line, state->port_line_len);
846 if (dyn_port == 0) {
847 goto tx_complete;
848 }
849 state->dyn_port = dyn_port;
850 state->active = true;
851 tx->dyn_port = state->dyn_port;
852 tx->active = true;
853 SCLogDebug("FTP active mode (v4): dynamic port %"PRIu16"", dyn_port);
854 break;
855
856 case FTP_COMMAND_PASV:
857 if (state->current_line_len >= 4 && SCMemcmp("227 ", state->current_line, 4) == 0) {
858 FTPParsePassiveResponse(f, ftp_state, state->current_line, state->current_line_len);
859 }
860 break;
861
862 case FTP_COMMAND_EPSV:
863 if (state->current_line_len >= 4 && SCMemcmp("229 ", state->current_line, 4) == 0) {
864 FTPParsePassiveResponseV6(f, ftp_state, state->current_line, state->current_line_len);
865 }
866 break;
867 default:
868 break;
869 }
870
871 if (likely(state->current_line_len)) {
872 FTPString *response = FTPStringAlloc();
873 if (likely(response)) {
874 response->len = CopyCommandLine(&response->str, state->current_line, state->current_line_len);
875 response->truncated = state->current_line_truncated;
876 TAILQ_INSERT_TAIL(&tx->response_list, response, next);
877 }
878 }
879
880 /* Handle preliminary replies -- keep tx open */
881 if (FTPIsPPR(state->current_line, state->current_line_len)) {
882 continue;
883 }
884 tx_complete:
885 tx->done = true;
886 }
887
888 SCReturnStruct(APP_LAYER_OK);
889}

945/**
946 * \brief This function returns the oldest open transaction; if none
947 * are open, then the oldest transaction is returned
948 * \param ftp_state the ftp state structure for the parser
949 * \param starttx the ftp transaction where to start looking
950 *
951 * \retval transaction pointer when a transaction was found; NULL otherwise.
952 */
953static FTPTransaction *FTPGetOldestTx(const FtpState *ftp_state, FTPTransaction *starttx)
954{
955 if (unlikely(!ftp_state)) {
956 SCLogDebug("NULL state object; no transactions available");
957 return NULL;
958 }
959 FTPTransaction *tx = starttx;
960 FTPTransaction *lasttx = NULL;
961 while(tx != NULL) {
962 /* Return oldest open tx */
963 if (!tx->done) {
964 SCLogDebug("Returning tx %p id %"PRIu64, tx, tx->tx_id);
965 return tx;
966 }
967 /* save for the end */
968 lasttx = tx;
969 tx = TAILQ_NEXT(tx, next);
970 }
971 /* All tx are closed; return last element */
972 if (lasttx)
973 SCLogDebug("Returning OLDEST tx %p id %"PRIu64, lasttx, lasttx->tx_id);
974 return lasttx;
975}

1082/**
1083 * \brief This function is called to retrieve a ftp request
1084 * \param ftp_state the ftp state structure for the parser
1085 * \param output the resulting output
1086 *
1087 * \retval 1 when the command is parsed, 0 otherwise
1088 */
1089static AppLayerResult FTPDataParse(Flow *f, FtpDataState *ftpdata_state,
1090 AppLayerParserState *pstate, StreamSlice stream_slice, void *local_data, uint8_t direction)
1091{
1092 const uint8_t *input = StreamSliceGetData(&stream_slice);
1093 uint32_t input_len = StreamSliceGetDataLen(&stream_slice);
1094 uint16_t flags = FileFlowToFlags(f, direction) | FILE_USE_DETECT;
1095 int ret = 0;
1096 const bool eof = (flags & STREAM_TOSERVER)
1097 ? AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF_TS) != 0
1098 : AppLayerParserStateIssetFlag(pstate, APP_LAYER_PARSER_EOF_TC) != 0;
1099
1100 SCLogDebug("FTP-DATA input_len %u flags %04x dir %d/%s EOF %s", input_len, flags, direction,
1101 (direction & STREAM_TOSERVER) ? "toserver" : "toclient", eof ? "true" : "false");
1102 if (input_len && ftpdata_state->files == NULL) {
1103 struct FtpTransferCmd *data =
1104 (struct FtpTransferCmd *)FlowGetStorageById(f, AppLayerExpectationGetFlowId());
1105 if (data == NULL) {
1106 SCReturnStruct(APP_LAYER_ERROR);
1107 }
1108
1109 /* we shouldn't get data in the wrong dir. Don't set things up for this dir */
1110 if ((direction & data->direction) == 0) {
1111 // TODO set event for data in wrong direction
1112 SCLogDebug("input %u not for our direction (%s): %s/%s", input_len,
1113 (direction & STREAM_TOSERVER) ? "toserver" : "toclient",
1114 data->cmd == FTP_COMMAND_STOR ? "STOR" : "RETR",
1115 (data->direction & STREAM_TOSERVER) ? "toserver" : "toclient");
1116 SCReturnStruct(APP_LAYER_OK);
1117 }
1118
1119 ftpdata_state->files = FileContainerAlloc();
1120 if (ftpdata_state->files == NULL) {
1121 FlowFreeStorageById(f, AppLayerExpectationGetFlowId());
1122 SCReturnStruct(APP_LAYER_ERROR);
1123 }
1124
1125 ftpdata_state->file_name = data->file_name;
1126 ftpdata_state->file_len = data->file_len;
1127 data->file_name = NULL;
1128 data->file_len = 0;
1129 f->parent_id = data->flow_id;
1130 ftpdata_state->command = data->cmd;
1131 switch (data->cmd) {
1132 case FTP_COMMAND_STOR:
1133 ftpdata_state->direction = data->direction;
1134 SCLogDebug("STOR data to %s",
1135 (ftpdata_state->direction & STREAM_TOSERVER) ? "toserver" : "toclient");
1136 break;
1137 case FTP_COMMAND_RETR:
1138 ftpdata_state->direction = data->direction;
1139 SCLogDebug("RETR data to %s",
1140 (ftpdata_state->direction & STREAM_TOSERVER) ? "toserver" : "toclient");
1141 break;
1142 default:
1143 break;
1144 }
1145
1146 /* open with fixed track_id 0 as we can have just one
1147 * file per ftp-data flow. */
1148 if (FileOpenFileWithId(ftpdata_state->files, &sbcfg,
1149 0ULL, (uint8_t *) ftpdata_state->file_name,
1150 ftpdata_state->file_len,
1151 input, input_len, flags) != 0) {
1152 SCLogDebug("Can't open file");
1153 ret = -1;
1154 }
1155 FlowFreeStorageById(f, AppLayerExpectationGetFlowId());
1156 ftpdata_state->tx_data.files_opened = 1;
1157 } else {
1158 if (ftpdata_state->state == FTPDATA_STATE_FINISHED) {
1159 SCLogDebug("state is already finished");
1160 DEBUG_VALIDATE_BUG_ON(input_len); // data after state finished is a bug.
1161 SCReturnStruct(APP_LAYER_OK);
1162 }
1163 if ((direction & ftpdata_state->direction) == 0) {
1164 if (input_len) {
1165 // TODO set event for data in wrong direction
1166 }
1167 SCLogDebug("input %u not for us (%s): %s/%s", input_len,
1168 (direction & STREAM_TOSERVER) ? "toserver" : "toclient",
1169 ftpdata_state->command == FTP_COMMAND_STOR ? "STOR" : "RETR",
1170 (ftpdata_state->direction & STREAM_TOSERVER) ? "toserver" : "toclient");
1171 SCReturnStruct(APP_LAYER_OK);
1172 }
1173 if (input_len != 0) {
1174 ret = FileAppendData(ftpdata_state->files, input, input_len);
1175 if (ret == -2) {
1176 ret = 0;
1177 SCLogDebug("FileAppendData() - file no longer being extracted");
1178 goto out;
1179 } else if (ret < 0) {
1180 SCLogDebug("FileAppendData() failed: %d", ret);
1181 ret = -2;
1182 goto out;
1183 }
1184 }
1185 }
1186
1187 BUG_ON((direction & ftpdata_state->direction) == 0); // should be unreachble
1188 if (eof) {
1189 ret = FileCloseFile(ftpdata_state->files, NULL, 0, flags);
1190 ftpdata_state->state = FTPDATA_STATE_FINISHED;
1191 SCLogDebug("closed because of eof");
1192 }
1193out:
1194 if (ret < 0) {
1195 SCReturnStruct(APP_LAYER_ERROR);
1196 }
1197 SCReturnStruct(APP_LAYER_OK);
1198}

1336 *
1337 * \retval txptr or NULL if no more txs in list
1338 */
1339static AppLayerGetTxIterTuple FTPGetTxIterator(const uint8_t ipproto, const AppProto alproto,
1340 void *alstate, uint64_t min_tx_id, uint64_t max_tx_id, AppLayerGetTxIterState *state)
1341{
1342 FtpState *ftp_state = (FtpState *)alstate;
1343 AppLayerGetTxIterTuple no_tuple = { NULL, 0, false };
1344 if (ftp_state) {
1345 FTPTransaction *tx_ptr;
1346 if (state->un.ptr == NULL) {
1347 tx_ptr = TAILQ_FIRST(&ftp_state->tx_list);
1348 } else {
1349 tx_ptr = (FTPTransaction *)state->un.ptr;
1350 }
1351 if (tx_ptr) {
1352 while (tx_ptr->tx_id < min_tx_id) {
1353 tx_ptr = TAILQ_NEXT(tx_ptr, next);
1354 if (!tx_ptr) {
1355 return no_tuple;
1356 }
1357 }
1358 if (tx_ptr->tx_id >= max_tx_id) {
1359 return no_tuple;
1360 }
1361 state->un.ptr = TAILQ_NEXT(tx_ptr, next);
1362 AppLayerGetTxIterTuple tuple = {
1363 .tx_ptr = tx_ptr,
1364 .tx_id = tx_ptr->tx_id,
1365 .has_next = (state->un.ptr != NULL),
1366 };
1367 return tuple;
1368 }
1369 }
1370 return no_tuple;

1459/*
1460 * \brief Returns the ending offset of the next line from a multi-line buffer.
1461 *
1462 * "Buffer" refers to a FTP response in a single buffer containing multiple lines.
1463 * Here, "next line" is defined as terminating on
1464 * - Newline character
1465 * - Null character
1466 *
1467 * \param buffer Contains zero or more characters.
1468 * \param len Size, in bytes, of buffer.
1469 *
1470 * \retval Offset from the start of buffer indicating the where the
1471 * next "line ends". The characters between the input buffer and this
1472 * value comprise the line.
1473 *
1474 * NULL is found first or a newline isn't found, then UINT16_MAX is returned.
1475 */
1476uint16_t JsonGetNextLineFromBuffer(const char *buffer, const uint16_t len)
1477{
1478 if (!buffer || *buffer == '\0') {
1479 return UINT16_MAX;
1480 }
1481
1482 char *c = strchr(buffer, '\n');
1483 return c == NULL ? len : (uint16_t)(c - buffer + 1);
1484}
