65#ifndef LANDLOCK_ACCESS_FS_REFER
66#define LANDLOCK_ACCESS_FS_REFER (1ULL << 13)

69#define _LANDLOCK_ACCESS_FS_WRITE \
70 (LANDLOCK_ACCESS_FS_WRITE_FILE | LANDLOCK_ACCESS_FS_REMOVE_DIR | \

83struct landlock_ruleset {
84 int fd;
85 struct landlock_ruleset_attr attr;
86};
87
88static inline struct landlock_ruleset *LandlockCreateRuleset(void)
89{
90 struct landlock_ruleset *ruleset = SCCalloc(1, sizeof(struct landlock_ruleset));
91 if (ruleset == NULL) {
92 SCLogError(SC_ERR_MEM_ALLOC, "Can't alloc landlock ruleset");
93 return NULL;
94 }
95
96 ruleset->attr.handled_access_fs =
97 _LANDLOCK_ACCESS_FS_READ | _LANDLOCK_ACCESS_FS_WRITE | LANDLOCK_ACCESS_FS_EXECUTE;
98
99 int abi = landlock_create_ruleset(NULL, 0, LANDLOCK_CREATE_RULESET_VERSION);
100 if (abi < 0) {
101 SCFree(ruleset);
102 return NULL;
103 }
104 if (abi < 2) {
105 if (RequiresFeature(FEATURE_OUTPUT_FILESTORE)) {
106 SCLogError(SC_ERR_NOT_SUPPORTED,
107 "Landlock disabled: need Linux 5.19+ for file store support");
108 SCFree(ruleset);
109 return NULL;
110 } else {
111 ruleset->attr.handled_access_fs &= ~LANDLOCK_ACCESS_FS_REFER;
112 }
113 }
114
115 ruleset->fd = landlock_create_ruleset(&ruleset->attr, sizeof(ruleset->attr), 0);
116 if (ruleset->fd < 0) {
117 SCFree(ruleset);
118 SCLogError(SC_ERR_CONF_YAML_ERROR, "Can't create landlock ruleset");
119 return NULL;
120 }
121 return ruleset;
122}

124static inline void LandlockEnforceRuleset(struct landlock_ruleset *ruleset)
125{
126 if (prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0) == -1) {
127 SCLogError(
128 SC_ERR_CONF_YAML_ERROR, "Can't self restrict (prctl phase): %s", strerror(errno));
129 return;
130 }
131 if (landlock_restrict_self(ruleset->fd, 0)) {
132 SCLogError(SC_ERR_CONF_YAML_ERROR, "Can't self restrict (landlock phase): %s",
133 strerror(errno));
134 }
135}

137static int LandlockSandboxingAddRule(
138 struct landlock_ruleset *ruleset, const char *directory, uint64_t permission)
139{
140 struct landlock_path_beneath_attr path_beneath = {
141 .allowed_access = permission & ruleset->attr.handled_access_fs,
142 };
143
144 int dir_fd = open(directory, O_PATH | O_CLOEXEC | O_DIRECTORY);
145 if (dir_fd == -1) {
146 SCLogError(SC_ERR_CONF_YAML_ERROR, "Can't open %s", directory);
147 return -1;
148 }
149 path_beneath.parent_fd = dir_fd;
150
151 if (landlock_add_rule(ruleset->fd, LANDLOCK_RULE_PATH_BENEATH, &path_beneath, 0)) {
152 SCLogError(SC_ERR_CONF_YAML_ERROR, "Can't add write rule: %s", strerror(errno));
153 close(dir_fd);
154 return -1;
155 }
156
157 close(dir_fd);
158 return 0;
159}

161static inline void LandlockSandboxingWritePath(
162 struct landlock_ruleset *ruleset, const char *directory)
163{
164 if (LandlockSandboxingAddRule(ruleset, directory, _LANDLOCK_SURI_ACCESS_FS_WRITE) == 0) {
165 SCLogConfig("Added write permission to '%s'", directory);
166 }

169static inline void LandlockSandboxingReadPath(
170 struct landlock_ruleset *ruleset, const char *directory)

172 if (LandlockSandboxingAddRule(ruleset, directory, _LANDLOCK_ACCESS_FS_READ) == 0) {
173 SCLogConfig("Added read permission to '%s'", directory);

177void LandlockSandboxing(SCInstance *suri)
178{
179 /* Read configuration variable and exit if no enforcement */
180 int conf_status;
181 ConfGetBool("security.landlock.enabled", &conf_status);
182 if (!conf_status) {
183 SCLogConfig("Landlock is not enabled in configuration");
184 return;
185 }
186 struct landlock_ruleset *ruleset = LandlockCreateRuleset();
187 if (ruleset == NULL) {
188 SCLogError(SC_ERR_NOT_SUPPORTED, "Kernel does not support Landlock");
189 return;
190 }
191
192 LandlockSandboxingWritePath(ruleset, ConfigGetLogDirectory());
193 struct stat sb;
194 if (stat(ConfigGetDataDirectory(), &sb) == 0) {
195 LandlockSandboxingAddRule(ruleset, ConfigGetDataDirectory(),
196 _LANDLOCK_SURI_ACCESS_FS_WRITE | _LANDLOCK_ACCESS_FS_READ);
197 }
198 if (suri->run_mode == RUNMODE_PCAP_FILE) {
199 const char *pcap_file;
200 ConfGet("pcap-file.file", &pcap_file);
201 char *file_name = SCStrdup(pcap_file);
202 if (file_name != NULL) {
203 struct stat statbuf;
204 if (stat(file_name, &statbuf) != -1) {
205 if (S_ISDIR(statbuf.st_mode)) {
206 LandlockSandboxingReadPath(ruleset, file_name);
207 } else {
208 LandlockSandboxingReadPath(ruleset, dirname(file_name));
209 }
210 } else {
211 SCLogError(SC_ERR_OPENING_FILE, "Can't open pcap file");
212 }
213 SCFree(file_name);
214 }
215 }
216 if (suri->sig_file) {
217 char *file_name = SCStrdup(suri->sig_file);
218 if (file_name != NULL) {
219 LandlockSandboxingReadPath(ruleset, dirname(file_name));
220 SCFree(file_name);
221 }
222 }
223 if (suri->pid_filename) {
224 char *file_name = SCStrdup(suri->pid_filename);
225 if (file_name != NULL) {
226 LandlockSandboxingWritePath(ruleset, dirname(file_name));
227 SCFree(file_name);
228 }
229 }
230 if (ConfUnixSocketIsEnable()) {
231 const char *socketname;
232 if (ConfGet("unix-command.filename", &socketname) == 1) {
233 if (PathIsAbsolute(socketname)) {
234 char *file_name = SCStrdup(socketname);
235 if (file_name != NULL) {
236 LandlockSandboxingWritePath(ruleset, dirname(file_name));
237 SCFree(file_name);
238 }
239 } else {
240 LandlockSandboxingWritePath(ruleset, LOCAL_STATE_DIR "/run/suricata/");
241 }
242 } else {
243 LandlockSandboxingWritePath(ruleset, LOCAL_STATE_DIR "/run/suricata/");
244 }
245 }
246 if (suri->sig_file_exclusive == FALSE) {
247 const char *rule_path;
248 ConfGet("default-rule-path", &rule_path);
249 if (rule_path) {
250 LandlockSandboxingReadPath(ruleset, rule_path);
251 }
252 }
253
254 ConfNode *read_dirs = ConfGetNode("security.landlock.directories.read");
255 if (read_dirs) {
256 if (!ConfNodeIsSequence(read_dirs)) {
257 SCLogWarning(SC_ERR_INVALID_ARGUMENT,
258 "Invalid security.landlock.directories.read configuration section: "
259 "expected a list of directory names.");
260 } else {
261 ConfNode *directory;
262 TAILQ_FOREACH (directory, &read_dirs->head, next) {
263 LandlockSandboxingReadPath(ruleset, directory->val);
264 }
265 }
266 }
267 ConfNode *write_dirs = ConfGetNode("security.landlock.directories.write");
268 if (write_dirs) {
269 if (!ConfNodeIsSequence(write_dirs)) {
270 SCLogWarning(SC_ERR_INVALID_ARGUMENT,
271 "Invalid security.landlock.directories.write configuration section: "
272 "expected a list of directory names.");
273 } else {
274 ConfNode *directory;
275 TAILQ_FOREACH (directory, &write_dirs->head, next) {
276 LandlockSandboxingWritePath(ruleset, directory->val);
277 }
278 }
279 }
280 LandlockEnforceRuleset(ruleset);
281 SCFree(ruleset);
282}
