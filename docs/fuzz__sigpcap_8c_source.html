92 initialized = 1;
93 }

95 /* TODO add yaml config
96 for (pos = 0; pos < size; pos++) {
97 if (data[pos] == 0) {

111 for (pos=0; pos < size; pos++) {
112 if (data[pos] == 0) {
113 break;
114 }
115 }
116 if (pos > 0 && pos < size) {
117 // dump signatures to a file so as to reuse SigLoadSignatures
118 if (TestHelperBufferToFile(surifuzz.sig_file, data, pos-1) < 0) {
119 return 0;
120 }
121 } else {
122 if (TestHelperBufferToFile(surifuzz.sig_file, data, pos) < 0) {
123 return 0;
124 }
125 }
126
127 if (DetectEngineReload(&surifuzz) < 0) {
128 return 0;
129 }
130 DetectEngineThreadCtx *old_det_ctx = FlowWorkerGetDetectCtxPtr(fwd);
131
132 DetectEngineCtx *de_ctx = DetectEngineGetCurrent();
133 de_ctx->ref_cnt--;
134 DetectEngineThreadCtx *new_det_ctx = DetectEngineThreadCtxInitForReload(&tv, de_ctx, 1);
135 FlowWorkerReplaceDetectCtx(fwd, new_det_ctx);
136
137 DetectEngineThreadCtxDeinit(NULL, old_det_ctx);
138
139 if (pos < size) {
140 //skip zero
141 pos++;
142 }
143 data += pos;
144 size -= pos;
145
146 //rewrite buffer to a file as libpcap does not have buffer inputs
147 if (TestHelperBufferToFile("/tmp/fuzz.pcap", data, size) < 0) {
148 return 0;
149 }
150
151 //initialize structure
152 pkts = pcap_open_offline("/tmp/fuzz.pcap", errbuf);
153 if (pkts == NULL) {
154 return 0;
155 }
156
157 //loop over packets
158 r = pcap_next_ex(pkts, &header, &pkt);
159 p = PacketGetFromAlloc();
160 p->ts.tv_sec = header->ts.tv_sec;
161 p->ts.tv_usec = header->ts.tv_usec % 1000000;
162 p->datalink = pcap_datalink(pkts);
163 while (r > 0) {
164 if (PacketCopyData(p, pkt, header->caplen) == 0) {
165 // DecodePcapFile
166 TmEcode ecode = tmm_modules[TMM_DECODEPCAPFILE].Func(&tv, p, dtv);
167 if (ecode == TM_ECODE_FAILED) {
168 break;
169 }
170 Packet *extra_p = PacketDequeueNoLock(&tv.decode_pq);
171 while (extra_p != NULL) {
172 PacketFreeOrRelease(extra_p);
173 extra_p = PacketDequeueNoLock(&tv.decode_pq);
174 }
175 tmm_modules[TMM_FLOWWORKER].Func(&tv, p, fwd);
176 extra_p = PacketDequeueNoLock(&tv.decode_pq);
177 while (extra_p != NULL) {
178 PacketFreeOrRelease(extra_p);
179 extra_p = PacketDequeueNoLock(&tv.decode_pq);
180 }
181 }
182 r = pcap_next_ex(pkts, &header, &pkt);
183 PACKET_RECYCLE(p);
184 p->ts.tv_sec = header->ts.tv_sec;
185 p->ts.tv_usec = header->ts.tv_usec % 1000000;
186 p->datalink = pcap_datalink(pkts);
187 pcap_cnt++;
188 p->pcap_cnt = pcap_cnt;
189 }
190 //close structure
191 pcap_close(pkts);
192 PacketFree(p);
193 FlowReset();
194
195 return 0;
196}
