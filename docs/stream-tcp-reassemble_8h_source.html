138static inline bool STREAM_LASTACK_GT_BASESEQ(const TcpStream *stream)
139{
140 /* last ack not yet initialized */
141 if (STREAM_BASE_OFFSET(stream) == 0 && (stream->tcp_flags & TH_ACK) == 0) {
142#ifdef UNITTESTS
143 if (RunmodeIsUnittests() && stream->last_ack == 0)
144 return false;
145#else
146 return false;
147#endif
148 }
149 if (SEQ_GT(stream->last_ack, stream->base_seq))
150 return true;
151 return false;
152}
