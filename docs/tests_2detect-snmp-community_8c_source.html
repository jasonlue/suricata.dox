27static int DetectSNMPCommunityTest(void)
28{
29 AppLayerParserThreadCtx *alp_tctx = AppLayerParserThreadCtxAlloc();
30 DetectEngineThreadCtx *det_ctx = NULL;
31 DetectEngineCtx *de_ctx = NULL;
32 Flow f;
33 Packet *p;
34 TcpSession tcp;
35 ThreadVars tv;
36 Signature *s;
37
38 uint8_t request[] = {
39 0x30, 0x27, 0x02, 0x01, 0x01, 0x04, 0x0b, 0x5b,
40 0x52, 0x30, 0x5f, 0x43, 0x40, 0x63, 0x74, 0x69,
41 0x21, 0x5d, 0xa1, 0x15, 0x02, 0x04, 0x2b, 0x13,
42 0x3f, 0x85, 0x02, 0x01, 0x00, 0x02, 0x01, 0x00,
43 0x30, 0x07, 0x30, 0x05, 0x06, 0x01, 0x01, 0x05,
44 0x00
45 };
46
47 /* Setup flow. */
48 memset(&f, 0, sizeof(Flow));
49 memset(&tcp, 0, sizeof(TcpSession));
50 memset(&tv, 0, sizeof(ThreadVars));
51 p = UTHBuildPacket(request, sizeof(request), IPPROTO_UDP);
52 FLOW_INITIALIZE(&f);
53 f.alproto = ALPROTO_SNMP;
54 f.protoctx = (void *)&tcp;
55 f.proto = IPPROTO_UDP;
56 f.protomap = FlowGetProtoMapping(f.proto);
57 f.flags |= FLOW_IPV4;
58 p->flow = &f;
59 p->flags |= PKT_HAS_FLOW | PKT_STREAM_EST;
60 p->flowflags |= FLOW_PKT_TOSERVER | FLOW_PKT_ESTABLISHED;
61 StreamTcpInitConfig(true);
62
63 de_ctx = DetectEngineCtxInit();
64 FAIL_IF_NULL(de_ctx);
65
66 /* This rule should match. */
67 s = DetectEngineAppendSig(de_ctx,
68 "alert snmp any any -> any any ("
69 "msg:\"SNMP Test Rule\"; "
70 "snmp.community; content:\"[R0_C@cti!]\"; "
71 "sid:1; rev:1;)");
72 FAIL_IF_NULL(s);
73
74 /* This rule should not match. */
75 s = DetectEngineAppendSig(de_ctx,
76 "alert snmp any any -> any any ("
77 "msg:\"SNMP Test Rule\"; "
78 "snmp.community; content:\"private\"; "
79 "sid:2; rev:1;)");
80 FAIL_IF_NULL(s);
81
82 SigGroupBuild(de_ctx);
83 DetectEngineThreadCtxInit(&tv, (void *)de_ctx, (void *)&det_ctx);
84
85 int r = AppLayerParserParse(NULL, alp_tctx, &f, ALPROTO_SNMP,
86 STREAM_TOSERVER, request, sizeof(request));
87 FAIL_IF(r != 0);
88
89 /* Check that we have app-layer state. */
90 FAIL_IF_NULL(f.alstate);
91
92 SigMatchSignatures(&tv, de_ctx, det_ctx, p);
93 FAIL_IF(!PacketAlertCheck(p, 1));
94 FAIL_IF(PacketAlertCheck(p, 2));
95
96 /* Cleanup. */
97 AppLayerParserThreadCtxFree(alp_tctx);
98 DetectEngineThreadCtxDeinit(&tv, det_ctx);
99 SigGroupCleanup(de_ctx);
100 DetectEngineCtxFree(de_ctx);
101 StreamTcpFreeConfig(true);
102 FLOW_DESTROY(&f);
103 UTHFreePacket(p);
104
105 PASS;
106}
