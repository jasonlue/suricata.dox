95int TLSGetIPInformations(const Packet *p, char* srcip, size_t srcip_len,
96 Port* sp, char* dstip, size_t dstip_len, Port* dp,
97 int ipproto)
98{
99 if ((PKT_IS_TOSERVER(p))) {
100 switch (ipproto) {
101 case AF_INET:
102 PrintInet(AF_INET, (const void *) GET_IPV4_SRC_ADDR_PTR(p),
103 srcip, srcip_len);
104 PrintInet(AF_INET, (const void *) GET_IPV4_DST_ADDR_PTR(p),
105 dstip, dstip_len);
106 break;
107 case AF_INET6:
108 PrintInet(AF_INET6, (const void *) GET_IPV6_SRC_ADDR(p), srcip,
109 srcip_len);
110 PrintInet(AF_INET6, (const void *) GET_IPV6_DST_ADDR(p), dstip,
111 dstip_len);
112 break;
113 default:
114 return 0;
115 }
116 *sp = p->sp;
117 *dp = p->dp;
118 } else {
119 switch (ipproto) {
120 case AF_INET:
121 PrintInet(AF_INET, (const void *) GET_IPV4_DST_ADDR_PTR(p),
122 srcip, srcip_len);
123 PrintInet(AF_INET, (const void *) GET_IPV4_SRC_ADDR_PTR(p),
124 dstip, dstip_len);
125 break;
126 case AF_INET6:
127 PrintInet(AF_INET6, (const void *) GET_IPV6_DST_ADDR(p), srcip,
128 srcip_len);
129 PrintInet(AF_INET6, (const void *) GET_IPV6_SRC_ADDR(p), dstip,
130 dstip_len);
131 break;
132 default:
133 return 0;
134 }
135 *sp = p->dp;
136 *dp = p->sp;
137 }
138 return 1;
139}

141static TmEcode LogTlsLogThreadInit(ThreadVars *t, const void *initdata,
142 void **data)
143{
144 LogTlsLogThread *aft = SCMalloc(sizeof(LogTlsLogThread));
145 if (unlikely(aft == NULL))
146 return TM_ECODE_FAILED;
147
148 memset(aft, 0, sizeof(LogTlsLogThread));
149
150 if (initdata == NULL) {
151 SCLogDebug("Error getting context for TLSLog. \"initdata\" argument NULL");
152 SCFree(aft);
153 return TM_ECODE_FAILED;
154 }
155
156 aft->buffer = MemBufferCreateNew(OUTPUT_BUFFER_SIZE);
157 if (aft->buffer == NULL) {
158 SCFree(aft);
159 return TM_ECODE_FAILED;
160 }
161
162 /* Use the Output Context (file pointer and mutex) */
163 aft->tlslog_ctx = ((OutputCtx *) initdata)->data;
164
165 *data = (void *)aft;
166 return TM_ECODE_OK;
167}

169static TmEcode LogTlsLogThreadDeinit(ThreadVars *t, void *data)
170{
171 LogTlsLogThread *aft = (LogTlsLogThread *)data;
172 if (aft == NULL) {
173 return TM_ECODE_OK;
174 }
175
176 MemBufferFree(aft->buffer);
177 memset(aft, 0, sizeof(LogTlsLogThread));
178
179 SCFree(aft);
180 return TM_ECODE_OK;

202/** \brief Create a new tls log LogFileCtx.
203 * \param conf Pointer to ConfNode containing this loggers configuration.
204 * \return NULL if failure, LogFileCtx* to the file_ctx if succesful
205 * */
206static OutputInitResult LogTlsLogInitCtx(ConfNode *conf)
207{
208 OutputInitResult result = { NULL, false };
209 LogFileCtx* file_ctx = LogFileNewCtx();
210
211 if (file_ctx == NULL) {
212 SCLogError(SC_ERR_TLS_LOG_GENERIC, "LogTlsLogInitCtx: Couldn't "
213 "create new file_ctx");
214 return result;
215 }
216
217 if (SCConfLogOpenGeneric(conf, file_ctx, DEFAULT_LOG_FILENAME, 1) < 0) {
218 goto filectx_error;
219 }
220
221 LogTlsFileCtx *tlslog_ctx = SCCalloc(1, sizeof(LogTlsFileCtx));
222 if (unlikely(tlslog_ctx == NULL)) {
223 goto filectx_error;
224 }
225 tlslog_ctx->file_ctx = file_ctx;
226
227 const char *extended = ConfNodeLookupChildValue(conf, "extended");
228 const char *custom = ConfNodeLookupChildValue(conf, "custom");
229 const char *customformat = ConfNodeLookupChildValue(conf, "customformat");
230
231 /* If custom logging format is selected, lets parse it */
232 if (custom != NULL && customformat != NULL && ConfValIsTrue(custom)) {
233 tlslog_ctx->cf = LogCustomFormatAlloc();
234 if (!tlslog_ctx->cf) {
235 goto tlslog_error;
236 }
237
238 tlslog_ctx->flags |= LOG_TLS_CUSTOM;
239
240 if (!LogCustomFormatParse(tlslog_ctx->cf, customformat)) {
241 goto parser_error;
242 }
243 } else {
244 if (extended == NULL) {
245 tlslog_ctx->flags |= LOG_TLS_DEFAULT;
246 } else {
247 if (ConfValIsTrue(extended)) {
248 tlslog_ctx->flags |= LOG_TLS_EXTENDED;
249 }
250 }
251 }
252
253 const char *resumption = ConfNodeLookupChildValue(conf,
254 "session-resumption");
255 if (resumption == NULL || ConfValIsTrue(resumption)) {
256 tlslog_ctx->flags |= LOG_TLS_SESSION_RESUMPTION;
257 }
258
259 OutputCtx *output_ctx = SCCalloc(1, sizeof(OutputCtx));
260 if (unlikely(output_ctx == NULL)) {
261 goto tlslog_error;
262 }
263 output_ctx->data = tlslog_ctx;
264 output_ctx->DeInit = LogTlsLogDeInitCtx;
265
266 SCLogDebug("TLS log output initialized");
267
268 /* Enable the logger for the app layer */
269 AppLayerParserRegisterLogger(IPPROTO_TCP, ALPROTO_TLS);
270
271 result.ctx = output_ctx;
272 result.ok = true;
273 return result;
274
275parser_error:
276 SCLogError(SC_ERR_INVALID_ARGUMENT, "Syntax error in custom tls log "
277 "format string.");
278tlslog_error:
279 LogCustomFormatFree(tlslog_ctx->cf);
280 SCFree(tlslog_ctx);
281filectx_error:
282 LogFileFreeCtx(file_ctx);
283 return result;
284}

309static void LogTlsLogBasic(LogTlsLogThread *aft, SSLState *ssl_state,
310 const struct timeval *ts, char *srcip, Port sp,
311 char *dstip, Port dp)
312{
313 char timebuf[64];
314 CreateTimeString(ts, timebuf, sizeof(timebuf));
315 MemBufferWriteString(aft->buffer,
316 "%s %s:%d -> %s:%d TLS:",
317 timebuf, srcip, sp, dstip, dp);
318
319 if (ssl_state->server_connp.cert0_subject != NULL) {
320 MemBufferWriteString(aft->buffer, " Subject='%s'",
321 ssl_state->server_connp.cert0_subject);
322 }
323
324 if (ssl_state->server_connp.cert0_issuerdn != NULL) {
325 MemBufferWriteString(aft->buffer, " Issuerdn='%s'",
326 ssl_state->server_connp.cert0_issuerdn);
327 }
328
329 if (ssl_state->flags & SSL_AL_FLAG_SESSION_RESUMED) {
330 /* Only log a session as 'resumed' if a certificate has not
331 been seen. */
332 if ((ssl_state->server_connp.cert0_issuerdn == NULL) &&
333 (ssl_state->server_connp.cert0_subject == NULL) &&
334 (ssl_state->flags & SSL_AL_FLAG_STATE_SERVER_HELLO) &&
335 ((ssl_state->flags & SSL_AL_FLAG_LOG_WITHOUT_CERT) == 0)) {
336 MemBufferWriteString(aft->buffer, " Session='resumed'");
337 }
338 }
339}

341static void LogTlsLogExtended(LogTlsLogThread *aft, SSLState *ssl_state,
342 const struct timeval *ts, char *srcip, Port sp,
343 char *dstip, Port dp)
344{
345 if (ssl_state->server_connp.cert0_fingerprint != NULL) {
346 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
347 LogTlsLogString(aft->buffer, "SHA1",
348 ssl_state->server_connp.cert0_fingerprint);
349 }
350 if (ssl_state->client_connp.sni != NULL) {
351 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
352 LogTlsLogString(aft->buffer, "SNI", ssl_state->client_connp.sni);
353 }
354 if (ssl_state->server_connp.cert0_serial != NULL) {
355 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
356 LogTlsLogString(aft->buffer, "SERIAL",
357 ssl_state->server_connp.cert0_serial);
358 }
359
360 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
361 LogTlsLogVersion(aft->buffer, ssl_state->server_connp.version);
362
363 if (ssl_state->server_connp.cert0_not_before != 0) {
364 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
365 LogTlsLogDate(aft->buffer, "NOTBEFORE",
366 &ssl_state->server_connp.cert0_not_before);
367 }
368 if (ssl_state->server_connp.cert0_not_after != 0) {
369 LOG_CF_WRITE_SPACE_SEPARATOR(aft->buffer);
370 LogTlsLogDate(aft->buffer, "NOTAFTER",
371 &ssl_state->server_connp.cert0_not_after);
372 }
373}

375/* Custom format logging */
376static void LogTlsLogCustom(LogTlsLogThread *aft, SSLState *ssl_state,
377 const struct timeval *ts, char *srcip, Port sp,
378 char *dstip, Port dp)
379{
380 LogTlsFileCtx *tlslog_ctx = aft->tlslog_ctx;
381 uint32_t i;
382 char buf[64];
383
384 for (i = 0; i < tlslog_ctx->cf->cf_n; i++)
385 {
386 LogCustomFormatNode *node = tlslog_ctx->cf->cf_nodes[i];
387 if (!node) /* Should never happen */
388 continue;
389
390 switch (node->type) {
391 case LOG_CF_LITERAL:
392 /* LITERAL */
393 MemBufferWriteString(aft->buffer, "%s", node->data);
394 break;
395 case LOG_CF_TIMESTAMP:
396 /* TIMESTAMP */
397 LogCustomFormatWriteTimestamp(aft->buffer, node->data, ts);
398 break;
399 case LOG_CF_TIMESTAMP_U:
400 /* TIMESTAMP USECONDS */
401 snprintf(buf, sizeof(buf), "%06u", (unsigned int) ts->tv_usec);
402 PrintRawUriBuf((char *)aft->buffer->buffer,
403 &aft->buffer->offset,
404 aft->buffer->size, (uint8_t *)buf,
405 MIN(strlen(buf),6));
406 break;
407 case LOG_CF_CLIENT_IP:
408 /* CLIENT IP ADDRESS */
409 PrintRawUriBuf((char *)aft->buffer->buffer,
410 &aft->buffer->offset, aft->buffer->size,
411 (uint8_t *)srcip,strlen(srcip));
412 break;
413 case LOG_CF_SERVER_IP:
414 /* SERVER IP ADDRESS */
415 PrintRawUriBuf((char *)aft->buffer->buffer,
416 &aft->buffer->offset, aft->buffer->size,
417 (uint8_t *)dstip, strlen(dstip));
418 break;
419 case LOG_CF_CLIENT_PORT:
420 /* CLIENT PORT */
421 MemBufferWriteString(aft->buffer, "%" PRIu16 "", sp);
422 break;
423 case LOG_CF_SERVER_PORT:
424 /* SERVER PORT */
425 MemBufferWriteString(aft->buffer, "%" PRIu16 "", dp);
426 break;
427 case LOG_TLS_CF_VERSION:
428 LogTlsLogVersion(aft->buffer, ssl_state->server_connp.version);
429 break;
430 case LOG_TLS_CF_DATE_NOT_BEFORE:
431 LogTlsLogDate(aft->buffer, "NOTBEFORE",
432 &ssl_state->server_connp.cert0_not_before);
433 break;
434 case LOG_TLS_CF_DATE_NOT_AFTER:
435 LogTlsLogDate(aft->buffer, "NOTAFTER",
436 &ssl_state->server_connp.cert0_not_after);
437 break;
438 case LOG_TLS_CF_SHA1:
439 if (ssl_state->server_connp.cert0_fingerprint != NULL) {
440 MemBufferWriteString(aft->buffer, "%s",
441 ssl_state->server_connp.cert0_fingerprint);
442 } else {
443 LOG_CF_WRITE_UNKNOWN_VALUE(aft->buffer);
444 }
445 break;
446 case LOG_TLS_CF_SNI:
447 if (ssl_state->client_connp.sni != NULL) {
448 MemBufferWriteString(aft->buffer, "%s",
449 ssl_state->client_connp.sni);
450 } else {
451 LOG_CF_WRITE_UNKNOWN_VALUE(aft->buffer);
452 }
453 break;
454 case LOG_TLS_CF_SUBJECT:
455 if (ssl_state->server_connp.cert0_subject != NULL) {
456 MemBufferWriteString(aft->buffer, "%s",
457 ssl_state->server_connp.cert0_subject);
458 } else {
459 LOG_CF_WRITE_UNKNOWN_VALUE(aft->buffer);
460 }
461 break;
462 case LOG_TLS_CF_ISSUER:
463 if (ssl_state->server_connp.cert0_issuerdn != NULL) {
464 MemBufferWriteString(aft->buffer, "%s",
465 ssl_state->server_connp.cert0_issuerdn);
466 } else {
467 LOG_CF_WRITE_UNKNOWN_VALUE(aft->buffer);
468 }
469 break;
470 case LOG_TLS_CF_EXTENDED:
471 /* Extended format */
472 LogTlsLogExtended(aft, ssl_state, ts, srcip, sp, dstip, dp);
473 break;
474 default:
475 /* NO MATCH */
476 MemBufferWriteString(aft->buffer, LOG_CF_NONE);
477 SCLogDebug("No matching parameter %%%c for custom tls log.",
478 node->type);
479 break;
480 }
481 }
482}

485static int LogTlsLogger(ThreadVars *tv, void *thread_data, const Packet *p,
486 Flow *f, void *state, void *tx, uint64_t tx_id)
487{
488 LogTlsLogThread *aft = (LogTlsLogThread *)thread_data;
489 LogTlsFileCtx *hlog = aft->tlslog_ctx;
490 int ipproto = (PKT_IS_IPV4(p)) ? AF_INET : AF_INET6;
491
492 SSLState *ssl_state = (SSLState *)state;
493 if (unlikely(ssl_state == NULL)) {
494 return 0;
495 }
496
497 if (((hlog->flags & LOG_TLS_SESSION_RESUMPTION) == 0 ||
498 (ssl_state->flags & SSL_AL_FLAG_SESSION_RESUMED) == 0) &&
499 (ssl_state->server_connp.cert0_issuerdn == NULL ||
500 ssl_state->server_connp.cert0_subject == NULL) &&
501 ((ssl_state->flags & SSL_AL_FLAG_LOG_WITHOUT_CERT) == 0)) {
502 return 0;
503 }
504
505 char srcip[PRINT_BUF_LEN], dstip[PRINT_BUF_LEN];
506
507 Port sp, dp;
508 if (!TLSGetIPInformations(p, srcip, PRINT_BUF_LEN, &sp, dstip,
509 PRINT_BUF_LEN, &dp, ipproto)) {
510 return 0;
511 }
512
513 MemBufferReset(aft->buffer);
514
515 if (hlog->flags & LOG_TLS_CUSTOM) {
516 LogTlsLogCustom(aft, ssl_state, &p->ts, srcip, sp, dstip, dp);
517 } else if (hlog->flags & LOG_TLS_EXTENDED) {
518 LogTlsLogBasic(aft, ssl_state, &p->ts, srcip, sp, dstip, dp);
519 LogTlsLogExtended(aft, ssl_state, &p->ts, srcip, sp, dstip, dp);
520 } else {
521 LogTlsLogBasic(aft, ssl_state, &p->ts, srcip, sp, dstip, dp);
522 }
523
524 MemBufferWriteString(aft->buffer, "\n");
525
526 aft->tls_cnt++;
527
528 hlog->file_ctx->Write((const char *)MEMBUFFER_BUFFER(aft->buffer),
529 MEMBUFFER_OFFSET(aft->buffer), hlog->file_ctx);
530
531 return 0;
532}
