43static int DetectKrb5TicketEncryptionSetup(
44 DetectEngineCtx *de_ctx, Signature *s, const char *krb5str)
45{
46 DetectKrb5TicketEncryptionData *krb5d = NULL;
47 SigMatch *sm = NULL;
48
49 if (DetectSignatureSetAppProto(s, ALPROTO_KRB5) != 0)
50 return -1;
51
52 krb5d = rs_krb5_detect_encryption_parse(krb5str);
53 if (krb5d == NULL)
54 goto error;
55
56 sm = SigMatchAlloc();
57 if (sm == NULL)
58 goto error;
59
60 sm->type = DETECT_AL_KRB5_TICKET_ENCRYPTION;
61 sm->ctx = (void *)krb5d;
62
63 SigMatchAppendSMToList(s, sm, g_krb5_ticket_encryption_list_id);
64
65 return 0;
66
67error:
68 if (krb5d != NULL)
69 DetectKrb5TicketEncryptionFree(de_ctx, krb5d);
70 if (sm != NULL)
71 SCFree(sm);
72 return -1;
73}
