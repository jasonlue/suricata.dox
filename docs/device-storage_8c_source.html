35/** \defgroup devicestorage Device storage API
36 *
37 * The device storage API is a per-device storage. It is a mean to extend
38 * the LiveDevice structure with arbitrary data.
39 *
40 * You have first to register the storage via LiveDevStorageRegister() during
41 * the init of your module. Then you can attach data via LiveDevSetStorageById()
42 * and access them via LiveDevGetStorageById().
43 * @{
44 */
45
46/**
47 * \brief Register a LiveDevice storage
48 *
49 * \param name the name of the storage
50 * \param size integer coding the size of the stored value (sizeof(void *) is best choice here)
51 * \param Alloc allocation function for the storage (can be null)
52 * \param Free free function for the new storage
53 *
54 * \retval The ID of the newly register storage that will be used to access data
55 *
56 * It has to be called once during the init of the sub system
57 */
58
59LiveDevStorageId LiveDevStorageRegister(const char *name, const unsigned int size,
60 void *(*Alloc)(unsigned int), void (*Free)(void *))
61{
62 int id = StorageRegister(STORAGE_DEVICE, name, size, Alloc, Free);
63 LiveDevStorageId ldsi = { .id = id };
64 return ldsi;
65}

67/**
68 * \brief Store a pointer in a given LiveDevice storage
69 *
70 * \param d a pointer to the LiveDevice
71 * \param id the id of the storage (return of HostStorageRegister() call)
72 * \param ptr pointer to the data to store

74
75int LiveDevSetStorageById(LiveDevice *d, LiveDevStorageId id, void *ptr)
76{
77 return StorageSetById((Storage *)((void *)d + sizeof(LiveDevice)), STORAGE_DEVICE, id.id, ptr);
