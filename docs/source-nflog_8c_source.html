140/**
141 * \brief NFLOG callback function
142 * This function setup a packet from a nflog message
143 */
144static int NFLOGCallback(struct nflog_g_handle *gh, struct nfgenmsg *msg,
145 struct nflog_data *nfa, void *data)
146{
147 NFLOGThreadVars *ntv = (NFLOGThreadVars *) data;
148 struct nfulnl_msg_packet_hdr *ph;
149 char *payload;
150 int ret;
151
152 /* grab a packet*/
153 Packet *p = PacketGetFromQueueOrAlloc();
154 if (p == NULL)
155 return -1;
156
157 PKT_SET_SRC(p, PKT_SRC_WIRE);
158
159 ph = nflog_get_msg_packet_hdr(nfa);
160 if (ph != NULL) {
161 p->nflog_v.hw_protocol = ph->hw_protocol;
162 }
163
164 p->nflog_v.ifi = nflog_get_indev(nfa);
165 p->nflog_v.ifo = nflog_get_outdev(nfa);
166
167 ret = nflog_get_payload(nfa, &payload);
168
169 if (ret > 0) {
170 if (ret > 65536) {
171 SCLogWarning(SC_ERR_INVALID_ARGUMENTS, "NFLOG sent too big packet");
172 SET_PKT_LEN(p, 0);
173 } else if (runmode_workers)
174 PacketSetData(p, (uint8_t *)payload, ret);
175 else
176 PacketCopyData(p, (uint8_t *)payload, ret);
177 } else if (ret == -1)
178 SET_PKT_LEN(p, 0);
179
180 ret = nflog_get_timestamp(nfa, &p->ts);
181 if (ret != 0) {
182 memset(&p->ts, 0, sizeof(struct timeval));
183 gettimeofday(&p->ts, NULL);
184 }
185
186 p->datalink = DLT_RAW;
187
188#ifdef COUNTERS
189 ntv->pkts++;
190 ntv->bytes += GET_PKT_LEN(p);
191#endif
192 (void) SC_ATOMIC_ADD(ntv->livedev->pkts, 1);
193
194 if (TmThreadsSlotProcessPkt(ntv->tv, ntv->slot, p) != TM_ECODE_OK) {
195 return -1;
196 }
197
198 return 0;
199}

201/**
202 * \brief Receives packet from a nflog group via libnetfilter_log
203 * This is a setup function for recieving packets via libnetfilter_log.
204 * \param tv pointer to ThreadVars
205 * \param initdata pointer to the group passed from the user
206 * \param data pointer gets populated with NFLOGThreadVars
207 * \retvalTM_ECODE_OK on success
208 * \retval TM_ECODE_FAILED on error
209 */
210TmEcode ReceiveNFLOGThreadInit(ThreadVars *tv, const void *initdata, void **data)
211{
212 NflogGroupConfig *nflconfig = (NflogGroupConfig *)initdata;
213
214 if (initdata == NULL) {
215 SCLogError(SC_ERR_INVALID_ARGUMENT, "initdata == NULL");
216 SCReturnInt(TM_ECODE_FAILED);
217 }
218
219 NFLOGThreadVars *ntv = SCMalloc(sizeof(NFLOGThreadVars));
220 if (unlikely(ntv == NULL)) {
221 nflconfig->DerefFunc(nflconfig);
222 SCReturnInt(TM_ECODE_FAILED);
223 }
224 memset(ntv, 0, sizeof(NFLOGThreadVars));
225
226 ntv->tv = tv;
227 ntv->group = nflconfig->group;
228 ntv->nlbufsiz = nflconfig->nlbufsiz;
229 ntv->nlbufsiz_max = nflconfig->nlbufsiz_max;
230 ntv->qthreshold = nflconfig->qthreshold;
231 ntv->qtimeout = nflconfig->qtimeout;
232 ntv->nful_overrun_warned = nflconfig->nful_overrun_warned;
233
234 ntv->h = nflog_open();
235 if (ntv->h == NULL) {
236 SCLogError(SC_ERR_NFLOG_OPEN, "nflog_open() failed");
237 SCFree(ntv);
238 return TM_ECODE_FAILED;
239 }
240
241 SCLogDebug("binding netfilter_log as nflog handler for AF_INET and AF_INET6");
242
243 if (nflog_bind_pf(ntv->h, AF_INET) < 0) {
244 FatalError(SC_ERR_FATAL, "nflog_bind_pf() for AF_INET failed");
245 }
246 if (nflog_bind_pf(ntv->h, AF_INET6) < 0) {
247 FatalError(SC_ERR_FATAL, "nflog_bind_pf() for AF_INET6 failed");
248 }
249
250 ntv->gh = nflog_bind_group(ntv->h, ntv->group);
251 if (!ntv->gh) {
252 SCLogError(SC_ERR_NFLOG_OPEN, "nflog_bind_group() failed");
253 SCFree(ntv);
254 return TM_ECODE_FAILED;
255 }
256
257 if (nflog_set_mode(ntv->gh, NFULNL_COPY_PACKET, 0xFFFF) < 0) {
258 SCLogError(SC_ERR_NFLOG_SET_MODE, "can't set packet_copy mode");
259 SCFree(ntv);
260 return TM_ECODE_FAILED;
261 }
262
263 nflog_callback_register(ntv->gh, &NFLOGCallback, (void *)ntv);
264
265 if (ntv->nlbufsiz < ntv->nlbufsiz_max)
266 ntv->nlbufsiz = nfnl_rcvbufsiz(nflog_nfnlh(ntv->h), ntv->nlbufsiz);
267 else {
268 SCLogError(SC_ERR_NFLOG_MAX_BUFSIZ, "Maximum buffer size (%d) in NFLOG "
269 "has been reached", ntv->nlbufsiz);
270 return TM_ECODE_FAILED;
271 }
272
273 if (nflog_set_qthresh(ntv->gh, ntv->qthreshold) >= 0)
274 SCLogDebug("NFLOG netlink queue threshold has been set to %d",
275 ntv->qthreshold);
276 else
277 SCLogDebug("NFLOG netlink queue threshold can't be set to %d",
278 ntv->qthreshold);
279
280 if (nflog_set_timeout(ntv->gh, ntv->qtimeout) >= 0)
281 SCLogDebug("NFLOG netlink queue timeout has been set to %d",
282 ntv->qtimeout);
283 else
284 SCLogDebug("NFLOG netlink queue timeout can't be set to %d",
285 ntv->qtimeout);
286
287 ntv->livedev = LiveGetDevice(nflconfig->numgroup);
288 if (ntv->livedev == NULL) {
289 SCLogError(SC_ERR_INVALID_VALUE, "Unable to find Live device");
290 SCFree(ntv);
291 SCReturnInt(TM_ECODE_FAILED);
292 }
293
294 /* set a timeout to the socket so we can check for a signal
295 * in case we don't get packets for a longer period. */
296 struct timeval timev;
297 timev.tv_sec = 1;
298 timev.tv_usec = 0;
299
300 int fd = nflog_fd(ntv->h);
301 if (setsockopt(fd, SOL_SOCKET, SO_RCVTIMEO, &timev, sizeof(timev)) == -1) {
302 SCLogWarning(SC_WARN_NFLOG_SETSOCKOPT, "can't set socket "
303 "timeout: %s", strerror(errno));
304 }
305
306#ifdef PACKET_STATISTICS
307 ntv->capture_kernel_packets = StatsRegisterCounter("capture.kernel_packets",
308 ntv->tv);
309 ntv->capture_kernel_drops = StatsRegisterCounter("capture.kernel_drops",
310 ntv->tv);
311#endif
312
313 char *active_runmode = RunmodeGetActive();
314 if (active_runmode && !strcmp("workers", active_runmode))
315 runmode_workers = 1;
316 else
317 runmode_workers = 0;
318
319#define T_DATA_SIZE 70000
320 ntv->data = SCMalloc(T_DATA_SIZE);
321 if (ntv->data == NULL) {
322 nflconfig->DerefFunc(nflconfig);
323 SCFree(ntv);
324 SCReturnInt(TM_ECODE_FAILED);
325 }
326
327 ntv->datalen = T_DATA_SIZE;
328#undef T_DATA_SIZE
329
330 DatalinkSetGlobalType(DLT_RAW);
331
332 *data = (void *)ntv;
333
334 nflconfig->DerefFunc(nflconfig);
335 SCReturnInt(TM_ECODE_OK);
336}

338/**
339 * \brief DeInit function unbind group and close nflog's handle
340 * \param tv pointer to ThreadVars
341 * \param data pointer that gets cast into NFLogThreadVars
342 * \retval TM_ECODE_OK is always returned
343 */
344TmEcode ReceiveNFLOGThreadDeinit(ThreadVars *tv, void *data)
345{
346 NFLOGThreadVars *ntv = (NFLOGThreadVars *)data;
347
348 SCLogDebug("closing nflog group %d", ntv->group);
349 if (nflog_unbind_pf(ntv->h, AF_INET) < 0) {
350 FatalError(SC_ERR_FATAL, "nflog_unbind_pf() for AF_INET failed");
351 }
352
353 if (nflog_unbind_pf(ntv->h, AF_INET6) < 0) {
354 FatalError(SC_ERR_FATAL, "nflog_unbind_pf() for AF_INET6 failed");
355 }
356
357 if (ntv->gh) {
358 nflog_unbind_group(ntv->gh);
359 ntv->gh = NULL;
360 }
361
362 if (ntv->h) {
363 nflog_close(ntv->h);
364 ntv->h = NULL;
365 }
366
367 if (ntv->data != NULL) {
368 SCFree(ntv->data);
369 ntv->data = NULL;
370 }
371 ntv->datalen = 0;
372
373 SCFree(ntv);
374
375 SCReturnInt(TM_ECODE_OK);
376}

378/**
379 * \brief Increases netlink buffer size
380 *
381 * This function netlink's buffer size until
382 * the max buffer size is reached
383 *
384 * \param data pointer that gets cast into NFLOGThreadVars
385 * \param size netlink buffer size
386 */
387static int NFLOGSetnlbufsiz(void *data, unsigned int size)
388{
389 SCEnter();
390 NFLOGThreadVars *ntv = (NFLOGThreadVars *)data;
391
392 if (size < ntv->nlbufsiz_max) {
393 ntv->nlbufsiz = nfnl_rcvbufsiz(nflog_nfnlh(ntv->h), ntv->nlbufsiz);
394 return 1;
395 }
396
397 SCLogWarning(SC_WARN_NFLOG_MAXBUFSIZ_REACHED,
398 "Maximum buffer size (%d) in NFLOG has been "
399 "reached. Please, consider raising "
400 "`buffer-size` and `max-size` in nflog configuration",
401 ntv->nlbufsiz);
402 return 0;
403
404}

406/**
407 * \brief Recieves packets from a group via libnetfilter_log.
408 *
409 * This function recieves packets from a group and passes
410 * the packet on to the nflog callback function.
411 *
412 * \param tv pointer to ThreadVars
413 * \param data pointer that gets cast into NFLOGThreadVars
414 * \param slot slot containing task information
415 * \retval TM_ECODE_OK on success
416 * \retval TM_ECODE_FAILED on failure
417 */
418TmEcode ReceiveNFLOGLoop(ThreadVars *tv, void *data, void *slot)
419{
420 SCEnter();
421 NFLOGThreadVars *ntv = (NFLOGThreadVars *)data;
422 int rv, fd;
423 int ret = -1;
424
425 ntv->slot = ((TmSlot *) slot)->slot_next;
426
427 fd = nflog_fd(ntv->h);
428 if (fd < 0) {
429 SCLogError(SC_ERR_NFLOG_FD, "Can't obtain a file descriptor");
430 SCReturnInt(TM_ECODE_FAILED);
431 }
432
433 while (1) {
434 if (suricata_ctl_flags != 0)
435 break;
436
437 rv = recv(fd, ntv->data, ntv->datalen, 0);
438 if (rv < 0) {
439 /*We received an error on socket read */
440 if (errno == EINTR || errno == EWOULDBLOCK) {
441 /*Nothing for us to process */
442 continue;
443 } else if (errno == ENOBUFS) {
444 if (!ntv->nful_overrun_warned) {
445 int s = ntv->nlbufsiz * 2;
446 if (NFLOGSetnlbufsiz((void *)ntv, s)) {
447 SCLogWarning(SC_WARN_NFLOG_LOSING_EVENTS,
448 "We are losing events, "
449 "increasing buffer size "
450 "to %d", ntv->nlbufsiz);
451 } else {
452 ntv->nful_overrun_warned = 1;
453 }
454 }
455 continue;
456 } else {
457 SCLogWarning(SC_WARN_NFLOG_RECV,
458 "Read from NFLOG fd failed: %s",
459 strerror(errno));
460 SCReturnInt(TM_ECODE_FAILED);
461 }
462 }
463
464 ret = nflog_handle_packet(ntv->h, ntv->data, rv);
465 if (ret != 0)
466 SCLogWarning(SC_ERR_NFLOG_HANDLE_PKT,
467 "nflog_handle_packet error %" PRId32 "", ret);
468
469 StatsSyncCountersIfSignalled(tv);
470 }
471
472 SCReturnInt(TM_ECODE_OK);
473}

490/**
491 * \brief Decode IPv4/v6 packets.
492 *
493 * \param tv pointer to ThreadVars
494 * \param p pointer to the current packet
495 * \param data pointer that gets cast into NFLOGThreadVars for ptv
496 *
497 * \retval TM_ECODE_OK is always returned
498 */
499TmEcode DecodeNFLOG(ThreadVars *tv, Packet *p, void *data)
500{
501 SCEnter();
502 IPV4Hdr *ip4h = (IPV4Hdr *)GET_PKT_DATA(p);
503 IPV6Hdr *ip6h = (IPV6Hdr *)GET_PKT_DATA(p);
504 DecodeThreadVars *dtv = (DecodeThreadVars *)data;
505
506 DecodeUpdatePacketCounters(tv, dtv, p);
507
508 if (IPV4_GET_RAW_VER(ip4h) == 4) {
509 if (unlikely(GET_PKT_LEN(p) > USHRT_MAX)) {
510 return TM_ECODE_FAILED;
511 }
512 SCLogDebug("IPv4 packet");
513 DecodeIPV4(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p));
514 } else if(IPV6_GET_RAW_VER(ip6h) == 6) {
515 if (unlikely(GET_PKT_LEN(p) > USHRT_MAX)) {
516 return TM_ECODE_FAILED;
517 }
518 SCLogDebug("IPv6 packet");
519 DecodeIPV6(tv, dtv, p, GET_PKT_DATA(p), GET_PKT_LEN(p));
520 } else {
521 SCLogDebug("packet unsupported by NFLOG, first byte: %02x", *GET_PKT_DATA(p));
522 }
523
524 PacketDecodeFinalize(tv, dtv, p);
525
526 SCReturnInt(TM_ECODE_OK);
527}
