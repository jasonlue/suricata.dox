59void *HttpHeaderThreadDataInit(void *data)
60{
61 HttpHeaderThreadData *td = SCCalloc(1, sizeof(*td));
62 if (td != NULL) {
63 if (data == NULL) {
64 td->size_step = 512;
65 } else {
66 HttpHeaderThreadDataConfig *c = data;
67 td->size_step = c->size_step;
68 }
69
70 /* initialize minimal buffers */
71 (void)HttpHeaderExpandBuffer(td, &td->buffer, 1);
72 }
73 return td;
74}

83int HttpHeaderExpandBuffer(HttpHeaderThreadData *td,
84 HttpHeaderBuffer *buf, uint32_t size)
85{
86 size_t extra = td->size_step;
87 while ((buf->size + extra) < (size + buf->len)) {
88 extra += td->size_step;
89 }
90 SCLogDebug("adding %"PRIuMAX" to the buffer", (uintmax_t)extra);
91
92 uint8_t *new_buffer = SCRealloc(buf->buffer, buf->size + extra);
93 if (unlikely(new_buffer == NULL)) {
94 buf->len = 0;
95 return -1;
96 }
97 buf->buffer = new_buffer;
98 buf->size += extra;
99 return 0;
100}
