70void DetectReplaceRegister (void)
71{
72 sigmatch_table[DETECT_REPLACE].name = "replace";
73 sigmatch_table[DETECT_REPLACE].desc = "only to be used in IPS-mode. Change the following content into another";
74 sigmatch_table[DETECT_REPLACE].url = "/rules/payload-keywords.html#replace";
75 sigmatch_table[DETECT_REPLACE].Match = DetectReplacePostMatch;
76 sigmatch_table[DETECT_REPLACE].Setup = DetectReplaceSetup;
77#ifdef UNITTESTS
78 sigmatch_table[DETECT_REPLACE].RegisterTests = DetectReplaceRegisterTests;
79#endif
80 sigmatch_table[DETECT_REPLACE].flags = (SIGMATCH_QUOTES_MANDATORY|SIGMATCH_HANDLE_NEGATION);
81}

83static int DetectReplacePostMatch(DetectEngineThreadCtx *det_ctx,
84 Packet *p, const Signature *s, const SigMatchCtx *ctx)
85{
86 if (det_ctx->replist) {
87 DetectReplaceExecuteInternal(p, det_ctx->replist);
88 det_ctx->replist = NULL;
89 }
90 return 1;
91}

93int DetectReplaceSetup(DetectEngineCtx *de_ctx, Signature *s, const char *replacestr)
94{
95 uint8_t *content = NULL;
96 uint16_t len = 0;
97
98 if (s->init_data->negated) {
99 SCLogError(SC_ERR_INVALID_VALUE, "Can't negate replacement string: %s",
100 replacestr);
101 return -1;
102 }
103
104 switch (run_mode) {
105 case RUNMODE_NFQ:
106 case RUNMODE_IPFW:
107 break;
108 default:
109 SCLogWarning(SC_ERR_RUNMODE,
110 "Can't use 'replace' keyword in non IPS mode: %s",
111 s->sig_str);
112 /* this is a success, having the alert is interesting */
113 return 0;
114 }
115
116 int ret = DetectContentDataParse("replace", replacestr, &content, &len);
117 if (ret == -1)
118 return -1;
119
120 /* add to the latest "content" keyword from pmatch */
121 const SigMatch *pm = DetectGetLastSMByListId(s, DETECT_SM_LIST_PMATCH,
122 DETECT_CONTENT, -1);
123 if (pm == NULL) {
124 SCLogError(SC_ERR_WITHIN_MISSING_CONTENT, "replace needs"
125 "preceding content option for raw sig");
126 SCFree(content);
127 return -1;
128 }
129
130 /* we can remove this switch now with the unified structure */
131 DetectContentData *ud = (DetectContentData *)pm->ctx;
132 if (ud == NULL) {
133 SCLogError(SC_ERR_INVALID_ARGUMENT, "invalid argument");
134 SCFree(content);
135 return -1;
136 }
137 if (ud->flags & DETECT_CONTENT_NEGATED) {
138 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a relative "
139 "negated keyword set along with a replacement");
140 goto error;
141 }
142 if (ud->content_len != len) {
143 SCLogError(SC_ERR_INVALID_SIGNATURE, "can't have a content "
144 "length different from replace length");
145 goto error;
146 }
147
148 ud->replace = SCMalloc(len);
149 if (ud->replace == NULL) {
150 goto error;
151 }
152 memcpy(ud->replace, content, len);
153 ud->replace_len = len;
154 ud->flags |= DETECT_CONTENT_REPLACE;
155 /* want packet matching only won't be able to replace data with
156 * a flow.
157 */
158 s->flags |= SIG_FLAG_REQUIRE_PACKET;
159 SCFree(content);
160 content = NULL;
161
162 SigMatch *sm = SigMatchAlloc();
163 if (unlikely(sm == NULL)) {
164 SCFree(ud->replace);
165 ud->replace = NULL;
166 goto error;
167 }
168 sm->type = DETECT_REPLACE;
169 sm->ctx = NULL;
170 SigMatchAppendSMToList(s, sm, DETECT_SM_LIST_POSTMATCH);
171 return 0;
172
173error:
174 SCFree(ud->replace);
175 ud->replace = NULL;
176 SCFree(content);
177 return -1;
178}
179
180/* Add to the head of the replace-list.
181 *
182 * The first to add to the replace-list has the highest priority. So,
183 * adding the the head of the list results in the newest modifications
184 * of content being applied first, so later changes can over ride
185 * earlier changes. Thus the highest priority modifications should be
186 * applied last.
187 */
188DetectReplaceList *DetectReplaceAddToList(DetectReplaceList *replist,
189 uint8_t *found,
190 DetectContentData *cd)
191{
192 DetectReplaceList *newlist;
193
194 if (cd->content_len != cd->replace_len)
195 return NULL;
196 SCLogDebug("replace: Adding match");
197
198 newlist = SCMalloc(sizeof(DetectReplaceList));
199 if (unlikely(newlist == NULL))
200 return replist;
201 newlist->found = found;
202 newlist->cd = cd;
203 /* Push new value onto the front of the list. */
204 newlist->next = replist;
205
206 return newlist;
207}

210void DetectReplaceExecuteInternal(Packet *p, DetectReplaceList *replist)
211{
212 DetectReplaceList *tlist = NULL;
213
214 SCLogDebug("replace: Executing match");
215 while (replist) {
216 memcpy(replist->found, replist->cd->replace, replist->cd->replace_len);
217 SCLogDebug("replace: replaced data");
218 p->flags |= PKT_STREAM_MODIFIED;
219 ReCalculateChecksum(p);
220 tlist = replist;
221 replist = replist->next;
222 SCFree(tlist);
223 }
224}
225
226
227void DetectReplaceFreeInternal(DetectReplaceList *replist)
228{
229 DetectReplaceList *tlist = NULL;
230 while (replist) {
231 SCLogDebug("replace: Freeing match");
232 tlist = replist;
233 replist = replist->next;
234 SCFree(tlist);
235 }
236}

240/**
241 * \test Test packet Matches
242 * \param raw_eth_pkt pointer to the ethernet packet
243 * \param pktsize size of the packet
244 * \param sig pointer to the signature to test
245 * \param sid sid number of the signature
246 * \retval return 1 if match
247 * \retval return 0 if not
248 */
249static
250int DetectReplaceLongPatternMatchTest(uint8_t *raw_eth_pkt, uint16_t pktsize,
251 const char *sig, uint32_t sid, uint8_t *pp,
252 uint16_t *len)
253{
254 int result = 0;
255
256 Packet *p = NULL;
257 p = PacketGetFromAlloc();
258 if (unlikely(p == NULL))
259 return 0;
260
261 DecodeThreadVars dtv;
262
263 ThreadVars th_v;
264 DetectEngineThreadCtx *det_ctx = NULL;
265
266 if (pp == NULL) {
267 SCLogDebug("replace: looks like a second run");
268 }
269
270 PacketCopyData(p, raw_eth_pkt, pktsize);
271 memset(&dtv, 0, sizeof(DecodeThreadVars));
272 memset(&th_v, 0, sizeof(th_v));
273 dtv.app_tctx = AppLayerGetCtxThread(&th_v);
274
275 FlowInitConfig(FLOW_QUIET);
276 DecodeEthernet(&th_v, &dtv, p, GET_PKT_DATA(p), pktsize);
277
278 DetectEngineCtx *de_ctx = DetectEngineCtxInit();
279 if (de_ctx == NULL) {
280 goto end;
281 }
282 de_ctx->flags |= DE_QUIET;
283
284 de_ctx->sig_list = SigInit(de_ctx, sig);
285 if (de_ctx->sig_list == NULL) {
286 goto end;
287 }
288 de_ctx->sig_list->next = NULL;
289
290 if (de_ctx->sig_list->sm_lists_tail[DETECT_SM_LIST_PMATCH]->type == DETECT_CONTENT) {
291 DetectContentData *co = (DetectContentData *)de_ctx->sig_list->sm_lists_tail[DETECT_SM_LIST_PMATCH]->ctx;
292 if (co->flags & DETECT_CONTENT_RELATIVE_NEXT) {
293 printf("relative next flag set on final match which is content: ");
294 goto end;
295 }
296 }
297
298 SigGroupBuild(de_ctx);
299 DetectEngineAddToMaster(de_ctx);
300 DetectEngineThreadCtxInit(&th_v, NULL, (void *)&det_ctx);
301
302 SigMatchSignatures(&th_v, de_ctx, det_ctx, p);
303 DetectEngineMoveToFreeList(de_ctx);
304
305 if (PacketAlertCheck(p, sid) != 1) {
306 SCLogDebug("replace: no alert on sig %d", sid);
307 goto end;
308 }
309
310 if (pp) {
311 memcpy(pp, GET_PKT_DATA(p), GET_PKT_LEN(p));
312 *len = pktsize;
313 SCLogDebug("replace: copying %d on %p", *len, pp);
314 }
315
316
317 result = 1;
318end:
319 if (dtv.app_tctx != NULL)
320 AppLayerDestroyCtxThread(dtv.app_tctx);
321 if (det_ctx != NULL)
322 DetectEngineThreadCtxDeinit(&th_v, (void *)det_ctx);
323 DetectEnginePruneFreeList();
324 PACKET_RECYCLE(p);
325 FlowShutdown();
326 SCFree(p);
327
328
329 return result;
330}
