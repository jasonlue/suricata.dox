55void DetectTemplateRustBufferRegister(void)
56{
57 /* TEMPLATE_START_REMOVE */
58 if (ConfGetNode("app-layer.protocols.template-rust") == NULL) {
59 return;
60 }
61 /* TEMPLATE_END_REMOVE */
62 sigmatch_table[DETECT_AL_TEMPLATE_RUST_BUFFER].name =
63 "template_rust_buffer";
64 sigmatch_table[DETECT_AL_TEMPLATE_RUST_BUFFER].desc =
65 "Template content modifier to match on the template buffers";
66 sigmatch_table[DETECT_AL_TEMPLATE_RUST_BUFFER].Setup =
67 DetectTemplateRustBufferSetup;
68#ifdef UNITTESTS
69 sigmatch_table[DETECT_AL_TEMPLATE_RUST_BUFFER].RegisterTests =
70 DetectTemplateRustBufferRegisterTests;
71#endif
72 sigmatch_table[DETECT_AL_TEMPLATE_RUST_BUFFER].flags |= SIGMATCH_NOOPT;
73
74 /* register inspect engines */
75 DetectAppLayerInspectEngineRegister2("template_rust_buffer", ALPROTO_TEMPLATE,
76 SIG_FLAG_TOSERVER, 0, DetectEngineInspectTemplateRustBuffer, NULL);
77 DetectAppLayerInspectEngineRegister2("template_rust_buffer", ALPROTO_TEMPLATE,
78 SIG_FLAG_TOCLIENT, 0, DetectEngineInspectTemplateRustBuffer, NULL);
79
80 g_template_rust_id = DetectBufferTypeGetByName("template_rust_buffer");
81
82 SCLogNotice("Template application layer detect registered.");
83}

85static int DetectTemplateRustBufferSetup(DetectEngineCtx *de_ctx, Signature *s,
86 const char *str)
87{
88 s->init_data->list = g_template_rust_id;
89
90 if (DetectSignatureSetAppProto(s, ALPROTO_TEMPLATE_RUST) != 0)
91 return -1;
92
93 return 0;
94}

96static uint8_t DetectEngineInspectTemplateRustBuffer(DetectEngineCtx *de_ctx,
97 DetectEngineThreadCtx *det_ctx, const struct DetectEngineAppInspectionEngine_ *engine,
98 const Signature *s, Flow *f, uint8_t flags, void *alstate, void *txv, uint64_t tx_id)
99{
100 uint8_t ret = 0;
101 const uint8_t *data = NULL;
102 uint32_t data_len = 0;
103
104 if (flags & STREAM_TOSERVER) {
105 rs_template_get_request_buffer(txv, &data, &data_len);
106 } else if (flags & STREAM_TOCLIENT) {
107 rs_template_get_response_buffer(txv, &data, &data_len);
108 }
109
110 if (data != NULL) {
111 ret = DetectEngineContentInspection(de_ctx, det_ctx, s, engine->smd, NULL, f,
112 (uint8_t *)data, data_len, 0, DETECT_CI_FLAGS_SINGLE,
113 DETECT_ENGINE_CONTENT_INSPECTION_MODE_STATE);
114 }
115
116 SCLogNotice("Returning %d.", ret);
117 return ret;
118}

130static int DetectTemplateRustBufferTest(void)
131{
132 AppLayerParserThreadCtx *alp_tctx = AppLayerParserThreadCtxAlloc();
133 DetectEngineThreadCtx *det_ctx = NULL;
134 DetectEngineCtx *de_ctx = NULL;
135 Flow f;
136 Packet *p;
137 TcpSession tcp;
138 ThreadVars tv;
139 Signature *s;
140
141 uint8_t request[] = "12:Hello World!";
142
143 /* Setup flow. */
144 memset(&f, 0, sizeof(Flow));
145 memset(&tcp, 0, sizeof(TcpSession));
146 memset(&tv, 0, sizeof(ThreadVars));
147 p = UTHBuildPacket(request, sizeof(request), IPPROTO_TCP);
148 FLOW_INITIALIZE(&f);
149 f.alproto = ALPROTO_TEMPLATE_RUST;
150 f.protoctx = (void *)&tcp;
151 f.proto = IPPROTO_TCP;
152 f.flags |= FLOW_IPV4;
153 p->flow = &f;
154 p->flags |= PKT_HAS_FLOW | PKT_STREAM_EST;
155 p->flowflags |= FLOW_PKT_TOSERVER | FLOW_PKT_ESTABLISHED;
156 StreamTcpInitConfig(true);
157
158 de_ctx = DetectEngineCtxInit();
159 FAIL_IF_NULL(de_ctx);
160
161 /* This rule should match. */
162 s = DetectEngineAppendSig(de_ctx,
163 "alert tcp any any -> any any ("
164 "msg:\"TEMPLATE Test Rule\"; "
165 "template_rust_buffer; content:\"World!\"; "
166 "sid:1; rev:1;)");
167 FAIL_IF_NULL(s);
168
169 /* This rule should not match. */
170 s = DetectEngineAppendSig(de_ctx,
171 "alert tcp any any -> any any ("
172 "msg:\"TEMPLATE Test Rule\"; "
173 "template_rust_buffer; content:\"W0rld!\"; "
174 "sid:2; rev:1;)");
175 FAIL_IF_NULL(s);
176
177 SigGroupBuild(de_ctx);
178 DetectEngineThreadCtxInit(&tv, (void *)de_ctx, (void *)&det_ctx);
179
180 AppLayerParserParse(NULL, alp_tctx, &f, ALPROTO_TEMPLATE_RUST,
181 STREAM_TOSERVER, request, sizeof(request));
182
183 /* Check that we have app-layer state. */
184 FAIL_IF_NULL(f.alstate);
185
186 SigMatchSignatures(&tv, de_ctx, det_ctx, p);
187 FAIL_IF(!PacketAlertCheck(p, 1));
188 FAIL_IF(PacketAlertCheck(p, 2));
189
190 /* Cleanup. */
191 if (alp_tctx != NULL)
192 AppLayerParserThreadCtxFree(alp_tctx);
193 if (det_ctx != NULL)
194 DetectEngineThreadCtxDeinit(&tv, det_ctx);
195 if (de_ctx != NULL)
196 SigGroupCleanup(de_ctx);
197 if (de_ctx != NULL)
198 DetectEngineCtxFree(de_ctx);
199 StreamTcpFreeConfig(true);
200 FLOW_DESTROY(&f);
201 UTHFreePacket(p);
202
203 PASS;
204}
