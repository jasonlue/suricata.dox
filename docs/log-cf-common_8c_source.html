90/**
91 * \brief Parses and saves format nodes for custom format
92 * \param LogCustomFormat * cf - custom format to build
93 * \param const char * format - string with format specification
94 */
95int LogCustomFormatParse(LogCustomFormat *cf, const char *format)
96{
97 const char *p, *np;
98 uint32_t n;
99 LogCustomFormatNode *node = NULL;
100
101 if (cf==NULL)
102 return 0;
103
104 if (format==NULL)
105 return 0;
106
107 p=format;
108
109 for (cf->cf_n = 0; cf->cf_n < LOG_MAXN_NODES-1 && p && *p != '\0';){
110
111 node = LogCustomFormatNodeAlloc();
112 if (node == NULL) {
113 goto parsererror;
114 }
115 node->maxlen = 0;
116
117 if (*p != '%'){
118 /* Literal found in format string */
119 node->type = LOG_CF_LITERAL;
120 np = strchr(p, '%');
121 if (np == NULL){
122 n = LOG_NODE_STRLEN-2;
123 np = NULL; /* End */
124 }else{
125 n = np-p;
126 }
127 strlcpy(node->data,p,n+1);
128 p = np;
129 } else {
130 /* Non Literal found in format string */
131 p++;
132 if (*p == '[') { /* Check if maxlength has been specified (ie: [25]) */
133 p++;
134 np = strchr(p, ']');
135 if (np != NULL) {
136 if (np-p > 0 && np-p < 10){
137 long maxlen = strtol(p,NULL,10);
138 if (maxlen > 0 && maxlen < LOG_NODE_MAXOUTPUTLEN) {
139 node->maxlen = (uint32_t) maxlen;
140 }
141 } else {
142 goto parsererror;
143 }
144 p = np + 1;
145 } else {
146 goto parsererror;
147 }
148 }
149 if (*p == '{') { /* Simple format char */
150 np = strchr(p, '}');
151 if (np != NULL && np-p > 1 && np-p < LOG_NODE_STRLEN-2) {
152 p++;
153 n = np-p;
154 strlcpy(node->data, p, n+1);
155 p = np;
156 } else {
157 goto parsererror;
158 }
159 p++;
160 } else {
161 node->data[0] = '\0';
162 }
163 node->type = *p;
164 if (*p == '%'){
165 node->type = LOG_CF_LITERAL;
166 strlcpy(node->data, "%", 2);
167 }
168 p++;
169 }
170 LogCustomFormatAddNode(cf, node);
171 }
172 return 1;
173
174parsererror:
175 LogCustomFormatNodeFree(node);
176 return 0;
177}

179/**
180 * \brief Adds a node to custom format
181 * \param LogCustomFormat * cf - custom format
182 * \param LogCustomFormatNode * node - node to add
183 */
184void LogCustomFormatAddNode(LogCustomFormat *cf, LogCustomFormatNode *node)
185{
186 if (cf == NULL || node == NULL)
187 return;
188
189 if (cf->cf_n == LOG_MAXN_NODES) {
190 SCLogWarning(SC_WARN_LOG_CF_TOO_MANY_NODES, "Too many options for custom format");
191 return;
192 }
193
194#ifdef DEBUG
195 SCLogDebug("%d-> n.type=[%d] n.maxlen=[%d] n.data=[%s]",
196 cf->cf_n, node->type, node->maxlen, node->data);
197#endif
198
199 cf->cf_nodes[cf->cf_n] = node;
200 cf->cf_n++;
201}

203/**
204 * \brief Writes a timestamp with given format into a MemBuffer
205 * \param MemBuffer * buffer - where to write
206 * \param const char * fmt - format to be used write timestamp
207 * \param const struct timeveal *ts - the timetstamp
208 *
209 */
210void LogCustomFormatWriteTimestamp(MemBuffer *buffer, const char *fmt, const struct timeval *ts) {
211
212 time_t time = ts->tv_sec;
213 struct tm local_tm;
214 struct tm *timestamp = SCLocalTime(time, &local_tm);
215 char buf[128] = {0};
216 const char * fmt_to_use = TIMESTAMP_DEFAULT_FORMAT;
217
218 if (fmt && *fmt != '\0') {
219 fmt_to_use = fmt;
220 }
221
222 CreateFormattedTimeString (timestamp, fmt_to_use, buf, sizeof(buf));
223 PrintRawUriBuf((char *)buffer->buffer, &buffer->offset,
224 buffer->size, (uint8_t *)buf,strlen(buf));
225}
